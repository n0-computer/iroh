25/// Per-nameserver timeout for a single attempt.
26const NAMESERVER_TIMEOUT: Duration = Duration::from_secs(2);
27
28/// Delay between launching queries to successive nameservers.
29/// Gives the preferred nameserver a head start before trying alternates.
30const NAMESERVER_STAGGER: Duration = Duration::from_millis(100);
31
32/// Number of UDP retry attempts per nameserver before giving up.
33/// UDP is unreliable, so a single dropped packet shouldn't be fatal.
34const UDP_ATTEMPTS: usize = 2;
35
36/// Default value for `ndots`: hostnames with at least this many dots are
37/// tried as absolute names first, before appending search domains.
38const DEFAULT_NDOTS: usize = 1;
39
40#[derive(Debug)]
41pub(super) struct SimpleDnsResolver {
42    nameservers: Vec<(SocketAddr, DnsProtocol)>,
43    search_domains: Vec<String>,
44    #[cfg(with_crypto_provider)]
45    tls_config: Option<Arc<rustls::ClientConfig>>,
46    /// Lazily initialized, cached reqwest client for DNS-over-HTTPS queries.
47    https_client: Mutex<Option<reqwest::Client>>,
48    cache: DnsCache,
49    builder: Builder,
50}

52impl SimpleDnsResolver {
53    pub(super) fn new(builder: Builder) -> Self {
54        let (nameservers, search_domains) = Self::build_config(&builder);
55        debug!(
56            nameservers = nameservers.len(),
57            search_domains = search_domains.len(),
58            "configured DNS resolver"
59        );
60        for (addr, proto) in &nameservers {
61            trace!(%addr, ?proto, "nameserver");
62        }
63        for domain in &search_domains {
64            trace!(%domain, "search domain");
65        }
66        #[cfg(with_crypto_provider)]
67        let tls_config = builder
68            .tls_client_config
69            .as_ref()
70            .map(|c| Arc::new(c.clone()));
71        Self {
72            nameservers,
73            search_domains,
74            #[cfg(with_crypto_provider)]
75            tls_config,
76            https_client: Mutex::new(None),
77            cache: DnsCache::new(),
78            builder,
79        }
80    }

82    fn build_config(builder: &Builder) -> (Vec<(SocketAddr, DnsProtocol)>, Vec<String>) {
83        let mut nameservers = Vec::new();
84        let mut search_domains = Vec::new();
85
86        if builder.use_system_defaults {
87            let config = system_config::system_config();
88            nameservers.extend(config.nameservers);
89            search_domains = config.search_domains;
90        }
91
92        nameservers.extend(builder.nameservers.iter().copied());
93
94        if nameservers.is_empty() {
95            nameservers.extend(system_config::fallback_nameservers());
96        }
97
98        (nameservers, search_domains)
99    }

101    /// Returns the list of candidate names to try for a given hostname,
102    /// applying search domain expansion per resolv.conf(5) semantics.
103    ///
104    /// - If the name ends with `.` (FQDN), it is used as-is.
105    /// - If the name has >= `ndots` dots, try the bare name first, then
106    ///   each search domain suffix.
107    /// - If the name has < `ndots` dots, try each search domain suffix
108    ///   first, then the bare name.
109    fn search_names(&self, host: &str) -> Vec<String> {
110        // Explicit FQDN: no search domain expansion.
111        if host.ends_with('.') || self.search_domains.is_empty() {
112            return vec![host.to_string()];
113        }
114
115        let dot_count = host.bytes().filter(|&b| b == b'.').count();
116        let mut names = Vec::with_capacity(self.search_domains.len() + 1);
117
118        if dot_count >= DEFAULT_NDOTS {
119            // Likely absolute -- try bare name first.
120            names.push(host.to_string());
121            for domain in &self.search_domains {
122                names.push(format!("{host}.{domain}"));
123            }
124        } else {
125            // Short name -- try search domains first.
126            for domain in &self.search_domains {
127                names.push(format!("{host}.{domain}"));
128            }
129            names.push(host.to_string());
130        }
131
132        names
133    }

135    /// Returns a clone of the cached reqwest client, creating it on first use.
136    ///
137    /// `reqwest::Client` uses an inner `Arc`, so cloning is cheap.
138    fn get_or_init_https_client(&self) -> Result<reqwest::Client, DnsError> {
139        let mut guard = self.https_client.lock().expect("poisoned");
140        if let Some(client) = guard.as_ref() {
141            return Ok(client.clone());
142        }
143        #[cfg(with_crypto_provider)]
144        let tls = self.tls_config.as_ref();
145        #[cfg(not(with_crypto_provider))]
146        let tls = None;
147        let client = transport::build_https_client(tls)?;
148        *guard = Some(client.clone());
149        Ok(client)
150    }

152    /// Run a future with [`NAMESERVER_TIMEOUT`].
153    async fn with_timeout<T>(
154        fut: impl Future<Output = Result<T, DnsError>>,
155    ) -> Result<T, DnsError> {
156        time::timeout(NAMESERVER_TIMEOUT, fut).await?
157    }

159    /// Query a single nameserver, with UDP retry and truncation fallback.
160    async fn query_nameserver(
161        &self,
162        addr: SocketAddr,
163        proto: DnsProtocol,
164        query_bytes: &[u8],
165    ) -> Result<Vec<u8>, DnsError> {
166        match proto {
167            DnsProtocol::Udp => {
168                let mut last_err = None;
169                for attempt in 0..UDP_ATTEMPTS {
170                    trace!(%addr, attempt, "sending UDP query");
171                    match Self::with_timeout(transport::udp_query(addr, query_bytes)).await {
172                        Ok(resp) if query::is_truncated(&resp) => {
173                            debug!(%addr, "UDP response truncated, retrying over TCP");
174                            return Self::with_timeout(transport::tcp_query(addr, query_bytes))
175                                .await;
176                        }
177                        Ok(resp) => return Ok(resp),
178                        Err(e) => {
179                            trace!(%addr, attempt, err = %e, "UDP query failed");
180                            last_err = Some(e);
181                        }
182                    }
183                }
184                Err(last_err.unwrap_or_else(|| e!(DnsError::NoResponse)))
185            }
186            DnsProtocol::Tcp => Self::with_timeout(transport::tcp_query(addr, query_bytes)).await,
187            #[cfg(with_crypto_provider)]
188            DnsProtocol::Tls => {
189                let tls_config = self.tls_config.as_ref().ok_or_else(|| {
190                    e!(DnsError::Transport {
191                        source: std::io::Error::other("TLS config required for DNS-over-TLS"),
192                    })
193                })?;
194                Self::with_timeout(transport::tls_query(addr, query_bytes, tls_config)).await
195            }
196            #[cfg(with_crypto_provider)]
197            DnsProtocol::Https => {
198                let client = self.get_or_init_https_client()?;
199                Self::with_timeout(transport::https_query(addr, query_bytes, &client)).await
200            }
201        }
202    }

204    /// Send a query to all nameservers in parallel with staggered starts.
205    ///
206    /// Each nameserver gets a staggered start delay to give the preferred
207    /// (first) nameserver a head start. The first successful response wins.
208    /// UDP queries are retried once per nameserver on failure.
209    async fn send_query(&self, query_bytes: &[u8]) -> Result<Vec<u8>, DnsError> {
210        if self.nameservers.is_empty() {
211            return Err(e!(DnsError::NoResponse));
212        }
213
214        let count = self.nameservers.len();
215        let mut futures = FuturesUnorderedBounded::new(count);
216
217        for (i, (addr, proto)) in self.nameservers.iter().copied().enumerate() {
218            let stagger = NAMESERVER_STAGGER * i as u32;
219            futures.push(async move {
220                if !stagger.is_zero() {
221                    time::sleep(stagger).await;
222                }
223                trace!(%addr, ?proto, "sending DNS query");
224                self.query_nameserver(addr, proto, query_bytes).await
225            });
226        }
227
228        let mut last_err = None;
229        while let Some(result) = futures.next().await {
230            match result {
231                Ok(resp) => return Ok(resp),
232                Err(e) => last_err = Some(e),
233            }
234        }
235        Err(last_err.unwrap_or_else(|| e!(DnsError::NoResponse)))
236    }

238    /// Send a query and follow CNAME chains recursively if the response contains
239    /// a CNAME but no records of the requested type.
240    async fn send_query_following_cnames(
241        &self,
242        host: String,
243        qtype: TYPE,
244    ) -> Result<(Vec<u8>, u16), DnsError> {
245        let mut current_host = host;
246        for _ in 0..MAX_CNAME_DEPTH {
247            let (id, query_bytes) = query::build_query(&current_host, qtype)?;
248            let response = self.send_query(&query_bytes).await?;
249
250            // Check if the response only has CNAMEs but no records of the
251            // requested type. If so, follow the CNAME to a new query.
252            let packet = simple_dns::Packet::parse(&response)?;
253            let has_target_records = packet.answers.iter().any(|rr| {
254                matches!(
255                    (&rr.rdata, qtype),
256                    (RData::A(_), TYPE::A)
257                        | (RData::AAAA(_), TYPE::AAAA)
258                        | (RData::TXT(_), TYPE::TXT)
259                )
260            });
261
262            if has_target_records {
263                return Ok((response, id));
264            }
265
266            // No target records -- check for a CNAME to follow.
267            match query::cname_target(&packet, &current_host) {
268                Some(target) => {
269                    debug!(
270                        from = %current_host,
271                        to = %target,
272                        "following CNAME"
273                    );
274                    current_host = target;
275                }
276                None => {
277                    // No CNAME either; return the response as-is (empty results).
278                    return Ok((response, id));
279                }
280            }
281        }
282        // Exceeded max CNAME depth.
283        Err(e!(DnsError::InvalidResponse))
284    }

286    pub(super) async fn lookup_ipv4(
287        &self,
288        host: String,
289    ) -> Result<impl Iterator<Item = Ipv4Addr> + use<>, DnsError> {
290        if let Some(CachedRecord::A(addrs)) = self.cache.get(&host, QueryType::A) {
291            trace!(%host, count = addrs.len(), "A lookup cache hit");
292            return Ok(addrs.into_iter());
293        }
294
295        let mut last_err = None;
296        for name in self.search_names(&host) {
297            trace!(%name, "resolving A record");
298            match self.send_query_following_cnames(name, TYPE::A).await {
299                Ok((response, id)) => {
300                    let (addrs, ttl) = query::parse_a_response(&response, id)?;
301                    if !addrs.is_empty() {
302                        debug!(%host, count = addrs.len(), ttl, "resolved A record");
303                        self.cache
304                            .insert(&host, QueryType::A, CachedRecord::A(addrs.clone()), ttl);
305                        return Ok(addrs.into_iter());
306                    }
307                }
308                Err(DnsError::NxDomain { .. }) => {
309                    last_err = Some(e!(DnsError::NxDomain));
310                }
311                Err(e) => return Err(e),
312            }
313        }
314        Err(last_err.unwrap_or_else(|| e!(DnsError::NoResponse)))
315    }

317    pub(super) async fn lookup_ipv6(
318        &self,
319        host: String,
320    ) -> Result<impl Iterator<Item = Ipv6Addr> + use<>, DnsError> {
321        if let Some(CachedRecord::Aaaa(addrs)) = self.cache.get(&host, QueryType::AAAA) {
322            trace!(%host, count = addrs.len(), "AAAA lookup cache hit");
323            return Ok(addrs.into_iter());
324        }
325
326        let mut last_err = None;
327        for name in self.search_names(&host) {
328            trace!(%name, "resolving AAAA record");
329            match self.send_query_following_cnames(name, TYPE::AAAA).await {
330                Ok((response, id)) => {
331                    let (addrs, ttl) = query::parse_aaaa_response(&response, id)?;
332                    if !addrs.is_empty() {
333                        debug!(%host, count = addrs.len(), ttl, "resolved AAAA record");
334                        self.cache.insert(
335                            &host,
336                            QueryType::AAAA,
337                            CachedRecord::Aaaa(addrs.clone()),
338                            ttl,
339                        );
340                        return Ok(addrs.into_iter());
341                    }
342                }
343                Err(DnsError::NxDomain { .. }) => {
344                    last_err = Some(e!(DnsError::NxDomain));
345                }
346                Err(e) => return Err(e),
347            }
348        }
349        Err(last_err.unwrap_or_else(|| e!(DnsError::NoResponse)))
350    }

352    pub(super) async fn lookup_txt(
353        &self,
354        host: String,
355    ) -> Result<impl Iterator<Item = TxtRecordData> + use<>, DnsError> {
356        {
357            if let Some(CachedRecord::Txt(records)) = self.cache.get(&host, QueryType::TXT) {
358                trace!(%host, count = records.len(), "TXT lookup cache hit");
359                return Ok(records.into_iter());
360            }
361        }
362
363        let mut last_err = None;
364        for name in self.search_names(&host) {
365            trace!(%name, "resolving TXT record");
366            match self.send_query_following_cnames(name, TYPE::TXT).await {
367                Ok((response, id)) => {
368                    let (records, ttl) = query::parse_txt_response(&response, id)?;
369                    if !records.is_empty() {
370                        debug!(%host, count = records.len(), ttl, "resolved TXT record");
371                        self.cache.insert(
372                            &host,
373                            QueryType::TXT,
374                            CachedRecord::Txt(records.clone()),
375                            ttl,
376                        );
377                        return Ok(records.into_iter());
378                    }
379                }
380                Err(DnsError::NxDomain { .. }) => {
381                    last_err = Some(e!(DnsError::NxDomain));
382                }
383                Err(e) => return Err(e),
384            }
385        }
386        Err(last_err.unwrap_or_else(|| e!(DnsError::NoResponse)))
387    }

389    pub(super) fn clear_cache(&self) {
390        self.cache.clear();
391    }

393    pub(super) fn reset(&mut self) {
394        let (nameservers, search_domains) = Self::build_config(&self.builder);
395        self.nameservers = nameservers;
396        self.search_domains = search_domains;
397        #[cfg(with_crypto_provider)]
398        {
399            self.tls_config = self
400                .builder
401                .tls_client_config
402                .as_ref()
403                .map(|c| Arc::new(c.clone()));
404        }
405        // Clear cached HTTPS client so it gets rebuilt with the new TLS config on next use
406        *self.https_client.lock().expect("poisoned") = None;
407        self.clear_cache();
408    }

426    fn with_proto(addr: SocketAddr, proto: DnsProtocol) -> DnsResolver {
427        let mut builder = DnsResolver::builder().with_nameserver(addr, proto);
428        if proto == DnsProtocol::Tls {
429            let root_store =
430                rustls::RootCertStore::from_iter(webpki_roots::TLS_SERVER_ROOTS.iter().cloned());
431            builder = builder.tls_client_config(
432                rustls::ClientConfig::builder()
433                    .with_root_certificates(root_store)
434                    .with_no_client_auth(),
435            );
436        }
437        builder.build()
438    }
