1//! Simple TTL-based DNS cache using LRU eviction.
2
3use std::{
4    hash::{Hash, Hasher},
5    net::{Ipv4Addr, Ipv6Addr},
6    time::{Duration, Instant},
7};
8
9use lru::LruCache;
10
11use super::TxtRecordData;
12
13/// Maximum number of entries in the DNS cache.
14const MAX_CACHE_ENTRIES: usize = 512;
15
16/// Query type key for cache entries.
17#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
18#[allow(clippy::upper_case_acronyms)]
19pub(super) enum QueryType {
20    A,
21    AAAA,
22    TXT,
23}
24
25/// Pre-hash `(host, qtype)` into a u64 key for allocation-free cache lookups.
26///
27/// Collisions are benign (worst case: wrong cache entry returned, causing a
28/// re-query). With 64-bit hashes and a 512-entry cache the probability is
29/// negligible.
30fn cache_key(host: &str, qtype: QueryType) -> u64 {
31    let mut hasher = std::hash::DefaultHasher::new();
32    host.hash(&mut hasher);
33    qtype.hash(&mut hasher);
34    hasher.finish()
35}
36
37/// A cached DNS result.
38#[derive(Debug, Clone)]
39pub(super) enum CachedRecord {
40    A(Vec<Ipv4Addr>),
41    Aaaa(Vec<Ipv6Addr>),
42    Txt(Vec<TxtRecordData>),
43}
44
45/// A cache entry with TTL expiry tracking.
46#[derive(Debug, Clone)]
47struct CacheEntry {
48    record: CachedRecord,
49    inserted_at: Instant,
50    ttl: Duration,
51}
52
53impl CacheEntry {
54    fn is_expired(&self) -> bool {
55        self.inserted_at.elapsed() > self.ttl
56    }
57}
58
59/// DNS cache with LRU eviction and TTL-based expiry.
60///
61/// Uses pre-hashed u64 keys to avoid allocating a `String` on every lookup.
62/// The only remaining per-hit allocation is the `record.clone()` on cache hit,
63/// which is necessary because the cache is behind a `std::sync::RwLock`.
64#[derive(Debug)]
65pub(super) struct DnsCache {
66    inner: LruCache<u64, CacheEntry>,
67}
68
69impl DnsCache {
70    pub(super) fn new() -> Self {
71        Self {
72            inner: LruCache::new(std::num::NonZeroUsize::new(MAX_CACHE_ENTRIES).expect("non-zero")),
73        }
74    }
75
76    /// Look up a cached record. Returns `None` if not found or expired.
77    pub(super) fn get(&mut self, host: &str, qtype: QueryType) -> Option<CachedRecord> {
78        let key = cache_key(host, qtype);
79        let entry = self.inner.get(&key)?;
80        if entry.is_expired() {
81            self.inner.pop(&key);
82            return None;
83        }
84        Some(entry.record.clone())
85    }
86
87    /// Insert a record into the cache with the given TTL.
88    ///
89    /// A TTL of 0 means don't cache. This also means that negative responses
90    /// (NXDOMAIN, NODATA) are never cached, since empty results have TTL 0.
91    /// This matches the old hickory-resolver configuration which set
92    /// `negative_max_ttl = Some(Duration::ZERO)`.
93    ///
94    /// **Known limitation:** Without negative caching, repeated queries for
95    /// non-existent domains always hit the network. Under high concurrency
96    /// this can become a thundering herd. A future improvement could cache
97    /// negative results for a short duration (e.g. 5-10 seconds).
98    pub(super) fn insert(&mut self, host: &str, qtype: QueryType, record: CachedRecord, ttl: u32) {
99        if ttl == 0 {
100            return;
101        }
102        let entry = CacheEntry {
103            record,
104            inserted_at: Instant::now(),
105            ttl: Duration::from_secs(ttl as u64),
106        };
107        self.inner.put(cache_key(host, qtype), entry);
108    }
109
110    /// Clear all cache entries.
111    pub(super) fn clear(&mut self) {
112        self.inner.clear();
113    }
114}
