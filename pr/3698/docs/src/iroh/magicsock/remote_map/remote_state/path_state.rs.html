14/// Map of all paths that we are aware of for a remote endpoint.
15///
16/// Also stores a list of resolve requests which are triggered once at least one path is known,
17/// or once this struct is notified of a failed discovery run.
18#[derive(Debug, Default)]
19pub(super) struct RemotePathState {
20    /// All possible paths we are aware of.
21    ///
22    /// These paths might be entirely impossible to use, since they are added by discovery
23    /// mechanisms. The are only potentially usable.
24    paths: FxHashMap<transports::Addr, PathState>,
25    /// Pending resolve requests from [`Self::resolve_remote`].
26    pending_resolve_requests: VecDeque<oneshot::Sender<Result<(), DiscoveryError>>>,
27}
28
29impl RemotePathState {
30    /// Insert a new address into our list of potential paths.
31    ///
32    /// This will emit pending resolve requests.
33    pub(super) fn insert(&mut self, addr: transports::Addr, source: Source) {
34        self.paths
35            .entry(addr)
36            .or_default()
37            .sources
38            .insert(source.clone(), Instant::now());
39        self.emit_pending_resolve_requests(None);
40    }
41
42    /// Inserts multiple addresses into our list of potential paths.
43    ///
44    /// This will emit pending resolve requests.
45    pub(super) fn insert_multiple(
46        &mut self,
47        addrs: impl Iterator<Item = transports::Addr>,
48        source: Source,
49    ) {
50        let now = Instant::now();
51        for addr in addrs {
52            self.paths
53                .entry(addr)
54                .or_default()
55                .sources
56                .insert(source.clone(), now);
57        }
58        trace!("added addressing information");
59        self.emit_pending_resolve_requests(None);
60    }
61
62    /// Triggers `tx` immediately if there are any known paths, or store in the list of pending requests.
63    ///
64    /// The pending requests will be resolved once a path becomes known, or once discovery
65    /// concludes without results, whichever comes first.
66    ///
67    /// Sends `Ok(())` over `tx` if there are any known paths, and a [`DiscoveryError`] if there are
68    /// no known paths by the time a discovery run finished with an error or without results.
69    pub(super) fn resolve_remote(&mut self, tx: oneshot::Sender<Result<(), DiscoveryError>>) {
70        if !self.paths.is_empty() {
71            tx.send(Ok(())).ok();
72        } else {
73            self.pending_resolve_requests.push_back(tx);
74        }
75    }
76
77    /// Notifies that a discovery run has finished.
78    ///
79    /// This will emit pending resolve requests.
80    pub(super) fn discovery_finished(&mut self, result: Result<(), DiscoveryError>) {
81        self.emit_pending_resolve_requests(result.err());
82    }
83
84    /// Returns an iterator over the addresses of all paths.
85    pub(super) fn addrs(&self) -> impl Iterator<Item = &transports::Addr> {
86        self.paths.keys()
87    }
88
89    /// Returns whether this stores any addresses.
90    pub(super) fn is_empty(&self) -> bool {
91        self.paths.is_empty()
92    }
93
94    /// Replies to all pending resolve requests.
95    ///
96    /// This is a no-op if no requests are queued. Replies `Ok` if we have any known paths,
97    /// otherwise with the provided `discovery_error` or with [`DiscoveryError::NoResults`].
98    fn emit_pending_resolve_requests(&mut self, discovery_error: Option<DiscoveryError>) {
99        if self.pending_resolve_requests.is_empty() {
100            return;
101        }
102        let result = match (self.paths.is_empty(), discovery_error) {
103            (false, _) => Ok(()),
104            (true, Some(err)) => Err(err),
105            (true, None) => Err(e!(DiscoveryError::NoResults)),
106        };
107        for tx in self.pending_resolve_requests.drain(..) {
108            tx.send(result.clone()).ok();
109        }
110    }
111}

113/// The state of a single path to the remote endpoint.
114///
115/// Each path is identified by the destination [`transports::Addr`] and they are stored in
116/// the [`RemotePathState`] map at [`RemoteStateActor::paths`].
117///
118/// [`RemoteStateActor::paths`]: super::RemoteStateActor::paths
119#[derive(Debug, Default)]
120pub(super) struct PathState {
121    /// How we learned about this path, and when.
122    ///
123    /// We keep track of only the latest [`Instant`] for each [`Source`], keeping the size
124    /// of the map of sources down to one entry per type of source.
125    pub(super) sources: HashMap<Source, Instant>,
126}
