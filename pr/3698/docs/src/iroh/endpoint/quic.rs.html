35/// Parameters governing the core QUIC state machine
36///
37/// Default values should be suitable for most internet applications. Applications protocols which
38/// forbid remotely-initiated streams should set `max_concurrent_bidi_streams` and
39/// `max_concurrent_uni_streams` to zero.
40///
41/// In some cases, performance or resource requirements can be improved by tuning these values to
42/// suit a particular application and/or network connection. In particular, data window sizes can be
43/// tuned for a particular expected round trip time, link capacity, and memory availability. Tuning
44/// for higher bandwidths and latencies increases worst-case memory consumption, but does not impair
45/// performance at lower bandwidths and latencies. The default configuration is tuned for a 100Mbps
46/// link with a 100ms round trip time.
47///
48/// In iroh, the config has some specific default values that make iroh's holepunching work
49/// well with QUIC multipath. Adjusting those settings may cause suboptimal usage.
50///
51/// Look at the following methods for more details:
52/// - [`QuicTransportConfig::default_path_keep_alive_interval`]
53/// - [`QuicTransportConfig::default_path_max_idle_timeout`]
54/// - [`QuicTransportConfig::max_concurrent_multipath_paths`]
55/// - [`QuicTransportConfig::set_max_remote_nat_traversal_addresses`]
56#[derive(Debug, Clone)]
57pub struct QuicTransportConfig(pub(crate) quinn::TransportConfig);
58
59impl Default for QuicTransportConfig {
60    fn default() -> Self {
61        let mut cfg = quinn::TransportConfig::default();
62        // Override some transport config settings.
63        cfg.keep_alive_interval(Some(HEARTBEAT_INTERVAL));
64        cfg.default_path_keep_alive_interval(Some(HEARTBEAT_INTERVAL));
65        cfg.default_path_max_idle_timeout(Some(PATH_MAX_IDLE_TIMEOUT));
66        cfg.max_concurrent_multipath_paths(MAX_MULTIPATH_PATHS + 1);
67        cfg.set_max_remote_nat_traversal_addresses(MAX_MULTIPATH_PATHS as u8);
68        Self(cfg)
69    }
70}

95    /// Maximum duration of inactivity to accept before timing out the connection.
96    ///
97    /// The true idle timeout is the minimum of this and the peer's own max idle timeout. `None`

192    /// The initial value to be used as the maximum UDP payload size before running MTU discovery
193    /// (see [`QuicTransportConfig::mtu_discovery_config`]).
194    ///
195    /// Must be at least 1200, which is the default, and known to be safe for typical internet

205    /// The maximum UDP payload size guaranteed to be supported by the network.
206    ///
207    /// Must be at least 1200, which is the default, and lower than or equal to
208    /// [`QuicTransportConfig::initial_mtu`].
209    ///
210    /// Real-world MTUs can vary according to ISP, VPN, and properties of intermediate network links
211    /// outside of either endpoint's control. Extreme care should be used when raising this value
212    /// outside of private networks where these factors are fully controlled. If the provided value
213    /// is higher than what the network path actually supports, the result will be unpredictable and
214    /// catastrophic packet loss, without a possibility of repair. Prefer
215    /// [`QuicTransportConfig::initial_mtu`] together with
216    /// [`QuicTransportConfig::mtu_discovery_config`] to set a maximum UDP payload size that robustly
217    /// adapts to the network.

367    /// Enables the Multipath Extension for QUIC.
368    ///
369    /// Setting this to any nonzero value will enable the Multipath Extension for QUIC,
370    /// <https://datatracker.ietf.org/doc/draft-ietf-quic-multipath/>.
371    ///
372    /// The value provided specifies the number maximum number of paths this endpoint may open
373    /// concurrently when multipath is negotiated. For any path to be opened, the remote must
374    /// enable multipath as well.
375    ///
376    /// Note: this method will ignore values less than the recommended 13 and will log a warning.
377    pub fn max_concurrent_multipath_paths(&mut self, max_concurrent: u32) -> &mut Self {
378        if max_concurrent < MAX_MULTIPATH_PATHS + 1 {
379            warn!(
380                "QuicTransportConfig::max_concurrent_multipath_paths must be at minimum {}, ignoring user supplied value",
381                MAX_MULTIPATH_PATHS + 1
382            );
383            return self;
384        }
385        self.0.max_concurrent_multipath_paths(max_concurrent);
386        self
387    }

390    ///
391    /// If the path is idle for this long the path will be abandoned. Bear in mind this will
392    /// interact with the [`QuicTransportConfig::max_idle_timeout`], if the last path is

403        }
404        self.0.default_path_max_idle_timeout(Some(timeout));
