17/// Whether this error was because we couldn't create a client or a send error.
18#[allow(missing_docs)]
19#[common_fields({
20    backtrace: Option<snafu::Backtrace>,
21    #[snafu(implicit)]
22    span_trace: n0_snafu::SpanTrace,
23})]
24#[derive(Debug, Snafu)]
25pub enum PingError {
26    #[snafu(display("failed to create IPv4 ping client"))]
27    CreateClientIpv4 { source: std::io::Error },
28    #[snafu(display("failed to create IPv6 ping client"))]
29    CreateClientIpv6 { source: std::io::Error },
30    /// Could not send ping.
31    #[snafu(display("failed to send ping"))]
32    Ping { source: surge_ping::SurgeError },
33}

35/// Allows sending ICMP echo requests to a host in order to determine network latency.
36/// Will gracefully handle both IPv4 and IPv6.
37#[derive(Debug, Clone, Default)]
38pub struct Pinger(Arc<Inner>);
39
40impl Debug for Inner {
41    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
42        f.debug_struct("Inner").finish()
43    }
44}
45
46#[derive(Default)]
47struct Inner {
48    client_v6: Mutex<Option<Client>>,
49    client_v4: Mutex<Option<Client>>,
50}
51
52impl Pinger {
53    /// Create a new [Pinger].
54    pub fn new() -> Self {
55        Default::default()
56    }
57
58    /// Lazily create the ping client.
59    ///
60    /// We do this because it means we do not bind a socket until we really try to send a
61    /// ping.  It makes it more transparent to use the pinger.
62    fn get_client(&self, kind: ICMP) -> Result<Client, PingError> {
63        let client = match kind {
64            ICMP::V4 => {
65                let mut opt_client = self.0.client_v4.lock().expect("poisoned");
66                match *opt_client {
67                    Some(ref client) => client.clone(),
68                    None => {
69                        let cfg = Config::builder().kind(kind).build();
70                        let client = Client::new(&cfg).context(CreateClientIpv4Snafu)?;
71                        *opt_client = Some(client.clone());
72                        client
73                    }
74                }
75            }
76            ICMP::V6 => {
77                let mut opt_client = self.0.client_v6.lock().expect("poisoned");
78                match *opt_client {
79                    Some(ref client) => client.clone(),
80                    None => {
81                        let cfg = Config::builder().kind(kind).build();
82                        let client = Client::new(&cfg).context(CreateClientIpv6Snafu)?;
83                        *opt_client = Some(client.clone());
84                        client
85                    }
86                }
87            }
88        };
89        Ok(client)
90    }
91
92    /// Send a ping request with associated data, returning the perceived latency.
93    pub async fn send(&self, addr: IpAddr, data: &[u8]) -> Result<Duration, PingError> {
94        let client = match addr {
95            IpAddr::V4(_) => self.get_client(ICMP::V4)?,
96            IpAddr::V6(_) => self.get_client(ICMP::V6)?,
97        };
98        let ident = PingIdentifier(rand::random());
99        debug!(%addr, %ident, "Creating pinger");
100        let mut pinger = client.pinger(addr, ident).await;
101        pinger.timeout(DEFAULT_TIMEOUT); // todo: timeout too large for net_report
102        match pinger
103            .ping(PingSequence(0), data)
104            .await
105            .context(PingSnafu)?
106        {
107            (IcmpPacket::V4(packet), dur) => {
108                debug!(
109                    "{} bytes from {}: icmp_seq={} ttl={:?} time={:0.2?}",
110                    packet.get_size(),
111                    packet.get_source(),
112                    packet.get_sequence(),
113                    packet.get_ttl(),
114                    dur
115                );
116                Ok(dur)
117            }
118
119            (IcmpPacket::V6(packet), dur) => {
120                debug!(
121                    "{} bytes from {}: icmp_seq={} hlim={} time={:0.2?}",
122                    packet.get_size(),
123                    packet.get_source(),
124                    packet.get_sequence(),
125                    packet.get_max_hop_limit(),
126                    dur
127                );
128                Ok(dur)
129            }
130        }
131    }
132}

134#[cfg(test)]
135mod tests {
136    use std::net::{Ipv4Addr, Ipv6Addr};
137
138    use tracing::error;
139    use tracing_test::traced_test;
140
141    use super::*;
142
143    // See net_report::reportgen::tests::test_icmp_probe_eu_relay for permissions to ping.
144    #[tokio::test]
145    #[traced_test]
146    async fn test_ping_localhost() {
147        let pinger = Pinger::new();
148
149        match pinger.send(Ipv4Addr::LOCALHOST.into(), b"data").await {
150            Ok(duration) => {
151                assert!(!duration.is_zero());
152            }
153            Err(
154                PingError::CreateClientIpv4 { source, .. }
155                | PingError::CreateClientIpv6 { source, .. },
156            ) => {
157                // We don't have permission, too bad.
158                error!("no ping permissions: {source:#}");
159            }
160            Err(PingError::Ping { source, .. }) => {
161                panic!("ping failed: {source:#}");
162            }
163        }
164
165        match pinger.send(Ipv6Addr::LOCALHOST.into(), b"data").await {
166            Ok(duration) => {
167                assert!(!duration.is_zero());
168            }
169            Err(
170                PingError::CreateClientIpv4 { source, .. }
171                | PingError::CreateClientIpv6 { source, .. },
172            ) => {
173                // We don't have permission, too bad.
174                error!("no ping permissions: {source:#}");
175            }
176            Err(PingError::Ping { source, .. }) => {
177                error!("ping failed, probably no IPv6 stack: {source:#}");
178            }
179        }
180    }
