21/// The mode how SSL certificates should be created.
22#[derive(Serialize, Deserialize, Debug, Clone, PartialEq, Eq, strum::Display)]
23#[serde(rename_all = "snake_case")]
24pub enum CertMode {
25    /// Certs are loaded from a the `cert_cache` path
26    Manual,
27    /// ACME with LetsEncrypt servers
28    LetsEncrypt,
29    /// Create self-signed certificates and store them in the `cert_cache` path
30    SelfSigned,
31}
32
33impl CertMode {
34    /// Build the [`TlsAcceptor`] for this mode.
35    pub(crate) async fn build(
36        &self,
37        domains: Vec<String>,
38        cert_cache: PathBuf,
39        letsencrypt_contact: Option<String>,
40        letsencrypt_prod: bool,
41    ) -> Result<TlsAcceptor> {
42        Ok(match self {
43            CertMode::Manual => TlsAcceptor::manual(domains, cert_cache).await?,
44            CertMode::SelfSigned => TlsAcceptor::self_signed(domains).await?,
45            CertMode::LetsEncrypt => {
46                let contact =
47                    letsencrypt_contact.context("contact is required for letsencrypt cert mode")?;
48                TlsAcceptor::letsencrypt(domains, &contact, letsencrypt_prod, cert_cache)?
49            }
50        })
51    }
52}

54/// TLS Certificate Authority acceptor.
55#[derive(Clone)]
56pub enum TlsAcceptor {
57    LetsEncrypt(AxumAcceptor),
58    Manual(RustlsAcceptor),
59}
60
61impl<I: AsyncRead + AsyncWrite + Unpin + Send + 'static, S: Send + 'static> Accept<I, S>
62    for TlsAcceptor
63{
64    type Stream = tokio_rustls::server::TlsStream<I>;
65    type Service = S;
66    type Future = BoxFuture<io::Result<(Self::Stream, Self::Service)>>;
67
68    fn accept(&self, stream: I, service: S) -> Self::Future {
69        match self {
70            Self::LetsEncrypt(a) => a.accept(stream, service).boxed(),
71            Self::Manual(a) => a.accept(stream, service).boxed(),
72        }
73    }
74}
75
76impl TlsAcceptor {
77    async fn self_signed(domains: Vec<String>) -> Result<Self> {
78        let rcgen::CertifiedKey { cert, key_pair } =
79            rcgen::generate_simple_self_signed(domains).e()?;
80        let config = RustlsConfig::from_der(vec![cert.der().to_vec()], key_pair.serialize_der())
81            .await
82            .e()?;
83        let acceptor = RustlsAcceptor::new(config);
84        Ok(Self::Manual(acceptor))
85    }
86
87    async fn manual(domains: Vec<String>, dir: PathBuf) -> Result<Self> {
88        let config = rustls::ServerConfig::builder().with_no_client_auth();
89        if domains.len() != 1 {
90            whatever!("Multiple domains in manual mode are not supported");
91        }
92
93        let keyname = escape_hostname(&domains[0]);
94        let cert_path = dir.join(format!("{keyname}.crt"));
95        let key_path = dir.join(format!("{keyname}.key"));
96
97        let certs = load_certs(cert_path).await?;
98        let secret_key = load_secret_key(key_path).await?;
99
100        let config = config.with_single_cert(certs, secret_key).e()?;
101        let config = RustlsConfig::from_config(Arc::new(config));
102        let acceptor = RustlsAcceptor::new(config);
103        Ok(Self::Manual(acceptor))
104    }
105
106    fn letsencrypt(
107        domains: Vec<String>,
108        contact: &str,
109        is_production: bool,
110        dir: PathBuf,
111    ) -> Result<Self> {
112        let config = rustls::ServerConfig::builder().with_no_client_auth();
113        let mut state = AcmeConfig::new(domains)
114            .contact([format!("mailto:{contact}")])
115            .cache_option(Some(DirCache::new(dir)))
116            .directory_lets_encrypt(is_production)
117            .state();
118
119        let config = config.with_cert_resolver(state.resolver());
120        let acceptor = state.acceptor();
121
122        tokio::spawn(
123            async move {
124                loop {
125                    match state.next().await.unwrap() {
126                        Ok(ok) => debug!("acme event: {:?}", ok),
127                        Err(err) => error!("error: {:?}", err),
128                    }
129                }
130            }
131            .instrument(info_span!("acme")),
132        );
133        let config = Arc::new(config);
134        let acceptor = AxumAcceptor::new(acceptor, config);
135        Ok(Self::LetsEncrypt(acceptor))
136    }
137}

139async fn load_certs(
140    filename: impl AsRef<Path>,
141) -> Result<Vec<rustls::pki_types::CertificateDer<'static>>> {
142    let certfile = tokio::fs::read(filename)
143        .await
144        .context("cannot open certificate file")?;
145    let mut reader = std::io::Cursor::new(certfile);
146    let certs: Result<Vec<_>, std::io::Error> = rustls_pemfile::certs(&mut reader).collect();
147    let certs = certs.e()?;
148
149    Ok(certs)
150}
151
152async fn load_secret_key(
153    filename: impl AsRef<Path>,
154) -> Result<rustls::pki_types::PrivateKeyDer<'static>> {
155    let keyfile = tokio::fs::read(filename.as_ref())
156        .await
157        .context("cannot open secret key file")?;
158    let mut reader = std::io::Cursor::new(keyfile);
159
160    loop {
161        match rustls_pemfile::read_one(&mut reader).context("cannot parse secret key .pem file")? {
162            Some(rustls_pemfile::Item::Pkcs1Key(key)) => {
163                return Ok(rustls::pki_types::PrivateKeyDer::Pkcs1(key));
164            }
165            Some(rustls_pemfile::Item::Pkcs8Key(key)) => {
166                return Ok(rustls::pki_types::PrivateKeyDer::Pkcs8(key));
167            }
168            Some(rustls_pemfile::Item::Sec1Key(key)) => {
169                return Ok(rustls::pki_types::PrivateKeyDer::Sec1(key));
170            }
171            None => break,
172            _ => {}
173        }
174    }
175
176    whatever!(
177        "no keys found in {} (encrypted keys not supported)",
178        filename.as_ref().display()
179    );
180}

182static UNSAFE_HOSTNAME_CHARACTERS: OnceLock<regex::Regex> = OnceLock::new();
183
184fn escape_hostname(hostname: &str) -> Cow<'_, str> {
185    let regex = UNSAFE_HOSTNAME_CHARACTERS
186        .get_or_init(|| regex::Regex::new(r"[^a-zA-Z0-9-\.]").expect("valid regex"));
187    regex.replace_all(hostname, "")
188}
