3use std::fmt::Debug;

5use aead::{AeadCore, AeadInOut, Buffer};

9pub(crate) const NONCE_LEN: usize = 24;
10
11const AEAD_DATA: &[u8] = &[];
12
13pub(super) fn public_ed_box(key: &ed25519_dalek::VerifyingKey) -> crypto_box::PublicKey {
14    crypto_box::PublicKey::from(key.to_montgomery())
15}
16
17pub(super) fn secret_ed_box(key: &ed25519_dalek::SigningKey) -> crypto_box::SecretKey {
18    crypto_box::SecretKey::from(key.to_scalar())
19}
20
21/// Shared Secret.
22pub struct SharedSecret(crypto_box::ChaChaBox);
23
24/// Errors that can occur during [`SharedSecret::open`].
25#[common_fields({
26    backtrace: Option<snafu::Backtrace>,
27    #[snafu(implicit)]
28    span_trace: n0_snafu::SpanTrace,
29})]
30#[derive(Debug, Snafu)]
31#[non_exhaustive]
32pub enum DecryptionError {
33    /// The nonce had the wrong size.
34    #[snafu(display("Invalid nonce"))]
35    InvalidNonce {},
36    /// AEAD decryption failed.
37    #[snafu(display("Aead error"))]
38    Aead { source: aead::Error },
39}
40
41impl Debug for SharedSecret {
42    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
43        write!(f, "SharedSecret(crypto_box::ChaChaBox)")
44    }
45}
46
47impl SharedSecret {
48    pub fn new(this: &crypto_box::SecretKey, other: &crypto_box::PublicKey) -> Self {
49        SharedSecret(crypto_box::ChaChaBox::from_clamped(other, this))
50    }
51
52    /// Seals the provided cleartext.
53    pub fn seal(&self, buffer: &mut dyn Buffer) {
54        let nonce = crypto_box::ChaChaBox::try_generate_nonce_with_rng(&mut rand::rngs::OsRng)
55            .expect("not enough randomness");
56
57        self.0
58            .encrypt_in_place(&nonce, AEAD_DATA, buffer)
59            .expect("encryption failed");
60
61        buffer.extend_from_slice(&nonce).expect("buffer too small");
62    }
63
64    /// Opens the ciphertext, which must have been created using `Self::seal`, and places the clear text into the provided buffer.
65    pub fn open(&self, buffer: &mut dyn Buffer) -> Result<(), DecryptionError> {
66        ensure!(buffer.len() >= NONCE_LEN, InvalidNonceSnafu);
67
68        let offset = buffer.len() - NONCE_LEN;
69        let nonce: [u8; NONCE_LEN] = buffer.as_ref()[offset..]
70            .try_into()
71            .map_err(|_| InvalidNonceSnafu.build())?;
72
73        buffer.truncate(offset);
74        self.0
75            .decrypt_in_place(&nonce.into(), AEAD_DATA, buffer)
76            .context(AeadSnafu)?;
77
78        Ok(())
79    }
80}

86    fn shared(this: &iroh_base::SecretKey, other: &iroh_base::PublicKey) -> SharedSecret {
87        let secret_key = secret_ed_box(this.secret());
88        let public_key = public_ed_box(&other.public());
89
90        SharedSecret::new(&secret_key, &public_key)

93    #[test]
94    fn test_seal_open_roundtrip() {
95        let mut rng = rand::rng();
96        let key_a = iroh_base::SecretKey::generate(&mut rng);
97        let key_b = iroh_base::SecretKey::generate(&mut rng);
98
99        println!("a -> a");
100        seal_open_roundtrip(&key_a, &key_a);
101        println!("b -> b");
102        seal_open_roundtrip(&key_b, &key_b);
103
104        println!("a -> b");
105        seal_open_roundtrip(&key_a, &key_b);
106        println!("b -> a");
107        seal_open_roundtrip(&key_b, &key_a);
108    }

110    fn seal_open_roundtrip(key_a: &iroh_base::SecretKey, key_b: &iroh_base::SecretKey) {
111        let msg = b"super secret message!!!!".to_vec();
112        let shared_a = shared(key_a, &key_b.public());
113        let mut sealed_message = msg.clone();
114        shared_a.seal(&mut sealed_message);
115
116        let shared_b = shared(key_b, &key_a.public());
117        let mut decrypted_message = sealed_message.clone();
118        shared_b.open(&mut decrypted_message).unwrap();
119        assert_eq!(&msg[..], &decrypted_message);
120    }
