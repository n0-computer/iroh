92/// The maximum latency of all nodes, if none are found yet.
93///
94/// Normally the max latency of all nodes is computed, but if we don't yet know any nodes
95/// latencies we return this as default.  This is the value of the initial QAD probe
96/// delays.  It is only used as time to wait for further latencies to arrive, which *should*
97/// never happen unless there already is at least one latency.  Yet here we are, defining a
98/// default which will never be used.
99const DEFAULT_MAX_LATENCY: Duration = Duration::from_millis(100);

129#[cfg(not(wasm_browser))]
130impl QadConns {
131    fn clear(&mut self) {
132        if let Some((_, conn)) = self.v4.take() {
133            conn.conn
134                .close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
135        }
136        if let Some((_, conn)) = self.v6.take() {
137            conn.conn
138                .close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
139        }
140    }
141
142    fn current(&self) -> Vec<ProbeReport> {
143        let mut reports = Vec::new();
144        if let Some((_, ref conn)) = self.v4 {
145            if let Some(mut r) = conn.observer.get() {
146                // grab latest rtt
147                r.latency = conn.conn.rtt();
148                reports.push(ProbeReport::QadIpv4(r));
149            }
150        }
151
152        if let Some((_, ref conn)) = self.v6 {
153            if let Some(mut r) = conn.observer.get() {
154                // grab latest rtt
155                r.latency = conn.conn.rtt();
156                reports.push(ProbeReport::QadIpv6(r));
157            }
158        }
159
160        reports
161    }
162
163    fn watch_v4(&self) -> impl n0_future::Stream<Item = Option<QadProbeReport>> + Unpin + use<> {
164        let watcher = self.v4.as_ref().map(|(_url, conn)| conn.observer.watch());
165
166        if let Some(watcher) = watcher {
167            watcher.stream_updates_only().boxed()
168        } else {
169            n0_future::stream::empty().boxed()
170        }
171    }
172
173    fn watch_v6(&self) -> impl n0_future::Stream<Item = Option<QadProbeReport>> + Unpin + use<> {
174        let watcher = self.v6.as_ref().map(|(_url, conn)| conn.observer.watch());
175        if let Some(watcher) = watcher {
176            watcher.stream_updates_only().boxed()
177        } else {
178            n0_future::stream::empty().boxed()
179        }
180    }

253    /// Generates a [`Report`].
254    ///
255    /// Look at [`Options`] for the different configuration options.
256    pub(crate) async fn get_report(&mut self, if_state: IfStateDetails, is_major: bool) -> Report {
257        let now = Instant::now();
258
259        let mut do_full = is_major
260            || self.reports.next_full
261            || now.duration_since(self.reports.last_full) > FULL_REPORT_INTERVAL;
262
263        debug!(%do_full, "net_report starting");
264
265        // If the last report had a captive portal and reported no UDP access,
266        // it's possible that we didn't get a useful net_report due to the
267        // captive portal blocking us. If so, make this report a full (non-incremental) one.
268        if !do_full {
269            if let Some(ref last) = self.reports.last {
270                if !last.has_udp() && last.captive_portal == Some(true) {
271                    do_full = true;
272                }
273            }
274        }
275        if do_full {
276            self.reports.last = None; // causes ProbePlan::new below to do a full (initial) plan
277            self.reports.next_full = false;
278            self.reports.last_full = now;
279            self.metrics.reports_full.inc();
280        }
281        self.metrics.reports.inc();
282
283        let enough_relays = std::cmp::min(self.relay_map.len(), ENOUGH_NODES);
284        #[cfg(wasm_browser)]
285        let if_state = IfStateDetails::default();
286        #[cfg(not(wasm_browser))]
287        let if_state = IfStateDetails {
288            have_v4: if_state.have_v4,
289            have_v6: if_state.have_v6,
290        };
291
292        let mut report = Report::default();
293
294        // Start the reportgen client to start any needed probes
295        let (actor, mut probe_rx) = reportgen::Client::new(
296            self.reports.last.clone(),
297            self.relay_map.clone(),
298            self.probes.clone(),
299            if_state.clone(),
300            #[cfg(not(wasm_browser))]
301            self.socket_state.clone(),
302            #[cfg(any(test, feature = "test-utils"))]
303            self.insecure_skip_relay_cert_verify,
304        );
305
306        #[cfg(not(wasm_browser))]
307        let reports = self
308            .spawn_qad_probes(&if_state, enough_relays, do_full)
309            .await;
310
311        #[cfg(not(wasm_browser))]
312        for r in reports {
313            report.update(&r);
314        }
315
316        let mut timeout_fut = std::pin::pin!(MaybeFuture::default());
317
318        #[cfg(not(wasm_browser))]
319        let mut qad_v4_stream = self.qad_conns.watch_v4();
320        #[cfg(wasm_browser)]
321        let mut qad_v4_stream = n0_future::stream::empty::<Option<()>>();
322        #[cfg(not(wasm_browser))]
323        let mut qad_v6_stream = self.qad_conns.watch_v6();
324        #[cfg(wasm_browser)]
325        let mut qad_v6_stream = n0_future::stream::empty::<Option<()>>();
326
327        loop {
328            tokio::select! {
329                biased;
330
331                _ = &mut timeout_fut, if timeout_fut.is_some() => {
332                    trace!("timeout done, shutting down");
333                    drop(actor); // shuts down the probes
334                    break;
335                }
336
337                Some(Some(r)) = qad_v4_stream.next() => {
338                    #[cfg(not(wasm_browser))]
339                    {
340                        trace!(?r, "new report from QAD V4");
341                        report.update(&ProbeReport::QadIpv4(r));
342                    }
343                }
344
345                Some(Some(r)) = qad_v6_stream.next() => {
346                    #[cfg(not(wasm_browser))]
347                    {
348                        trace!(?r, "new report from QAD V6");
349                        report.update(&ProbeReport::QadIpv6(r));
350                    }
351                }
352
353                maybe_probe = probe_rx.recv() => {
354                    let Some(probe_res) = maybe_probe else {
355                        break;
356                    };
357                    trace!(?probe_res, "handling probe");
358                    match probe_res {
359                        ProbeFinished::Regular(probe) => match probe {
360                            Ok(probe) => {
361                                report.update(&probe);
362                                if timeout_fut.is_none() {
363                                    if let Some(timeout) = self.have_enough_reports(enough_relays, &report) {
364                                        timeout_fut.as_mut().set_future(time::sleep(timeout));
365                                    }
366                                }
367                            }
368                            Err(err) => {
369                                trace!("probe errored: {:?}", err);
370                            }
371                        },
372                        #[cfg(not(wasm_browser))]
373                        ProbeFinished::CaptivePortal(portal) => {
374                            report.captive_portal = portal;
375                        }
376                    }
377                }
378            }
379        }
380
381        self.add_report_history_and_set_preferred_relay(&mut report);
382        debug!(
383            ?report,
384            "generated report in {:02}ms",
385            now.elapsed().as_millis()
386        );
387
388        report
389    }

391    #[cfg(not(wasm_browser))]
392    async fn spawn_qad_probes(
393        &mut self,
394        if_state: &IfStateDetails,
395        enough_relays: usize,
396        do_full: bool,
397    ) -> Vec<ProbeReport> {
398        use tracing::{Instrument, info_span};
399
400        debug!("spawning QAD probes");
401
402        let Some(ref quic_client) = self.socket_state.quic_client else {
403            return Vec::new();
404        };
405
406        if do_full {
407            // clear out existing connections if we are doing a full reset
408            self.qad_conns.clear();
409        }
410
411        if let Some((url, conn)) = &self.qad_conns.v4 {
412            // verify conn is still around
413            if let Some(reason) = conn.conn.close_reason() {
414                trace!(?url, "QAD v4 conn closed: {}", reason);
415                self.qad_conns.v4.take();
416            }
417        }
418        if let Some((url, conn)) = &self.qad_conns.v6 {
419            // verify conn is still around
420            if let Some(reason) = conn.conn.close_reason() {
421                trace!(?url, "QAD v6 conn closed: {}", reason);
422                self.qad_conns.v6.take();
423            }
424        }
425        if self.qad_conns.v4.is_some() && self.qad_conns.v6.is_some() == if_state.have_v6 {
426            trace!("not spawning QAD, already have probes");
427            return self.qad_conns.current();
428        }
429
430        // TODO: randomize choice?
431        const MAX_RELAYS: usize = 5;
432
433        let mut v4_buf = JoinSet::new();
434        let cancel_v4 = CancellationToken::new();
435        let mut v6_buf = JoinSet::new();
436        let cancel_v6 = CancellationToken::new();
437
438        for relay_node in self.relay_map.nodes().take(MAX_RELAYS) {
439            if if_state.have_v4 {
440                debug!(?relay_node.url, "v4 QAD probe");
441                let ip_mapped_addrs = self.socket_state.ip_mapped_addrs.clone();
442                let relay_node = relay_node.clone();
443                let dns_resolver = self.socket_state.dns_resolver.clone();
444                let quic_client = quic_client.clone();
445                let relay_url = relay_node.url.clone();
446                v4_buf.spawn(
447                    cancel_v4
448                        .child_token()
449                        .run_until_cancelled_owned(time::timeout(
450                            PROBES_TIMEOUT,
451                            run_probe_v4(ip_mapped_addrs, relay_node, quic_client, dns_resolver),
452                        ))
453                        .instrument(info_span!("QAD-IPv4", %relay_url)),
454                );
455            }
456
457            if if_state.have_v6 {
458                debug!(?relay_node.url, "v6 QAD probe");
459                let ip_mapped_addrs = self.socket_state.ip_mapped_addrs.clone();
460                let relay_node = relay_node.clone();
461                let dns_resolver = self.socket_state.dns_resolver.clone();
462                let quic_client = quic_client.clone();
463                let relay_url = relay_node.url.clone();
464                v6_buf.spawn(
465                    cancel_v6
466                        .child_token()
467                        .run_until_cancelled_owned(time::timeout(
468                            PROBES_TIMEOUT,
469                            run_probe_v6(ip_mapped_addrs, relay_node, quic_client, dns_resolver),
470                        ))
471                        .instrument(info_span!("QAD-IPv6", %relay_url)),
472                );
473            }
474        }
475
476        let mut reports = Vec::new();
477
478        // We set _pending to true if at least one report was started for each category.
479        // If we did not start any report for either category, _pending is set to false right away
480        // (it "completed" in the sense that nothing will ever run). If we did start at least one report,
481        // _pending is set to true, and will be set to false further down once the first task
482        // completed.
483        let mut ipv4_pending = !v4_buf.is_empty();
484        let mut ipv6_pending = !v6_buf.is_empty();
485        loop {
486            // We early-abort the tasks once we have at least `enough_relays` reports,
487            // and at least one ipv4 and one ipv6 report completed (if they were started, see comment above).
488            if reports.len() >= enough_relays && !ipv4_pending && !ipv6_pending {
489                debug!("enough probes: {}", reports.len());
490                cancel_v4.cancel();
491                cancel_v6.cancel();
492                break;
493            }
494
495            tokio::select! {
496                biased;
497
498                val = v4_buf.join_next(), if !v4_buf.is_empty() => {
499                    ipv4_pending = false;
500                    match val {
501                        Some(Ok(Some(Ok(res)))) => {
502                            match res {
503                                Ok((r, conn)) => {
504                                    debug!(?r, "got v4 QAD conn");
505                                    let url = r.node.clone();
506                                    reports.push(ProbeReport::QadIpv4(r));
507                                    if self.qad_conns.v4.is_none() {
508                                        self.qad_conns.v4.replace((url, conn));
509                                    } else {
510                                        conn.conn.close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
511                                    }
512                                }
513                                Err(err) => {
514                                    debug!("probe v4 failed: {err:?}");
515                                }
516                            }
517                        }
518                        Some(Err(err)) => {
519                            if err.is_panic() {
520                                panic!("probe v4 panicked: {err:?}");
521                            }
522                            warn!("probe v4 failed: {err:?}");
523                        }
524                        Some(Ok(None)) => {
525                            debug!("probe v4 canceled");
526                        }
527                        Some(Ok(Some(Err(time::Elapsed { .. })))) => {
528                            debug!("probe v4 timed out");
529                        }
530                        None => {}
531                    }
532                }
533                val = v6_buf.join_next(), if !v6_buf.is_empty() => {
534                    ipv6_pending = false;
535                    match val {
536                        Some(Ok(Some(Ok(res)))) => {
537                            match res {
538                                Ok((r, conn)) => {
539                                    debug!(?r, "got v6 QAD conn");
540                                    let url = r.node.clone();
541                                    reports.push(ProbeReport::QadIpv6(r));
542                                    if self.qad_conns.v6.is_none() {
543                                        self.qad_conns.v6.replace((url, conn));
544                                    } else {
545                                        conn.conn.close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
546                                    }
547                                }
548                                Err(err) => {
549                                    debug!("probe v6 failed: {err:?}");
550                                }
551                            }
552                        }
553                        Some(Err(err)) => {
554                            if err.is_panic() {
555                                panic!("probe v6 panicked: {err:?}");
556                            }
557                            warn!("probe v6 failed: {err:?}");
558                        }
559                        Some(Ok(None)) => {
560                            debug!("probe v6 canceled");
561                        }
562                        Some(Ok(Some(Err(time::Elapsed { .. })))) => {
563                            debug!("probe v6 timed out");
564                        }
565                        None => {}
566                    }
567                }
568                else => {
569                    break;
570                }
571            }
572        }
573
574        reports
575    }

577    fn have_enough_reports(&self, enough_relays: usize, report: &Report) -> Option<Duration> {
578        // Once we've heard from enough relay servers (3), start a timer to give up on the other
579        // probes. The timer's duration is a function of whether this is our initial full
580        // probe or an incremental one. For incremental ones, wait for the duration of the
581        // slowest relay. For initial ones, double that.
582        let latencies: Vec<Duration> = report.relay_latency.iter().map(|(_, l)| l).collect();
583        let have_enough_latencies = latencies.len() >= enough_relays;
584
585        if have_enough_latencies {
586            let timeout = match self.reports.last.is_some() {
587                true => Duration::from_secs(0),
588                false => latencies
589                    .iter()
590                    .max()
591                    .copied()
592                    .unwrap_or(DEFAULT_MAX_LATENCY),
593            };
594            debug!(
595                reports=latencies.len(),
596                delay=?timeout,
597                "Have enough probe reports, aborting further probes soon",
598            );
599
600            Some(timeout)
601        } else {
602            None
603        }
604    }

606    /// Adds `r` to the set of recent Reports and mutates `r.preferred_relay` to contain the best recent one.
607    fn add_report_history_and_set_preferred_relay(&mut self, r: &mut Report) {
608        let mut prev_relay = None;
609        if let Some(ref last) = self.reports.last {
610            prev_relay.clone_from(&last.preferred_relay);
611
612            // If we don't have new information, copy this from the last report
613            if r.mapping_varies_by_dest_ipv4.is_none() {
614                r.mapping_varies_by_dest_ipv4 = last.mapping_varies_by_dest_ipv4;
615            }
616            if r.mapping_varies_by_dest_ipv6.is_none() {
617                r.mapping_varies_by_dest_ipv6 = last.mapping_varies_by_dest_ipv6;
618            }
619        }
620
621        let now = Instant::now();
622        const MAX_AGE: Duration = Duration::from_secs(5 * 60);
623
624        // relay ID => its best recent latency in last MAX_AGE
625        let mut best_recent = RelayLatencies::default();
626
627        // chain the current report as we are still mutating it
628        let prevs_iter = self
629            .reports
630            .prev
631            .iter()
632            .map(|(a, b)| -> (&Instant, &Report) { (a, b) });
633
634        let mut to_remove = Vec::new();
635        for (t, pr) in prevs_iter {
636            if now.duration_since(*t) > MAX_AGE {
637                to_remove.push(*t);
638                continue;
639            }
640            best_recent.merge(&pr.relay_latency);
641        }
642        // merge in current run
643        best_recent.merge(&r.relay_latency);
644
645        for t in to_remove {
646            self.reports.prev.remove(&t);
647        }
648
649        // Then, pick which currently-alive relay server from the
650        // current report has the best latency over the past MAX_AGE.
651        let mut best_any = Duration::default();
652        let mut old_relay_cur_latency = Duration::default();
653        {
654            for (url, duration) in r.relay_latency.iter() {
655                if Some(url) == prev_relay.as_ref() {
656                    old_relay_cur_latency = duration;
657                }
658                if let Some(best) = best_recent.get(url) {
659                    if r.preferred_relay.is_none() || best < best_any {
660                        best_any = best;
661                        r.preferred_relay.replace(url.clone());
662                    }
663                }
664            }
665
666            // If we're changing our preferred relay but the old one's still
667            // accessible and the new one's not much better, just stick with
668            // where we are.
669            if prev_relay.is_some()
670                && r.preferred_relay != prev_relay
671                && !old_relay_cur_latency.is_zero()
672                && best_any > old_relay_cur_latency / 3 * 2
673            {
674                r.preferred_relay = prev_relay;
675            }
676        }
677
678        self.reports.prev.insert(now, r.clone());
679        self.reports.last = Some(r.clone());
680    }
681}

684async fn run_probe_v4(
685    ip_mapped_addrs: Option<IpMappedAddresses>,
686    relay_node: Arc<RelayNode>,
687    quic_client: QuicClient,
688    dns_resolver: DnsResolver,
689) -> n0_snafu::Result<(QadProbeReport, QadConn)> {
690    use n0_snafu::ResultExt;
691
692    let relay_addr_orig = reportgen::get_relay_addr_ipv4(&dns_resolver, &relay_node).await?;
693    let relay_addr =
694        reportgen::maybe_to_mapped_addr(ip_mapped_addrs.as_ref(), relay_addr_orig.into());
695
696    debug!(?relay_addr_orig, ?relay_addr, "relay addr v4");
697    let host = relay_node.url.host_str().context("missing host url")?;
698    let conn = quic_client.create_conn(relay_addr, host).await?;
699    let mut receiver = conn.observed_external_addr();
700
701    // wait for an addr
702    let addr = receiver
703        .wait_for(|addr| addr.is_some())
704        .await
705        .context("receiver dropped")?
706        .expect("known");
707    let report = QadProbeReport {
708        node: relay_node.url.clone(),
709        addr: SocketAddr::new(addr.ip().to_canonical(), addr.port()),
710        latency: conn.rtt(),
711    };
712
713    let observer = Watchable::new(None);
714    let node = relay_node.url.clone();
715    let handle = task::spawn({
716        let conn = conn.clone();
717        let observer = observer.clone();
718        async move {
719            loop {
720                let val = *receiver.borrow();
721                // if we've sent to an ipv4 address, but received an observed address
722                // that is ivp6 then the address is an [IPv4-Mapped IPv6 Addresses](https://doc.rust-lang.org/beta/std/net/struct.Ipv6Addr.html#ipv4-mapped-ipv6-addresses)
723                let val = val.map(|val| SocketAddr::new(val.ip().to_canonical(), val.port()));
724                let latency = conn.rtt();
725                trace!(?val, ?relay_addr, ?latency, "got addr V4");
726                observer
727                    .set(val.map(|addr| QadProbeReport {
728                        node: node.clone(),
729                        addr,
730                        latency,
731                    }))
732                    .ok();
733                if receiver.changed().await.is_err() {
734                    break;
735                }
736            }
737        }
738    });
739    let handle = AbortOnDropHandle::new(handle);
740
741    Ok((
742        report,
743        QadConn {
744            conn,
745            observer,
746            _handle: handle,
747        },
748    ))
749}

825        let server = server::Server::spawn(server::testing::server_config())
826            .await

837    }

839    /// Create a [`crate::RelayMap`] of the given size.
840    ///
841    /// This function uses [`relay`]. Note that the returned map uses internal order that will
842    /// often _not_ match the order of the servers.
843    pub(crate) async fn relay_map(relays: usize) -> (Vec<server::Server>, crate::RelayMap) {
844        let mut servers = Vec::with_capacity(relays);
845        let mut nodes = Vec::with_capacity(relays);
846        for _ in 0..relays {
847            let (relay_server, node) = relay().await;
848            servers.push(relay_server);
849            nodes.push(node);
850        }
851        (servers, crate::RelayMap::from_iter(nodes))
852    }
