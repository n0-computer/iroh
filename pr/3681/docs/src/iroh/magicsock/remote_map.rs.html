38/// Map containing all the state for endpoints.
39///
40/// - Has actors which each manage all the connection state for a remote endpoint.
41///
42/// - Has the mapped addresses we use to refer to non-IP transports destinations into IPv6
43///   addressing space that is used by Quinn.
44#[derive(Debug)]
45pub(crate) struct RemoteMap {
46    //
47    // State we keep about remote endpoints.
48    //
49    /// The actors tracking each remote endpoint.
50    actor_senders: Mutex<FxHashMap<EndpointId, mpsc::Sender<RemoteStateMessage>>>,
51    /// The mapping between [`EndpointId`]s and [`EndpointIdMappedAddr`]s.
52    pub(super) endpoint_mapped_addrs: AddrMap<EndpointId, EndpointIdMappedAddr>,
53    /// The mapping between endpoints via a relay and their [`RelayMappedAddr`]s.
54    pub(super) relay_mapped_addrs: AddrMap<(RelayUrl, EndpointId), RelayMappedAddr>,
55
56    //
57    // State needed to start a new RemoteStateHandle.
58    //
59    /// The endpoint ID of the local endpoint.
60    local_endpoint_id: EndpointId,
61    metrics: Arc<MagicsockMetrics>,
62    local_addrs: n0_watcher::Direct<BTreeSet<DirectAddr>>,
63    disco: DiscoState,
64    sender: TransportsSender,
65    discovery: ConcurrentDiscovery,
66    actor_tasks: Mutex<JoinSet<(EndpointId, Vec<RemoteStateMessage>)>>,

98    /// Removes the handles for terminated [`RemoteStateActor`]s from the endpoint map.
99    ///
100    /// This should be called periodically to remove handles to endpoint state actors
101    /// that have shutdown after their idle timeout expired.
102    pub(super) fn remove_closed_remote_state_actors(&self) {
103        let mut senders = self.actor_senders.lock().expect("poisoned");
104        senders.retain(|_eid, sender| !sender.is_closed());
105        while let Some(result) = self.actor_tasks.lock().expect("poisoned").try_join_next() {
106            match result {
107                Ok((eid, leftover_msgs)) => {
108                    let entry = senders.entry(eid);
109                    if leftover_msgs.is_empty() {
110                        match entry {
111                            hash_map::Entry::Occupied(occupied_entry) => occupied_entry.remove(),
112                            hash_map::Entry::Vacant(_) => {
113                                panic!("this should be impossible TODO(matheus23)");
114                            }
115                        };
116                    } else {
117                        // The remote actor got messages while it was closing, so we're restarting
118                        debug!(%eid, "restarting terminated remote state actor: messages received during shutdown");
119                        let sender = self.start_remote_state_actor(eid, leftover_msgs);
120                        entry.insert_entry(sender);
121                    }
122                }
123                Err(err) => {
124                    if let Ok(panic) = err.try_into_panic() {
125                        error!("RemoteStateActor panicked.");
126                        std::panic::resume_unwind(panic);
127                    }
128                }
129            }
130        }
131    }

133    /// Returns the sender for the [`RemoteStateActor`].
134    ///
135    /// If needed a new actor is started on demand.
136    ///
137    /// [`RemoteStateActor`]: remote_state::RemoteStateActor
138    pub(super) fn remote_state_actor(&self, eid: EndpointId) -> mpsc::Sender<RemoteStateMessage> {
139        let mut handles = self.actor_senders.lock().expect("poisoned");
140        match handles.entry(eid) {
141            hash_map::Entry::Occupied(entry) => entry.get().clone(),
142            hash_map::Entry::Vacant(entry) => {
143                let sender = self.start_remote_state_actor(eid, vec![]);
144                entry.insert(sender.clone());
145                sender
146            }
147        }
148    }

150    /// Starts a new remote state actor and returns a handle and a sender.
151    ///
152    /// The handle is not inserted into the endpoint map, this must be done by the caller of this function.
153    fn start_remote_state_actor(
154        &self,
155        eid: EndpointId,
156        initial_msgs: Vec<RemoteStateMessage>,
157    ) -> mpsc::Sender<RemoteStateMessage> {
158        // Ensure there is a RemoteMappedAddr for this EndpointId.
159        self.endpoint_mapped_addrs.get(&eid);
160        RemoteStateActor::new(
161            eid,
162            self.local_endpoint_id,
163            self.local_addrs.clone(),
164            self.disco.clone(),
165            self.relay_mapped_addrs.clone(),
166            self.metrics.clone(),
167            self.sender.clone(),
168            self.discovery.clone(),
169        )
170        .start(
171            initial_msgs,
172            self.actor_tasks.lock().expect("poisoned").deref_mut(),
173        )
174    }

176    pub(super) fn handle_ping(&self, msg: disco::Ping, sender: EndpointId, src: transports::Addr) {
177        if msg.endpoint_key != sender {
178            warn!("DISCO Ping EndpointId mismatch, ignoring ping");
179            return;
180        }
181        let remote_state = self.remote_state_actor(sender);
182        if let Err(err) = remote_state.try_send(RemoteStateMessage::PingReceived(msg, src)) {
183            // TODO: This is really, really bad and will drop pings under load.  But
184            //    DISCO pings are going away with QUIC-NAT-TRAVERSAL so I don't care.
185            warn!("DISCO Ping dropped: {err:#}");
186        }
187    }

217/// The origin or *source* through which an address associated with a remote endpoint
218/// was discovered.
219///
220/// An aggregate of the [`Source`]s of all the addresses of an endpoint describe the
221/// [`Source`]s of the endpoint itself.
222///
223/// A [`Source`] helps track how and where an address was learned. Multiple
224/// sources can be associated with a single address, if we have discovered this
225/// address through multiple means.
226#[derive(Serialize, Deserialize, strum::Display, Debug, Clone, Eq, PartialEq, Hash)]
227#[strum(serialize_all = "kebab-case")]
228#[allow(private_interfaces)]
229pub enum Source {
230    /// An endpoint communicated with us first via UDP.
231    Udp,
232    /// An endpoint communicated with us first via relay.
233    Relay,
234    /// Application layer added the address directly.
235    App,
236    /// The address was discovered by a discovery service.
237    #[strum(serialize = "{name}")]
238    Discovery {
239        /// The name of the discovery service that discovered the address.
240        name: String,
241    },
242    /// Application layer with a specific name added the endpoint directly.
243    #[strum(serialize = "{name}")]
244    NamedApp {
245        /// The name of the application that added the endpoint
246        name: String,
247    },
248    /// The address was advertised by a call-me-maybe DISCO message.
249    #[strum(serialize = "CallMeMaybe")]
250    CallMeMaybe {
251        /// private marker
252        _0: Private,
253    },
254    /// We received a ping on the path.
255    #[strum(serialize = "Ping")]
256    Ping {
257        /// private marker
258        _0: Private,
259    },
260    /// We established a connection on this address.
261    ///
262    /// Currently this means the path was in uses as [`PathId::ZERO`] when the a connection
263    /// was added to the `RemoteStateActor`.
264    ///
265    /// [`PathId::ZERO`]: quinn_proto::PathId::ZERO
266    #[strum(serialize = "Connection")]
267    Connection {
268        /// private marker
269        _0: Private,
270    },
271}

277/// An (Ip, Port) pair.
278///
279/// NOTE: storing an [`IpPort`] is safer than storing a [`SocketAddr`] because for IPv6 socket
280/// addresses include fields that can't be assumed consistent even within a single connection.
281#[derive(Debug, derive_more::Display, Clone, Copy, Hash, PartialEq, Eq, PartialOrd, Ord)]
282#[display("{}", SocketAddr::from(*self))]
283pub struct IpPort {
284    ip: IpAddr,
285    port: u16,
286}
287
288impl From<SocketAddr> for IpPort {
289    fn from(socket_addr: SocketAddr) -> Self {
290        Self {
291            ip: socket_addr.ip(),
292            port: socket_addr.port(),
293        }
294    }
295}
