12/// Custom transport.
13///
14/// Usually a transport will only deal with a single custom address type, but
15/// the signature allows for dealing with multiple custom address types.
16///
17/// A transport is a factory for custom endpoints. Whenever an iroh endpoint is
18/// created using [crate::endpoint::Builder::bind], a new custom endpoint will
19/// be created using [CustomTransport::bind].
20pub trait CustomTransport: std::fmt::Debug + Send + Sync + 'static {
21    /// Create a custom endpoint
22    ///
23    /// Analogously to [std::net::UdpSocket::bind], this is where the actual
24    /// underlying hardware resource is created.
25    fn bind(&self) -> io::Result<Box<dyn CustomEndpoint>>;
26}
27
28/// Custom endpoint created by a [CustomTransport].
29///
30/// An endpoint has a local address (or multiple local addresses), can receive
31/// packets, and can create senders to send packets.
32pub trait CustomEndpoint: std::fmt::Debug + Send + Sync + 'static {
33    /// A watcher for local addresses for this custom endpoint.
34    fn watch_local_addrs(&self) -> n0_watcher::Direct<Vec<CustomAddr>>;
35    /// Create a custom sender for this custom endpoint.
36    fn create_sender(&self) -> Arc<dyn CustomSender>;
37    /// poll receiving a packet on this custom endpoint.
38    ///
39    /// This will be called with `bufs`, `metas` and `source_addrs` of the same length.
40    /// It is acceptable to panic if this is not the case.
41    ///
42    /// The maximum length of the slices is [`noq_udp::BATCH_SIZE`].
43    /// It is acceptable to panic if this is exceeded.
44    ///
45    /// On success, all three slices must be filled up to the returned length,
46    /// and the returned length must be less than or equal to the length of the slices.
47    ///
48    /// It does not make much sense to return addresses unrelated to this transport.
49    fn poll_recv(
50        &mut self,
51        cx: &mut Context,
52        bufs: &mut [io::IoSliceMut<'_>],
53        metas: &mut [noq_udp::RecvMeta],
54        source_addrs: &mut [Addr],
55    ) -> Poll<io::Result<usize>>;
56
57    /// Maximum number of segments to transmit (GSO).
58    ///
59    /// This controls how many datagrams Noq will batch into a single transmit.
60    /// The default is 1 (no batching). Custom transports that support batching
61    /// can override this to allow more efficient transmission.
62    fn max_transmit_segments(&self) -> NonZeroUsize {
63        NonZeroUsize::MIN
64    }
65}

67/// Custom sender
68///
69/// A sender provides a poll based interface to send packets to custom addresses.
70/// It can decide whether it wants to send to a given custom address type.
71///
72/// This is not enforced at type level, but [CustomSender::poll_send] should
73/// only be called with addresses for which [CustomSender::is_valid_send_addr]
74/// returns true.
75pub trait CustomSender: std::fmt::Debug + Send + Sync + 'static {
76    /// True if this sender can send to the given address.
77    fn is_valid_send_addr(&self, addr: &CustomAddr) -> bool;
78    /// poll sending a packet on this sender.
79    ///
80    /// This will only be called from iroh with addresses for which [CustomSender::is_valid_send_addr] returns true.
81    ///
82    /// You should handle invalid addresses by returning an error.
83    fn poll_send(
84        &self,
85        cx: &mut std::task::Context,
86        dst: &CustomAddr,
87        transmit: &Transmit<'_>,
88    ) -> Poll<io::Result<()>>;
89}
