40/// Exposes a JVM to iroh so that we can read the system's DNS configuration.
41///
42/// This calls [`ndk_context::initialize_android_context`] to expose a
43/// `JavaVM` and Application Context to Rust code so that we can use JNI.
44/// This is required to get the configured nameservers on Android.
45///
46/// If this function is not called, fetching the configured nameservers will fail
47/// and the default [`DnsResolver`] will use fallback nameservers instead.
48///
49/// If you call [`ndk_context::initialize_android_context`] already somewhere
50/// up the stack in your app, or use a crate like `ndk-glue` or `android-activity`
51/// that do this for you, then there's no need to call this function.
52///
53/// If you don't use a glue crate, a typical way to initialize the context is
54/// via `JNI_OnLoad`:
55///
56/// *Note: `install_android_jni_context` is reexported from `iroh`, so you can substitute
57/// `iroh_dns` for `iroh` below.*
58///
59/// ```ignore
60/// #[cfg(target_os = "android")]
61/// #[no_mangle]
62/// pub extern "C" fn JNI_OnLoad(
63///     vm: jni::JavaVM,
64///     res: *mut std::os::raw::c_void,
65/// ) -> jni::sys::jint {
66///     use std::ffi::c_void;
67///
68///     let vm = vm.get_java_vm_pointer() as *mut c_void;
69///     unsafe {
70///         iroh_dns::install_android_jni_context(vm, res);
71///     }
72///     jni::JNIVersion::V6.into()
73/// }
74/// ```
75///
76/// # Safety
77///
78/// Both the `java_vm` and `context_jobject` pointers must remain valid until the process exits.
79/// See also [`ndk_context::initialize_android_context`].
80///
81/// [`ndk_context`]: https://docs.rs/ndk-context
82/// [`ndk_context::initialize_android_context`]: https://docs.rs/ndk-context/latest/ndk_context/fn.initialize_android_context.html
83//
84// Inlined rather than re-exported from `n0-dns-resolver`: a `pub use` of the
85// resolver crate's item does not resolve on a `doc`-only build (the dependency
86// is compiled without the `doc` cfg, so its item is absent), so we mirror the
87// signature and cfg here and delegate to it on Android.
88#[cfg(any(target_os = "android", doc))]
89pub unsafe fn install_android_jni_context(
90    java_vm: *mut std::ffi::c_void,
91    application_context: *mut std::ffi::c_void,
92) {
93    #[cfg(target_os = "android")]
94    unsafe {
95        n0_dns_resolver::install_android_jni_context(java_vm, application_context);
96    }
97    #[cfg(not(target_os = "android"))]
98    let _ = (java_vm, application_context);
99}

104/// Trait for DNS resolvers used in iroh.
105pub trait Resolver: fmt::Debug + Send + Sync + 'static {
106    /// Looks up an IPv4 address.
107    fn lookup_ipv4(&self, host: String) -> BoxFuture<Result<BoxIter<Ipv4Addr>, DnsError>>;
108
109    /// Looks up an IPv6 address.
110    fn lookup_ipv6(&self, host: String) -> BoxFuture<Result<BoxIter<Ipv6Addr>, DnsError>>;
111
112    /// Looks up TXT records.
113    fn lookup_txt(&self, host: String) -> BoxFuture<Result<BoxIter<TxtRecordData>, DnsError>>;
114
115    /// Clears the internal cache.
116    fn clear_cache(&self);
117
118    /// Returns a freshly-built resolver to replace `self` after a network change.
119    ///
120    /// The returned resolver replaces the previous one inside [`DnsResolver`] via an
121    /// atomic swap. Build a new instance with re-bound sockets and re-read nameserver
122    /// configuration rather than mutating in place. Must not perform IO: defer DNS
123    /// queries and socket binds until the new resolver is first used. May be called
124    /// concurrently, in which case all but one allocated replacement is dropped unused.
125    fn reset(&self) -> Box<dyn Resolver>;
126}

133/// Potential errors related to DNS operations.
134///
135/// Distinct, matchable causes have their own variant, such as [`Self::NxDomain`]
136/// (a nameserver reported the name does not exist, unlike a transient transport
137/// or timeout failure). Any other cause is reported as [`Self::Resolve`], whose
138/// [`AnyError`] source carries the underlying error.
139///
140/// The resolver backing this crate is an implementation detail and its error
141/// types are not part of the public API. If you depend on `n0-dns-resolver`
142/// directly and need the exact reason, downcast a [`Self::Resolve`] source:
143///
144/// ```ignore
145/// if let DnsError::Resolve { source, .. } = &err {
146///     if let Some(inner) = source.downcast_ref::<n0_dns_resolver::Error>() {
147///         // match on `inner` for the specific failure
148///     }
149/// }
150/// ```
151#[allow(missing_docs)]
152#[stack_error(derive, add_meta, std_sources)]
153#[non_exhaustive]
154pub enum DnsError {
155    #[error("Request timed out")]
156    Timeout {},
157    #[error("No response")]
158    NoResponse {},
159    #[error("Resolve failed, IPv4: {ipv4}, IPv6: {ipv6}")]
160    ResolveBoth {
161        ipv4: Box<DnsError>,
162        ipv6: Box<DnsError>,
163    },
164    #[error("Missing host")]
165    MissingHost {},
166    /// A resolution failure not covered by another variant. Downcast the source
167    /// (see the type-level docs) to recover the specific cause.
168    #[error("Failed to resolve")]
169    Resolve {
170        #[error(from)]
171        source: AnyError,
172    },
173    #[error("invalid DNS response: not a query for _iroh.z32encodedpubkey")]
174    InvalidResponse {},
175    /// The domain name does not exist (NXDOMAIN).
176    ///
177    /// A nameserver authoritatively reported that the name does not exist, as
178    /// opposed to a transient transport or timeout failure, so retrying will not
179    /// make the lookup succeed. Appended after [`Self::InvalidResponse`] so the
180    /// pre-existing variants keep their discriminants (API compatibility).
181    #[error("domain name does not exist (NXDOMAIN)")]
182    NxDomain {},
183}

221/// How the resolver uses its fallback nameservers relative to the primary ones.
222///
223/// The *primary* nameservers come from the system DNS configuration and the
224/// nameservers added on the [`Builder`]. The *fallback* nameservers default to a
225/// set of public resolvers, which [`Builder::with_fallback_nameservers`] can
226/// override. Select the mode with [`Builder::with_fallback_mode`].
227#[derive(Debug, Default, Copy, Clone, Eq, PartialEq)]
228#[non_exhaustive]
229pub enum FallbackMode {
230    /// Never query the fallback nameservers.
231    Never,
232    /// Race the fallback nameservers alongside the primary ones from the start.
233    Always,
234    /// Use the fallback nameservers only when the system DNS configuration could
235    /// not be loaded or configured no nameservers.
236    IfSystemUnavailable,
237    /// Keep the fallback nameservers as a lower-priority tier, queried only once
238    /// every primary nameserver has failed or timed out. This is the default.
239    #[default]
240    Deferred,
241}
242
243impl FallbackMode {
244    /// Converts into the resolver crate's fallback mode.
245    fn into_inner(self) -> n0_dns_resolver::FallbackMode {
246        match self {
247            FallbackMode::Never => n0_dns_resolver::FallbackMode::Never,
248            FallbackMode::Always => n0_dns_resolver::FallbackMode::Always,
249            FallbackMode::IfSystemUnavailable => n0_dns_resolver::FallbackMode::IfSystemUnavailable,
250            FallbackMode::Deferred => n0_dns_resolver::FallbackMode::Deferred,
251        }
252    }
253}

255/// A configured nameserver: its address, transport, and an optional TLS server
256/// name for DNS-over-TLS / DNS-over-HTTPS.
257///
258/// The connection is always made to `addr`. When `server_name` is set it is
259/// used for the TLS SNI and certificate validation (and as the DoH URL
260/// authority, with the address pinned); otherwise DoT/DoH are addressed by IP.
261#[derive(Debug, Clone)]
262struct Nameserver {
263    addr: SocketAddr,
264    protocol: DnsProtocol,
265    /// Only used for DoT/DoH.
266    server_name: Option<String>,
267}
268
269impl Nameserver {
270    /// A nameserver addressed by IP, with no TLS server name.
271    fn new(addr: SocketAddr, protocol: DnsProtocol) -> Self {
272        Self {
273            addr,
274            protocol,
275            server_name: None,
276        }
277    }
278
279    /// Converts into the resolver crate's nameserver type.
280    fn into_inner(self) -> n0_dns_resolver::Nameserver {
281        if let Some(server_name) = self.server_name {
282            return n0_dns_resolver::Nameserver::with_server_name(
283                self.addr,
284                self.protocol.into_inner(),
285                server_name,
286            );
287        }
288        n0_dns_resolver::Nameserver::new(self.addr, self.protocol.into_inner())
289    }
290
291    /// Adds this nameserver as a primary nameserver on a resolver builder.
292    fn add_to(self, builder: n0_dns_resolver::Builder) -> n0_dns_resolver::Builder {
293        if let Some(server_name) = self.server_name {
294            return builder.nameserver_with_name(
295                self.addr,
296                self.protocol.into_inner(),
297                server_name,
298            );
299        }
300        builder.nameserver(self.addr, self.protocol.into_inner())
301    }
302}

304/// Protocols over which DNS records can be resolved.
305#[derive(Debug, Default, Copy, Clone, Eq, PartialEq)]
306#[non_exhaustive]
307pub enum DnsProtocol {
308    /// DNS over UDP
309    ///
310    /// This is the classic DNS protocol and supported by most DNS servers.
311    #[default]
312    Udp,
313    /// DNS over TCP
314    ///
315    /// This is specified in the original DNS RFCs, but is not supported by all DNS servers.
316    Tcp,
317    /// DNS over TLS
318    ///
319    /// Performs DNS lookups over TLS-encrypted TCP connections, as defined in [RFC 7858].
320    ///
321    /// [RFC 7858]: https://www.rfc-editor.org/rfc/rfc7858.html
322    Tls,
323    /// DNS over HTTPS
324    ///
325    /// Performs DNS lookups over HTTPS, as defined in [RFC 8484].
326    ///
327    /// [RFC 8484]: https://www.rfc-editor.org/rfc/rfc8484.html
328    Https,
329}
330
331impl DnsProtocol {
332    /// Converts into the resolver crate's protocol type.
333    fn into_inner(self) -> n0_dns_resolver::DnsProtocol {
334        match self {
335            DnsProtocol::Udp => n0_dns_resolver::DnsProtocol::Udp,
336            DnsProtocol::Tcp => n0_dns_resolver::DnsProtocol::Tcp,
337            DnsProtocol::Tls => n0_dns_resolver::DnsProtocol::Tls,
338            DnsProtocol::Https => n0_dns_resolver::DnsProtocol::Https,
339        }
340    }
341}

419    /// Sets how the fallback nameservers are used relative to the primary ones.
420    ///
421    /// The default is [`FallbackMode::Deferred`]: the fallback nameservers are a
422    /// lower-priority tier, queried only when the primary nameservers fail or

430    ///
431    /// Shorthand for [`Self::with_fallback_mode`] with [`FallbackMode::Never`].

436    /// Replaces the default public-resolver fallback with the given nameservers,
437    /// each addressed by IP.
438    ///
439    /// Has no effect when the fallback mode is [`FallbackMode::Never`].
440    pub fn with_fallback_nameservers(
441        mut self,
442        nameservers: impl IntoIterator<Item = (SocketAddr, DnsProtocol)>,
443    ) -> Self {
444        self.fallback_nameservers.extend(
445            nameservers
446                .into_iter()
447                .map(|(addr, protocol)| Nameserver::new(addr, protocol)),
448        );
449        self
450    }
451
452    /// Builds the DNS resolver.
453    pub fn build(self) -> DnsResolver {
454        let mut builder =
455            n0_dns_resolver::DnsResolver::builder().fallback_mode(self.fallback_mode.into_inner());
456        if !self.use_system_defaults {
457            builder = builder.without_system_defaults();
458        }
459        for nameserver in self.nameservers {
460            builder = nameserver.add_to(builder);
461        }
462        if !self.fallback_nameservers.is_empty() {
463            builder = builder.fallback_nameservers(
464                self.fallback_nameservers
465                    .into_iter()
466                    .map(Nameserver::into_inner),
467            );
468        }
469        #[cfg(not(wasm_browser))]
470        if let Some(tls_client_config) = self.tls_client_config {
471            builder = builder.tls_client_config(tls_client_config);
472        }
473        DnsResolver::custom(DefaultResolver(Arc::new(builder.build())))
474    }

477/// Adapts the [`n0_dns_resolver::DnsResolver`] to the [`Resolver`] trait,

485        let this = self.0.clone();
486        Box::pin(async move {
487            let iter = this.lookup_ipv4(host).await.map_err(map_resolve_error)?;
488            let iter: BoxIter<_> = Box::new(iter);

495        Box::pin(async move {

497            let iter: BoxIter<_> = Box::new(iter);

500    }

503        let this = self.0.clone();
504        Box::pin(async move {
505            let iter = this.lookup_txt(host).await.map_err(map_resolve_error)?;

520/// Maps an [`n0_dns_resolver::Error`] onto this crate's [`DnsError`].
521///
522/// A private function rather than a `From` impl, so `n0_dns_resolver`'s error
523/// types stay out of this crate's public API. The distinct, matchable causes map
524/// to their own variant; every other cause becomes [`DnsError::Resolve`] carrying
525/// the original error as its [`AnyError`] source, which callers can downcast (see
526/// the [`DnsError`] docs).
527fn map_resolve_error(err: n0_dns_resolver::Error) -> DnsError {
528    use n0_dns_resolver::Error as E;
529    match err {
530        E::Timeout { .. } => e!(DnsError::Timeout),
531        E::NoResponse { .. } => e!(DnsError::NoResponse),
532        E::NxDomain { .. } => e!(DnsError::NxDomain),
533        E::InvalidResponse { .. } => e!(DnsError::InvalidResponse),
534        other => e!(DnsError::Resolve, AnyError::from_stack(other)),
535    }
536}

542fn convert_txt(txt: n0_dns_resolver::TxtRecordData) -> TxtRecordData {

546/// The DNS resolver used throughout `iroh`.
547///
548/// By default, we use a built-in resolver that reads the system's DNS configuration.
549/// The nameservers can be customized by constructing the resolver with [`Self::builder`].
550/// Alternatively, you can create a fully custom DNS resolver by implementing the [`Resolver`]
551/// trait and creating the resolver with [`Self::custom`].
552///
553/// # Usage on Android
554///
555/// The system-defaults reader uses JNI through [`ndk_context`], which must be
556/// initialized with a `JavaVM` and `Application` context before the resolver
557/// is constructed. Glue crates like ndk-glue and android-activity do this
558/// before `main`. Apps that don't use either should call [`install_android_jni_context`]
559/// once at startup, see docs there for details.
560///
561/// If `ndk_context` is not initialized, fetching the system config on Android will fail
562/// and the resolver will use Google's fallback DNS servers. Due to how things are
563/// implemented in `ndk_context`, detecting the failure relies on unwinding a panic.
564/// If your app uses `panic = "abort"` in its compilation profile, this doesn't work,
565/// so in that case your app will panic if no JNI context is initialized.
566/// Therefore, either make sure that the JNI context is installed, or don't use
567/// `panic = "abort"`.
568///
569/// [`install_android_jni_context`]: crate::install_android_jni_context
570/// [`ndk_context`]: https://docs.rs/ndk-context
571#[derive(Debug, Clone)]
572pub struct DnsResolver {
573    inner: Arc<Inner>,
574}
575
576/// Shared state behind [`DnsResolver`].
577#[derive(Debug)]
578struct Inner {
579    /// Wakes in-flight [`Self::op`] calls when the resolver is swapped.
580    notify_reset: Notify,
581    resolver: ArcSwap<Box<dyn Resolver>>,
582}
583
584impl Inner {
585    fn new(inner: Box<dyn Resolver>) -> Self {
586        Self {
587            notify_reset: Notify::new(),
588            resolver: ArcSwap::from_pointee(inner),
589        }
590    }
591
592    /// Atomically swaps the resolver and wakes in-flight [`Self::op`] calls.
593    ///
594    /// The swap happens before the wake. An op that observes or misses the wake is
595    /// then guaranteed to load the new resolver. Non-blocking.
596    ///
597    /// Under contention only the first concurrent caller's swap lands; the others
598    /// drop their freshly-built resolver. The winner's notification is enough since
599    /// every in-flight op will pick up the new resolver on its next iteration.
600    fn reset(&self) {
601        let current = self.resolver.load();
602        let new = Arc::new(current.reset());
603        let prev = self.resolver.compare_and_swap(&current, new);
604        if Arc::ptr_eq(&current, &prev) {
605            self.notify_reset.notify_waiters();
606        }
607    }
608

613    /// Runs `f(resolver)` with a timeout, restarting against the new resolver if
614    /// [`Self::reset`] fires.
615    ///
616    /// Three things race in `biased` order: the lookup completes (returned even if a
617    /// reset happened concurrently, since a successful result is still valid), a reset
618    /// is observed (drop the in-flight future and re-run `f`), or the timeout elapses.
619    ///
620    /// `timeout` is per-attempt. Each retry starts a fresh sleep, so the wall-clock
621    /// total can exceed it if many resets fire. This is intentional: a fresh attempt
622    /// against a just-changed network should not inherit the previous attempt's
623    /// remaining budget.
624    ///
625    /// `f` may be invoked more than once and so must be `Fn`. Captured state must be
626    /// reusable across calls, typically by cloning inside the closure body.
627    ///
628    /// `notified` is enabled before `load_full`. Combined with [`Self::reset`]'s
629    /// swap-then-notify ordering: a wake missed before `enable()` had already been
630    /// preceded by the swap, so the following `load_full` returns the new resolver.
631    async fn op<F, Fut, R, E>(&self, timeout: Duration, f: F) -> Result<R, DnsError>
632    where
633        E: 'static + Send + Into<DnsError>,
634        R: 'static + Send,
635        F: 'static + Send + Fn(Arc<Box<dyn Resolver>>) -> Fut,
636        Fut: 'static + Send + Future<Output = Result<R, E>>,
637    {
638        loop {
639            let notified = self.notify_reset.notified();
640            tokio::pin!(notified);
641            notified.as_mut().enable();
642
643            let timeout = n0_future::time::sleep(timeout);
644            tokio::pin!(timeout);
645
646            let resolver = self.resolver.load_full();
647            let fut = f(resolver);
648            tokio::pin!(fut);
649
650            tokio::select! {
651                biased;
652                res = fut => return res.map_err(Into::into),
653                _ = notified => continue,
654                _ = timeout => return Err(e!(DnsError::Timeout)),
655            }
656        }
657    }
658}

757        timeout: Duration,

759        let host = host.to_string();

761            self.lookup_ipv4(host.clone(), timeout),
762            self.lookup_ipv6(host, timeout)
763        );
764
765        match res {
766            (Ok(ipv4), Ok(ipv6)) => Ok(LookupIter::Both(ipv4.chain(ipv6))),
767            (Ok(ipv4), Err(_)) => Ok(LookupIter::Ipv4(ipv4)),
768            (Err(_), Ok(ipv6)) => Ok(LookupIter::Ipv6(ipv6)),
769            (Err(ipv4_err), Err(ipv6_err)) => Err(e!(DnsError::ResolveBoth {
770                ipv4: Box::new(ipv4_err),
771                ipv6: Box::new(ipv6_err)
772            })),
773        }
774    }

776    /// Resolves a hostname from a URL to an IP address.
777    pub async fn resolve_host(
778        &self,
779        url: &Url,
780        prefer_ipv6: bool,
781        timeout: Duration,
782    ) -> Result<IpAddr, DnsError> {
783        let host = url.host().ok_or_else(|| e!(DnsError::MissingHost))?;
784        match host {
785            url::Host::Domain(domain) => {
786                // Need to do a DNS lookup
787                let lookup = tokio::join!(
788                    self.lookup_ipv4(domain, timeout),
789                    self.lookup_ipv6(domain, timeout)
790                );
791                let (v4, v6) = match lookup {
792                    (Err(ipv4_err), Err(ipv6_err)) => {
793                        return Err(e!(DnsError::ResolveBoth {
794                            ipv4: Box::new(ipv4_err),
795                            ipv6: Box::new(ipv6_err)
796                        }));
797                    }
798                    (Err(_), Ok(mut v6)) => (None, v6.next()),
799                    (Ok(mut v4), Err(_)) => (v4.next(), None),
800                    (Ok(mut v4), Ok(mut v6)) => (v4.next(), v6.next()),
801                };
802                if prefer_ipv6 {
803                    v6.or(v4).ok_or_else(|| e!(DnsError::NoResponse))
804                } else {
805                    v4.or(v6).ok_or_else(|| e!(DnsError::NoResponse))
806                }
807            }
808            url::Host::Ipv4(ip) => Ok(IpAddr::V4(ip)),
809            url::Host::Ipv6(ip) => Ok(IpAddr::V6(ip)),
810        }
811    }

813    /// Resolves a hostname from a URL to its IP addresses, streamed as they resolve.
814    ///
815    /// IPv4 and IPv6 are looked up concurrently and each address is yielded as
816    /// soon as its lookup completes, so a caller can start dialing without
817    /// waiting for both families.
818    ///
819    /// A lookup failure is swallowed as long as the other family yields an address.
820    /// Only if both lookups fail does the stream yield a single [`DnsError`].
821    ///
822    /// The stream ends once both lookups have finished and every address or the error
823    /// have been yielded.
824    pub fn resolve_host_all<'a>(
825        &'a self,
826        url: &Url,
827        timeout: Duration,
828    ) -> impl Stream<Item = Result<IpAddr, DnsError>> + Send + 'a {
829        let host = match url.host() {
830            None => {
831                return Either::Left(stream::once(Err(e!(DnsError::MissingHost))));
832            }
833            Some(url::Host::Ipv4(ip)) => {
834                return Either::Left(stream::once(Ok(IpAddr::V4(ip))));
835            }
836            Some(url::Host::Ipv6(ip)) => {
837                return Either::Left(stream::once(Ok(IpAddr::V6(ip))));
838            }
839            Some(url::Host::Domain(domain)) => domain.to_string(),
840        };
841
842        type Lookup<'a, A> =
843            Pin<Box<dyn Future<Output = Result<BoxIter<A>, DnsError>> + Send + 'a>>;
844
845        struct State<'a> {
846            v4_fut: MaybeFuture<Lookup<'a, Ipv4Addr>>,
847            v6_fut: MaybeFuture<Lookup<'a, Ipv6Addr>>,
848            v4_err: Option<DnsError>,
849            v6_err: Option<DnsError>,
850            queue: VecDeque<IpAddr>,
851            closed: bool,
852            yielded: bool,
853        }
854
855        let state = State {
856            v4_fut: MaybeFuture::Some(Box::pin({
857                let host = host.clone();
858                self.inner.op(timeout, move |r| r.lookup_ipv4(host.clone()))
859            })),
860            v6_fut: MaybeFuture::Some(Box::pin({
861                let host = host.clone();
862                self.inner.op(timeout, move |r| r.lookup_ipv6(host.clone()))
863            })),
864            v4_err: None,
865            v6_err: None,
866            queue: VecDeque::new(),
867            closed: false,
868            yielded: false,
869        };
870
871        Either::Right(stream::unfold(state, async |mut state| {
872            loop {
873                if state.closed {
874                    return None;
875                }
876
877                if let Some(item) = state.queue.pop_front() {
878                    state.yielded = true;
879                    return Some((Ok(item), state));
880                }
881
882                // Return final error item once both futures completed, or None if items were yielded.
883                if state.v4_fut.is_none() && state.v6_fut.is_none() {
884                    state.closed = true;
885                    if let (Some(v4), Some(v6)) = (state.v4_err.take(), state.v6_err.take()) {
886                        let error = e!(DnsError::ResolveBoth {
887                            ipv4: Box::new(v4),
888                            ipv6: Box::new(v6),
889                        });
890                        return Some((Err(error), state));
891                    } else if !state.yielded {
892                        return Some((Err(e!(DnsError::NoResponse)), state));
893                    } else {
894                        return None;
895                    }
896                }
897                tokio::select! {
898                    // We don't actually care about polling order, but `biased` saves the randomization cost.
899                    biased;
900                    res = &mut state.v4_fut => {
901                        match res {
902                            Ok(items) => state.queue.extend(items.map(IpAddr::V4)),
903                            Err(err) => state.v4_err = Some(err),
904                        }
905                    }
906                    res = &mut state.v6_fut => {
907                        match res {
908                            Ok(items) => state.queue.extend(items.map(IpAddr::V6)),
909                            Err(err) => state.v6_err = Some(err),
910                        }
911                    }
912                }
913            }
914        }))
915    }

1090/// Helper enum to give a unified type to the iterators of [`DnsResolver::lookup_ipv4_ipv6`].
1091enum LookupIter<A, B> {
1092    Ipv4(A),
1093    Ipv6(B),
1094    Both(std::iter::Chain<A, B>),
1095}
1096
1097impl<A: Iterator<Item = IpAddr>, B: Iterator<Item = IpAddr>> Iterator for LookupIter<A, B> {
1098    type Item = IpAddr;
1099
1100    fn next(&mut self) -> Option<Self::Item> {
1101        match self {
1102            LookupIter::Ipv4(iter) => iter.next(),
1103            LookupIter::Ipv6(iter) => iter.next(),
1104            LookupIter::Both(iter) => iter.next(),
1105        }
1106    }

1109/// Staggers calls to the future F with the given delays.
1110///
1111/// The first call is performed immediately. The first call to succeed generates an Ok result
1112/// ignoring any previous error. If all calls fail, an error summarizing all errors is returned.
1113async fn stagger_call<
1114    T,
1115    E: StackError + 'static,
1116    F: Fn() -> Fut,
1117    Fut: Future<Output = Result<T, E>>,
1118>(
1119    f: F,
1120    delays_ms: &[u64],
1121) -> Result<T, StaggeredError<E>> {
1122    let mut calls = n0_future::FuturesUnorderedBounded::new(delays_ms.len() + 1);
1123    // NOTE: we add the 0 delay here to have a uniform set of futures. This is more performant than
1124    // using alternatives that allow futures of different types.
1125    for delay in std::iter::once(&0u64).chain(delays_ms) {
1126        let delay = add_jitter(delay);
1127        let fut = f();
1128        let staggered_fut = async move {
1129            time::sleep(delay).await;
1130            fut.await
1131        };
1132        calls.push(staggered_fut)
1133    }
1134
1135    let mut errors = vec![];
1136    while let Some(call_result) = calls.next().await {
1137        match call_result {
1138            Ok(t) => return Ok(t),
1139            Err(e) => errors.push(e),
1140        }
1141    }
1142
1143    Err(e!(StaggeredError { errors }))
1144}
1145
1146fn add_jitter(delay: &u64) -> Duration {
1147    // If delay is 0, return 0 immediately.
1148    if *delay == 0 {
1149        return Duration::ZERO;
1150    }
1151
1152    // Calculate jitter as a random value in the range of +/- MAX_JITTER_PERCENT of the delay.
1153    let max_jitter = delay.saturating_mul(MAX_JITTER_PERCENT * 2) / 100;
1154    let jitter = rand::random::<u64>() % max_jitter;
1155
1156    Duration::from_millis(delay.saturating_sub(max_jitter / 2).saturating_add(jitter))
1157}
