38pub fn run_iroh(opt: Opt) -> Result<()> {
39    let server_span = tracing::error_span!("server");
40    let runtime = rt();
41    #[cfg(feature = "qlog")]
42    let qlog = QlogFileGroup::from_env("iroh-bench-iroh");
43
44    #[cfg(feature = "local-relay")]
45    let (relay_url, relay_server) = if opt.only_relay {
46        let (_, relay_url, relay_server) =
47            runtime.block_on(::iroh::test_utils::run_relay_server())?;
48
49        (Some(relay_url), Some(relay_server))
50    } else {
51        (None, None)
52    };
53    #[cfg(not(feature = "local-relay"))]
54    let relay_url = None;
55
56    let (server_addr, endpoint) = {
57        let _guard = server_span.enter();
58        iroh::server_endpoint(
59            &runtime,
60            &relay_url,
61            &opt,
62            #[cfg(feature = "qlog")]
63            &qlog,
64        )
65    };
66
67    #[cfg(feature = "metrics")]
68    let endpoint_metrics = endpoint.metrics().clone();
69
70    let server_thread = std::thread::spawn(move || {
71        let _guard = server_span.entered();
72        if let Err(e) = runtime.block_on(iroh::server(endpoint, opt)) {
73            eprintln!("server failed: {e:#}");
74        }
75    });
76
77    let mut handles = Vec::new();
78    for id in 0..opt.clients {
79        let server_addr = server_addr.clone();
80        let relay_url = relay_url.clone();
81        #[cfg(feature = "qlog")]
82        let qlog = qlog.clone();
83        handles.push(std::thread::spawn(move || {
84            let _guard = tracing::error_span!("client", id).entered();
85            let runtime = rt();
86            match runtime.block_on(iroh::client(
87                server_addr,
88                relay_url.clone(),
89                opt,
90                #[cfg(feature = "qlog")]
91                &qlog,
92            )) {
93                Ok(stats) => Ok(stats),
94                Err(e) => {
95                    eprintln!("client failed: {e:#}");
96                    Err(e)
97                }
98            }
99        }));
100    }
101
102    for (id, handle) in handles.into_iter().enumerate() {
103        // We print all stats at the end of the test sequentially to avoid
104        // them being garbled due to being printed concurrently
105        if let Ok(stats) = handle.join().expect("client thread") {
106            stats.print(id);
107        }
108    }
109
110    #[cfg(feature = "metrics")]
111    if opt.metrics {
112        // print metrics
113        println!("\nMetrics:");
114        collect_and_print("MagicsockMetrics", &*endpoint_metrics.magicsock);
115        collect_and_print("NetReportMetrics", &*endpoint_metrics.net_report);
116        collect_and_print("PortmapMetrics", &*endpoint_metrics.portmapper);
117        #[cfg(feature = "local-relay")]
118        if let Some(relay_server) = relay_server.as_ref() {
119            collect_and_print("RelayServerMetrics", &*relay_server.metrics().server);
120        }
121    }
122
123    server_thread.join().expect("server thread");
124
125    Ok(())
126}

129pub fn run_quinn(opt: Opt) -> Result<()> {
130    use rustls::pki_types::{CertificateDer, PrivatePkcs8KeyDer};
131
132    #[cfg(feature = "qlog")]
133    let qlog = QlogFileGroup::from_env("iroh-bench-quinn");
134
135    let server_span = tracing::error_span!("server");
136    let runtime = rt();
137    let cert = rcgen::generate_simple_self_signed(vec!["localhost".into()]).unwrap();
138    let key = PrivatePkcs8KeyDer::from(cert.signing_key.serialize_der());
139    let cert = CertificateDer::from(cert.cert);
140
141    let (server_addr, endpoint) = {
142        let _guard = server_span.enter();
143        quinn::server_endpoint(
144            &runtime,
145            cert.clone(),
146            key.into(),
147            &opt,
148            #[cfg(feature = "qlog")]
149            &qlog,
150        )
151    };
152
153    let server_thread = std::thread::spawn(move || {
154        let _guard = server_span.entered();
155        if let Err(e) = runtime.block_on(quinn::server(endpoint, opt)) {
156            eprintln!("server failed: {e:#}");
157        }
158    });
159
160    let mut handles = Vec::new();
161    for id in 0..opt.clients {
162        let cert = cert.clone();
163        #[cfg(feature = "qlog")]
164        let qlog = qlog.clone();
165        handles.push(std::thread::spawn(move || {
166            let _guard = tracing::error_span!("client", id).entered();
167            let runtime = rt();
168            match runtime.block_on(quinn::client(
169                server_addr,
170                cert,
171                opt,
172                #[cfg(feature = "qlog")]
173                &qlog,
174            )) {
175                Ok(stats) => Ok(stats),
176                Err(e) => {
177                    eprintln!("client failed: {e:#}");
178                    Err(e)
179                }
180            }
181        }));
182    }
183
184    for (id, handle) in handles.into_iter().enumerate() {
185        // We print all stats at the end of the test sequentially to avoid
186        // them being garbled due to being printed concurrently
187        if let Ok(stats) = handle.join().expect("client thread") {
188            stats.print(id);
189        }
190    }
191
192    server_thread.join().expect("server thread");
193
194    Ok(())
195}

201#[cfg(feature = "metrics")]
202fn collect_and_print(category: &'static str, metrics: &dyn MetricsGroup) {
203    let mut map = BTreeMap::new();
204    for item in metrics.iter() {
205        let value: i64 = match item.value() {
206            MetricValue::Counter(v) => v as i64,
207            MetricValue::Gauge(v) => v,
208            _ => continue,
209        };
210        map.insert(item.name().to_string(), value);
211    }
212    println!("{category}: {map:#?}");
213}
