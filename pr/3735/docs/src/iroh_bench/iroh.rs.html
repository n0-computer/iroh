22/// Creates a server endpoint which runs on the given runtime
23pub fn server_endpoint(
24    rt: &tokio::runtime::Runtime,
25    relay_url: &Option<RelayUrl>,
26    opt: &Opt,
27    #[cfg(feature = "qlog")] qlog: &QlogFileGroup,
28) -> (EndpointAddr, Endpoint) {
29    let _guard = rt.enter();
30    rt.block_on(async move {
31        let relay_mode = relay_url
32            .clone()
33            .map_or(RelayMode::Disabled, |url| RelayMode::Custom(url.into()));
34
35        #[allow(unused_mut)]
36        let mut builder = Endpoint::builder();
37        #[cfg(feature = "local-relay")]
38        {
39            builder = builder.insecure_skip_relay_cert_verify(relay_url.is_some());
40            if opt.only_relay {
41                builder = builder.clear_ip_transports();
42            }
43        }
44        #[cfg(feature = "qlog")]
45        let config = qlog
46            .transport_config("server".into(), VantagePointType::Server)
47            .unwrap();
48        #[cfg(not(feature = "qlog"))]
49        let config = Default::default();
50        let ep = builder
51            .alpns(vec![ALPN.to_vec()])
52            .relay_mode(relay_mode)
53            .transport_config(transport_config(config, opt.max_streams, opt.initial_mtu))
54            .bind()
55            .await
56            .unwrap();
57
58        if relay_url.is_some() {
59            ep.online().await;
60        }
61
62        let addr = ep.bound_sockets();
63        let addr = SocketAddr::new("127.0.0.1".parse().unwrap(), addr[0].port());
64        let mut addr = EndpointAddr::new(ep.id()).with_ip_addr(addr);
65        if let Some(relay_url) = relay_url {
66            addr = addr.with_relay_url(relay_url.clone());
67        }
68        (addr, ep)
69    })
70}

72/// Create and run a client
73pub async fn client(
74    server_addr: EndpointAddr,
75    relay_url: Option<RelayUrl>,
76    opt: Opt,
77    #[cfg(feature = "qlog")] qlog: &QlogFileGroup,
78) -> Result<ClientStats> {
79    let client_start = std::time::Instant::now();
80    let (endpoint, connection) = connect_client(
81        server_addr,
82        relay_url,
83        opt,
84        #[cfg(feature = "qlog")]
85        qlog,
86    )
87    .await?;
88    let client_connect_time = client_start.elapsed();
89    let mut res = client_handler(
90        EndpointSelector::Iroh(endpoint),
91        ConnectionSelector::Iroh(connection),
92        opt,
93    )
94    .await?;
95    res.connect_time = client_connect_time;
96    Ok(res)
97}

149pub fn transport_config(
150    mut config: QuinnTransportConfig,
151    max_streams: usize,
152    initial_mtu: u16,
153) -> QuinnTransportConfig {
154    // High stream windows are chosen because the amount of concurrent streams
155    // is configurable as a parameter.
156    config.max_concurrent_uni_streams(max_streams.try_into().unwrap());
157    config.initial_mtu(initial_mtu);
158
159    let mut acks = quinn::AckFrequencyConfig::default();
160    acks.ack_eliciting_threshold(10u32.into());
161    config.ack_frequency_config(Some(acks));
162
163    config
164}

166async fn drain_stream(
167    stream: &mut RecvStream,
168    read_unordered: bool,
169) -> Result<(usize, Duration, u64)> {
170    let mut read = 0;
171
172    let download_start = Instant::now();
173    let mut first_byte = true;
174    let mut ttfb = download_start.elapsed();
175
176    let mut num_chunks: u64 = 0;
177
178    if read_unordered {
179        while let Some(chunk) = stream.read_chunk(usize::MAX, false).await.anyerr()? {
180            if first_byte {
181                ttfb = download_start.elapsed();
182                first_byte = false;
183            }
184            read += chunk.bytes.len();
185            num_chunks += 1;
186        }
187    } else {
188        // These are 32 buffers, for reading approximately 32kB at once
189        #[rustfmt::skip]
190        let mut bufs = [
191            Bytes::new(), Bytes::new(), Bytes::new(), Bytes::new(),
192            Bytes::new(), Bytes::new(), Bytes::new(), Bytes::new(),
193            Bytes::new(), Bytes::new(), Bytes::new(), Bytes::new(),
194            Bytes::new(), Bytes::new(), Bytes::new(), Bytes::new(),
195            Bytes::new(), Bytes::new(), Bytes::new(), Bytes::new(),
196            Bytes::new(), Bytes::new(), Bytes::new(), Bytes::new(),
197            Bytes::new(), Bytes::new(), Bytes::new(), Bytes::new(),
198            Bytes::new(), Bytes::new(), Bytes::new(), Bytes::new(),
199        ];
200
201        while let Some(n) = stream.read_chunks(&mut bufs[..]).await.anyerr()? {
202            if first_byte {
203                ttfb = download_start.elapsed();
204                first_byte = false;
205            }
206            read += bufs.iter().take(n).map(|buf| buf.len()).sum::<usize>();
207            num_chunks += 1;
208        }
209    }
210
211    Ok((read, ttfb, num_chunks))
212}

214async fn send_data_on_stream(stream: &mut SendStream, stream_size: u64) -> Result<()> {
215    const DATA: &[u8] = &[0xAB; 1024 * 1024];
216    let bytes_data = Bytes::from_static(DATA);
217
218    let full_chunks = stream_size / (DATA.len() as u64);
219    let remaining = (stream_size % (DATA.len() as u64)) as usize;
220
221    for _ in 0..full_chunks {
222        stream
223            .write_chunk(bytes_data.clone())
224            .await
225            .std_context("failed sending data")?;
226    }
227
228    if remaining != 0 {
229        stream
230            .write_chunk(bytes_data.slice(0..remaining))
231            .await
232            .std_context("failed sending data")?;
233    }
234
235    stream.finish().std_context("failed finishing stream")?;
236    stream
237        .stopped()
238        .await
239        .std_context("failed to wait for stream to be stopped")?;
240
241    Ok(())
242}

244pub async fn handle_client_stream(
245    connection: &Connection,
246    upload_size: u64,
247    read_unordered: bool,
248) -> Result<(TransferResult, TransferResult)> {
249    let start = Instant::now();
250
251    let (mut send_stream, mut recv_stream) = connection
252        .open_bi()
253        .await
254        .std_context("failed to open stream")?;
255
256    send_data_on_stream(&mut send_stream, upload_size).await?;
257
258    let upload_result = TransferResult::new(start.elapsed(), upload_size, Duration::default(), 0);
259
260    let start = Instant::now();
261    let (size, ttfb, num_chunks) = drain_stream(&mut recv_stream, read_unordered).await?;
262    let download_result = TransferResult::new(start.elapsed(), size as u64, ttfb, num_chunks);
263
264    Ok((upload_result, download_result))
265}

267/// Take the provided endpoint and run the server
268pub async fn server(endpoint: Endpoint, opt: Opt) -> Result<()> {
269    let mut server_tasks = Vec::new();
270
271    // Handle only the expected amount of clients
272    for _ in 0..opt.clients {
273        let incoming = endpoint.accept().await.unwrap();
274        let accepting = match incoming.accept() {
275            Ok(accepting) => accepting,
276            Err(err) => {
277                warn!("incoming connection failed: {err:#}");
278                // we can carry on in these cases:
279                // this can be caused by retransmitted datagrams
280                continue;
281            }
282        };
283        let connection = accepting.await.context("handshake failed")?;
284
285        server_tasks.push(tokio::spawn(async move {
286            loop {
287                let (mut send_stream, mut recv_stream) = match connection.accept_bi().await {
288                    Err(ConnectionError::ApplicationClosed(_)) => break,
289                    Err(e) => {
290                        eprintln!("accepting stream failed: {e:?}");
291                        break;
292                    }
293                    Ok(stream) => stream,
294                };
295                trace!("stream established");
296
297                tokio::spawn(async move {
298                    drain_stream(&mut recv_stream, opt.read_unordered).await?;
299                    send_data_on_stream(&mut send_stream, opt.download_size).await?;
300                    n0_error::Ok(())
301                });
302            }
303
304            if opt.stats {
305                println!("\nServer connection stats:\n{:#?}", connection.stats());
306            }
307        }));
308    }
309
310    // Await all the tasks. We have to do this to prevent the runtime getting dropped
311    // and all server tasks to be cancelled
312    for handle in server_tasks {
313        if let Err(e) = handle.await {
314            eprintln!("Server task error: {e:?}");
315        };
316    }
317
318    Ok(())
