26/// Potential errors related to dns.
27#[common_fields({
28    backtrace: Option<Backtrace>,
29    #[snafu(implicit)]
30    span_trace: n0_snafu::SpanTrace,
31})]
32#[allow(missing_docs)]
33#[derive(Debug, Snafu)]
34#[non_exhaustive]
35#[snafu(visibility(pub(crate)))]
36pub enum DnsError {
37    #[snafu(transparent)]
38    Timeout { source: tokio::time::error::Elapsed },
39    #[snafu(display("No response"))]
40    NoResponse {},
41    #[snafu(display("Resolve failed ipv4: {ipv4}, ipv6 {ipv6}"))]
42    ResolveBoth {
43        ipv4: Box<DnsError>,
44        ipv6: Box<DnsError>,
45    },
46    #[snafu(display("missing host"))]
47    MissingHost {},
48    #[snafu(transparent)]
49    Resolve {
50        source: hickory_resolver::ResolveError,
51    },
52    #[snafu(display("invalid DNS response: not a query for _iroh.z32encodedpubkey"))]
53    InvalidResponse {},
54}

56/// Error returned when an input value is too long for [`crate::node_info::UserData`].
57#[allow(missing_docs)]
58#[derive(Debug, Snafu)]
59#[snafu(module)]
60#[snafu(display("no calls succeeded: [{}]", errors.iter().map(|e| e.to_string()).collect::<Vec<_>>().join("")))]
61pub struct StaggeredError<E: std::fmt::Debug + std::fmt::Display> {
62    backtrace: Option<Backtrace>,
63    #[snafu(implicit)]
64    span_trace: n0_snafu::SpanTrace,
65    errors: Vec<E>,
66}

69    pub(crate) fn new(errors: Vec<E>) -> Self {
70        Self {
71            errors,
72            backtrace: GenerateImplicitData::generate(),
73            span_trace: n0_snafu::SpanTrace::generate(),
74        }
75    }

83    /// Create a new DNS resolver with sensible cross-platform defaults.
84    ///
85    /// We first try to read the system's resolver from `/etc/resolv.conf`.
86    /// This does not work at least on some Androids, therefore we fallback
87    /// to the default `ResolverConfig` which uses eg. to google's `8.8.8.8` or `8.8.4.4`.
88    pub fn new() -> Self {
89        let (system_config, mut options) =
90            hickory_resolver::system_conf::read_system_conf().unwrap_or_default();
91
92        // Copy all of the system config, but strip the bad windows nameservers.  Unfortunately
93        // there is no easy way to do this.
94        let mut config = hickory_resolver::config::ResolverConfig::new();
95        if let Some(name) = system_config.domain() {
96            config.set_domain(name.clone());
97        }
98        for name in system_config.search() {
99            config.add_search(name.clone());
100        }
101        for nameserver_cfg in system_config.name_servers() {
102            if !WINDOWS_BAD_SITE_LOCAL_DNS_SERVERS.contains(&nameserver_cfg.socket_addr.ip()) {
103                config.add_name_server(nameserver_cfg.clone());
104            }
105        }
106
107        // see [`DnsResolver::lookup_ipv4_ipv6`] for info on why we avoid `LookupIpStrategy::Ipv4AndIpv6`
108        options.ip_strategy = hickory_resolver::config::LookupIpStrategy::Ipv4thenIpv6;
109
110        let mut builder =
111            TokioResolver::builder_with_config(config, TokioConnectionProvider::default());
112        *builder.options_mut() = options;
113        DnsResolver(builder.build())
114    }

127        DnsResolver(builder.build())

168    /// Resolve IPv4 and IPv6 in parallel with a timeout.
169    ///
170    /// `LookupIpStrategy::Ipv4AndIpv6` will wait for ipv6 resolution timeout, even if it is
171    /// not usable on the stack, so we manually query both lookups concurrently and time them out
172    /// individually.
173    pub async fn lookup_ipv4_ipv6(
174        &self,
175        host: impl ToString,
176        timeout: Duration,
177    ) -> Result<impl Iterator<Item = IpAddr>, DnsError> {
178        let host = host.to_string();
179        let res = tokio::join!(
180            self.lookup_ipv4(host.clone(), timeout),
181            self.lookup_ipv6(host, timeout)
182        );
183
184        match res {
185            (Ok(ipv4), Ok(ipv6)) => Ok(LookupIter::Both(ipv4.chain(ipv6))),
186            (Ok(ipv4), Err(_)) => Ok(LookupIter::Ipv4(ipv4)),
187            (Err(_), Ok(ipv6)) => Ok(LookupIter::Ipv6(ipv6)),
188            (Err(ipv4_err), Err(ipv6_err)) => Err(ResolveBothSnafu {
189                ipv4: Box::new(ipv4_err),
190                ipv6: Box::new(ipv6_err),
191            }
192            .build()),
193        }
194    }

196    /// Resolve a hostname from a URL to an IP address.
197    pub async fn resolve_host(
198        &self,
199        url: &Url,
200        prefer_ipv6: bool,
201        timeout: Duration,
202    ) -> Result<IpAddr, DnsError> {
203        let host = url.host().context(MissingHostSnafu)?;
204        match host {
205            url::Host::Domain(domain) => {
206                // Need to do a DNS lookup
207                let lookup = tokio::join!(
208                    self.lookup_ipv4(domain, timeout),
209                    self.lookup_ipv6(domain, timeout)
210                );
211                let (v4, v6) = match lookup {
212                    (Err(ipv4_err), Err(ipv6_err)) => {
213                        return Err(ResolveBothSnafu {
214                            ipv4: Box::new(ipv4_err),
215                            ipv6: Box::new(ipv6_err),
216                        }
217                        .build());
218                    }
219                    (Err(_), Ok(mut v6)) => (None, v6.next()),
220                    (Ok(mut v4), Err(_)) => (v4.next(), None),
221                    (Ok(mut v4), Ok(mut v6)) => (v4.next(), v6.next()),
222                };
223                if prefer_ipv6 {
224                    v6.or(v4).context(NoResponseSnafu)
225                } else {
226                    v4.or(v6).context(NoResponseSnafu)
227                }
228            }
229            url::Host::Ipv4(ip) => Ok(IpAddr::V4(ip)),
230            url::Host::Ipv6(ip) => Ok(IpAddr::V6(ip)),
231        }
232    }

234    /// Perform an ipv4 lookup with a timeout in a staggered fashion.
235    ///
236    /// From the moment this function is called, each lookup is scheduled after the delays in
237    /// `delays_ms` with the first call being done immediately. `[200ms, 300ms]` results in calls
238    /// at T+0ms, T+200ms and T+300ms. The `timeout` is applied to each call individually. The
239    /// result of the first successful call is returned, or a summary of all errors otherwise.
240    pub async fn lookup_ipv4_staggered(
241        &self,
242        host: impl ToString,
243        timeout: Duration,
244        delays_ms: &[u64],
245    ) -> Result<impl Iterator<Item = IpAddr>, StaggeredError<DnsError>> {
246        let host = host.to_string();
247        let f = || self.lookup_ipv4(host.clone(), timeout);
248        stagger_call(f, delays_ms).await
249    }

393/// Deprecated IPv6 site-local anycast addresses still configured by windows.
394///
395/// Windows still configures these site-local addresses as soon even as an IPv6 loopback
396/// interface is configured.  We do not want to use these DNS servers, the chances of them
397/// being usable are almost always close to zero, while the chance of DNS configuration
398/// **only** relying on these servers and not also being configured normally are also almost
399/// zero.  The chance of the DNS resolver accidentally trying one of these and taking a
400/// bunch of timeouts to figure out they're no good are on the other hand very high.
401const WINDOWS_BAD_SITE_LOCAL_DNS_SERVERS: [IpAddr; 3] = [
402    IpAddr::V6(Ipv6Addr::new(0xfec0, 0, 0, 0xffff, 0, 0, 0, 1)),
403    IpAddr::V6(Ipv6Addr::new(0xfec0, 0, 0, 0xffff, 0, 0, 0, 2)),
404    IpAddr::V6(Ipv6Addr::new(0xfec0, 0, 0, 0xffff, 0, 0, 0, 3)),
405];

407/// Helper enum to give a unified type to the iterators of [`DnsResolver::lookup_ipv4_ipv6`].
408enum LookupIter<A, B> {
409    Ipv4(A),
410    Ipv6(B),
411    Both(std::iter::Chain<A, B>),
412}
413
414impl<A: Iterator<Item = IpAddr>, B: Iterator<Item = IpAddr>> Iterator for LookupIter<A, B> {
415    type Item = IpAddr;
416
417    fn next(&mut self) -> Option<Self::Item> {
418        match self {
419            LookupIter::Ipv4(iter) => iter.next(),
420            LookupIter::Ipv6(iter) => iter.next(),
421            LookupIter::Both(iter) => iter.next(),
422        }
423    }
424}

426/// Staggers calls to the future F with the given delays.
427///
428/// The first call is performed immediately. The first call to succeed generates an Ok result
429/// ignoring any previous error. If all calls fail, an error summarizing all errors is returned.
430async fn stagger_call<
431    T,
432    E: std::fmt::Debug + std::fmt::Display,
433    F: Fn() -> Fut,
434    Fut: Future<Output = Result<T, E>>,
435>(
436    f: F,
437    delays_ms: &[u64],
438) -> Result<T, StaggeredError<E>> {
439    let mut calls = n0_future::FuturesUnorderedBounded::new(delays_ms.len() + 1);
440    // NOTE: we add the 0 delay here to have a uniform set of futures. This is more performant than
441    // using alternatives that allow futures of different types.
442    for delay in std::iter::once(&0u64).chain(delays_ms) {
443        let delay = Duration::from_millis(*delay);
444        let fut = f();
445        let staggered_fut = async move {
446            time::sleep(delay).await;
447            fut.await
448        };
449        calls.push(staggered_fut)
450    }
451
452    let mut errors = vec![];
453    while let Some(call_result) = calls.next().await {
454        match call_result {
455            Ok(t) => return Ok(t),
456            Err(e) => errors.push(e),
457        }
458    }
459
460    Err(StaggeredError::new(errors))
461}

473    async fn stagger_basic() {
474        const CALL_RESULTS: &[Result<u8, u8>] = &[Err(2), Ok(3), Ok(5), Ok(7)];
475        static DONE_CALL: AtomicUsize = AtomicUsize::new(0);
476        let f = || {
477            let r_pos = DONE_CALL.fetch_add(1, std::sync::atomic::Ordering::Relaxed);

479                tracing::info!(r_pos, "call");

485        let result = stagger_call(f, &delays).await.unwrap();
