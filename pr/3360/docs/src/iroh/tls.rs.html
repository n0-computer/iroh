1//! TLS configuration for iroh.
2//!
3//! Currently there is one mechanisms available
4//! - Raw Public Keys, using the TLS extension described in [RFC 7250]
5//!
6//! [RFC 7250]: https://datatracker.ietf.org/doc/html/rfc7250
7

20/// Maximum amount of TLS tickets we will cache (by default) for 0-RTT connection
21/// establishment.
22///
23/// 8 tickets per remote endpoint, 32 different endpoints would max out the required storage:
24/// ~200 bytes per session + certificates (which are ~387 bytes)
25/// So 8 * 32 * (200 + 387) = 150.272 bytes, assuming pointers to certificates
26/// are never aliased pointers (they're Arc'ed).
27/// I think 150KB is an acceptable default upper limit for such a cache.
28pub(crate) const DEFAULT_MAX_TLS_TICKETS: usize = 8 * 32;
29
30/// Configuration for TLS.
31///
32/// The main point of this struct is to keep state that should be kept the same
33/// over multiple TLS sessions the same.
34/// E.g. the `server_verifier` and `client_verifier` Arc pointers are checked to be
35/// the same between different TLS session calls with 0-RTT data in rustls.
36/// This makes sure that's the case.
37#[derive(Debug)]
38pub(crate) struct TlsConfig {
39    pub(crate) secret_key: SecretKey,
40    cert_resolver: Arc<AlwaysResolvesCert>,
41    server_verifier: Arc<verifier::ServerCertificateVerifier>,
42    client_verifier: Arc<verifier::ClientCertificateVerifier>,
43    session_store: Arc<dyn rustls::client::ClientSessionStore>,
44}
45
46impl TlsConfig {
47    pub(crate) fn new(secret_key: SecretKey, max_tls_tickets: usize) -> Self {
48        let cert_resolver = Arc::new(
49            AlwaysResolvesCert::new(&secret_key).expect("Client cert key DER is valid; qed"),
50        );
51        Self {
52            secret_key,
53            cert_resolver,
54            server_verifier: Arc::new(verifier::ServerCertificateVerifier),
55            client_verifier: Arc::new(verifier::ClientCertificateVerifier),
56            session_store: Arc::new(rustls::client::ClientSessionMemoryCache::new(
57                max_tls_tickets,
58            )),
59        }
60    }
61
62    /// Create a TLS client configuration.
63    ///
64    /// If *keylog* is `true` this will enable logging of the pre-master key to the file in the
65    /// `SSLKEYLOGFILE` environment variable.  This can be used to inspect the traffic for
66    /// debugging purposes.
67    pub(crate) fn make_client_config(
68        &self,
69        alpn_protocols: Vec<Vec<u8>>,
70        keylog: bool,
71    ) -> QuicClientConfig {
72        let mut crypto = rustls::ClientConfig::builder_with_provider(Arc::new(
73            rustls::crypto::ring::default_provider(),
74        ))
75        .with_protocol_versions(verifier::PROTOCOL_VERSIONS)
76        .expect("version supported by ring")
77        .dangerous()
78        .with_custom_certificate_verifier(self.server_verifier.clone())
79        .with_client_cert_resolver(self.cert_resolver.clone());
80        crypto.alpn_protocols = alpn_protocols;
81
82        // TODO: enable/disable 0-RTT/storing tickets
83        crypto.resumption = rustls::client::Resumption::store(self.session_store.clone());
84        crypto.enable_early_data = true;
85
86        if keylog {
87            warn!("enabling SSLKEYLOGFILE for TLS pre-master keys");
88            crypto.key_log = Arc::new(rustls::KeyLogFile::new());
89        }
90
91        crypto
92            .try_into()
93            .expect("expected to have a TLS1.3-compatible crypto provider set (hardcoded)")
94    }
95
96    /// Create a TLS server configuration.
97    ///
98    /// If *keylog* is `true` this will enable logging of the pre-master key to the file in the
99    /// `SSLKEYLOGFILE` environment variable.  This can be used to inspect the traffic for
100    /// debugging purposes.
101    pub(crate) fn make_server_config(
102        &self,
103        alpn_protocols: Vec<Vec<u8>>,
104        keylog: bool,
105    ) -> QuicServerConfig {
106        let mut crypto = rustls::ServerConfig::builder_with_provider(Arc::new(
107            rustls::crypto::ring::default_provider(),
108        ))
109        .with_protocol_versions(verifier::PROTOCOL_VERSIONS)
110        .expect("fixed config")
111        .with_client_cert_verifier(self.client_verifier.clone())
112        .with_cert_resolver(self.cert_resolver.clone());
113        crypto.alpn_protocols = alpn_protocols;
114        if keylog {
115            warn!("enabling SSLKEYLOGFILE for TLS pre-master keys");
116            crypto.key_log = Arc::new(rustls::KeyLogFile::new());
117        }
118
119        // must be u32::MAX or 0 (the default). Any other value panics with QUIC
120        // This is specified in RFC 9001: https://www.rfc-editor.org/rfc/rfc9001#section-4.6.1
121        crypto.max_early_data_size = u32::MAX;
122        crypto
123            .try_into()
124            .expect("expected to have a TLS1.3-compatible crypto provider set (hardcoded)")
125    }
126}
