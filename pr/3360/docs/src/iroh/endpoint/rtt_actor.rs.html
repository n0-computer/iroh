1//! Actor which coordinates the congestion controller for the magic socket
2
3use std::{pin::Pin, sync::Arc, task::Poll};
4
5use iroh_base::NodeId;
6use n0_future::{
7    MergeUnbounded, Stream, StreamExt,
8    task::{self, AbortOnDropHandle},
9};
10use tokio::sync::mpsc;
11use tracing::{Instrument, debug, info_span};
12
13use crate::{magicsock::ConnectionType, metrics::MagicsockMetrics};
14
15#[derive(Debug)]
16pub(super) struct RttHandle {
17    // We should and some point use this to propagate panics and errors.
18    pub(super) _handle: AbortOnDropHandle<()>,
19    pub(super) msg_tx: mpsc::Sender<RttMessage>,
20}
21
22impl RttHandle {
23    pub(super) fn new(metrics: Arc<MagicsockMetrics>) -> Self {
24        let mut actor = RttActor {
25            connection_events: Default::default(),
26            metrics,
27        };
28        let (msg_tx, msg_rx) = mpsc::channel(16);
29        let handle = task::spawn(
30            async move {
31                actor.run(msg_rx).await;
32            }
33            .instrument(info_span!("rtt-actor")),
34        );
35        Self {
36            _handle: AbortOnDropHandle::new(handle),
37            msg_tx,
38        }
39    }
40}
41
42/// Messages to send to the [`RttActor`].
43#[derive(Debug)]
44pub(super) enum RttMessage {
45    /// Informs the [`RttActor`] of a new connection is should monitor.
46    NewConnection {
47        /// The connection.
48        connection: quinn::WeakConnectionHandle,
49        /// Path changes for this connection from the magic socket.
50        conn_type_changes: n0_watcher::Stream<n0_watcher::Direct<ConnectionType>>,
51        /// For reporting-only, the Node ID of this connection.
52        node_id: NodeId,
53    },
54}
55
56/// Actor to coordinate congestion controller state with magic socket state.
57///
58/// The magic socket can change the underlying network path, between two nodes.  If we can
59/// inform the QUIC congestion controller of this event it will work much more efficiently.
60#[derive(derive_more::Debug)]
61struct RttActor {
62    /// Stream of connection type changes.
63    #[debug("MergeUnbounded<WatcherStream<ConnectionType>>")]
64    connection_events: MergeUnbounded<MappedStream>,
65    metrics: Arc<MagicsockMetrics>,
66}
67
68#[derive(Debug)]
69struct MappedStream {
70    stream: n0_watcher::Stream<n0_watcher::Direct<ConnectionType>>,
71    node_id: NodeId,
72    /// Reference to the connection.
73    connection: quinn::WeakConnectionHandle,
74    /// This an indiciator of whether this connection was direct before.
75    /// This helps establish metrics on number of connections that became direct.
76    was_direct_before: bool,
77}
78
79struct ConnectionEvent {
80    became_direct: bool,
81}
82
83impl Stream for MappedStream {
84    type Item = ConnectionEvent;
85
86    /// Performs the congestion controller reset for a magic socket path change.
87    ///
88    /// Regardless of which kind of path we are changed to, the congestion controller needs
89    /// resetting.  Even when switching to mixed we should reset the state as e.g. switching
90    /// from direct to mixed back to direct should be a rare exception and is a bug if this
91    /// happens commonly.
92    fn poll_next(
93        mut self: Pin<&mut Self>,
94        cx: &mut std::task::Context<'_>,
95    ) -> Poll<Option<Self::Item>> {
96        match Pin::new(&mut self.stream).poll_next(cx) {
97            Poll::Ready(Some(new_conn_type)) => {
98                let mut became_direct = false;
99                if self.connection.network_path_changed() {
100                    debug!(
101                        node_id = %self.node_id.fmt_short(),
102                        new_type = ?new_conn_type,
103                        "Congestion controller state reset",
104                    );
105                    if !self.was_direct_before && matches!(new_conn_type, ConnectionType::Direct(_))
106                    {
107                        self.was_direct_before = true;
108                        became_direct = true
109                    }
110                };
111                Poll::Ready(Some(ConnectionEvent { became_direct }))
112            }
113            Poll::Ready(None) => Poll::Ready(None),
114            Poll::Pending => Poll::Pending,
115        }
116    }
117}
118
119impl RttActor {
120    /// Runs the actor main loop.
121    ///
122    /// The main loop will finish when the sender is dropped.
123    async fn run(&mut self, mut msg_rx: mpsc::Receiver<RttMessage>) {
124        loop {
125            tokio::select! {
126                biased;
127                msg = msg_rx.recv() => {
128                    match msg {
129                        Some(msg) => self.handle_msg(msg),
130                        None => break,
131                    }
132                }
133                event = self.connection_events.next(), if !self.connection_events.is_empty() => {
134                    if event.map(|e| e.became_direct).unwrap_or(false) {
135                        self.metrics.connection_became_direct.inc();
136                    }
137                }
138            }
139        }
140        debug!("rtt-actor finished");
141    }
142
143    /// Handle actor messages.
144    fn handle_msg(&mut self, msg: RttMessage) {
145        match msg {
146            RttMessage::NewConnection {
147                connection,
148                conn_type_changes,
149                node_id,
150            } => {
151                self.handle_new_connection(connection, conn_type_changes, node_id);
152            }
153        }
154    }
155
156    /// Handles the new connection message.
157    fn handle_new_connection(
158        &mut self,
159        connection: quinn::WeakConnectionHandle,
160        conn_type_changes: n0_watcher::Stream<n0_watcher::Direct<ConnectionType>>,
161        node_id: NodeId,
162    ) {
163        self.connection_events.push(MappedStream {
164            stream: conn_type_changes,
165            connection,
166            node_id,
167            was_direct_before: false,
168        });
169        self.metrics.connection_handshake_success.inc();
170    }
171}
