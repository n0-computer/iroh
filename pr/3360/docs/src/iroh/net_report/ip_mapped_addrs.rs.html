12/// Can occur when converting a [`SocketAddr`] to an [`IpMappedAddr`]
13#[derive(Debug, Snafu)]
14#[snafu(display("Failed to convert"))]
15pub struct IpMappedAddrError;
16
17/// A map fake Ipv6 address with an actual IP address.
18///
19/// It is essentially a lookup key for an IP that iroh's magicsocket knows about.
20#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, Ord, PartialOrd)]
21pub(crate) struct IpMappedAddr(Ipv6Addr);
22
23/// Counter to always generate unique addresses for [`IpMappedAddr`].
24static IP_ADDR_COUNTER: AtomicU64 = AtomicU64::new(1);
25
26impl IpMappedAddr {
27    /// The Prefix/L of our Unique Local Addresses.
28    const ADDR_PREFIXL: u8 = 0xfd;
29    /// The Global ID used in our Unique Local Addresses.
30    const ADDR_GLOBAL_ID: [u8; 5] = [21, 7, 10, 81, 11];
31    /// The Subnet ID used in our Unique Local Addresses.
32    const ADDR_SUBNET: [u8; 2] = [0, 1];
33
34    /// The dummy port used for all mapped addresses.
35    const MAPPED_ADDR_PORT: u16 = 12345;
36
37    /// Generates a globally unique fake UDP address.
38    ///
39    /// This generates a new IPv6 address in the Unique Local Address range (RFC 4193)
40    /// which is recognised by iroh as an IP mapped address.
41    pub(super) fn generate() -> Self {
42        let mut addr = [0u8; 16];
43        addr[0] = Self::ADDR_PREFIXL;
44        addr[1..6].copy_from_slice(&Self::ADDR_GLOBAL_ID);
45        addr[6..8].copy_from_slice(&Self::ADDR_SUBNET);
46
47        let counter = IP_ADDR_COUNTER.fetch_add(1, Ordering::Relaxed);
48        addr[8..16].copy_from_slice(&counter.to_be_bytes());
49
50        Self(Ipv6Addr::from(addr))
51    }
52
53    /// Returns a consistent [`SocketAddr`] for the [`IpMappedAddr`].
54    ///
55    /// This does not have a routable IP address.
56    ///
57    /// This uses a made-up, but fixed port number.  The [IpMappedAddresses`] map this is
58    /// made for creates a unique [`IpMappedAddr`] for each IP+port and thus does not use
59    /// the port to map back to the original [`SocketAddr`].
60    pub(crate) fn private_socket_addr(&self) -> SocketAddr {
61        SocketAddr::new(IpAddr::from(self.0), Self::MAPPED_ADDR_PORT)
62    }
63}
64
65impl TryFrom<Ipv6Addr> for IpMappedAddr {
66    type Error = IpMappedAddrError;
67
68    fn try_from(value: Ipv6Addr) -> std::result::Result<Self, Self::Error> {
69        let octets = value.octets();
70        if octets[0] == Self::ADDR_PREFIXL
71            && octets[1..6] == Self::ADDR_GLOBAL_ID
72            && octets[6..8] == Self::ADDR_SUBNET
73        {
74            return Ok(Self(value));
75        }
76        Err(IpMappedAddrError)
77    }
78}

80impl std::fmt::Display for IpMappedAddr {
81    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
82        write!(f, "IpMappedAddr({})", self.0)
83    }
84}

86/// A Map of [`IpMappedAddresses`] to [`SocketAddr`].
87// TODO(ramfox): before this is ready to be used beyond QAD, we should add
88// mechanisms for keeping track of "aliveness" and pruning address, as we do
89// with the `NodeMap`
90#[derive(Debug, Clone, Default)]
91pub(crate) struct IpMappedAddresses(Arc<std::sync::Mutex<Inner>>);
92
93#[derive(Debug, Default)]
94pub(super) struct Inner {
95    by_mapped_addr: BTreeMap<IpMappedAddr, SocketAddr>,
96    /// Because [`std::net::SocketAddrV6`] contains extra fields besides the IP
97    /// address and port (ie, flow_info and scope_id), the a [`std::net::SocketAddrV6`]
98    /// with the same IP addr and port might Hash to something different.
99    /// So to get a hashable key for the map, we are using `(IpAddr, u6)`.
100    by_ip_port: BTreeMap<(IpAddr, u16), IpMappedAddr>,
101}
102
103impl IpMappedAddresses {
104    /// Adds a [`SocketAddr`] to the map and returns the generated [`IpMappedAddr`].
105    ///
106    /// If this [`SocketAddr`] already exists in the map, it returns its
107    /// associated [`IpMappedAddr`].
108    ///
109    /// Otherwise a new [`IpMappedAddr`] is generated for it and returned.
110    pub(super) fn get_or_register(&self, socket_addr: SocketAddr) -> IpMappedAddr {
111        let ip_port = (socket_addr.ip(), socket_addr.port());
112        let mut inner = self.0.lock().expect("poisoned");
113        if let Some(mapped_addr) = inner.by_ip_port.get(&ip_port) {
114            return *mapped_addr;
115        }
116        let ip_mapped_addr = IpMappedAddr::generate();
117        inner.by_mapped_addr.insert(ip_mapped_addr, socket_addr);
118        inner.by_ip_port.insert(ip_port, ip_mapped_addr);
119        ip_mapped_addr
120    }
121
122    /// Returns the [`IpMappedAddr`] for the given [`SocketAddr`].
123    pub(crate) fn get_mapped_addr(&self, socket_addr: &SocketAddr) -> Option<IpMappedAddr> {
124        let ip_port = (socket_addr.ip(), socket_addr.port());
125        let inner = self.0.lock().expect("poisoned");
126        inner.by_ip_port.get(&ip_port).copied()
127    }
128
129    /// Returns the [`SocketAddr`] for the given [`IpMappedAddr`].
130    pub(crate) fn get_ip_addr(&self, mapped_addr: &IpMappedAddr) -> Option<SocketAddr> {
131        let inner = self.0.lock().expect("poisoned");
132        inner.by_mapped_addr.get(mapped_addr).copied()
133    }
134}
