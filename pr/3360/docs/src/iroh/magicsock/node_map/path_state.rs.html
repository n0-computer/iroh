21/// The minimum time between pings to an endpoint.
22///
23/// Except in the case of CallMeMaybe frames resetting the counter, as the first pings
24/// likely didn't through the firewall.
25const DISCO_PING_INTERVAL: Duration = Duration::from_secs(5);
26
27/// State about a particular path to another [`NodeState`].
28///
29/// This state is used for both the relay path and any direct UDP paths.
30///
31/// [`NodeState`]: super::node_state::NodeState
32#[derive(Debug, Clone)]
33pub(super) struct PathState {
34    /// The node for which this path exists.
35    node_id: NodeId,
36    /// The path this applies for.
37    path: SendAddr,
38    /// The last (outgoing) ping time.
39    pub(super) last_ping: Option<Instant>,
40
41    /// If non-zero, means that this was an endpoint that we learned about at runtime (from an
42    /// incoming ping). If so, we keep the time updated and use it to discard old candidates.
43    // NOTE: tx_id Originally added in tailscale due to <https://github.com/tailscale/tailscale/issues/7078>.
44    last_got_ping: Option<(Instant, stun_rs::TransactionId)>,
45
46    /// The time this endpoint was last advertised via a call-me-maybe DISCO message.
47    pub(super) call_me_maybe_time: Option<Instant>,
48
49    /// Tracks whether this path is valid.
50    ///
51    /// Also stores the latest [`PongReply`], if there is one.
52    ///
53    /// See [`PathValidity`] docs.
54    pub(super) validity: PathValidity,
55    /// When the last payload data was **received** via this path.
56    ///
57    /// This excludes DISCO messages.
58    pub(super) last_payload_msg: Option<Instant>,
59    /// Sources is a map of [`Source`]s to [`Instant`]s, keeping track of all the ways we have
60    /// learned about this path
61    ///
62    /// We keep track of only the latest [`Instant`] for each [`Source`], keeping the size of
63    /// the map of sources down to one entry per type of source.
64    pub(super) sources: HashMap<Source, Instant>,
65}

67impl PathState {
68    pub(super) fn new(node_id: NodeId, path: SendAddr, source: Source, now: Instant) -> Self {
69        let mut sources = HashMap::new();
70        sources.insert(source, now);
71        Self {
72            node_id,
73            path,
74            last_ping: None,
75            last_got_ping: None,
76            call_me_maybe_time: None,
77            validity: PathValidity::empty(),
78            last_payload_msg: None,
79            sources,
80        }
81    }
82
83    pub(super) fn with_last_payload(
84        node_id: NodeId,
85        path: SendAddr,
86        source: Source,
87        now: Instant,
88    ) -> Self {
89        let mut sources = HashMap::new();
90        sources.insert(source, now);
91        PathState {
92            node_id,
93            path,
94            last_ping: None,
95            last_got_ping: None,
96            call_me_maybe_time: None,
97            validity: PathValidity::empty(),
98            last_payload_msg: Some(now),
99            sources,
100        }
101    }
102
103    pub(super) fn with_ping(
104        node_id: NodeId,
105        path: SendAddr,
106        tx_id: stun_rs::TransactionId,
107        source: Source,
108        now: Instant,
109    ) -> Self {
110        let mut new = PathState::new(node_id, path, source, now);
111        new.handle_ping(tx_id, now);
112        new
113    }
114
115    pub(super) fn add_pong_reply(&mut self, r: PongReply) {
116        if let SendAddr::Udp(ref path) = self.path {
117            if self.validity.is_empty() {
118                event!(
119                    target: "iroh::_events::holepunched",
120                    Level::DEBUG,
121                    remote_node = %self.node_id.fmt_short(),
122                    path = ?path,
123                    direction = "outgoing",
124                );
125            }
126        }
127
128        self.validity = PathValidity::new(r.pong_at, r.latency);
129    }
130
131    pub(super) fn receive_payload(&mut self, now: Instant) {
132        self.last_payload_msg = Some(now);
133        self.validity
134            .receive_payload(now, path_validity::Source::QuicPayload);
135    }

137    #[cfg(test)]
138    pub(super) fn with_pong_reply(node_id: NodeId, r: PongReply) -> Self {
139        PathState {
140            node_id,
141            path: r.from.clone(),
142            last_ping: None,
143            last_got_ping: None,
144            call_me_maybe_time: None,
145            validity: PathValidity::new(r.pong_at, r.latency),
146            last_payload_msg: None,
147            sources: HashMap::new(),
148        }
149    }
150
151    /// Check whether this path is considered active.
152    ///
153    /// Active means the path has received payload messages within the last
154    /// [`SESSION_ACTIVE_TIMEOUT`].
155    ///
156    /// Note that a path might be alive but not active if it's contactable but not in
157    /// use.
158    pub(super) fn is_active(&self) -> bool {
159        self.last_payload_msg
160            .as_ref()
161            .map(|instant| instant.elapsed() <= SESSION_ACTIVE_TIMEOUT)
162            .unwrap_or(false)
163    }
164
165    /// Returns the instant the last incoming ping was received.
166    pub(super) fn last_incoming_ping(&self) -> Option<&Instant> {
167        self.last_got_ping.as_ref().map(|(time, _tx_id)| time)
168    }
169
170    /// Reports the last instant this path was considered alive.
171    ///
172    /// Alive means the path is considered in use by the remote endpoint.  Either because we
173    /// received a payload message, a DISCO message (ping, pong) or it was advertised in a
174    /// call-me-maybe message.
175    ///
176    /// This is the most recent instant between:
177    /// - when last pong was received.
178    /// - when this path was last advertised in a received CallMeMaybe message.
179    /// - When the last payload transmission occurred.
180    /// - when the last ping from them was received.
181    pub(super) fn last_alive(&self) -> Option<Instant> {
182        self.validity
183            .latest_pong()
184            .into_iter()
185            .chain(self.last_payload_msg)
186            .chain(self.call_me_maybe_time)
187            .chain(self.last_incoming_ping().cloned())
188            .max()
189    }
190
191    /// The last control or DISCO message **about** this path.
192    ///
193    /// This is the most recent instant among:
194    /// - when last pong was received.
195    /// - when this path was last advertised in a received CallMeMaybe message.
196    /// - when the last ping from them was received.
197    ///
198    /// Returns the time elapsed since the last control message, and the type of control message.
199    pub(super) fn last_control_msg(&self, now: Instant) -> Option<(Duration, ControlMsg)> {
200        // get every control message and assign it its kind
201        let last_pong = self
202            .validity
203            .latest_pong()
204            .map(|pong_at| (pong_at, ControlMsg::Pong));
205        let last_call_me_maybe = self
206            .call_me_maybe_time
207            .as_ref()
208            .map(|call_me| (*call_me, ControlMsg::CallMeMaybe));
209        let last_ping = self
210            .last_incoming_ping()
211            .map(|ping| (*ping, ControlMsg::Ping));
212
213        last_pong
214            .into_iter()
215            .chain(last_call_me_maybe)
216            .chain(last_ping)
217            .max_by_key(|(instant, _kind)| *instant)
218            .map(|(instant, kind)| (now.duration_since(instant), kind))
219    }
220
221    /// Returns the latency from the most recent pong, if available.
222    pub(super) fn latency(&self) -> Option<Duration> {
223        self.validity.latency()
224    }
225
226    pub(super) fn needs_ping(&self, now: &Instant) -> bool {
227        match self.last_ping {
228            None => true,
229            Some(last_ping) => {
230                let elapsed = now.duration_since(last_ping);
231
232                // TODO: remove!
233                // This logs "ping is too new" for each send whenever the endpoint does *not* need
234                // a ping. Pretty sure this is not a useful log, but maybe there was a reason?
235                // if !needs_ping {
236                //     debug!("ping is too new: {}ms", elapsed.as_millis());
237                // }
238                elapsed > DISCO_PING_INTERVAL
239            }
240        }
241    }
242
243    pub(super) fn handle_ping(&mut self, tx_id: stun_rs::TransactionId, now: Instant) -> PingRole {
244        if Some(&tx_id) == self.last_got_ping.as_ref().map(|(_t, tx_id)| tx_id) {
245            PingRole::Duplicate
246        } else {
247            let prev = self.last_got_ping.replace((now, tx_id));
248            let heartbeat_deadline = HEARTBEAT_INTERVAL + (HEARTBEAT_INTERVAL / 2);
249            match prev {
250                Some((prev_time, _tx)) if now.duration_since(prev_time) <= heartbeat_deadline => {
251                    PingRole::LikelyHeartbeat
252                }
253                Some((prev_time, _tx)) => {
254                    debug!(
255                        elapsed = ?now.duration_since(prev_time),
256                        "heartbeat missed, reactivating",
257                    );
258                    PingRole::Activate
259                }
260                None => {
261                    if let SendAddr::Udp(ref addr) = self.path {
262                        event!(
263                            target: "iroh::_events::holepunched",
264                            Level::DEBUG,
265                            remote_node = %self.node_id.fmt_short(),
266                            path = ?addr,
267                            direction = "incoming",
268                        );
269                    }
270                    PingRole::Activate
271                }
272            }
273        }
274    }
275
276    pub(super) fn add_source(&mut self, source: Source, now: Instant) {
277        self.sources.insert(source, now);
278    }
279
280    pub(super) fn clear(&mut self) {
281        self.last_ping = None;
282        self.last_got_ping = None;
283        self.call_me_maybe_time = None;
284        self.validity = PathValidity::empty();
285    }
286
287    fn summary(&self, mut w: impl std::fmt::Write) -> std::fmt::Result {
288        write!(w, "{{ ")?;
289        if self.is_active() {
290            write!(w, "active ")?;
291        }
292        if let Some(pong_at) = self.validity.latest_pong() {
293            write!(w, "pong-received({:?} ago) ", pong_at.elapsed())?;
294        }
295        if let Some(when) = self.last_incoming_ping() {
296            write!(w, "ping-received({:?} ago) ", when.elapsed())?;
297        }
298        if let Some(ref when) = self.last_ping {
299            write!(w, "ping-sent({:?} ago) ", when.elapsed())?;
300        }
301        if let Some(last_source) = self.sources.iter().max_by_key(|&(_, instant)| instant) {
302            write!(
303                w,
304                "last-source: {}({:?} ago)",
305                last_source.0,
306                last_source.1.elapsed()
307            )?;
308        }
309        write!(w, "}}")
310    }

314pub(super) fn summarize_node_paths(paths: &BTreeMap<IpPort, PathState>) -> String {
315    use std::fmt::Write;
316
317    let mut w = String::new();
318    write!(&mut w, "[").ok();
319    for (i, (ipp, state)) in paths.iter().enumerate() {
320        if i > 0 {
321            write!(&mut w, ", ").ok();
322        }
323        write!(&mut w, "{ipp}").ok();
324        state.summary(&mut w).ok();
325    }
326    write!(&mut w, "]").ok();
327    w
