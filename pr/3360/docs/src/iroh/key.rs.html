9pub(crate) const NONCE_LEN: usize = 24;
10
11pub(super) fn public_ed_box(key: &ed25519_dalek::VerifyingKey) -> crypto_box::PublicKey {
12    crypto_box::PublicKey::from(key.to_montgomery())
13}
14
15pub(super) fn secret_ed_box(key: &ed25519_dalek::SigningKey) -> crypto_box::SecretKey {
16    crypto_box::SecretKey::from(key.to_scalar())
17}
18
19/// Shared Secret.
20pub struct SharedSecret(crypto_box::ChaChaBox);
21
22/// Errors that can occur during [`SharedSecret::open`].
23#[common_fields({
24    backtrace: Option<snafu::Backtrace>,
25    #[snafu(implicit)]
26    span_trace: n0_snafu::SpanTrace,
27})]
28#[derive(Debug, Snafu)]
29#[non_exhaustive]
30pub enum DecryptionError {
31    /// The nonce had the wrong size.
32    #[snafu(display("Invalid nonce"))]
33    InvalidNonce {},
34    /// AEAD decryption failed.
35    #[snafu(display("Aead error"))]
36    Aead { source: aead::Error },
37}
38
39impl Debug for SharedSecret {
40    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
41        write!(f, "SharedSecret(crypto_box::ChaChaBox)")
42    }
43}
44
45impl SharedSecret {
46    pub fn new(this: &crypto_box::SecretKey, other: &crypto_box::PublicKey) -> Self {
47        SharedSecret(crypto_box::ChaChaBox::new(other, this))
48    }
49
50    /// Seals the provided cleartext.
51    pub fn seal(&self, buffer: &mut dyn Buffer) {
52        use aead::{AeadCore, AeadInPlace, OsRng};
53
54        let nonce = crypto_box::ChaChaBox::generate_nonce(&mut OsRng);
55        self.0
56            .encrypt_in_place(&nonce, &[], buffer)
57            .expect("encryption failed");
58
59        buffer.extend_from_slice(&nonce).expect("buffer too small");
60    }
61
62    /// Opens the ciphertext, which must have been created using `Self::seal`, and places the clear text into the provided buffer.
63    pub fn open(&self, buffer: &mut dyn Buffer) -> Result<(), DecryptionError> {
64        use aead::AeadInPlace;
65        ensure!(buffer.len() >= NONCE_LEN, InvalidNonceSnafu);
66
67        let offset = buffer.len() - NONCE_LEN;
68        let nonce: [u8; NONCE_LEN] = buffer.as_ref()[offset..]
69            .try_into()
70            .map_err(|_| InvalidNonceSnafu.build())?;
71
72        buffer.truncate(offset);
73        self.0
74            .decrypt_in_place(&nonce.into(), &[], buffer)
75            .context(AeadSnafu)?;
76
77        Ok(())
78    }
79}

85    fn shared(this: &iroh_base::SecretKey, other: &iroh_base::PublicKey) -> SharedSecret {
86        let secret_key = secret_ed_box(this.secret());
87        let public_key = public_ed_box(&other.public());
88
89        SharedSecret::new(&secret_key, &public_key)
90    }

92    #[test]
93    fn test_seal_open_roundtrip() {
94        let mut rng = rand::thread_rng();
95        let key_a = iroh_base::SecretKey::generate(&mut rng);
96        let key_b = iroh_base::SecretKey::generate(&mut rng);
97
98        seal_open_roundtrip(&key_a, &key_b);
99        seal_open_roundtrip(&key_b, &key_a);
100        seal_open_roundtrip(&key_a, &key_a);
101    }
102
103    fn seal_open_roundtrip(key_a: &iroh_base::SecretKey, key_b: &iroh_base::SecretKey) {
104        let msg = b"super secret message!!!!".to_vec();
105        let shared_a = shared(key_a, &key_b.public());
106        let mut sealed_message = msg.clone();
107        shared_a.seal(&mut sealed_message);
108        let shared_b = shared(key_b, &key_a.public());
109        let mut decrypted_message = sealed_message.clone();
110        shared_b.open(&mut decrypted_message).unwrap();
111        assert_eq!(&msg[..], &decrypted_message);
112    }
