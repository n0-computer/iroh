15/// A drop guard to clean up test infrastructure.
16///
17/// After dropping the test infrastructure will asynchronously shutdown and release its
18/// resources.
19// Nightly sees the sender as dead code currently, but we only rely on Drop of the
20// sender.
21#[derive(Debug)]
22#[allow(dead_code)]
23pub struct CleanupDropGuard(pub(crate) oneshot::Sender<()>);

25/// Runs a relay server with QUIC enabled suitable for tests.
26///
27/// The returned `Url` is the url of the relay server in the returned [`RelayMap`].
28/// When dropped, the returned [`Server`] does will stop running.
29pub async fn run_relay_server() -> Result<(RelayMap, RelayUrl, Server), SpawnError> {
30    run_relay_server_with(true).await
31}

33/// Runs a relay server.
34///
35/// If `quic` is set to `true`, it will make the appropriate [`QuicConfig`] from the generated tls certificates and run the quic server at a random free port.
36///
37///
38/// The return value is similar to [`run_relay_server`].
39pub async fn run_relay_server_with(quic: bool) -> Result<(RelayMap, RelayUrl, Server), SpawnError> {
40    let (certs, server_config) = iroh_relay::server::testing::self_signed_tls_certs_and_config();
41
42    let tls = TlsConfig {
43        cert: CertConfig::<(), ()>::Manual { certs },
44        https_bind_addr: (Ipv4Addr::LOCALHOST, 0).into(),
45        quic_bind_addr: (Ipv4Addr::LOCALHOST, 0).into(),
46        server_config,
47    };
48    let quic = if quic {
49        Some(QuicConfig {
50            server_config: tls.server_config.clone(),
51            bind_addr: tls.quic_bind_addr,
52        })
53    } else {
54        None
55    };
56    let config = ServerConfig {
57        relay: Some(RelayConfig {
58            http_bind_addr: (Ipv4Addr::LOCALHOST, 0).into(),
59            tls: Some(tls),
60            limits: Default::default(),
61            key_cache_capacity: Some(1024),
62            access: AccessConfig::Everyone,
63        }),
64        quic,
65        ..Default::default()
66    };
67    let server = Server::spawn(config).await?;
68    let url: RelayUrl = format!("https://{}", server.https_addr().expect("configured"))
69        .parse()
70        .expect("invalid relay url");
71
72    let quic = server
73        .quic_addr()
74        .map(|addr| RelayQuicConfig { port: addr.port() });
75    let n: RelayMap = RelayNode {
76        url: url.clone(),
77        quic,
78    }
79    .into();
80    Ok((n, url, server))
81}

98    /// Handle and drop guard for test DNS and Pkarr servers.
99    ///
100    /// Once the struct is dropped the servers will shut down.

102    pub struct DnsPkarrServer {
103        /// The node origin domain.
104        pub node_origin: String,
105        /// The shared state of the DNS and Pkarr servers.
106        state: State,
107        /// The socket address of the DNS server.
108        pub nameserver: SocketAddr,
109        /// The HTTP URL of the Pkarr server.
110        pub pkarr_url: Url,
111        _dns_drop_guard: CleanupDropGuard,
112        _pkarr_drop_guard: CleanupDropGuard,

115    impl DnsPkarrServer {
116        /// Run DNS and Pkarr servers on localhost.
117        pub async fn run() -> std::io::Result<Self> {
118            Self::run_with_origin("dns.iroh.test".to_string()).await
119        }
120
121        /// Run DNS and Pkarr servers on localhost with the specified `node_origin` domain.
122        pub async fn run_with_origin(node_origin: String) -> std::io::Result<Self> {
123            let state = State::new(node_origin.clone());
124            let (nameserver, dns_drop_guard) = run_dns_server(state.clone()).await?;
125            let (pkarr_url, pkarr_drop_guard) = run_pkarr_relay(state.clone()).await?;
126            Ok(Self {
127                node_origin,
128                nameserver,
129                pkarr_url,
130                state,
131                _dns_drop_guard: dns_drop_guard,
132                _pkarr_drop_guard: pkarr_drop_guard,
133            })
134        }
135
136        /// Create a [`ConcurrentDiscovery`] with [`DnsDiscovery`] and [`PkarrPublisher`]
137        /// configured to use the test servers.
138        pub fn discovery(&self, secret_key: SecretKey) -> ConcurrentDiscovery {
139            ConcurrentDiscovery::from_services(vec![
140                // Enable DNS discovery by default
141                Box::new(
142                    DnsDiscovery::builder(self.node_origin.clone())
143                        .dns_resolver(self.dns_resolver())
144                        .build(),
145                ),
146                // Enable pkarr publishing by default
147                Box::new(PkarrPublisher::builder(self.pkarr_url.clone()).build(secret_key)),
148            ])
149        }
150
151        /// Create a [`DnsResolver`] configured to use the test DNS server.
152        pub fn dns_resolver(&self) -> DnsResolver {
153            DnsResolver::with_nameserver(self.nameserver)
154        }
155
156        /// Wait until a Pkarr announce for a node is published to the server.
157        ///
158        /// If `timeout` elapses an error is returned.
159        pub async fn on_node(&self, node_id: &NodeId, timeout: Duration) -> std::io::Result<()> {
160            self.state.on_node(node_id, timeout).await
161        }

181    /// Trait used by [`run_dns_server`] for answering DNS queries.
182    pub trait QueryHandler: Send + Sync + 'static {
183        fn resolve(
184            &self,
185            query: &Message,
186            reply: &mut Message,
187        ) -> impl Future<Output = std::io::Result<()>> + Send;
188    }
189
190    pub type QueryHandlerFunction = Box<
191        dyn Fn(&Message, &mut Message) -> BoxFuture<std::io::Result<()>> + Send + Sync + 'static,
192    >;
193
194    impl QueryHandler for QueryHandlerFunction {
195        fn resolve(
196            &self,
197            query: &Message,
198            reply: &mut Message,
199        ) -> impl Future<Output = std::io::Result<()>> + Send {
200            (self)(query, reply)
201        }
202    }
203
204    /// Run a DNS server.
205    ///
206    /// Must pass a [`QueryHandler`] that answers queries.
207    pub async fn run_dns_server(
208        resolver: impl QueryHandler,
209    ) -> std::io::Result<(SocketAddr, CleanupDropGuard)> {
210        let bind_addr = SocketAddr::from((Ipv4Addr::LOCALHOST, 0));
211        let socket = UdpSocket::bind(bind_addr).await?;
212        let bound_addr = socket.local_addr()?;
213        let s = TestDnsServer { socket, resolver };
214        let (tx, mut rx) = oneshot::channel();
215        tokio::task::spawn(async move {
216            tokio::select! {
217                _ = &mut rx => {
218                    debug!("shutting down dns server");
219                }
220                res = s.run() => {
221                    if let Err(e) = res {
222                        error!("error running dns server {e:?}");
223                    }
224                }
225            }
226        });
227        Ok((bound_addr, CleanupDropGuard(tx)))
228    }
229
230    struct TestDnsServer<R> {
231        resolver: R,
232        socket: UdpSocket,
233    }
234
235    impl<R: QueryHandler> TestDnsServer<R> {
236        async fn run(self) -> std::io::Result<()> {
237            let mut buf = [0; 1450];
238            loop {
239                let res = self.socket.recv_from(&mut buf).await;
240                let (len, from) = res?;
241                if let Err(err) = self.handle_datagram(from, &buf[..len]).await {
242                    warn!(?err, %from, "failed to handle incoming datagram");
243                }
244            }
245        }
246
247        async fn handle_datagram(&self, from: SocketAddr, buf: &[u8]) -> std::io::Result<()> {
248            let packet = Message::from_bytes(buf)?;
249            debug!(queries = ?packet.queries(), %from, "received query");
250            let mut reply = packet.clone();
251            reply.set_message_type(MessageType::Response);
252            self.resolver.resolve(&packet, &mut reply).await?;
253            debug!(?reply, %from, "send reply");
254            let buf = reply.to_vec()?;
255            let len = self.socket.send_to(&buf, from).await?;
256            assert_eq!(len, buf.len(), "failed to send complete packet");
257            Ok(())
258        }
259    }

282    pub async fn run_pkarr_relay(state: AppState) -> std::io::Result<(Url, CleanupDropGuard)> {
283        let bind_addr = SocketAddr::from((Ipv4Addr::LOCALHOST, 0));
284        let app = Router::new()
285            .route("/pkarr/{key}", put(pkarr_put))
286            .with_state(state);
287        let listener = tokio::net::TcpListener::bind(bind_addr).await?;
288        let bound_addr = listener.local_addr()?;
289        let url: Url = format!("http://{bound_addr}/pkarr")
290            .parse()
291            .expect("valid url");
292
293        let (tx, mut rx) = oneshot::channel();
294        tokio::spawn(async move {
295            let serve = axum::serve(listener, app);
296            tokio::select! {
297                _ = &mut rx => {
298                    debug!("shutting down pkarr server");
299                }
300                res = serve.into_future() => {
301                    if let Err(e) = res {
302                        error!("pkarr server error: {e:?}");
303                    }
304                }
305            }
306        });
307        Ok((url, CleanupDropGuard(tx)))
308    }
309
310    async fn pkarr_put(
311        State(state): State<AppState>,
312        Path(key): Path<String>,
313        body: Bytes,
314    ) -> Result<impl IntoResponse, AppError> {
315        let key = pkarr::PublicKey::try_from(key.as_str()).map_err(std::io::Error::other)?;
316        let signed_packet =
317            pkarr::SignedPacket::from_relay_payload(&key, &body).map_err(std::io::Error::other)?;
318        let _updated = state.upsert(signed_packet)?;
319        Ok(http::StatusCode::NO_CONTENT)
320    }
321
322    #[derive(Debug)]
323    struct AppError(std::io::Error);
324    impl<T: Into<std::io::Error>> From<T> for AppError {
325        fn from(value: T) -> Self {
326            Self(value.into())
327        }
328    }
329    impl IntoResponse for AppError {
330        fn into_response(self) -> axum::response::Response {
331            warn!(err = ?self, "request failed");
332            (http::StatusCode::INTERNAL_SERVER_ERROR, self.0.to_string()).into_response()
333        }
334    }

352    #[derive(Debug, Clone)]
353    pub struct State {
354        packets: Arc<Mutex<HashMap<NodeId, SignedPacket>>>,
355        origin: String,
356        notify: Arc<tokio::sync::Notify>,
357    }
358
359    impl State {
360        pub fn new(origin: String) -> Self {
361            Self {
362                packets: Default::default(),
363                origin,
364                notify: Arc::new(tokio::sync::Notify::new()),
365            }
366        }
367
368        pub fn on_update(&self) -> tokio::sync::futures::Notified<'_> {
369            self.notify.notified()
370        }
371
372        pub async fn on_node(&self, node: &NodeId, timeout: Duration) -> std::io::Result<()> {
373            let timeout = tokio::time::sleep(timeout);
374            tokio::pin!(timeout);
375            while self.get(node, |p| {
376                let node_info = p
377                    .as_ref()
378                    .and_then(|p| NodeInfo::from_pkarr_signed_packet(p).ok());
379                debug!("got info {:#?}", node_info);
380                p.is_none()
381            }) {
382                tokio::select! {
383                    _ = &mut timeout => return Err(std::io::Error::other("timeout")),
384                    _ = self.on_update() => {}
385                }
386            }
387            Ok(())
388        }
389
390        pub fn upsert(&self, signed_packet: SignedPacket) -> std::io::Result<bool> {
391            let node_id = NodeId::from_bytes(&signed_packet.public_key().to_bytes())
392                .map_err(std::io::Error::other)?;
393            let mut map = self.packets.lock().expect("poisoned");
394            let updated = match map.entry(node_id) {
395                hash_map::Entry::Vacant(e) => {
396                    e.insert(signed_packet);
397                    true
398                }
399                hash_map::Entry::Occupied(mut e) => {
400                    if signed_packet.more_recent_than(e.get()) {
401                        e.insert(signed_packet);
402                        true
403                    } else {
404                        false
405                    }
406                }
407            };
408            if updated {
409                self.notify.notify_waiters();
410            }
411            Ok(updated)
412        }
413
414        /// Returns a mutex guard, do not hold over await points
415        pub fn get<F, T>(&self, node_id: &NodeId, cb: F) -> T
416        where
417            F: FnOnce(Option<&mut SignedPacket>) -> T,
418        {
419            let mut map = self.packets.lock().expect("poisoned");
420            let packet = map.get_mut(node_id);
421            cb(packet)
422        }
423
424        pub fn resolve_dns(
425            &self,
426            query: &hickory_resolver::proto::op::Message,
427            reply: &mut hickory_resolver::proto::op::Message,
428            ttl: u32,
429        ) -> std::io::Result<()> {
430            for query in query.queries() {
431                let domain_name = query.name().to_string();
432                let Some(node_id) = node_id_from_domain_name(&domain_name) else {
433                    continue;
434                };
435
436                self.get(&node_id, |packet| {
437                    if let Some(packet) = packet {
438                        let node_info = NodeInfo::from_pkarr_signed_packet(packet)
439                            .map_err(std::io::Error::other)?;
440                        for record in node_info_to_hickory_records(&node_info, &self.origin, ttl) {
441                            reply.add_answer(record);
442                        }
443                    }
444                    Ok::<_, std::io::Error>(())
445                })?;
446            }
447            Ok(())
448        }
449    }

451    impl QueryHandler for State {
452        fn resolve(
453            &self,
454            query: &hickory_resolver::proto::op::Message,
455            reply: &mut hickory_resolver::proto::op::Message,
456        ) -> impl Future<Output = std::io::Result<()>> + Send {
457            const TTL: u32 = 30;
458            let res = self.resolve_dns(query, reply, TTL);
459            std::future::ready(res)
460        }

463    /// Parses a [`NodeId`] from a DNS domain name.
464    ///
465    /// Splits the domain name into labels on each dot. Expects the first label to be
466    /// [`IROH_TXT_NAME`] and the second label to be a z32 encoded [`NodeId`]. Ignores
467    /// subsequent labels.
468    ///
469    /// Returns a [`NodeId`] if parsed successfully, otherwise `None`.
470    fn node_id_from_domain_name(name: &str) -> Option<NodeId> {
471        let mut labels = name.split(".");
472        let label = labels.next()?;
473        if label != IROH_TXT_NAME {
474            return None;
475        }
476        let label = labels.next()?;
477        let node_id = NodeId::from_z32(label).ok()?;
478        Some(node_id)
479    }

481    /// Converts a [`NodeInfo`]into a [`hickory_resolver::proto::rr::Record`] DNS record.
482    fn node_info_to_hickory_records(
483        node_info: &NodeInfo,
484        origin: &str,
485        ttl: u32,
486    ) -> impl Iterator<Item = hickory_resolver::proto::rr::Record> + 'static {
487        let txt_strings = node_info.to_txt_strings();
488        let records = to_hickory_records(txt_strings, node_info.node_id, origin, ttl);
489        records.collect::<Vec<_>>().into_iter()
490    }
491
492    /// Converts to a list of [`hickory_resolver::proto::rr::Record`] resource records.
493    fn to_hickory_records(
494        txt_strings: Vec<String>,
495        node_id: NodeId,
496        origin: &str,
497        ttl: u32,
498    ) -> impl Iterator<Item = hickory_resolver::proto::rr::Record> + '_ {
499        use hickory_resolver::proto::rr;
500        let name = format!("{IROH_TXT_NAME}.{}.{origin}", node_id.to_z32());
501        let name = rr::Name::from_utf8(name).expect("invalid name");
502        txt_strings.into_iter().map(move |s| {
503            let txt = rr::rdata::TXT::new(vec![s]);
504            let rdata = rr::RData::TXT(txt);
505            rr::Record::from_rdata(name.clone(), ttl, rdata)
506        })
507    }
