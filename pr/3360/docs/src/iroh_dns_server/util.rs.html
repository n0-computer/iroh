24impl PublicKeyBytes {
25    pub fn new(bytes: [u8; 32]) -> Self {
26        Self(bytes)
27    }
28
29    pub fn from_z32(s: &str) -> Result<Self> {
30        let bytes = z32::decode(s.as_bytes()).e()?;
31        let bytes = TryInto::<[u8; 32]>::try_into(&bytes[..]).context("invalid length")?;
32        Ok(Self(bytes))
33    }
34
35    pub fn to_z32(self) -> String {
36        z32::encode(&self.0)
37    }
38
39    pub fn to_bytes(self) -> [u8; 32] {
40        self.0
41    }
42
43    pub fn as_bytes(&self) -> &[u8; 32] {
44        &self.0
45    }
46
47    pub fn from_signed_packet(packet: &SignedPacket) -> Self {
48        Self(packet.public_key().to_bytes())
49    }
50}

90pub fn signed_packet_to_hickory_message(signed_packet: &SignedPacket) -> Result<Message> {
91    let encoded = signed_packet.encoded_packet();
92    let message = Message::from_bytes(&encoded).e()?;
93    Ok(message)

96pub fn signed_packet_to_hickory_records_without_origin(
97    signed_packet: &SignedPacket,
98    filter: impl Fn(&Record) -> bool,
99) -> Result<(Label, BTreeMap<RrKey, Arc<RecordSet>>)> {
100    let common_zone = Label::from_utf8(&signed_packet.public_key().to_z32()).e()?;
101    let mut message = signed_packet_to_hickory_message(signed_packet)?;
102    let answers = message.take_answers();
103    let mut output: BTreeMap<RrKey, Arc<RecordSet>> = BTreeMap::new();
104    for mut record in answers.into_iter() {
105        // disallow SOA and NS records
106        if matches!(record.record_type(), RecordType::SOA | RecordType::NS) {
107            continue;
108        }
109        // expect the z32 encoded pubkey as root name
110        let name = record.name();
111        if name.num_labels() < 1 {
112            continue;
113        }
114        let zone = name.iter().next_back().unwrap().into_label().e()?;
115        if zone != common_zone {
116            continue;
117        }
118        if !filter(&record) {
119            continue;
120        }
121
122        let name_without_zone =
123            Name::from_labels(name.iter().take(name.num_labels() as usize - 1)).e()?;
124        record.set_name(name_without_zone);
125
126        let rrkey = RrKey::new(record.name().into(), record.record_type());
127        match output.entry(rrkey) {
128            btree_map::Entry::Vacant(e) => {
129                let set: RecordSet = record.into();
130                e.insert(Arc::new(set));
131            }
132            btree_map::Entry::Occupied(mut e) => {
133                let set = e.get_mut();
134                let serial = set.serial();
135                // safe because we just created the arc and are sync iterating
136                Arc::get_mut(set).unwrap().insert(record, serial);
137            }
138        }
139    }
140    Ok((common_zone, output))
141}

143pub fn record_set_append_origin(
144    input: &RecordSet,
145    origin: &Name,
146    serial: u32,
147) -> Result<RecordSet> {
148    let new_name = input.name().clone().append_name(origin).e()?;
149    let mut output = RecordSet::new(new_name.clone(), input.record_type(), serial);
150    // TODO: less clones
151    for record in input.records_without_rrsigs() {
152        let mut record = record.clone();
153        record.set_name(new_name.clone());
154        output.insert(record, serial);
155    }
156    Ok(output)
