11/// Config for http server rate limit.
12#[derive(Debug, Deserialize, Default, Serialize, Clone)]
13#[serde(rename_all = "lowercase")]
14pub enum RateLimitConfig {
15    /// Disable rate limit.
16    Disabled,
17    /// Enable rate limit based on the connection's peer IP address.
18    ///
19    /// <https://docs.rs/tower_governor/latest/tower_governor/key_extractor/struct.PeerIpKeyExtractor.html>
20    #[default]
21    Simple,
22    /// Enable rate limit based on headers commonly used by reverse proxies.
23    ///
24    /// Uses headers commonly used by reverse proxies to extract the original IP address,
25    /// falling back to the connection's peer IP address.
26    /// <https://docs.rs/tower_governor/latest/tower_governor/key_extractor/struct.SmartIpKeyExtractor.html>
27    Smart,
28}
29
30impl Default for &RateLimitConfig {
31    fn default() -> Self {
32        &RateLimitConfig::Simple
33    }
34}
35
36/// Create the default rate-limiting layer.
37///
38/// This spawns a background thread to clean up the rate limiting cache.
39pub fn create(
40    rate_limit_config: &RateLimitConfig,
41) -> Option<GovernorLayer<PeerIpKeyExtractor, NoOpMiddleware<QuantaInstant>>> {
42    let use_smart_extractor = match rate_limit_config {
43        RateLimitConfig::Disabled => {
44            tracing::info!("Rate limiting disabled");
45            return None;
46        }
47        RateLimitConfig::Simple => false,
48        RateLimitConfig::Smart => true,
49    };
50
51    tracing::info!("Rate limiting enabled ({rate_limit_config:?})");
52
53    // Configure rate limiting:
54    // * allow bursts with up to five requests per IP address
55    // * replenish one element every two seconds
56    let mut governor_conf_builder = GovernorConfigBuilder::default();
57    // governor_conf_builder.use_headers()
58    governor_conf_builder.per_second(4);
59    governor_conf_builder.burst_size(2);
60
61    if use_smart_extractor {
62        governor_conf_builder.key_extractor(SmartIpKeyExtractor);
63    }
64
65    let governor_conf = governor_conf_builder
66        .finish()
67        .expect("failed to build rate-limiting governor");
68
69    let governor_conf = Arc::new(governor_conf);
70
71    // The governor needs a background task for garbage collection (to clear expired records)
72    let gc_interval = Duration::from_secs(60);
73    let governor_limiter = governor_conf.limiter().clone();
74    std::thread::spawn(move || {
75        loop {
76            std::thread::sleep(gc_interval);
77            tracing::debug!("rate limiting storage size: {}", governor_limiter.len());
78            governor_limiter.retain_recent();
79        }
80    });
81
82    Some(GovernorLayer {
83        config: governor_conf,
84    })
85}
