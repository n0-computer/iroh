35/// Standard DNS port (Do53). DoH fallbacks use the HTTPS port instead.
36const DNS_PORT: u16 = 53;
37/// HTTPS port, used for the DNS-over-HTTPS fallback nameservers.
38#[cfg(with_crypto_provider)]
39const HTTPS_PORT: u16 = 443;
40
41// Public DNS providers used for the last-resort fallback.
42//
43// The DNS-over-HTTPS entries below are addressed by IP (see
44// transport::https_query). This works because Cloudflare, Google and Quad9 all
45// list their anycast IPs (including every IP used here) as iPAddress SANs in
46// their DoH certificates, so IP-addressed DoH validates without a hostname.
47const CLOUDFLARE_V4_PRIMARY: Ipv4Addr = Ipv4Addr::new(1, 1, 1, 1);
48const CLOUDFLARE_V4_SECONDARY: Ipv4Addr = Ipv4Addr::new(1, 0, 0, 1);
49const CLOUDFLARE_V6_PRIMARY: Ipv6Addr = Ipv6Addr::new(0x2606, 0x4700, 0x4700, 0, 0, 0, 0, 0x1111);
50const CLOUDFLARE_V6_SECONDARY: Ipv6Addr = Ipv6Addr::new(0x2606, 0x4700, 0x4700, 0, 0, 0, 0, 0x1001);
51const GOOGLE_V4_PRIMARY: Ipv4Addr = Ipv4Addr::new(8, 8, 8, 8);
52const GOOGLE_V4_SECONDARY: Ipv4Addr = Ipv4Addr::new(8, 8, 4, 4);
53const GOOGLE_V6_PRIMARY: Ipv6Addr = Ipv6Addr::new(0x2001, 0x4860, 0x4860, 0, 0, 0, 0, 0x8888);
54const GOOGLE_V6_SECONDARY: Ipv6Addr = Ipv6Addr::new(0x2001, 0x4860, 0x4860, 0, 0, 0, 0, 0x8844);
55const QUAD9_V4_PRIMARY: Ipv4Addr = Ipv4Addr::new(9, 9, 9, 9);
56const QUAD9_V4_SECONDARY: Ipv4Addr = Ipv4Addr::new(149, 112, 112, 112);
57const QUAD9_V6_PRIMARY: Ipv6Addr = Ipv6Addr::new(0x2620, 0x00fe, 0, 0, 0, 0, 0, 0x00fe);
58const QUAD9_V6_SECONDARY: Ipv6Addr = Ipv6Addr::new(0x2620, 0x00fe, 0, 0, 0, 0, 0, 0x0009);

60/// Parsed DNS configuration: the nameservers to query and resolv.conf options.
61#[derive(Debug, Clone, Default)]
62pub(super) struct DnsConfig {
63    pub(super) nameservers: Vec<Nameserver>,
64    /// Search domains from resolv.conf `search` or `domain` directives.
65    ///
66    /// When resolving a short hostname (one with fewer dots than `ndots`,
67    /// default 1), the resolver should try appending each search domain
68    /// before querying the bare name.
69    pub(super) search_domains: Vec<String>,
70    /// The `ndots` option from resolv.conf.
71    ///
72    /// Names with at least this many dots are tried as absolute first.
73    /// `None` means use the default (1).
74    /// See <https://man7.org/linux/man-pages/man5/resolv.conf.5.html>.
75    pub(super) ndots: Option<usize>,
76}
77
78impl DnsConfig {
79    /// Reads the system DNS configuration using the platform-specific reader.
80    pub(super) fn system() -> Result<Self, std::io::Error> {
81        read_system_dns()
82    }
83
84    /// The public-resolver fallback configuration (Cloudflare, Google, Quad9).
85    pub(super) fn fallback() -> Self {
86        Self::from_nameservers(fallback_nameservers())
87    }
88
89    /// Builds a config from an explicit nameserver list, with no search domains
90    /// and the default `ndots`.
91    fn from_nameservers(nameservers: Vec<Nameserver>) -> Self {
92        Self {
93            nameservers,
94            search_domains: Vec::new(),
95            ndots: None,
96        }
97    }
98}

100/// Public resolvers used as a last resort when no nameservers are configured.
101///
102/// Spans multiple providers (Cloudflare, Google, Quad9) and transports so that
103/// resolution still works when one provider is down or plain DNS is blocked.
104///
105/// The order matters: the resolver races the first few entries together (see
106/// `MAX_CONCURRENT_QUERIES`), so the DNS-over-HTTPS entries sit within that
107/// first wave rather than behind every UDP server. On a network that filters
108/// port 53 the UDP entries hang and DoH gets raced right away instead of
109/// waiting out a dozen UDP timeouts; on a working network the UDP entries
110/// answer first, before the staggered DoH attempts are even started. The
111/// resolver also tracks per-server RTT, so the servers that work on the current
112/// network float to the front over time.
113fn fallback_nameservers() -> Vec<Nameserver> {
114    let udp = |ip: IpAddr| Nameserver::new(SocketAddr::new(ip, DNS_PORT), DnsProtocol::Udp);
115    let mut servers = vec![
116        udp(IpAddr::V4(CLOUDFLARE_V4_PRIMARY)),
117        udp(IpAddr::V4(GOOGLE_V4_PRIMARY)),
118    ];
119    // DoH right after the two fastest UDP primaries, so it lands in the first
120    // raced wave when plain DNS is blocked.
121    #[cfg(with_crypto_provider)]
122    {
123        let doh = |ip: IpAddr| Nameserver::new(SocketAddr::new(ip, HTTPS_PORT), DnsProtocol::Https);
124        servers.extend([
125            doh(IpAddr::V4(CLOUDFLARE_V4_PRIMARY)),
126            doh(IpAddr::V4(GOOGLE_V4_PRIMARY)),
127            doh(IpAddr::V4(QUAD9_V4_PRIMARY)),
128        ]);
129    }
130    servers.extend([
131        udp(IpAddr::V4(QUAD9_V4_PRIMARY)),
132        udp(IpAddr::V6(CLOUDFLARE_V6_PRIMARY)),
133        udp(IpAddr::V6(GOOGLE_V6_PRIMARY)),
134        udp(IpAddr::V6(QUAD9_V6_PRIMARY)),
135        udp(IpAddr::V4(CLOUDFLARE_V4_SECONDARY)),
136        udp(IpAddr::V4(GOOGLE_V4_SECONDARY)),
137        udp(IpAddr::V4(QUAD9_V4_SECONDARY)),
138        udp(IpAddr::V6(CLOUDFLARE_V6_SECONDARY)),
139        udp(IpAddr::V6(GOOGLE_V6_SECONDARY)),
140        udp(IpAddr::V6(QUAD9_V6_SECONDARY)),
141    ]);
142    servers
143}

145#[cfg(test)]
146mod tests {
147    use super::*;
148
149    #[test]
150    fn fallback_nameservers_span_providers_and_families() {
151        let servers = fallback_nameservers();
152        assert!(servers.len() >= 6);
153        // Both address families are represented.
154        assert!(servers.iter().any(|ns| ns.addr.ip().is_ipv4()));
155        assert!(servers.iter().any(|ns| ns.addr.ip().is_ipv6()));
156        // UDP entries use the standard DNS port.
157        assert!(
158            servers
159                .iter()
160                .filter(|ns| ns.protocol == DnsProtocol::Udp)
161                .all(|ns| ns.addr.port() == DNS_PORT)
162        );
163        // Cloudflare, Google and Quad9 are all present.
164        let ips: Vec<IpAddr> = servers.iter().map(|ns| ns.addr.ip()).collect();
165        assert!(ips.contains(&IpAddr::V4(CLOUDFLARE_V4_PRIMARY)));
166        assert!(ips.contains(&IpAddr::V4(GOOGLE_V4_PRIMARY)));
167        assert!(ips.contains(&IpAddr::V4(QUAD9_V4_PRIMARY)));
168    }
169
170    /// DoH must land in the first raced wave (see `MAX_CONCURRENT_QUERIES`, 3),
171    /// otherwise it would never be tried before the lookup times out on a
172    /// network that silently drops UDP/53.
173    #[cfg(with_crypto_provider)]
174    #[test]
175    fn fallback_races_doh_in_first_wave() {
176        let servers = fallback_nameservers();
177        assert!(
178            servers[..3]
179                .iter()
180                .any(|ns| ns.protocol == DnsProtocol::Https),
181            "expected a DoH entry within the first 3 fallback servers, got {:?}",
182            &servers[..3],
183        );
184    }
