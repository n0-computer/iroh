1//! Static name-to-address mappings from the system hosts file.
2//!
3//! On Unix this is `/etc/hosts`, on Windows the per-system hosts file under
4//! `%SystemRoot%`. Entries are consulted ahead of the cache and the network so
5//! that an operator can pin a relay or discovery origin to a fixed address, the
6//! way the old hickory-backed resolver did via its `use_hosts_file` default.
7
8use std::{
9    collections::HashMap,
10    net::{IpAddr, Ipv4Addr, Ipv6Addr},
11};
12
13use tracing::warn;
14
15/// The A and AAAA addresses mapped to a single name.
16#[derive(Debug, Default)]
17struct Entry {
18    a: Vec<Ipv4Addr>,
19    aaaa: Vec<Ipv6Addr>,
20}
21
22/// Static host-to-address mappings parsed from the system hosts file.
23#[derive(Debug, Default)]
24pub(crate) struct Hosts {
25    by_name: HashMap<String, Entry>,
26}
27
28impl Hosts {
29    /// Reads and parses the system hosts file.
30    ///
31    /// Returns an empty mapping when the file is missing, unreadable, or the
32    /// platform has no hosts file, so a missing file is never an error.
33    pub(crate) fn from_system() -> Self {
34        match hosts_path().and_then(|path| std::fs::read_to_string(path).ok()) {
35            Some(content) => Self::parse(&content),
36            None => Self::default(),
37        }
38    }
39
40    /// Parses hosts-file content into a name-to-address mapping.
41    ///
42    /// Each non-comment line has the form `address host [host ...]`. Names are
43    /// lowercased; comments (`#` to end of line) and unparsable addresses are
44    /// skipped.
45    fn parse(content: &str) -> Self {
46        let mut by_name: HashMap<String, Entry> = HashMap::new();
47        for line in content.lines() {
48            let line = match line.split_once('#') {
49                Some((before, _)) => before,
50                None => line,
51            }
52            .trim();
53            if line.is_empty() {
54                continue;
55            }
56            let mut fields = line.split_whitespace();
57            let Some(addr) = fields.next() else {
58                continue;
59            };
60            let Ok(addr) = addr.parse::<IpAddr>() else {
61                warn!(%addr, "ignoring unparsable address in hosts file");
62                continue;
63            };
64            for name in fields {
65                let entry = by_name.entry(name.to_ascii_lowercase()).or_default();
66                match addr {
67                    IpAddr::V4(ip) => entry.a.push(ip),
68                    IpAddr::V6(ip) => entry.aaaa.push(ip),
69                }
70            }
71        }
72        Self { by_name }
73    }
74
75    /// Normalizes a query name to the hosts-file key form: lowercased, with any
76    /// trailing dot removed.
77    fn normalize(name: &str) -> String {
78        name.strip_suffix('.').unwrap_or(name).to_ascii_lowercase()
79    }
80
81    /// Returns the mapped IPv4 addresses for `name`, if any.
82    pub(crate) fn lookup_ipv4(&self, name: &str) -> Option<Vec<Ipv4Addr>> {
83        let entry = self.by_name.get(&Self::normalize(name))?;
84        (!entry.a.is_empty()).then(|| entry.a.clone())
85    }
86
87    /// Returns the mapped IPv6 addresses for `name`, if any.
88    pub(crate) fn lookup_ipv6(&self, name: &str) -> Option<Vec<Ipv6Addr>> {
89        let entry = self.by_name.get(&Self::normalize(name))?;
90        (!entry.aaaa.is_empty()).then(|| entry.aaaa.clone())
91    }
92
93    /// Builds a hosts map directly from file content, for tests.
94    #[cfg(test)]
95    pub(crate) fn from_content(content: &str) -> Self {
96        Self::parse(content)
97    }
98}
99
100#[cfg(unix)]
101fn hosts_path() -> Option<std::path::PathBuf> {
102    Some(std::path::PathBuf::from("/etc/hosts"))
103}
104
105#[cfg(windows)]
106fn hosts_path() -> Option<std::path::PathBuf> {
107    let system_root = std::env::var_os("SystemRoot")?;
108    Some(std::path::Path::new(&system_root).join("System32\\drivers\\etc\\hosts"))
109}
110
111#[cfg(not(any(unix, windows)))]
112fn hosts_path() -> Option<std::path::PathBuf> {
113    None
114}

116#[cfg(test)]
117mod tests {
118    use super::*;
119
120    #[test]
121    fn parse_basic_and_lookup() {
122        let hosts = Hosts::parse(
123            "127.0.0.1 localhost\n10.0.1.10 myrelay.test relay\n::1 localhost myrelay.test\n",
124        );
125        assert_eq!(
126            hosts.lookup_ipv4("myrelay.test"),
127            Some(vec![Ipv4Addr::new(10, 0, 1, 10)])
128        );
129        // Alias on the same line resolves too.
130        assert_eq!(
131            hosts.lookup_ipv4("relay"),
132            Some(vec![Ipv4Addr::new(10, 0, 1, 10)])
133        );
134        assert_eq!(
135            hosts.lookup_ipv6("myrelay.test"),
136            Some(vec![Ipv6Addr::LOCALHOST])
137        );
138        // No AAAA entry for a name that only has an A record.
139        assert_eq!(hosts.lookup_ipv6("relay"), None);
140    }
141
142    #[test]
143    fn lookup_is_case_insensitive_and_fqdn_tolerant() {
144        let hosts = Hosts::parse("10.0.1.10 MyRelay.Test\n");
145        assert_eq!(
146            hosts.lookup_ipv4("myrelay.test."),
147            Some(vec![Ipv4Addr::new(10, 0, 1, 10)])
148        );
149    }
150
151    #[test]
152    fn parse_skips_comments_and_garbage() {
153        let hosts = Hosts::parse(
154            "# a comment\n\n  10.0.1.10  host1  # trailing comment\nnot-an-ip host2\n",
155        );
156        assert_eq!(
157            hosts.lookup_ipv4("host1"),
158            Some(vec![Ipv4Addr::new(10, 0, 1, 10)])
159        );
160        assert_eq!(hosts.lookup_ipv4("host2"), None);
161    }
162
163    #[test]
164    fn multiple_addresses_accumulate() {
165        let hosts = Hosts::parse("10.0.0.1 host\n10.0.0.2 host\n");
166        assert_eq!(
167            hosts.lookup_ipv4("host"),
168            Some(vec![Ipv4Addr::new(10, 0, 0, 1), Ipv4Addr::new(10, 0, 0, 2)])
169        );
170    }
171
172    #[test]
173    fn empty_lookup_returns_none() {
174        let hosts = Hosts::default();
175        assert_eq!(hosts.lookup_ipv4("anything"), None);
176    }
