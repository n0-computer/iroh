42/// Maps a transport-layer failure onto the public [`DnsError`].
43impl From<TransportError> for DnsError {
44    fn from(err: TransportError) -> Self {
45        e!(DnsError::Transport, AnyError::from_stack(err))
46    }
47}

49/// Maps a query build or response-parse failure onto the public [`DnsError`].
50impl From<QueryError> for DnsError {
51    fn from(err: QueryError) -> Self {
52        match err {
53            QueryError::BuildQuery { source, .. } => e!(DnsError::InvalidQuery, source),
54            QueryError::Malformed { .. } | QueryError::Unexpected { .. } => {
55                e!(DnsError::InvalidResponse)
56            }
57            QueryError::NxDomain { .. } => e!(DnsError::NxDomain),
58            QueryError::ServerFailure { rcode, .. } => e!(DnsError::ServerError { rcode }),
59        }
60    }
61}

63/// Per-nameserver timeout for a single attempt.
64const NAMESERVER_TIMEOUT: Duration = Duration::from_secs(2);
65
66/// Maximum number of nameserver queries in flight at once.
67///
68/// Bounds how many servers we race so that growing the nameserver list does
69/// not turn every lookup into an N-way fan-out.
70const MAX_CONCURRENT_QUERIES: usize = 3;
71
72/// Delay before starting the next nameserver attempt, unless the in-flight
73/// attempt fails first. Gives faster servers a head start without blasting
74/// the whole list at once (happy-eyeballs style).
75const QUERY_ATTEMPT_DELAY: Duration = Duration::from_millis(100);
76
77/// Number of UDP retry attempts per nameserver before giving up.
78/// UDP is unreliable, so a single dropped packet shouldn't be fatal.
79const UDP_ATTEMPTS: usize = 2;
80
81/// Default value for `ndots` per resolv.conf(5).
82///
83/// Names with at least this many dots are tried as absolute names first,
84/// before appending search domains. Names with fewer dots try search
85/// domains first. See <https://man7.org/linux/man-pages/man5/resolv.conf.5.html>.
86const DEFAULT_NDOTS: usize = 1;
87
88/// RFC 6761 Section 6.3: "localhost" and names under it resolve to loopback.
89fn is_localhost(host: &str) -> bool {
90    let host = host.strip_suffix('.').unwrap_or(host);
91    host.eq_ignore_ascii_case("localhost") || host.ends_with(".localhost")
92}

122    /// Builds a resolver from `builder`, reusing an existing [`DnsCache`].
123    ///
124    /// Used by [`Self::reset`] to rebuild the resolver on a network change while
125    /// carrying the cache across, so lookups keep hitting cached records while
126    /// the new nameservers settle (see issue #4037).
127    fn with_cache(builder: Builder, cache: DnsCache) -> Self {
128        let config = Self::build_config(&builder);
129        debug!(
130            nameservers = ?config.nameservers,
131            search_domains = ?config.search_domains,
132            ndots = ?config.ndots,
133            "configured DNS resolver"
134        );
135        #[cfg(with_crypto_provider)]
136        let tls_config = builder
137            .tls_client_config
138            .as_ref()
139            .map(|c| Arc::new(c.clone()));
140        let health = RttMap::new(config.nameservers.len());
141        // The hosts file is part of the system resolver configuration, so we
142        // only consult it when the caller opted into system defaults. Reading
143        // it here mirrors reading /etc/resolv.conf in `build_config`.
144        let hosts = if builder.use_system_defaults {
145            Hosts::from_system()
146        } else {
147            Hosts::default()
148        };
149        Self {
150            nameservers: config.nameservers,
151            search_domains: config.search_domains,
152            ndots: config.ndots.unwrap_or(DEFAULT_NDOTS),
153            #[cfg(with_crypto_provider)]
154            tls_config,
155            #[cfg(with_crypto_provider)]
156            https_client: Mutex::new(None),
157            rtt_map: health,
158            conn_pool: ConnPool::new(),
159            cache,
160            hosts,
161            builder,
162        }
163    }

165    /// Builds the effective DNS config: system defaults (when requested) plus
166    /// any explicitly configured nameservers, falling back to public resolvers
167    /// only when neither provides any.
168    fn build_config(builder: &Builder) -> DnsConfig {
169        let mut config = if builder.use_system_defaults {
170            DnsConfig::system().unwrap_or_default()
171        } else {
172            DnsConfig::default()
173        };
174        config
175            .nameservers
176            .extend(builder.nameservers.iter().cloned());
177        if config.nameservers.is_empty() {
178            config.nameservers = DnsConfig::fallback().nameservers;
179        }
180        config
181    }

183    /// Returns the list of candidate names to try for a given hostname,
184    /// applying search domain expansion per resolv.conf(5) semantics.
185    ///
186    /// - If the name ends with `.` (FQDN), it is used as-is.
187    /// - If the name has more labels than `ndots`, try the bare name first,
188    ///   then each search domain suffix.
189    /// - Otherwise, try each search domain suffix first, then the bare name.
190    ///
191    /// See <https://man7.org/linux/man-pages/man5/resolv.conf.5.html>.
192    fn search_names(&self, host: &str) -> Vec<String> {
193        // Explicit FQDN: no search domain expansion.
194        if host.ends_with('.') || self.search_domains.is_empty() {
195            return vec![host.to_string()];
196        }
197
198        // Label count = dots + 1 (e.g. "foo.bar" has 2 labels).
199        // resolv.conf(5): "if the name has more dots than ndots, try as absolute first"
200        // which is equivalent to num_labels > ndots.
201        let num_labels = host.bytes().filter(|&b| b == b'.').count() + 1;
202        let bare_first = num_labels > self.ndots;
203
204        let mut names = Vec::with_capacity(self.search_domains.len() + 1);
205        let mut push = |name: String| {
206            if !names.contains(&name) {
207                names.push(name);
208            }
209        };
210
211        if bare_first {
212            push(host.to_string());
213        }
214        for domain in &self.search_domains {
215            push(format!("{host}.{domain}"));
216        }
217        if !bare_first {
218            push(host.to_string());
219        }
220
221        names
222    }

224    /// Returns a clone of the cached reqwest client, creating it on first use.
225    ///
226    /// `reqwest::Client` uses an inner `Arc`, so cloning is cheap.
227    #[cfg(with_crypto_provider)]
228    fn get_or_init_https_client(&self) -> Result<reqwest::Client, DnsError> {
229        let mut guard = self.https_client.lock().expect("poisoned");
230        match guard.as_ref() {
231            Some(client) => Ok(client.clone()),
232            None => {
233                // Pin each named DoH server to its address so reqwest does not
234                // recursively resolve the hostname.
235                let resolves: Vec<(String, std::net::SocketAddr)> = self
236                    .nameservers
237                    .iter()
238                    .filter(|ns| ns.protocol == DnsProtocol::Https)
239                    .filter_map(|ns| ns.server_name.clone().map(|name| (name, ns.addr)))
240                    .collect();
241                let client = transport::build_https_client(self.tls_config.as_ref(), &resolves)?;
242                *guard = Some(client.clone());
243                Ok(client)
244            }
245        }
246    }

248    /// Run a future with [`NAMESERVER_TIMEOUT`].
249    async fn with_timeout<T, E: Into<AnyError>>(
250        fut: impl Future<Output = Result<T, E>>,
251    ) -> Result<T, DnsError> {
252        time::timeout(NAMESERVER_TIMEOUT, fut)
253            .await
254            .map(|r| r.map_err(|e| e!(DnsError::Transport, e.into())))
255            .map_err(|_| e!(DnsError::Timeout))?
256    }

258    /// Query a single nameserver, with UDP retry and truncation fallback.
259    async fn query_nameserver(
260        &self,
261        ns: &Nameserver,
262        query_bytes: &[u8],
263    ) -> Result<Vec<u8>, DnsError> {
264        let addr = ns.addr;
265        match ns.protocol {
266            DnsProtocol::Udp => {
267                let mut last_err = None;
268                for attempt in 0..UDP_ATTEMPTS {
269                    trace!(%addr, attempt, "sending UDP query");
270                    match Self::with_timeout(transport::udp_query(addr, query_bytes)).await {
271                        Ok(resp) if query::is_truncated(&resp) => {
272                            debug!(%addr, "UDP response truncated, retrying over TCP");
273                            return Self::with_timeout(transport::tcp_query(
274                                &self.conn_pool,
275                                addr,
276                                query_bytes,
277                            ))
278                            .await;
279                        }
280                        Ok(resp) => return Ok(resp),
281                        Err(e) => {
282                            trace!(%addr, attempt, err = %e, "UDP query failed");
283                            last_err = Some(e);
284                        }
285                    }
286                }
287                Err(last_err.unwrap_or_else(|| e!(DnsError::NoResponse)))
288            }
289            DnsProtocol::Tcp => {
290                Self::with_timeout(transport::tcp_query(&self.conn_pool, addr, query_bytes)).await
291            }
292            #[cfg(with_crypto_provider)]
293            DnsProtocol::Tls => {
294                let tls_config = self.tls_config.as_ref().ok_or_else(|| {
295                    e!(DnsError::Resolve {
296                        source: std::io::Error::other("TLS config required for DNS-over-TLS")
297                            .into(),
298                    })
299                })?;
300                Self::with_timeout(transport::tls_query(
301                    &self.conn_pool,
302                    addr,
303                    query_bytes,
304                    tls_config,
305                    ns.server_name.as_deref(),
306                ))
307                .await
308            }
309            #[cfg(with_crypto_provider)]
310            DnsProtocol::Https => {
311                let client = self.get_or_init_https_client()?;
312                Self::with_timeout(transport::https_query(
313                    addr,
314                    ns.server_name.as_deref(),
315                    query_bytes,
316                    &client,
317                ))
318                .await
319            }
320        }
321    }

323    /// Returns nameserver indices ordered fastest-first by smoothed RTT.
324    fn nameserver_order(&self) -> Vec<usize> {
325        let mut order: Vec<usize> = (0..self.nameservers.len()).collect();
326        order.sort_by(|&a, &b| {
327            self.rtt_map
328                .get_decayed(a)
329                .total_cmp(&self.rtt_map.get_decayed(b))
330        });
331        order
332    }

334    /// Sends a query to the configured nameservers, racing them happy-eyeballs
335    /// style: tries the historically fastest first, starts the next either
336    /// [`QUERY_ATTEMPT_DELAY`] later or as soon as the in-flight attempt fails
337    /// (fail-fast), and caps in-flight attempts at [`MAX_CONCURRENT_QUERIES`].
338    ///
339    /// The first successful response wins; UDP queries are retried per
340    /// nameserver on failure. Per-server success and failure update the
341    /// smoothed RTT used for ordering, so the server list is self-healing.
342    async fn send_query(&self, query_bytes: &[u8]) -> Result<Vec<u8>, DnsError> {
343        if self.nameservers.is_empty() {
344            return Err(e!(DnsError::NoResponse));
345        }
346
347        let order = self.nameserver_order();
348        // Index into `order` of the next nameserver to try.
349        let mut next = 0;
350        // In-flight attempts, each yielding (nameserver index, start, result).
351        let mut dials = FuturesUnordered::new();
352        let mut last_err = None;
353        // Timer after which to start the next attempt, or `None` for immediately.
354        let next_attempt = MaybeFuture::None;
355        tokio::pin!(next_attempt);
356
357        loop {
358            // Start the next attempt if one is due (no pending delay), we are
359            // under the concurrency cap, and a nameserver remains.
360            if next_attempt.is_none() && dials.len() < MAX_CONCURRENT_QUERIES && next < order.len()
361            {
362                let idx = order[next];
363                next += 1;
364                let start = Instant::now();
365                dials.push(async move {
366                    let ns = &self.nameservers[idx];
367                    (idx, start, self.query_nameserver(ns, query_bytes).await)
368                });
369                // Pace the following attempt, unless this was the last server.
370                if next < order.len() {
371                    next_attempt
372                        .as_mut()
373                        .set_future(time::sleep(QUERY_ATTEMPT_DELAY));
374                }
375            }
376
377            if dials.is_empty() && next >= order.len() {
378                return Err(last_err.unwrap_or_else(|| e!(DnsError::NoResponse)));
379            }
380
381            tokio::select! {
382                biased;
383                // A dial attempt completed.
384                Some((idx, start, res)) = dials.next(), if !dials.is_empty() => match res {
385                    Ok(resp) => {
386                        // A SERVFAIL or REFUSED response means this server cannot
387                        // answer for the name (overloaded, not authoritative, policy
388                        // block). Treat it like a transport failure and race the next
389                        // server rather than making it the final answer; another
390                        // nameserver may still resolve the name.
391                        if let Some(rcode) = query::server_failure_rcode(&resp) {
392                            self.rtt_map.record_failure(idx);
393                            last_err =
394                                Some(e!(DnsError::ServerError { rcode: format!("{rcode:?}") }));
395                            // Fail fast: start the next attempt now rather than waiting.
396                            next_attempt.as_mut().set_none();
397                        } else {
398                            self.rtt_map.record_success(idx, start.elapsed());
399                            return Ok(resp);
400                        }
401                    }
402                    Err(e) => {
403                        self.rtt_map.record_failure(idx);
404                        last_err = Some(e);
405                        // Fail fast: start the next attempt now rather than waiting.
406                        next_attempt.as_mut().set_none();
407                    }
408                },
409                // The next attempt is due.
410                () = &mut next_attempt, if next_attempt.is_some() => {
411                    next_attempt.as_mut().set_none();
412                }
413            }
414        }
415    }

417    /// Send a query and follow CNAME chains recursively if the response contains
418    /// a CNAME but no records of the requested type.
419    async fn send_query_following_cnames(
420        &self,
421        host: String,
422        qtype: TYPE,
423    ) -> Result<Vec<u8>, DnsError> {
424        let mut current_host = host;
425        for _ in 0..MAX_CNAME_DEPTH {
426            let name = simple_dns::Name::new(&current_host)
427                .map_err(|e| e!(DnsError::InvalidQuery, AnyError::from_std(e)))?;
428            let (id, query_bytes) = query::build_query(&current_host, qtype)?;
429            let response = self.send_query(&query_bytes).await?;
430            let packet =
431                simple_dns::Packet::parse(&response).map_err(|_| e!(DnsError::InvalidResponse))?;
432
433            // Validate the id, QR bit, question, and RCODE before trusting the
434            // packet to decide the answer or the next CNAME target. This is the
435            // only check of the response against the name we actually asked for.
436            query::check_response(&packet, id, &name, qtype)?;
437
438            let has_answer = packet
439                .answers
440                .iter()
441                .any(|rr| rr.rdata.type_code() == qtype);
442
443            if has_answer {
444                return Ok(response);
445            }
446
447            // No records of the requested type -- follow CNAME if present.
448            let Some(target) = query::cname_target(&packet, &current_host) else {
449                return Ok(response);
450            };
451            debug!(from = %current_host, to = %target, "following CNAME");
452            current_host = target;
453        }
454        Err(e!(DnsError::InvalidResponse))
455    }

457    /// Shared lookup logic: check cache, try search names, parse response, cache result.
458    #[allow(clippy::type_complexity)]
459    async fn lookup<T: Clone + std::fmt::Debug>(
460        &self,
461        host: &str,
462        qtype: QueryType,
463        dns_type: TYPE,
464        from_cache: fn(CachedRecord) -> Option<Vec<T>>,
465        parse: fn(&[u8]) -> Result<(Vec<T>, u32), QueryError>,
466        to_cache: fn(Vec<T>) -> CachedRecord,
467    ) -> Result<Vec<T>, DnsError> {
468        if let Some(cached) = self.cache.get(host, qtype).and_then(from_cache) {
469            trace!(%host, records = cached.len(), ?qtype, "cache hit");
470            return Ok(cached);
471        }
472
473        let mut last_err = None;
474        let names = self.search_names(host);
475        let total = names.len();
476        for (i, name) in names.into_iter().enumerate() {
477            trace!(%name, ?qtype, "resolving");
478            let res = match self
479                .send_query_following_cnames(name.clone(), dns_type)
480                .await
481            {
482                Ok(response) => parse(&response).map_err(DnsError::from),
483                Err(e) => Err(e),
484            };
485            match res {
486                Ok((results, ttl)) if !results.is_empty() => {
487                    debug!(%host, ?qtype, ?results, ttl, "resolved");
488                    self.cache
489                        .insert(host, qtype, to_cache(results.clone()), ttl);
490                    return Ok(results);
491                }
492                Ok(_) => {}
493                Err(ref e @ DnsError::NxDomain { .. })
494                | Err(ref e @ DnsError::ServerError { .. }) => {
495                    let remaining = total - i - 1;
496                    trace!(%name, ?qtype, remaining, reason = %e, "lookup failed");
497                    last_err = Some(e!(DnsError::NxDomain));
498                }
499                Err(e) => {
500                    debug!(%name, ?qtype, reason = %e, "lookup failed");
501                    return Err(e);
502                }
503            }
504        }
505        let err = last_err.unwrap_or_else(|| e!(DnsError::NoResponse));
506        debug!(%host, ?qtype, reason = %err, "resolve failed");
507        Err(err)
508    }

510    pub(super) async fn lookup_ipv4(
511        &self,
512        host: String,
513    ) -> Result<impl Iterator<Item = Ipv4Addr> + use<>, DnsError> {
514        // RFC 6761: localhost always resolves to loopback.
515        if is_localhost(&host) {
516            return Ok(vec![Ipv4Addr::LOCALHOST].into_iter());
517        }
518        // A hosts-file entry overrides DNS, so check it ahead of the cache.
519        if let Some(addrs) = self
520            .search_names(&host)
521            .iter()
522            .find_map(|name| self.hosts.lookup_ipv4(name))
523        {
524            trace!(%host, ?addrs, "resolved from hosts file");
525            return Ok(addrs.into_iter());
526        }
527        let addrs = self
528            .lookup(
529                &host,
530                QueryType::A,
531                TYPE::A,
532                CachedRecord::into_a,
533                query::parse_a_response,
534                CachedRecord::A,
535            )
536            .await?;
537        Ok(addrs.into_iter())
538    }

591    pub(super) fn reset(&self) -> Self {
592        // Carry the cache across so a network change does not start DNS cold,

598impl Resolver for Arc<SimpleDnsResolver> {
599    fn lookup_ipv4(&self, host: String) -> BoxFuture<Result<BoxIter<Ipv4Addr>, DnsError>> {
600        let this = self.clone();
601        Box::pin(async move {
602            let iter = (*this).lookup_ipv4(host).await?;
603            let iter: BoxIter<_> = Box::new(iter);
604            Ok(iter)
605        })

655    fn with_proto(addr: SocketAddr, proto: DnsProtocol) -> DnsResolver {
656        #[cfg_attr(not(with_crypto_provider), allow(unused_mut))]
657        let mut builder = DnsResolver::builder().with_nameserver(addr, proto);
658        #[cfg(with_crypto_provider)]
659        if proto == DnsProtocol::Tls {
660            let root_store =
661                rustls::RootCertStore::from_iter(webpki_roots::TLS_SERVER_ROOTS.iter().cloned());
662            builder = builder.tls_client_config(
663                rustls::ClientConfig::builder()
664                    .with_root_certificates(root_store)
665                    .with_no_client_auth(),
666            );
667        }
668        builder.build()
669    }

844    /// Spawns a mock UDP nameserver that answers one query with `rcode`,
845    /// echoing the question and adding `answer` as an A record when given.
846    async fn spawn_mock_ns(
847        rcode: simple_dns::RCODE,
848        answer: Option<Ipv4Addr>,
849    ) -> (SocketAddr, tokio::task::JoinHandle<()>) {
850        use simple_dns::{
851            CLASS, Packet, PacketFlag, ResourceRecord,
852            rdata::{A, RData},
853        };
854
855        let socket = tokio::net::UdpSocket::bind("127.0.0.1:0").await.unwrap();
856        let addr = socket.local_addr().unwrap();
857        let handle = tokio::spawn(async move {
858            let mut buf = vec![0u8; 512];
859            let (len, peer) = socket.recv_from(&mut buf).await.unwrap();
860            let query = Packet::parse(&buf[..len]).unwrap();
861            let question = query.questions[0].clone();
862            let mut reply = Packet::new_reply(query.id());
863            reply.set_flags(PacketFlag::RECURSION_DESIRED | PacketFlag::RECURSION_AVAILABLE);
864            *reply.rcode_mut() = rcode;
865            if let Some(ip) = answer {
866                reply.answers.push(ResourceRecord::new(
867                    question.qname.clone(),
868                    CLASS::IN,
869                    300,
870                    RData::A(A {
871                        address: u32::from(ip),
872                    }),
873                ));
874            }
875            reply.questions.push(question);
876            socket
877                .send_to(&reply.build_bytes_vec().unwrap(), peer)
878                .await
879                .unwrap();
880        });
881        (addr, handle)
882    }

884    /// A SERVFAIL or REFUSED response from the fastest nameserver must not be
885    /// the final answer: the resolver races on to a nameserver that can answer.
886    #[tokio::test]
887    async fn servfail_winner_falls_through_to_next_nameserver() {
888        let (bad, bad_handle) = spawn_mock_ns(simple_dns::RCODE::ServerFailure, None).await;
889        let (good, good_handle) =
890            spawn_mock_ns(simple_dns::RCODE::NoError, Some(Ipv4Addr::new(10, 1, 2, 3))).await;
891
892        // `bad` is listed first, so it is the fastest by default ordering and
893        // wins the race with a SERVFAIL; the lookup must fall through to `good`.
894        let resolver = DnsResolver::builder()
895            .with_nameserver(bad, DnsProtocol::Udp)
896            .with_nameserver(good, DnsProtocol::Udp)
897            .build();
898
899        let addrs: Vec<_> = resolver
900            .lookup_ipv4("test.example", TIMEOUT)
901            .await
902            .unwrap()
903            .collect();
904        assert_eq!(addrs, [IpAddr::V4(Ipv4Addr::new(10, 1, 2, 3))]);
905
906        bad_handle.await.unwrap();
907        good_handle.await.unwrap();
908    }

934    /// the DNS cache must carry across so reconnects keep resolving while the new

936    #[test]

940            "example.com",

942            CachedRecord::A(vec![Ipv4Addr::LOCALHOST]),

946        let reset = r.reset();
947

949        let survived = matches!(&cached, Some(CachedRecord::A(addrs)) if addrs.as_slice() == [Ipv4Addr::LOCALHOST]);
