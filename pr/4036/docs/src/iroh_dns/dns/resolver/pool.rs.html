1//! A small connection pool for TCP and DNS-over-TLS queries.
2//!
3//! UDP is connectionless and DNS-over-HTTPS pools inside `reqwest`, so only
4//! plain-TCP and DoT connections are pooled here. Reusing a connection
5//! amortizes the TCP (and, for DoT, TLS) handshake across repeated queries to
6//! the same nameserver.
7//!
8//! A connection is checked out for exclusive use (one in-flight query at a
9//! time, no pipelining) and returned on success. Two things keep idle
10//! connections from accumulating:
11//!
12//! - On checkout, connections older than [`IDLE_TIMEOUT`] are discarded rather
13//!   than handed out, and at most [`MAX_IDLE_PER_KEY`] are kept per nameserver.
14//! - A background task ([`reap_loop`]) sweeps every [`REAP_INTERVAL`] and drops
15//!   idle connections that no checkout has touched, so a nameserver queried
16//!   once and never again does not pin a socket open forever.
17//!
18//! The task is spawned lazily on the first check-in (always inside an async
19//! query, hence inside a runtime). It holds only a [`Weak`] reference, so it
20//! self-terminates once the pool is dropped (and dropping the pool closes its
21//! idle sockets right away rather than waiting on the task).
22
23use std::{
24    collections::HashMap,
25    net::SocketAddr,
26    sync::{Arc, Mutex, OnceLock, Weak},
27};
28
29use n0_future::{
30    task::spawn,
31    time::{self, Duration, Instant},
32};
33use tokio::net::TcpStream;
34
35/// Idle connections older than this are discarded instead of reused.
36const IDLE_TIMEOUT: Duration = Duration::from_secs(10);
37/// Maximum idle connections kept per pool key.
38const MAX_IDLE_PER_KEY: usize = 2;
39/// How often the background task sweeps idle connections.
40const REAP_INTERVAL: Duration = IDLE_TIMEOUT;
41
42/// An established DNS-over-TLS stream.
43#[cfg(with_crypto_provider)]
44pub(super) type TlsStream = tokio_rustls::client::TlsStream<TcpStream>;
45
46/// Pool key for a DoT connection: the address plus the TLS server name, so a
47/// connection is never reused for a different SNI name.
48#[cfg(with_crypto_provider)]
49pub(super) type TlsKey = (SocketAddr, Option<String>);
50
51/// A pooled idle connection and when it was last returned to the pool.
52struct Idle<S> {
53    stream: S,
54    last_used: Instant,
55}
56
57impl<S> Idle<S> {
58    fn new(stream: S) -> Self {
59        Self {
60            stream,
61            last_used: Instant::now(),
62        }
63    }
64
65    fn is_stale(&self) -> bool {
66        self.last_used.elapsed() >= IDLE_TIMEOUT
67    }
68}
69
70/// The pooled connections, shared between the [`ConnPool`] and its reaper task.
71#[derive(Default)]
72struct Inner {
73    tcp: Mutex<HashMap<SocketAddr, Vec<Idle<TcpStream>>>>,
74    #[cfg(with_crypto_provider)]
75    tls: Mutex<HashMap<TlsKey, Vec<Idle<TlsStream>>>>,
76}
77
78impl Inner {
79    /// Drops every idle connection past [`IDLE_TIMEOUT`] and any key left empty.
80    fn reap(&self) {
81        reap_map(&self.tcp);
82        #[cfg(with_crypto_provider)]
83        reap_map(&self.tls);
84    }
85}
86
87/// A pool of idle TCP and DoT connections, keyed by nameserver.
88pub(super) struct ConnPool {
89    inner: Arc<Inner>,
90    /// Set once the idle-connection reaper has been spawned.
91    reaper_spawned: OnceLock<()>,
92}
93
94impl std::fmt::Debug for ConnPool {
95    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
96        f.debug_struct("ConnPool").finish_non_exhaustive()
97    }
98}
99
100impl ConnPool {
101    pub(super) fn new() -> Self {
102        Self {
103            inner: Arc::new(Inner::default()),
104            reaper_spawned: OnceLock::new(),
105        }
106    }
107
108    /// Takes an idle TCP connection to `addr`, if a fresh one is pooled.
109    pub(super) fn checkout_tcp(&self, addr: SocketAddr) -> Option<TcpStream> {
110        take_fresh(self.inner.tcp.lock().expect("poisoned").get_mut(&addr)?)
111    }
112
113    /// Returns a healthy TCP connection to the pool for reuse.
114    pub(super) fn checkin_tcp(&self, addr: SocketAddr, stream: TcpStream) {
115        push_capped(
116            self.inner
117                .tcp
118                .lock()
119                .expect("poisoned")
120                .entry(addr)
121                .or_default(),
122            stream,
123        );
124        self.ensure_reaper();
125    }
126
127    /// Takes an idle DoT connection for `key`, if a fresh one is pooled.
128    #[cfg(with_crypto_provider)]
129    pub(super) fn checkout_tls(&self, key: &TlsKey) -> Option<TlsStream> {
130        take_fresh(self.inner.tls.lock().expect("poisoned").get_mut(key)?)
131    }
132
133    /// Returns a healthy DoT connection to the pool for reuse.
134    #[cfg(with_crypto_provider)]
135    pub(super) fn checkin_tls(&self, key: TlsKey, stream: TlsStream) {
136        push_capped(
137            self.inner
138                .tls
139                .lock()
140                .expect("poisoned")
141                .entry(key)
142                .or_default(),
143            stream,
144        );
145        self.ensure_reaper();
146    }
147
148    /// Spawns the reaper task on first use. Check-in always runs inside an async
149    /// query, so a runtime is guaranteed to be present.
150    fn ensure_reaper(&self) {
151        self.reaper_spawned.get_or_init(|| {
152            spawn(reap_loop(Arc::downgrade(&self.inner)));
153        });
154    }
155}
156
157/// Periodically drops idle connections, exiting once the pool is gone.
158async fn reap_loop(inner: Weak<Inner>) {
159    loop {
160        time::sleep(REAP_INTERVAL).await;
161        let Some(inner) = inner.upgrade() else {
162            return;
163        };
164        inner.reap();
165    }
166}
167
168/// Pops the most-recently-used connection still within [`IDLE_TIMEOUT`],
169/// discarding any staler ones encountered along the way.
170fn take_fresh<S>(idle: &mut Vec<Idle<S>>) -> Option<S> {
171    while let Some(conn) = idle.pop() {
172        if !conn.is_stale() {
173            return Some(conn.stream);
174        }
175    }
176    None
177}
178
179/// Returns a connection to the pool, dropping the oldest if the key is at cap.
180fn push_capped<S>(idle: &mut Vec<Idle<S>>, stream: S) {
181    idle.push(Idle::new(stream));
182    if idle.len() > MAX_IDLE_PER_KEY {
183        idle.remove(0);
184    }
185}
186
187/// Drops every stale connection in `map`, then removes any key left empty.
188fn reap_map<K, S>(map: &Mutex<HashMap<K, Vec<Idle<S>>>>) {
189    map.lock().expect("poisoned").retain(|_, idle| {
190        idle.retain(|conn| !conn.is_stale());
191        !idle.is_empty()
192    });
193}

195#[cfg(test)]
196mod tests {
197    use super::*;
198
199    /// A TCP connection plus its accepted server end, kept alive so the client
200    /// side is not reset.
201    async fn loopback_pair() -> (TcpStream, SocketAddr, TcpStream) {
202        let listener = tokio::net::TcpListener::bind("127.0.0.1:0").await.unwrap();
203        let addr = listener.local_addr().unwrap();
204        let (client, accepted) = tokio::join!(TcpStream::connect(addr), listener.accept());
205        (client.unwrap(), addr, accepted.unwrap().0)
206    }
207
208    fn tcp_idle_count(pool: &ConnPool) -> usize {
209        pool.inner.tcp.lock().unwrap().values().map(Vec::len).sum()
210    }
211
212    #[tokio::test]
213    async fn take_fresh_returns_mru_and_drops_stale() {
214        let mut idle = vec![
215            Idle {
216                stream: 1u32,
217                last_used: Instant::now() - IDLE_TIMEOUT * 2,
218            },
219            Idle::new(2u32),
220        ];
221        // The most-recently-used (back) fresh connection comes out first.
222        assert_eq!(take_fresh(&mut idle), Some(2));
223        // Only the stale one is left, and it is discarded rather than returned.
224        assert_eq!(take_fresh(&mut idle), None);
225        assert!(idle.is_empty());
226    }
227
228    #[tokio::test]
229    async fn push_capped_drops_oldest_over_cap() {
230        let mut idle = Vec::new();
231        for i in 0..(MAX_IDLE_PER_KEY as u32 + 1) {
232            push_capped(&mut idle, i);
233        }
234        assert_eq!(idle.len(), MAX_IDLE_PER_KEY);
235        // The very first (oldest) connection was evicted.
236        assert_eq!(idle.first().map(|c| c.stream), Some(1));
237    }
238
239    #[tokio::test]
240    async fn reap_map_drops_stale_then_empty_keys() {
241        let map: Mutex<HashMap<u8, Vec<Idle<u32>>>> = Mutex::new(HashMap::new());
242        {
243            let mut m = map.lock().unwrap();
244            m.insert(
245                1,
246                vec![Idle {
247                    stream: 10,
248                    last_used: Instant::now() - IDLE_TIMEOUT * 2,
249                }],
250            );
251            m.insert(2, vec![Idle::new(20)]);
252        }
253        reap_map(&map);
254        let m = map.lock().unwrap();
255        // Key 1 had only a stale connection, so it is gone entirely.
256        assert!(!m.contains_key(&1));
257        // Key 2's fresh connection survives.
258        assert_eq!(m.get(&2).map(Vec::len), Some(1));
259    }
260
261    #[tokio::test]
262    async fn tcp_checkout_checkin_roundtrip() {
263        let (client, addr, _server) = loopback_pair().await;
264
265        let pool = ConnPool::new();
266        assert!(pool.checkout_tcp(addr).is_none());
267        pool.checkin_tcp(addr, client);
268        assert!(pool.checkout_tcp(addr).is_some());
269        assert!(pool.checkout_tcp(addr).is_none());
270    }
271
272    #[tokio::test(start_paused = true)]
273    async fn reaper_evicts_untouched_idle_connections() {
274        let (client, addr, _server) = loopback_pair().await;
275
276        let pool = ConnPool::new();
277        pool.checkin_tcp(addr, client);
278        assert_eq!(tcp_idle_count(&pool), 1);
279
280        // Advance past the idle timeout so the reaper sweeps the connection even
281        // though nothing checks it out. (`start_paused` auto-advances the clock.)
282        time::sleep(IDLE_TIMEOUT + REAP_INTERVAL).await;
283        tokio::task::yield_now().await;
284
285        assert_eq!(tcp_idle_count(&pool), 0);
286    }
