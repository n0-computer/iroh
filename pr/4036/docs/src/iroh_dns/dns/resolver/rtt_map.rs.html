1//! Smoothed round-trip-time tracking used to order nameservers fastest-first.
2
3use std::{sync::Mutex, time::Instant};
4
5use n0_future::time::Duration;
6
7/// EWMA weight for folding a new RTT sample into a nameserver's smoothed RTT.
8const SRTT_ALPHA: f64 = 0.3;
9/// Neutral smoothed-RTT baseline (microseconds) for a never-probed nameserver,
10/// and the value the estimate decays toward. Measured-fast servers sit below it
11/// and are preferred; failed servers sit above it and are demoted. Decaying
12/// toward this (rather than toward zero) keeps a measured-fast server ahead of
13/// an idle or recovering one instead of the other way round.
14const SRTT_BASELINE_MICROS: f64 = 50_000.0;
15/// Penalty added to a nameserver's smoothed RTT on a failed attempt, in
16/// microseconds. Large enough to demote it below currently-healthy servers.
17const SRTT_FAILURE_PENALTY_MICROS: f64 = 150_000.0;
18/// Upper bound on a nameserver's smoothed RTT, in microseconds.
19const SRTT_MAX_MICROS: f64 = 5_000_000.0;
20/// Time constant (seconds) of the read-time decay of the smoothed RTT toward
21/// the baseline, so demoted servers recover and idle estimates lapse to neutral.
22const SRTT_DECAY_SECS: f64 = 180.0;
23
24/// Smoothed round-trip time estimate for one nameserver.
25///
26/// Used to order nameservers fastest-first and to demote ones that fail. A
27/// read-time exponential decay pulls the estimate back toward zero so that a
28/// demoted server eventually gets re-probed, and a once-fast server that has
29/// gone away does not stay preferred forever.
30#[derive(Debug)]
31struct Srtt {
32    /// Smoothed estimate in microseconds, as of `updated`.
33    micros: f64,
34    /// When `micros` was last written.
35    updated: Instant,
36}
37
38impl Srtt {
39    fn new() -> Self {
40        Self {
41            micros: SRTT_BASELINE_MICROS,
42            updated: Instant::now(),
43        }
44    }
45
46    /// The decayed estimate at `now`, used for ordering. Relaxes toward
47    /// [`SRTT_BASELINE_MICROS`] as the estimate ages.
48    fn decayed(&self, now: Instant) -> f64 {
49        let dt = now.saturating_duration_since(self.updated).as_secs_f64();
50        SRTT_BASELINE_MICROS + (self.micros - SRTT_BASELINE_MICROS) * (-dt / SRTT_DECAY_SECS).exp()
51    }
52
53    /// Folds a successful round-trip time into the estimate.
54    fn record_success(&mut self, rtt: Duration, now: Instant) {
55        let sample = rtt.as_micros() as f64;
56        let base = self.decayed(now);
57        self.micros = (SRTT_ALPHA * sample + (1.0 - SRTT_ALPHA) * base).min(SRTT_MAX_MICROS);
58        self.updated = now;
59    }
60
61    /// Penalizes the estimate after a failed attempt.
62    fn record_failure(&mut self, now: Instant) {
63        let base = self.decayed(now);
64        self.micros = (base + SRTT_FAILURE_PENALTY_MICROS).min(SRTT_MAX_MICROS);
65        self.updated = now;
66    }
67}
68
69/// Smoothed-RTT estimates for a fixed set of nameservers, indexed in parallel
70/// to the resolver's nameserver list.
71///
72/// Behind a single mutex so the resolver can read and update health from
73/// concurrent queries without threading a lock through the call sites.
74#[derive(Debug)]
75pub(super) struct RttMap {
76    entries: Mutex<Vec<Srtt>>,
77}
78
79impl RttMap {
80    /// Creates a map with `len` nameservers, each at the neutral baseline.
81    pub(super) fn new(len: usize) -> Self {
82        Self {
83            entries: Mutex::new((0..len).map(|_| Srtt::new()).collect()),
84        }
85    }
86
87    /// Returns the decayed smoothed RTT for nameserver `idx`, used for ordering.
88    pub(super) fn get_decayed(&self, idx: usize) -> f64 {
89        self.entries.lock().expect("poisoned")[idx].decayed(Instant::now())
90    }
91
92    /// Folds a successful round-trip time for nameserver `idx` into its estimate.
93    pub(super) fn record_success(&self, idx: usize, rtt: Duration) {
94        self.entries.lock().expect("poisoned")[idx].record_success(rtt, Instant::now());
95    }
96
97    /// Penalizes nameserver `idx` after a failed attempt.
98    pub(super) fn record_failure(&self, idx: usize) {
99        self.entries.lock().expect("poisoned")[idx].record_failure(Instant::now());
100    }
101}
