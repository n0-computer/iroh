12/// Errors that can occur while sending a query to a single nameserver and
13/// reading its response.
14///
15/// Mapped to [`crate::dns::DnsError::Transport`] by the resolver layer.
16#[allow(missing_docs)]
17#[stack_error(derive, add_meta, std_sources)]
18pub(super) enum TransportError {
19    #[error("transport I/O failed")]
20    Io {
21        #[error(from)]
22        source: io::Error,
23    },
24    #[error("response from unexpected source {actual}, expected {expected}")]
25    UnexpectedSource {
26        expected: SocketAddr,
27        actual: SocketAddr,
28    },
29    #[error("query too large for TCP framing")]
30    QueryTooLarge {},
31    #[cfg(with_crypto_provider)]
32    #[error("invalid TLS server name")]
33    InvalidServerName { source: AnyError },
34    #[cfg(with_crypto_provider)]
35    #[error("DNS-over-HTTPS request failed")]
36    Http {
37        #[error(from)]
38        source: reqwest::Error,
39    },
40    #[cfg(with_crypto_provider)]
41    #[error("failed to build HTTPS client")]
42    BuildClient { source: AnyError },
43}

45// Known limitation: TCP and TLS connections are not reused across queries.
46// Each query opens a fresh connection, which means a full TLS handshake per
47// DNS-over-TLS query. This adds significant latency for DoT/DoH workloads.
48// The default UDP-only configuration is not affected.
49//
50// A future improvement could maintain a per-nameserver connection pool for
51// TCP/TLS, similar to how hickory-resolver multiplexes queries over
52// persistent connections.
53//
54// UDP sockets are intentionally not reused (new random source port per query
55// prevents cache poisoning).
56
57/// Send a DNS query over UDP and receive the response.
58///
59/// Each query uses a fresh socket with a random ephemeral source port to
60/// prevent cache poisoning. The response source address is validated against
61/// the target nameserver.
62pub(super) async fn udp_query(addr: SocketAddr, query: &[u8]) -> Result<Vec<u8>, TransportError> {
63    let unspecified: std::net::IpAddr = if addr.is_ipv6() {
64        std::net::Ipv6Addr::UNSPECIFIED.into()
65    } else {
66        std::net::Ipv4Addr::UNSPECIFIED.into()
67    };
68    let bind_addr = SocketAddr::new(unspecified, 0);
69    let socket = tokio::net::UdpSocket::bind(bind_addr).await?;
70    socket.send_to(query, addr).await?;
71
72    let mut buf = vec![0u8; 4096];
73    let (len, src) = socket.recv_from(&mut buf).await?;
74    if src != addr {
75        return Err(e!(TransportError::UnexpectedSource {
76            expected: addr,
77            actual: src,
78        }));
79    }
80    buf.truncate(len);
81    Ok(buf)
82}

84/// Send a DNS query over TCP (RFC 1035 Section 4.2.2: 2-byte length prefix).
85pub(super) async fn tcp_query(addr: SocketAddr, query: &[u8]) -> Result<Vec<u8>, TransportError> {
86    let mut stream = tokio::net::TcpStream::connect(addr).await?;
87
88    // Write length-prefixed query
89    let len = u16::try_from(query.len())
90        .map_err(|_| e!(TransportError::QueryTooLarge))?
91        .to_be_bytes();
92    stream.write_all(&len).await?;
93    stream.write_all(query).await?;
94    stream.flush().await?;
95
96    // Read length-prefixed response
97    let resp_len = stream.read_u16().await? as usize;
98    let mut buf = vec![0u8; resp_len];
99    stream.read_exact(&mut buf).await?;
100    Ok(buf)
101}

103/// Send a DNS query over TLS (DNS-over-TLS, RFC 7858).
104///
105/// With `server_name`, that name is used for SNI and certificate validation;
106/// without it the certificate is validated against the IP address, which works
107/// for providers that include IP SANs (e.g. Google `8.8.8.8`, Cloudflare
108/// `1.1.1.1`) but not for those whose certificates only cover a hostname.
109#[cfg(with_crypto_provider)]
110pub(super) async fn tls_query(
111    addr: SocketAddr,
112    query: &[u8],
113    tls_config: &Arc<rustls::ClientConfig>,
114    server_name: Option<&str>,
115) -> Result<Vec<u8>, TransportError> {
116    let connector = tokio_rustls::TlsConnector::from(tls_config.clone());
117    let tcp_stream = tokio::net::TcpStream::connect(addr).await?;
118
119    // Use the explicit server name for SNI and validation if given, otherwise
120    // validate against the IP the connection was made to.
121    let server_name = match server_name {
122        Some(name) => rustls::pki_types::ServerName::try_from(name.to_string())
123            .map_err(|e| e!(TransportError::InvalidServerName, AnyError::from_std(e)))?,
124        None => rustls::pki_types::ServerName::IpAddress(addr.ip().into()),
125    };
126    let mut stream = connector.connect(server_name, tcp_stream).await?;
127
128    // Write length-prefixed query (same framing as TCP)
129    let len = u16::try_from(query.len())
130        .map_err(|_| e!(TransportError::QueryTooLarge))?
131        .to_be_bytes();
132    stream.write_all(&len).await?;
133    stream.write_all(query).await?;
134    stream.flush().await?;
135
136    // Read length-prefixed response
137    let resp_len = stream.read_u16().await? as usize;
138    let mut buf = vec![0u8; resp_len];
139    stream.read_exact(&mut buf).await?;
140    Ok(buf)
141}

143/// Build a [`reqwest::Client`] for DNS-over-HTTPS queries.
144///
145/// `resolves` pins each named DoH host to a fixed address, so a hostname-based
146/// DoH URL connects to that IP instead of being resolved recursively.
147#[cfg(with_crypto_provider)]
148pub(super) fn build_https_client(
149    tls_config: Option<&Arc<rustls::ClientConfig>>,
150    resolves: &[(String, SocketAddr)],
151) -> Result<reqwest::Client, TransportError> {
152    let mut builder = reqwest::Client::builder();
153    if let Some(config) = tls_config {
154        builder = builder.use_preconfigured_tls(config.clone());
155    }
156    for (host, addr) in resolves {
157        builder = builder.resolve(host, *addr);
158    }
159    builder
160        .build()
161        .map_err(|e| e!(TransportError::BuildClient, AnyError::from_std(e)))
162}

164/// Send a DNS query over HTTPS (DNS-over-HTTPS, RFC 8484).
165///
166/// With `server_name`, the URL is addressed by hostname (the client pins it to
167/// `addr`); without it the URL is addressed by IP (e.g.
168/// `https://1.1.1.1/dns-query`), which works only for providers whose
169/// certificates include the IP as a SAN.
170#[cfg(with_crypto_provider)]
171pub(super) async fn https_query(
172    addr: SocketAddr,
173    server_name: Option<&str>,
174    query: &[u8],
175    client: &reqwest::Client,
176) -> Result<Vec<u8>, TransportError> {
177    // With a server name, address the URL by hostname (the client pins it to
178    // `addr`); otherwise address it by IP.
179    let url = match server_name {
180        Some(name) => format!("https://{name}:{}/dns-query", addr.port()),
181        None => format!("https://{addr}/dns-query"),
182    };
183    let response = client
184        .post(&url)
185        .header("content-type", "application/dns-message")
186        .header("accept", "application/dns-message")
187        .body(query.to_vec())
188        .send()
189        .await?;
190
191    let bytes = response.error_for_status()?.bytes().await?;
192    Ok(bytes.to_vec())
193}

207    fn build_a_response(id: u16, addrs: &[Ipv4Addr]) -> Vec<u8> {
208        let mut packet = Packet::new_reply(id);
209        packet.set_flags(PacketFlag::RECURSION_DESIRED | PacketFlag::RECURSION_AVAILABLE);
210        for addr in addrs {
211            let rdata = RData::A(A {
212                address: u32::from(*addr),
213            });
214            packet.answers.push(ResourceRecord::new(
215                Name::new_unchecked("example.com"),
216                CLASS::IN,
217                300,
218                rdata,
219            ));
220        }
221        packet.build_bytes_vec().unwrap()
222    }

224    /// Spawn a mock UDP server that echoes back an A response for any query.
225    async fn mock_udp_server(addrs: &[Ipv4Addr]) -> (SocketAddr, tokio::task::JoinHandle<()>) {
226        let server = tokio::net::UdpSocket::bind("127.0.0.1:0").await.unwrap();
227        let server_addr = server.local_addr().unwrap();
228        let addrs = addrs.to_vec();
229        let handle = tokio::spawn(async move {
230            let mut buf = vec![0u8; 4096];
231            let (len, client_addr) = server.recv_from(&mut buf).await.unwrap();
232            let id = Packet::parse(&buf[..len]).unwrap().id();
233            server
234                .send_to(&build_a_response(id, &addrs), client_addr)
235                .await
236                .unwrap();
237        });
238        (server_addr, handle)
239    }

241    /// Spawn a mock TCP server that echoes back an A response for any query.
242    async fn mock_tcp_server(addrs: &[Ipv4Addr]) -> (SocketAddr, tokio::task::JoinHandle<()>) {
243        let listener = tokio::net::TcpListener::bind("127.0.0.1:0").await.unwrap();
244        let server_addr = listener.local_addr().unwrap();
245        let addrs = addrs.to_vec();
246        let handle = tokio::spawn(async move {
247            let (mut stream, _) = listener.accept().await.unwrap();
248            let query_len = stream.read_u16().await.unwrap() as usize;
249            let mut query_buf = vec![0u8; query_len];
250            stream.read_exact(&mut query_buf).await.unwrap();
251            let id = Packet::parse(&query_buf).unwrap().id();

253            stream
254                .write_all(&(resp.len() as u16).to_be_bytes())
255                .await
256                .unwrap();
257            stream.write_all(&resp).await.unwrap();
258            stream.flush().await.unwrap();
259        });
260        (server_addr, handle)
261    }
