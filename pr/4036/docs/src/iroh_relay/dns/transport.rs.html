14/// Send a DNS query over UDP and receive the response.
15pub(super) async fn udp_query(addr: SocketAddr, query: &[u8]) -> Result<Vec<u8>, DnsError> {
16    let bind_addr = SocketAddr::new(
17        if addr.is_ipv6() {
18            std::net::IpAddr::V6(std::net::Ipv6Addr::UNSPECIFIED)
19        } else {
20            std::net::IpAddr::V4(std::net::Ipv4Addr::UNSPECIFIED)
21        },
22        0,
23    );
24    let socket = tokio::net::UdpSocket::bind(bind_addr).await?;
25    socket.send_to(query, addr).await?;
26
27    let mut buf = vec![0u8; 4096];
28    let len = socket.recv(&mut buf).await?;
29    buf.truncate(len);
30    Ok(buf)
31}

33/// Send a DNS query over TCP (RFC 1035 Section 4.2.2: 2-byte length prefix).
34pub(super) async fn tcp_query(addr: SocketAddr, query: &[u8]) -> Result<Vec<u8>, DnsError> {
35    let mut stream = tokio::net::TcpStream::connect(addr).await?;
36
37    // Write length-prefixed query
38    let len = (query.len() as u16).to_be_bytes();
39    stream.write_all(&len).await?;
40    stream.write_all(query).await?;
41    stream.flush().await?;
42
43    // Read length-prefixed response
44    let resp_len = stream.read_u16().await? as usize;
45    let mut buf = vec![0u8; resp_len];
46    stream.read_exact(&mut buf).await?;
47    Ok(buf)
48}

50/// Send a DNS query over TLS (DNS-over-TLS, RFC 7858).
51pub(super) async fn tls_query(
52    addr: SocketAddr,
53    query: &[u8],
54    tls_config: &Arc<rustls::ClientConfig>,
55) -> Result<Vec<u8>, DnsError> {
56    let connector = tokio_rustls::TlsConnector::from(tls_config.clone());
57    let tcp_stream = tokio::net::TcpStream::connect(addr).await?;
58
59    // Use the IP address as the server name for SNI.
60    let server_name = rustls::pki_types::ServerName::IpAddress(addr.ip().into());
61    let mut stream = connector.connect(server_name, tcp_stream).await?;
62
63    // Write length-prefixed query (same framing as TCP)
64    let len = (query.len() as u16).to_be_bytes();
65    stream.write_all(&len).await?;
66    stream.write_all(query).await?;
67    stream.flush().await?;
68
69    // Read length-prefixed response
70    let resp_len = stream.read_u16().await? as usize;
71    let mut buf = vec![0u8; resp_len];
72    stream.read_exact(&mut buf).await?;
73    Ok(buf)
74}

76/// Build a [`reqwest::Client`] for DNS-over-HTTPS queries.
77pub(super) fn build_https_client(
78    tls_config: Option<&Arc<rustls::ClientConfig>>,
79) -> Result<reqwest::Client, DnsError> {
80    let mut builder = reqwest::Client::builder();
81    if let Some(config) = tls_config {
82        builder = builder.use_preconfigured_tls(config.clone());
83    }
84    Ok(builder.build()?)
85}

87/// Send a DNS query over HTTPS (DNS-over-HTTPS, RFC 8484).
88pub(super) async fn https_query(
89    addr: SocketAddr,
90    query: &[u8],
91    client: &reqwest::Client,
92) -> Result<Vec<u8>, DnsError> {
93    let url = format!("https://{}/dns-query", addr);
94    let response = client
95        .post(&url)
96        .header("content-type", "application/dns-message")
97        .header("accept", "application/dns-message")
98        .body(query.to_vec())
99        .send()
100        .await?;
101
102    let bytes = response.bytes().await?;
103    Ok(bytes.to_vec())
104}

118    fn build_a_response(id: u16, addrs: &[Ipv4Addr]) -> Vec<u8> {
119        let mut packet = Packet::new_reply(id);
120        packet.set_flags(PacketFlag::RECURSION_DESIRED | PacketFlag::RECURSION_AVAILABLE);
121        for addr in addrs {
122            let rdata = RData::A(A {
123                address: u32::from(*addr),
124            });
125            packet.answers.push(ResourceRecord::new(
126                Name::new_unchecked("example.com"),
127                CLASS::IN,
128                300,
129                rdata,
130            ));
131        }
132        packet.build_bytes_vec().unwrap()
133    }

135    /// Spawn a mock UDP server that echoes back an A response for any query.
136    async fn mock_udp_server(addrs: &[Ipv4Addr]) -> (SocketAddr, tokio::task::JoinHandle<()>) {
137        let server = tokio::net::UdpSocket::bind("127.0.0.1:0").await.unwrap();
138        let server_addr = server.local_addr().unwrap();
139        let addrs = addrs.to_vec();
140        let handle = tokio::spawn(async move {
141            let mut buf = vec![0u8; 4096];
142            let (len, client_addr) = server.recv_from(&mut buf).await.unwrap();
143            let id = Packet::parse(&buf[..len]).unwrap().id();
144            server
145                .send_to(&build_a_response(id, &addrs), client_addr)
146                .await
147                .unwrap();
148        });
149        (server_addr, handle)
150    }

152    /// Spawn a mock TCP server that echoes back an A response for any query.
153    async fn mock_tcp_server(addrs: &[Ipv4Addr]) -> (SocketAddr, tokio::task::JoinHandle<()>) {
154        let listener = tokio::net::TcpListener::bind("127.0.0.1:0").await.unwrap();
155        let server_addr = listener.local_addr().unwrap();
156        let addrs = addrs.to_vec();
157        let handle = tokio::spawn(async move {
158            let (mut stream, _) = listener.accept().await.unwrap();
159            let query_len = stream.read_u16().await.unwrap() as usize;
160            let mut query_buf = vec![0u8; query_len];
161            stream.read_exact(&mut query_buf).await.unwrap();
162            let id = Packet::parse(&query_buf).unwrap().id();
163            let resp = build_a_response(id, &addrs);
164            stream
165                .write_all(&(resp.len() as u16).to_be_bytes())
166                .await
167                .unwrap();
168            stream.write_all(&resp).await.unwrap();
169            stream.flush().await.unwrap();
170        });
171        (server_addr, handle)
172    }
