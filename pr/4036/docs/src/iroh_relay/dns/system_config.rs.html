7/// Deprecated IPv6 site-local anycast addresses still configured by Windows.
8///
9/// Windows still configures these site-local addresses as soon as an IPv6 loopback
10/// interface is configured. We do not want to use these DNS servers, the chances of them
11/// being usable are almost always close to zero, while the chance of DNS configuration
12/// **only** relying on these servers and not also being configured normally are also almost
13/// zero. The chance of the DNS resolver accidentally trying one of these and taking a
14/// bunch of timeouts to figure out they're no good are on the other hand very high.
15#[cfg(windows)]
16const WINDOWS_BAD_SITE_LOCAL_DNS_SERVERS: [IpAddr; 3] = [
17    IpAddr::V6(Ipv6Addr::new(0xfec0, 0, 0, 0xffff, 0, 0, 0, 1)),
18    IpAddr::V6(Ipv6Addr::new(0xfec0, 0, 0, 0xffff, 0, 0, 0, 2)),
19    IpAddr::V6(Ipv6Addr::new(0xfec0, 0, 0, 0xffff, 0, 0, 0, 3)),
20];
21
22/// Standard DNS port.
23const DNS_PORT: u16 = 53;
24
25/// Google Public DNS IPv4 primary (8.8.8.8).
26const GOOGLE_DNS_IPV4_PRIMARY: Ipv4Addr = Ipv4Addr::new(8, 8, 8, 8);
27/// Google Public DNS IPv4 secondary (8.8.4.4).
28const GOOGLE_DNS_IPV4_SECONDARY: Ipv4Addr = Ipv4Addr::new(8, 8, 4, 4);
29/// Google Public DNS IPv6 primary (2001:4860:4860::8888).
30const GOOGLE_DNS_IPV6_PRIMARY: Ipv6Addr = Ipv6Addr::new(0x2001, 0x4860, 0x4860, 0, 0, 0, 0, 0x8888);
31/// Google Public DNS IPv6 secondary (2001:4860:4860::8844).
32const GOOGLE_DNS_IPV6_SECONDARY: Ipv6Addr =
33    Ipv6Addr::new(0x2001, 0x4860, 0x4860, 0, 0, 0, 0, 0x8844);
34
35/// Parsed system DNS configuration.
36#[derive(Debug, Clone, Default)]
37pub(super) struct SystemDnsConfig {
38    pub(super) nameservers: Vec<(SocketAddr, DnsProtocol)>,
39    /// Search domains from resolv.conf `search` or `domain` directives.
40    ///
41    /// When resolving a short hostname (one with fewer dots than `ndots`,
42    /// default 1), the resolver should try appending each search domain
43    /// before querying the bare name.
44    pub(super) search_domains: Vec<String>,
45    /// The `ndots` option from resolv.conf.
46    ///
47    /// Names with at least this many dots are tried as absolute first.
48    /// `None` means use the default (1).
49    /// See <https://man7.org/linux/man-pages/man5/resolv.conf.5.html>.
50    pub(super) ndots: Option<usize>,
51}
52
53/// Parse system DNS configuration.
54///
55/// On Unix, reads `/etc/resolv.conf` for `nameserver` and `search`/`domain` lines.
56/// On Windows, uses the `ipconfig` crate to enumerate network adapters.
57/// Falls back to Google DNS if parsing fails or no servers are found.
58pub(super) fn system_config() -> SystemDnsConfig {
59    match read_system_dns() {
60        Ok(config) if !config.nameservers.is_empty() => config,
61        _ => SystemDnsConfig {
62            nameservers: fallback_nameservers(),
63            search_domains: Vec::new(),
64            ndots: None,
65        },
66    }
67}

81/// Google DNS fallback nameservers.
82pub(super) fn fallback_nameservers() -> Vec<(SocketAddr, DnsProtocol)> {
83    vec![
84        (
85            SocketAddr::new(IpAddr::V4(GOOGLE_DNS_IPV4_PRIMARY), DNS_PORT),
86            DnsProtocol::Udp,
87        ),
88        (
89            SocketAddr::new(IpAddr::V4(GOOGLE_DNS_IPV4_SECONDARY), DNS_PORT),
90            DnsProtocol::Udp,
91        ),
92        (
93            SocketAddr::new(IpAddr::V6(GOOGLE_DNS_IPV6_PRIMARY), DNS_PORT),
94            DnsProtocol::Udp,
95        ),
96        (
97            SocketAddr::new(IpAddr::V6(GOOGLE_DNS_IPV6_SECONDARY), DNS_PORT),
98            DnsProtocol::Udp,
99        ),
100    ]
101}

103/// Read DNS servers from Windows network adapter configuration.
104#[cfg(windows)]
105fn read_from_ipconfig() -> Result<SystemDnsConfig, std::io::Error> {
106    let adapters =
107        ipconfig::get_adapters().map_err(|e| std::io::Error::new(std::io::ErrorKind::Other, e))?;
108
109    let mut servers = Vec::new();
110    for adapter in adapters {
111        // Only consider adapters that are up
112        if adapter.oper_status() != ipconfig::OperStatus::IfOperStatusUp {
113            continue;
114        }
115        for dns_server in adapter.dns_servers() {
116            let ip = IpAddr::from(*dns_server);
117            if !WINDOWS_BAD_SITE_LOCAL_DNS_SERVERS.contains(&ip) {
118                servers.push((SocketAddr::new(ip, DNS_PORT), DnsProtocol::Udp));
119            }
120        }
121    }
122
123    // Deduplicate -- multiple adapters may report the same DNS server.
124    // Use a HashSet since dedup_by_key only removes consecutive duplicates.
125    let mut seen = std::collections::HashSet::new();
126    servers.retain(|(addr, _)| seen.insert(*addr));
127
128    // Read search domains from the Windows registry (comma-separated SearchList key).
129    // Falls back to the primary domain if no search list is configured.
130    let search_domains = ipconfig::computer::get_search_list()
131        .unwrap_or_default()
132        .into_iter()
133        .filter(|s| !s.is_empty())
134        .collect::<Vec<_>>();
135    let search_domains = if search_domains.is_empty() {
136        ipconfig::computer::get_domain()
137            .ok()
138            .flatten()
139            .filter(|s| !s.is_empty())
140            .into_iter()
141            .collect()
142    } else {
143        search_domains
144    };
145
146    Ok(SystemDnsConfig {
147        nameservers: servers,
148        search_domains,
149        ndots: None,
150    })
151}

153/// Read `/etc/resolv.conf` and extract nameserver addresses and search domains.
154#[cfg(not(windows))]
155fn read_resolv_conf() -> Result<SystemDnsConfig, std::io::Error> {
156    let content = std::fs::read_to_string("/etc/resolv.conf")?;
157    Ok(parse_resolv_conf(&content))
158}
159
160/// Parse resolv.conf content and extract nameserver addresses and search domains.
161///
162/// Recognized directives:
163/// - `nameserver <ip>` -- adds a DNS nameserver
164/// - `search <domain> [<domain> ...]` -- sets the search domain list
165/// - `domain <domain>` -- equivalent to a single-entry search list
166///
167/// The `search` and `domain` directives are mutually exclusive per resolv.conf(5);
168/// the last one seen wins, matching standard resolver behavior.
169#[cfg(any(not(windows), test))]
170fn parse_resolv_conf(content: &str) -> SystemDnsConfig {
171    let mut servers = Vec::new();
172    let mut search_domains = Vec::new();
173    let mut ndots = None;
174
175    for line in content.lines() {
176        let line = line.trim();
177        // Skip comments
178        if line.starts_with('#') || line.starts_with(';') {
179            continue;
180        }
181        let mut parts = line.split_whitespace();
182        match parts.next() {
183            Some("nameserver") => {
184                if let Some(addr_str) = parts.next() {
185                    // Try parsing as SocketAddr first (supports custom ports like
186                    // 8.8.8.8:5353 or [::1]:5353), then fall back to IpAddr with
187                    // the default DNS port.
188                    let addr = addr_str.parse::<SocketAddr>().ok().or_else(|| {
189                        addr_str
190                            .parse::<IpAddr>()
191                            .ok()
192                            .map(|ip| SocketAddr::new(ip, DNS_PORT))
193                    });
194                    if let Some(addr) = addr {
195                        servers.push((addr, DnsProtocol::Udp));
196                    }
197                }
198            }
199            Some("search") => {
200                // `search` replaces any previous search/domain list.
201                search_domains = parts.map(|s| s.to_string()).collect();
202            }
203            Some("domain") => {
204                // `domain` is equivalent to a single-entry search list.
205                if let Some(domain) = parts.next() {
206                    search_domains = vec![domain.to_string()];
207                }
208            }
209            Some("options") => {
210                for opt in parts {
211                    if let Some(n) = opt.strip_prefix("ndots:").and_then(|v| v.parse().ok()) {
212                        ndots = Some(n);
213                    }
214                }
215            }
216            _ => {}
217        }
218    }
219
220    SystemDnsConfig {
221        nameservers: servers,
222        search_domains,
223        ndots,
224    }
225}

231    fn ips(config: &SystemDnsConfig) -> Vec<IpAddr> {
232        config.nameservers.iter().map(|(a, _)| a.ip()).collect()
233    }
234
235    fn ipv4(a: u8, b: u8, c: u8, d: u8) -> IpAddr {
236        IpAddr::V4(Ipv4Addr::new(a, b, c, d))
237    }

239    #[test]
240    fn parse_basic() {
241        let config = parse_resolv_conf("nameserver 8.8.8.8\nnameserver 8.8.4.4\n");
242        assert_eq!(ips(&config), [ipv4(8, 8, 8, 8), ipv4(8, 8, 4, 4)]);
243        assert!(
244            config
245                .nameservers
246                .iter()
247                .all(|(_, p)| *p == DnsProtocol::Udp)
248        );
249    }

255        assert!(config.nameservers[1].0.ip().is_ipv6());
256    }
