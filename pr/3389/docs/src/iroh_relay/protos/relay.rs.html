1//! This module implements the send/recv relaying protocol.
2//!
3//! Protocol flow:
4//!  * server occasionally sends [`FrameType::Ping`]
5//!  * client responds to any [`FrameType::Ping`] with a [`FrameType::Pong`]
6//!  * clients sends [`FrameType::ClientToRelayDatagrams`]
7//!  * server then sends [`FrameType::RelayToClientDatagrams`] to recipient
8//!  * server sends [`FrameType::NodeGone`] when the other client disconnects
9

12use n0_future::time::{self, Duration};
13use nested_enum_utils::common_fields;
14use snafu::{Backtrace, ResultExt, Snafu};

17use crate::KeyCache;

19/// The maximum size of a packet sent over relay.
20/// (This only includes the data bytes visible to magicsock, not
21/// including its on-wire framing overhead)
22pub const MAX_PACKET_SIZE: usize = 64 * 1024;
23
24/// Maximum size a datagram payload is allowed to be.
25///
26/// This is [`MAX_PACKET_SIZE`] minus the length of an encoded public key minus 3 bytes,
27/// one for ECN, and two for the segment size.
28pub const MAX_PAYLOAD_SIZE: usize = MAX_PACKET_SIZE - NodeId::LENGTH - 3;
29
30/// The maximum frame size.
31///
32/// This is also the minimum burst size that a rate-limiter has to accept.
33#[cfg(not(wasm_browser))]
34pub(crate) const MAX_FRAME_SIZE: usize = 1024 * 1024;
35
36/// Interval in which we ping the relay server to ensure the connection is alive.
37///
38/// The default QUIC max_idle_timeout is 30s, so setting that to half this time gives some
39/// chance of recovering.
40#[cfg(feature = "server")]
41pub(crate) const PING_INTERVAL: Duration = Duration::from_secs(15);
42
43/// The number of packets buffered for sending per client
44#[cfg(feature = "server")]
45pub(crate) const PER_CLIENT_SEND_QUEUE_DEPTH: usize = 512;

47/// Protocol send errors.
48#[common_fields({
49    backtrace: Option<Backtrace>,
50    #[snafu(implicit)]
51    span_trace: n0_snafu::SpanTrace,
52})]
53#[allow(missing_docs)]
54#[derive(Debug, Snafu)]
55#[non_exhaustive]
56pub enum Error {
57    #[snafu(display("unexpected frame: got {got}, expected {expected}"))]
58    UnexpectedFrame { got: FrameType, expected: FrameType },
59    #[snafu(display("Frame is too large, has {frame_len} bytes"))]
60    FrameTooLarge { frame_len: usize },
61    #[snafu(transparent)]
62    Timeout { source: time::Elapsed },
63    #[snafu(transparent)]
64    SerDe { source: postcard::Error },
65    #[snafu(transparent)]
66    FrameTypeError { source: FrameTypeError },
67    #[snafu(display("Invalid public key"))]
68    InvalidPublicKey { source: SignatureError },
69    #[snafu(display("Invalid frame encoding"))]
70    InvalidFrame {},
71    #[snafu(display("Invalid frame type: {frame_type}"))]
72    InvalidFrameType { frame_type: FrameType },
73    #[snafu(display("Invalid protocol message encoding"))]
74    InvalidProtocolMessageEncoding { source: std::str::Utf8Error },
75    #[snafu(display("Too few bytes"))]
76    TooSmall {},
77}

79/// The messages that a relay sends to clients or the clients receive from the relay.
80#[derive(derive_more::Debug, Clone, PartialEq, Eq)]
81pub enum RelayToClientMsg {
82    /// Represents datagrams sent from relays (originally sent to them by another client).
83    Datagrams {
84        /// The [`NodeId`] of the original sender.
85        remote_node_id: NodeId,
86        /// The datagrams and related metadata.
87        datagrams: Datagrams,
88    },
89    /// Indicates that the client identified by the underlying public key had previously sent you a
90    /// packet but has now disconnected from the relay.
91    NodeGone(NodeId),
92    /// A one-way message from relay to client, declaring the connection health state.
93    Health {
94        /// If set, is a description of why the connection is unhealthy.
95        ///
96        /// If `None` means the connection is healthy again.
97        ///
98        /// The default condition is healthy, so the relay doesn't broadcast a [`RelayToClientMsg::Health`]
99        /// until a problem exists.
100        problem: String,
101    },
102    /// A one-way message from relay to client, advertising that the relay is restarting.
103    Restarting {
104        /// An advisory duration that the client should wait before attempting to reconnect.
105        /// It might be zero. It exists for the relay to smear out the reconnects.
106        reconnect_in: Duration,
107        /// An advisory duration for how long the client should attempt to reconnect
108        /// before giving up and proceeding with its normal connection failure logic. The interval
109        /// between retries is undefined for now. A relay should not send a `try_for` duration more
110        /// than a few seconds.
111        try_for: Duration,
112    },
113    /// Request from the relay to reply to the
114    /// other side with a [`ClientToRelayMsg::Pong`] with the given payload.
115    Ping([u8; 8]),
116    /// Reply to a [`ClientToRelayMsg::Ping`] from a client
117    /// with the payload sent previously in the ping.
118    Pong([u8; 8]),
119}
120
121/// Messages that clients send to relays.
122#[derive(Debug, Clone, PartialEq, Eq)]
123pub enum ClientToRelayMsg {
124    /// Request from the client to the server to reply to the
125    /// other side with a [`RelayToClientMsg::Pong`] with the given payload.
126    Ping([u8; 8]),
127    /// Reply to a [`RelayToClientMsg::Ping`] from a server
128    /// with the payload sent previously in the ping.
129    Pong([u8; 8]),
130    /// Request from the client to relay datagrams to given remote node.
131    Datagrams {
132        /// The remote node to relay to.
133        dst_node_id: NodeId,
134        /// The datagrams and related metadata to relay.
135        datagrams: Datagrams,
136    },
137}

139/// One or multiple datagrams being transferred via the relay.
140///
141/// This type is modeled after [`quinn_proto::Transmit`]
142/// (or even more similarly `quinn_udp::Transmit`, but we don't depend on that library here).
143#[derive(derive_more::Debug, Clone, PartialEq, Eq)]
144pub struct Datagrams {
145    /// Explicit congestion notification bits
146    pub ecn: Option<quinn_proto::EcnCodepoint>,
147    /// The segment size if this transmission contains multiple datagrams.
148    /// This is `None` if the transmit only contains a single datagram
149    pub segment_size: Option<u16>,
150    /// The contents of the datagram(s)
151    #[debug(skip)]
152    pub contents: Bytes,
153}
154
155impl<T: AsRef<[u8]>> From<T> for Datagrams {
156    fn from(bytes: T) -> Self {
157        Self {
158            ecn: None,
159            segment_size: None,
160            contents: Bytes::copy_from_slice(bytes.as_ref()),
161        }
162    }
163}
164
165impl Datagrams {
166    fn write_to<O: BufMut>(&self, mut dst: O) -> O {
167        let ecn = self.ecn.map_or(0, |ecn| ecn as u8);
168        dst.put_u8(ecn);
169        if let Some(segment_size) = self.segment_size {
170            dst.put_u16(segment_size);
171        }
172        dst.put(self.contents.as_ref());
173        dst
174    }
175
176    fn encoded_len(&self) -> usize {
177        1 // ECN byte
178        + self.segment_size.map_or(0, |_| 2) // segment size, when None, then a packed representation is assumed
179        + self.contents.len()
180    }
181
182    fn from_bytes(mut bytes: Bytes, is_batch: bool) -> Result<Self, Error> {
183        if is_batch {
184            // 1 bytes ECN, 2 bytes segment size
185            snafu::ensure!(bytes.len() >= 3, InvalidFrameSnafu);
186        } else {
187            snafu::ensure!(bytes.len() >= 1, InvalidFrameSnafu);
188        }
189
190        let ecn_byte = bytes.get_u8();
191        let ecn = quinn_proto::EcnCodepoint::from_bits(ecn_byte);
192
193        let segment_size = if is_batch {
194            let segment_size = bytes.get_u16(); // length checked above
195            if segment_size == 0 {
196                None
197            } else {
198                Some(segment_size)
199            }
200        } else {
201            None
202        };
203
204        Ok(Self {
205            ecn,
206            segment_size,
207            contents: bytes,
208        })
209    }
210}

212impl RelayToClientMsg {
213    /// Returns this frame's corresponding frame type.
214    pub fn typ(&self) -> FrameType {
215        match self {
216            Self::Datagrams { datagrams, .. } => {
217                if datagrams.segment_size.is_some() {
218                    FrameType::RelayToClientDatagrams
219                } else {
220                    FrameType::RelayToClientDatagram
221                }
222            }
223            Self::NodeGone { .. } => FrameType::NodeGone,
224            Self::Ping { .. } => FrameType::Ping,
225            Self::Pong { .. } => FrameType::Pong,
226            Self::Health { .. } => FrameType::Health,
227            Self::Restarting { .. } => FrameType::Restarting,
228        }
229    }
230
231    #[cfg(feature = "server")]
232    pub(crate) fn to_bytes(&self) -> BytesMut {
233        self.write_to(BytesMut::with_capacity(self.encoded_len()))
234    }
235
236    /// Encodes this frame for sending over websockets.
237    ///
238    /// Specifically meant for being put into a binary websocket message frame.
239    #[cfg(feature = "server")]
240    pub(crate) fn write_to<O: BufMut>(&self, mut dst: O) -> O {
241        dst = self.typ().write_to(dst);
242        match self {
243            Self::Datagrams {
244                remote_node_id,
245                datagrams,
246            } => {
247                dst.put(remote_node_id.as_ref());
248                dst = datagrams.write_to(dst);
249            }
250            Self::NodeGone(node_id) => {
251                dst.put(node_id.as_ref());
252            }
253            Self::Ping(data) => {
254                dst.put(&data[..]);
255            }
256            Self::Pong(data) => {
257                dst.put(&data[..]);
258            }
259            Self::Health { problem } => {
260                dst.put(problem.as_ref());
261            }
262            Self::Restarting {
263                reconnect_in,
264                try_for,
265            } => {
266                dst.put_u32(reconnect_in.as_millis() as u32);
267                dst.put_u32(try_for.as_millis() as u32);
268            }
269        }
270        dst
271    }
272
273    #[cfg(feature = "server")]
274    pub(crate) fn encoded_len(&self) -> usize {
275        let payload_len = match self {
276            Self::Datagrams { datagrams, .. } => {
277                32 // nodeid
278                + datagrams.encoded_len()
279            }
280            Self::NodeGone(_) => 32,
281            Self::Ping(_) | Self::Pong(_) => 8,
282            Self::Health { problem } => problem.len(),
283            Self::Restarting { .. } => {
284                4 // u32
285                + 4 // u32
286            }
287        };
288        1 // frame type
289        + payload_len
290    }
291
292    /// Tries to decode a frame received over websockets.
293    ///
294    /// Specifically, bytes received from a binary websocket message frame.
295    #[allow(clippy::result_large_err)]
296    pub(crate) fn from_bytes(mut content: Bytes, cache: &KeyCache) -> Result<Self, Error> {
297        let frame_type = FrameType::from_bytes(&mut content)?;
298        let frame_len = content.len();
299        snafu::ensure!(
300            frame_len <= MAX_PACKET_SIZE,
301            FrameTooLargeSnafu { frame_len }
302        );
303
304        let res = match frame_type {
305            FrameType::RelayToClientDatagram | FrameType::RelayToClientDatagrams => {
306                snafu::ensure!(content.len() >= NodeId::LENGTH, InvalidFrameSnafu);
307
308                let remote_node_id = cache
309                    .key_from_slice(&content[..NodeId::LENGTH])
310                    .context(InvalidPublicKeySnafu)?;
311                let datagrams = Datagrams::from_bytes(
312                    content.slice(NodeId::LENGTH..),
313                    frame_type == FrameType::RelayToClientDatagrams,
314                )?;
315                Self::Datagrams {
316                    remote_node_id,
317                    datagrams,
318                }
319            }
320            FrameType::NodeGone => {
321                snafu::ensure!(content.len() == NodeId::LENGTH, InvalidFrameSnafu);
322                let node_id = cache
323                    .key_from_slice(content.as_ref())
324                    .context(InvalidPublicKeySnafu)?;
325                Self::NodeGone(node_id)
326            }
327            FrameType::Ping => {
328                snafu::ensure!(content.len() == 8, InvalidFrameSnafu);
329                let mut data = [0u8; 8];
330                data.copy_from_slice(&content[..8]);
331                Self::Ping(data)
332            }
333            FrameType::Pong => {
334                snafu::ensure!(content.len() == 8, InvalidFrameSnafu);
335                let mut data = [0u8; 8];
336                data.copy_from_slice(&content[..8]);
337                Self::Pong(data)
338            }
339            FrameType::Health => {
340                let problem = std::str::from_utf8(&content)
341                    .context(InvalidProtocolMessageEncodingSnafu)?
342                    .to_owned();
343                Self::Health { problem }
344            }
345            FrameType::Restarting => {
346                snafu::ensure!(content.len() == 4 + 4, InvalidFrameSnafu);
347                let reconnect_in = u32::from_be_bytes(
348                    content[..4]
349                        .try_into()
350                        .map_err(|_| InvalidFrameSnafu.build())?,
351                );
352                let try_for = u32::from_be_bytes(
353                    content[4..]
354                        .try_into()
355                        .map_err(|_| InvalidFrameSnafu.build())?,
356                );
357                let reconnect_in = Duration::from_millis(reconnect_in as u64);
358                let try_for = Duration::from_millis(try_for as u64);
359                Self::Restarting {
360                    reconnect_in,
361                    try_for,
362                }
363            }
364            _ => {
365                return Err(InvalidFrameTypeSnafu { frame_type }.build());
366            }
367        };
368        Ok(res)
369    }
370}

644    fn secret_key() -> impl Strategy<Value = SecretKey> {
645        prop::array::uniform32(any::<u8>()).prop_map(SecretKey::from)
646    }
647
648    fn key() -> impl Strategy<Value = NodeId> {
649        secret_key().prop_map(|key| key.public())
650    }
651
652    fn ecn() -> impl Strategy<Value = Option<quinn_proto::EcnCodepoint>> {
653        (0..=3).prop_map(|n| match n {
654            1 => Some(quinn_proto::EcnCodepoint::Ce),
655            2 => Some(quinn_proto::EcnCodepoint::Ect0),
656            3 => Some(quinn_proto::EcnCodepoint::Ect1),
657            _ => None,
658        })
659    }
660
661    fn datagrams() -> impl Strategy<Value = Datagrams> {
662        (
663            ecn(),
664            prop::option::of(MAX_PAYLOAD_SIZE / 20..MAX_PAYLOAD_SIZE),
665            prop::collection::vec(any::<u8>(), 0..MAX_PAYLOAD_SIZE),
666        )
667            .prop_map(|(ecn, segment_size, data)| Datagrams {
668                ecn,
669                segment_size: segment_size.map(|ss| std::cmp::min(data.len(), ss) as u16),
670                contents: Bytes::from(data),
671            })
672    }
673
674    /// Generates a random valid frame
675    fn server_client_frame() -> impl Strategy<Value = RelayToClientMsg> {
676        let recv_packet = (key(), datagrams()).prop_map(|(remote_node_id, datagrams)| {
677            RelayToClientMsg::Datagrams {
678                remote_node_id,
679                datagrams,
680            }
681        });
682        let node_gone = key().prop_map(RelayToClientMsg::NodeGone);
683        let ping = prop::array::uniform8(any::<u8>()).prop_map(RelayToClientMsg::Ping);
684        let pong = prop::array::uniform8(any::<u8>()).prop_map(RelayToClientMsg::Pong);
685        let health = ".{0,65536}"
686            .prop_filter("exceeds MAX_PAYLOAD_SIZE", |s| {
687                s.len() < MAX_PAYLOAD_SIZE // a single unicode character can match a regex "." but take up multiple bytes
688            })
689            .prop_map(|problem| RelayToClientMsg::Health { problem });
690        let restarting = (any::<u32>(), any::<u32>()).prop_map(|(reconnect_in, try_for)| {
691            RelayToClientMsg::Restarting {
692                reconnect_in: Duration::from_millis(reconnect_in.into()),
693                try_for: Duration::from_millis(try_for.into()),
694            }
695        });
696        prop_oneof![recv_packet, node_gone, ping, pong, health, restarting]
697    }
698
699    fn client_server_frame() -> impl Strategy<Value = ClientToRelayMsg> {
700        let send_packet =
701            (key(), datagrams()).prop_map(|(dst_node_id, datagrams)| ClientToRelayMsg::Datagrams {
702                dst_node_id,
703                datagrams,
704            });
705        let ping = prop::array::uniform8(any::<u8>()).prop_map(ClientToRelayMsg::Ping);
706        let pong = prop::array::uniform8(any::<u8>()).prop_map(ClientToRelayMsg::Pong);
707        prop_oneof![send_packet, ping, pong]
708    }
