21/// The relay protocol version negotiated between client and server.
22///
23/// Sent as the websocket sub-protocol header `Sec-Websocket-Protocol` from
24/// the client. The server picks the best supported version and replies with it.
25///
26/// Variants are ordered by preference (highest first), so the [`Ord`] impl
27/// can be used during negotiation to pick the best version.
28#[derive(Clone, Copy, Debug, PartialEq, Eq, PartialOrd, Ord, Default, VariantArray)]
29pub enum ProtocolVersion {
30    /// Added in iroh 0.97.0.
31    /// - Deprecated `Health` frame (id 11)
32    /// - Added new `Health` frame (id 13)
33    /// - Changed behavior such that unknown frames are allowed
34    #[default]
35    V2,
36    /// Deprecated version 1 (before iroh 0.97.0)
37    V1,
38}
39
40impl ProtocolVersion {
41    /// Returns a comma-separated string of all supported protocol version identifiers.
42    pub fn all() -> String {
43        Self::VARIANTS
44            .iter()
45            .map(ProtocolVersion::to_str)
46            .collect::<Vec<_>>()
47            .join(", ")
48    }
49
50    /// Returns all supported protocol versions in a comma-seperated string as an HTTP header value.
51    pub fn all_as_header_value() -> HeaderValue {
52        HeaderValue::from_bytes(Self::all().as_bytes()).expect("valid header name")
53    }
54
55    /// Returns the protocol version identifier string.
56    pub fn to_str(&self) -> &'static str {
57        match self {
58            ProtocolVersion::V1 => "iroh-relay-v1",
59            ProtocolVersion::V2 => "iroh-relay-v2",
60        }
61    }
62
63    /// Returns this protocol version as an HTTP header value.
64    pub fn to_header_value(&self) -> HeaderValue {
65        HeaderValue::from_static(self.to_str())
66    }
67}
68
69impl TryFrom<&str> for ProtocolVersion {
70    type Error = UnsupportedRelayProtocolVersion;
71
72    fn try_from(value: &str) -> Result<Self, Self::Error> {
73        match value {
74            "iroh-relay-v1" => Ok(Self::V1),
75            "iroh-relay-v2" => Ok(Self::V2),
76            _ => Err(UnsupportedRelayProtocolVersion),
77        }
78    }
79}
