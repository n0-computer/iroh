21/// The relay protocol version negotiated between client and server.
22///
23/// Sent as the websocket sub-protocol header `Sec-Websocket-Protocol` from
24/// the client. The server picks the best supported version and replies with it.
25///
26/// Variants are ordered by preference (highest first), so the [`Ord`] impl
27/// can be used during negotiation to pick the best version.
28#[derive(
29    Clone,
30    Copy,
31    Debug,
32    PartialEq,
33    Eq,
34    PartialOrd,
35    Ord,
36    Default,
37    strum::VariantArray,
38    strum::EnumString,
39    strum::Display,
40    strum::IntoStaticStr,
41)]
42#[strum(parse_err_ty = UnsupportedRelayProtocolVersion, parse_err_fn = strum_err_fn)]
43// Needs to be ordered with latest version last, so that the `Ord` impl orders by latest version as max.
44pub enum ProtocolVersion {
45    /// Version 1 (before iroh 0.97.0)
46    #[strum(serialize = "iroh-relay-v1")]
47    V1,
48    /// Version 2 (added in iroh 0.97.0)
49    /// - Removed `Health` frame (id 11)
50    /// - Added new `Status` frame (id 13)
51    /// - Changed behavior such that unknown frames are allowed
52    #[default]
53    #[strum(serialize = "iroh-relay-v2")]
54    V2,
55}
56
57impl ProtocolVersion {
58    /// Returns an iterator of all supported protocol version identifiers, in order of preference.
59    pub fn all() -> impl Iterator<Item = &'static str> {
60        Self::VARIANTS
61            .iter()
62            .map(ProtocolVersion::to_str)
63            // We reverse the order so that the latest version comes first:
64            // `Self::VARIANTS` is ordered in definition order, where the latest version comes last
65            // so that the `Ord` derive correctly orderes by "latest is max".
66            .rev()
67    }
68
69    /// Returns a comma-separated string of all supported protocol version identifiers.
70    pub fn all_joined() -> String {
71        Self::all().collect::<Vec<_>>().join(", ")
72    }
73
74    /// Returns all supported protocol versions in a comma-seperated string as an HTTP header value.
75    pub fn all_as_header_value() -> HeaderValue {
76        HeaderValue::from_bytes(Self::all_joined().as_bytes()).expect("valid header name")
77    }
78
79    /// Returns the protocol version identifier string.
80    pub fn to_str(&self) -> &'static str {
81        self.into()
82    }
83
84    /// Tries to parse a [`ProtocolVersion`] from `s`.
85    ///
86    /// Returns `None` if `s` is not a valid protocol version string.
87    pub fn match_from_str(s: &str) -> Option<Self> {
88        Self::try_from(s).ok()
89    }
90
91    /// Returns this protocol version as an HTTP header value.
92    pub fn to_header_value(&self) -> HeaderValue {
93        HeaderValue::from_static(self.to_str())
94    }
95}

97/// Error returned when the relay protocol version is not recognized.
98#[stack_error(derive)]
99#[error("Relay protocol version is not supported")]

102fn strum_err_fn(_item: &str) -> UnsupportedRelayProtocolVersion {
103    UnsupportedRelayProtocolVersion::new()
