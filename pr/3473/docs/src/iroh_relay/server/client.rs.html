49/// The [`Server`] side representation of a [`Client`]'s connection.
50///
51/// [`Server`]: crate::server::Server
52/// [`Client`]: crate::client::Client
53#[derive(Debug)]
54pub(super) struct Client {
55    /// Identity of the connected peer.
56    node_id: NodeId,
57    /// Connection identifier.
58    connection_id: u64,
59    /// Used to close the connection loop.
60    done: CancellationToken,
61    /// Actor handle.
62    handle: AbortOnDropHandle<()>,
63    /// Queue of packets intended for the client.
64    send_queue: mpsc::Sender<Packet>,
65    /// Queue of disco packets intended for the client.
66    disco_send_queue: mpsc::Sender<Packet>,
67    /// Channel to notify the client that a previous sender has disconnected.
68    peer_gone: mpsc::Sender<NodeId>,
69}
70
71impl Client {
72    /// Creates a client from a connection & starts a read and write loop to handle io to and from
73    /// the client
74    /// Call [`Client::shutdown`] to close the read and write loops before dropping the [`Client`]
75    pub(super) fn new(
76        config: Config,
77        connection_id: u64,
78        clients: &Clients,
79        metrics: Arc<Metrics>,
80    ) -> Client {
81        let Config {
82            node_id,
83            stream,
84            write_timeout,
85            channel_capacity,
86        } = config;
87
88        let done = CancellationToken::new();
89        let (send_queue_s, send_queue_r) = mpsc::channel(channel_capacity);
90
91        let (disco_send_queue_s, disco_send_queue_r) = mpsc::channel(channel_capacity);
92        let (peer_gone_s, peer_gone_r) = mpsc::channel(channel_capacity);
93
94        let actor = Actor {
95            stream,
96            timeout: write_timeout,
97            send_queue: send_queue_r,
98            disco_send_queue: disco_send_queue_r,
99            node_gone: peer_gone_r,
100            node_id,
101            connection_id,
102            clients: clients.clone(),
103            client_counter: ClientCounter::default(),
104            ping_tracker: PingTracker::default(),
105            metrics,
106        };
107
108        // start io loop
109        let io_done = done.clone();
110        let handle = tokio::task::spawn(actor.run(io_done).instrument(tracing::info_span!(
111            "client-connection-actor",
112            remote_node = %node_id.fmt_short(),
113            connection_id = connection_id
114        )));
115
116        Client {
117            node_id,
118            connection_id,
119            handle: AbortOnDropHandle::new(handle),
120            done,
121            send_queue: send_queue_s,
122            disco_send_queue: disco_send_queue_s,
123            peer_gone: peer_gone_s,
124        }
125    }
126
127    pub(super) fn connection_id(&self) -> u64 {
128        self.connection_id
129    }
130
131    /// Shutdown the reader and writer loops and closes the connection.
132    ///
133    /// Any shutdown errors will be logged as warnings.
134    pub(super) async fn shutdown(self) {
135        self.start_shutdown();
136        if let Err(e) = self.handle.await {
137            warn!(
138                remote_node = %self.node_id.fmt_short(),
139                "error closing actor loop: {e:#?}",
140            );
141        };
142    }
143
144    /// Starts the process of shutdown.
145    pub(super) fn start_shutdown(&self) {
146        self.done.cancel();
147    }

150        &self,
151        src: NodeId,
152        data: Datagrams,
153    ) -> Result<(), TrySendError<Packet>> {
154        self.send_queue.try_send(Packet { src, data })
155    }
156
157    pub(super) fn try_send_disco_packet(
158        &self,
159        src: NodeId,
160        data: Datagrams,

265/// Manages all the reads and writes to this client. It periodically sends a `KEEP_ALIVE`
266/// message to the client to keep the connection alive.
267///
268/// Call `run` to manage the input and output to and from the connection and the server.
269/// Once it hits its first write error or error receiving off a channel,
270/// it errors on return.
271/// If writes do not complete in the given `timeout`, it will also error.
272///
273/// On the "write" side, the [`Actor`] can send the client:
274///  - a KEEP_ALIVE frame
275///  - a PEER_GONE frame to inform the client that a peer they have previously sent messages to
276///    is gone from the network
277///  - packets from other peers
278///
279/// On the "read" side, it can:
280///     - receive a ping and write a pong back
281///     to speak to the node ID associated with that client.
282#[derive(Debug)]
283struct Actor {
284    /// IO Stream to talk to the client
285    stream: RelayedStream,
286    /// Maximum time we wait to complete a write to the client
287    timeout: Duration,
288    /// Packets queued to send to the client
289    send_queue: mpsc::Receiver<Packet>,
290    /// Important packets queued to send to the client
291    disco_send_queue: mpsc::Receiver<Packet>,
292    /// Notify the client that a previous sender has disconnected
293    node_gone: mpsc::Receiver<NodeId>,
294    /// [`NodeId`] of this client
295    node_id: NodeId,
296    /// Connection identifier.
297    connection_id: u64,
298    /// Reference to the other connected clients.
299    clients: Clients,
300    /// Statistics about the connected clients
301    client_counter: ClientCounter,
302    ping_tracker: PingTracker,
303    metrics: Arc<Metrics>,
304}
305
306impl Actor {
307    async fn run(mut self, done: CancellationToken) {
308        // Note the accept and disconnects metrics must be in a pair.  Technically the
309        // connection is accepted long before this in the HTTP server, but it is clearer to
310        // handle the metric here.
311        self.metrics.accepts.inc();
312        if self.client_counter.update(self.node_id) {
313            self.metrics.unique_client_keys.inc();
314        }
315        match self.run_inner(done).await {
316            Err(e) => {
317                warn!("actor errored {e:#}, exiting");
318            }
319            Ok(()) => {
320                debug!("actor finished, exiting");
321            }
322        }
323
324        self.clients.unregister(self.connection_id, self.node_id);
325        self.metrics.disconnects.inc();
326    }
327
328    async fn run_inner(&mut self, done: CancellationToken) -> Result<(), RunError> {
329        use snafu::ResultExt;
330
331        // Add some jitter to ping pong interactions, to avoid all pings being sent at the same time
332        let next_interval = || {
333            let random_secs = rand::rngs::OsRng.gen_range(1..=5);
334            Duration::from_secs(random_secs) + PING_INTERVAL
335        };
336
337        let mut ping_interval = tokio::time::interval(next_interval());
338        // ticks immediately
339        ping_interval.set_missed_tick_behavior(MissedTickBehavior::Delay);
340        ping_interval.tick().await;
341
342        loop {
343            tokio::select! {
344                biased;
345
346                _ = done.cancelled() => {
347                    trace!("actor loop cancelled, exiting");
348                    // final flush
349                    self.stream.flush().await.map_err(|_| FlushSnafu.build())?;
350                    break;
351                }
352                maybe_frame = self.stream.next() => {
353                    self.handle_frame(maybe_frame).await?;
354                    // reset the ping interval, we just received a message
355                    ping_interval.reset();
356                }
357                // First priority, disco packets
358                packet = self.disco_send_queue.recv() => {
359                    let packet = packet.ok_or(DiscoSendQueuePacketDropSnafu.build())?;
360                    self.send_disco_packet(packet).await.context(DiscoPacketSendSnafu)?;
361                }
362                // Second priority, sending regular packets
363                packet = self.send_queue.recv() => {
364                    let packet = packet.ok_or(SendQueuePacketDropSnafu.build())?;
365                    self.send_packet(packet).await.context(PacketSendSnafu)?;
366                }
367                // Last priority, sending left nodes
368                node_id = self.node_gone.recv() => {
369                    let node_id = node_id.ok_or(NodeGoneDropSnafu.build())?;
370                    trace!("node_id gone: {:?}", node_id);
371                    self.write_frame(RelayToClientMsg::NodeGone(node_id)).await.context(NodeGoneWriteFrameSnafu)?;
372                }
373                _ = self.ping_tracker.timeout() => {
374                    trace!("pong timed out");
375                    break;
376                }
377                _ = ping_interval.tick() => {
378                    trace!("keep alive ping");
379                    // new interval
380                    ping_interval.reset_after(next_interval());
381                    let data = self.ping_tracker.new_ping();
382                    self.write_frame(RelayToClientMsg::Ping(data)).await.context(KeepAliveWriteFrameSnafu)?;
383                }
384            }
385
386            self.stream
387                .flush()
388                .await
389                .map_err(|_| TickFlushSnafu.build())?;
390        }
391        Ok(())
392    }
393
394    /// Writes the given frame to the connection.
395    ///
396    /// Errors if the send does not happen within the `timeout` duration
397    async fn write_frame(&mut self, frame: RelayToClientMsg) -> Result<(), WriteFrameError> {
398        tokio::time::timeout(self.timeout, self.stream.send(frame)).await??;
399        Ok(())
400    }
401
402    /// Writes contents to the client in a `RECV_PACKET` frame.
403    ///
404    /// Errors if the send does not happen within the `timeout` duration
405    /// Does not flush.
406    async fn send_raw(&mut self, packet: Packet) -> Result<(), WriteFrameError> {
407        let remote_node_id = packet.src;
408        let datagrams = packet.data;
409
410        if let Ok(len) = datagrams.contents.len().try_into() {
411            self.metrics.bytes_sent.inc_by(len);
412        }
413        self.write_frame(RelayToClientMsg::Datagrams {
414            remote_node_id,
415            datagrams,
416        })
417        .await
418    }
419
420    async fn send_packet(&mut self, packet: Packet) -> Result<(), WriteFrameError> {
421        trace!("send packet");
422        match self.send_raw(packet).await {
423            Ok(()) => {
424                self.metrics.send_packets_sent.inc();
425                Ok(())
426            }
427            Err(err) => {
428                self.metrics.send_packets_dropped.inc();
429                Err(err)
430            }
431        }
432    }
433
434    async fn send_disco_packet(&mut self, packet: Packet) -> Result<(), WriteFrameError> {
435        trace!("send disco packet");
436        match self.send_raw(packet).await {
437            Ok(()) => {
438                self.metrics.disco_packets_sent.inc();
439                Ok(())
440            }
441            Err(err) => {
442                self.metrics.disco_packets_dropped.inc();
443                Err(err)
444            }
445        }
446    }
447
448    /// Handles frame read results.
449    async fn handle_frame(
450        &mut self,
451        maybe_frame: Option<Result<ClientToRelayMsg, RelayRecvError>>,
452    ) -> Result<(), HandleFrameError> {
453        trace!(?maybe_frame, "handle incoming frame");
454        let frame = match maybe_frame {
455            Some(frame) => frame?,
456            None => return Err(StreamTerminatedSnafu.build()),
457        };
458
459        match frame {
460            ClientToRelayMsg::Datagrams {
461                dst_node_id: dst_key,
462                datagrams,
463            } => {
464                let packet_len = datagrams.contents.len();
465                if let Err(err @ ForwardPacketError { .. }) =
466                    self.handle_frame_send_packet(dst_key, datagrams)
467                {
468                    warn!("failed to handle send packet frame: {err:#}");
469                }
470                self.metrics.bytes_recv.inc_by(packet_len as u64);
471            }
472            ClientToRelayMsg::Ping(data) => {
473                self.metrics.got_ping.inc();
474                // TODO: add rate limiter
475                self.write_frame(RelayToClientMsg::Pong(data)).await?;
476                self.metrics.sent_pong.inc();
477            }
478            ClientToRelayMsg::Pong(data) => {
479                self.ping_tracker.pong_received(data);
480            }
481        }
482        Ok(())
483    }
484
485    fn handle_frame_send_packet(
486        &self,
487        dst: NodeId,
488        data: Datagrams,
489    ) -> Result<(), ForwardPacketError> {
490        if disco::looks_like_disco_wrapper(&data.contents) {
491            self.metrics.disco_packets_recv.inc();
492            self.clients
493                .send_disco_packet(dst, data, self.node_id, &self.metrics)?;
494        } else {
495            self.metrics.send_packets_recv.inc();
496            self.clients
497                .send_packet(dst, data, self.node_id, &self.metrics)?;
498        }
499        Ok(())
500    }
501}

533/// Tracks how many unique nodes have been seen during the last day.
534#[derive(Debug)]
535struct ClientCounter {
536    clients: HashSet<NodeId>,
537    last_clear_date: Date,
538}
539
540impl Default for ClientCounter {
541    fn default() -> Self {
542        Self {
543            clients: HashSet::new(),
544            last_clear_date: OffsetDateTime::now_utc().date(),
545        }
546    }
547}
548
549impl ClientCounter {
550    fn check_and_clear(&mut self) {
551        let today = OffsetDateTime::now_utc().date();
552        if today != self.last_clear_date {
553            self.clients.clear();
554            self.last_clear_date = today;
555        }
556    }
557
558    /// Marks this node as seen, returns whether it is new today or not.
559    fn update(&mut self, client: NodeId) -> bool {
560        self.check_and_clear();
561        self.clients.insert(client)
562    }
563}

576    async fn recv_frame<
577        E: snafu::Error + Sync + Send + 'static,
578        S: Stream<Item = Result<RelayToClientMsg, E>> + Unpin,
579    >(
580        frame_type: FrameType,
581        mut stream: S,
582    ) -> Result<RelayToClientMsg> {
583        match stream.next().await {
584            Some(Ok(frame)) => {
585                if frame_type != frame.typ() {
586                    snafu::whatever!(
587                        "Unexpected frame, got {:?}, but expected {:?}",
588                        frame.typ(),
589                        frame_type
590                    );
591                }
592                Ok(frame)
593            }
594            Some(Err(err)) => Err(err).e(),
595            None => snafu::whatever!("Unexpected EOF, expected frame {frame_type:?}"),
596        }
597    }

705        disco_data.extend_from_slice(target.as_bytes());
706        disco_data.extend_from_slice(data);
707        io_rw
708            .send(ClientToRelayMsg::Datagrams {
709                dst_node_id: target,
710                datagrams: disco_data.clone().into(),
711            })
712            .await
713            .context("send")?;
714
715        done.cancel();
716        handle.await.context("join")?;
717        Ok(())
718    }

721    #[traced_test]
722    async fn test_rate_limit() -> Result {
723        const LIMIT: u32 = 50;
724        const MAX_FRAMES: u32 = 100;
725
726        // Build the rate limited stream.
727        let (io_read, io_write) = tokio::io::duplex((LIMIT * MAX_FRAMES) as _);
728        let mut frame_writer = Conn::test(io_write);
729        // Rate limiter allowing LIMIT bytes/s
730        let mut stream = RelayedStream::test_limited(io_read, LIMIT / 10, LIMIT)?;
731
732        // Prepare a frame to send, assert its size.
733        let data = Datagrams::from(b"hello world!!!!!");
734        let target = SecretKey::generate(rand::thread_rng()).public();
735        let frame = ClientToRelayMsg::Datagrams {
736            dst_node_id: target,
737            datagrams: data.clone(),
738        };
739        let frame_len = frame.to_bytes().len();
740        assert_eq!(frame_len, LIMIT as usize);
741
742        // Send a frame, it should arrive.
743        info!("-- send packet");
744        frame_writer.send(frame.clone()).await.context("send")?;
745        frame_writer.flush().await.context("flush")?;
746        let recv_frame = tokio::time::timeout(Duration::from_millis(500), stream.next())
747            .await
748            .expect("timeout")
749            .expect("option")
750            .expect("ok");
751        assert_eq!(recv_frame, frame);
752
753        // Next frame does not arrive.
754        info!("-- send packet");
755        frame_writer.send(frame.clone()).await.context("send")?;
756        frame_writer.flush().await.context("flush")?;
757        let res = tokio::time::timeout(Duration::from_millis(100), stream.next()).await;
758        assert!(res.is_err(), "expecting a timeout");
759        info!("-- timeout happened");
760
761        // Wait long enough.
762        info!("-- sleep");
763        tokio::time::sleep(Duration::from_secs(1)).await;
764
765        // Frame arrives.
766        let recv_frame = tokio::time::timeout(Duration::from_millis(500), stream.next())
767            .await
768            .expect("timeout")
769            .expect("option")
770            .expect("ok");
771        assert_eq!(recv_frame, frame);
772
773        Ok(())
774    }
