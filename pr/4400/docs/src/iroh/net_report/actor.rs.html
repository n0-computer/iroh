1//! The actor that runs network probes and publishes a [`Report`].
2//!
3//! [`NetReportActor`] runs in the background for as long as the endpoint is
4//! alive. It learns about the network in three ways and writes what it finds
5//! into a [`Report`] that callers watch:
6//!
7//! - QAD (QUIC Address Discovery) opens a QUIC connection to a relay. The
8//!   relay reports the public address it sees us coming from, and the round
9//!   trip measures our latency to that relay.
10//! - An HTTPS probe measures latency to a relay with a plain GET request. It
11//!   finds no address, but it still works on networks that block QUIC.
12//! - The captive portal check looks for a network that intercepts HTTP.
13//!
14//! QAD connections are long-lived, and that shapes the rest. For each
15//! address family the actor races a probe to several relays, keeps the
16//! connection that answers first, and closes the others. It then holds that
17//! connection open: the relay reports a new address whenever ours changes,
18//! so an address change reaches callers without a new probe. These
19//! observations arrive on the same [`ProbeEvent`] channel as probe results,
20//! so everything the actor learns lands in one place.
21//!
22//! A probe cycle is a round of probing triggered by a request. Because the
23//! open QAD connections already keep addresses current, a cycle is mostly
24//! about the rest: measuring latency to every relay over HTTPS, picking the
25//! preferred relay, and checking for a captive portal. [`ProbeScope`] sets
26//! how much of that a cycle does, and [`ProbeCycle`] holds the one in
27//! flight. A cycle publishes a first report within [`FIRST_REPORT_TIMEOUT`]
28//! even while probes are still running, and gives up on any stragglers after
29//! [`ABORT_TIMEOUT`].

65/// A probe request waiting for the actor to pick it up.
66///
67/// Several [`Client::run_probes`](super::Client::run_probes) calls can
68/// arrive before the actor handles them. They collapse into this one
69/// request, which takes the most urgent [`ProbeScope`] of the batch and
70/// waits here until the actor takes it.
71pub(super) struct ProbeRequest {
72    pub if_state: IfState,
73    pub scope: ProbeScope,
74}

76/// A one-slot mailbox that carries a [`ProbeRequest`] from a
77/// [`Client`](super::Client) to the [`NetReportActor`].
78///
79/// It holds at most one request. A second request that arrives before the
80/// actor has taken the first does not queue behind it; it merges into it,
81/// raising the [`ProbeScope`] to the more urgent of the two and keeping the
82/// newer interface state. This is why it is a hand-written slot and not a
83/// channel: a channel would queue the requests, and probing twice in a row
84/// wastes work when one probe with the combined scope would do.
85pub(super) struct RequestSlot {
86    slot: std::sync::Mutex<Option<ProbeRequest>>,
87    notify: tokio::sync::Notify,
88}

90impl std::fmt::Debug for RequestSlot {
91    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
92        f.debug_struct("RequestSlot").finish_non_exhaustive()
93    }
94}

96impl RequestSlot {
97    pub(super) fn new() -> Self {
98        Self {
99            slot: std::sync::Mutex::new(None),
100            notify: tokio::sync::Notify::new(),
101        }
102    }
103
104    /// Merges a probe request into the slot.
105    ///
106    /// If a request is already pending, the scope is escalated to the more
107    /// urgent of the two and `if_state` is overwritten with the latest
108    /// value. If no request is pending, a new one is created.
109    pub(super) fn request(&self, if_state: IfState, scope: ProbeScope) {
110        let mut guard = self.slot.lock().expect("not poisoned");
111        match guard.as_mut() {
112            Some(pending) => {
113                pending.scope = pending.scope.max(scope);
114                pending.if_state = if_state;
115            }
116            None => {
117                *guard = Some(ProbeRequest { if_state, scope });
118            }
119        }
120        drop(guard);
121        self.notify.notify_one();
122    }
123
124    /// Takes the pending request, leaving the slot empty.
125    fn take(&self) -> Option<ProbeRequest> {
126        self.slot.lock().expect("not poisoned").take()
127    }
128}

130/// A message from a probe task to the [`NetReportActor`].
131///
132/// Each probe a cycle starts sends exactly one final result: `QadResult`,
133/// `Https`, or `CaptivePortal`. The actor counts these to know when the
134/// cycle is done. An open QAD connection is different: it sends a
135/// `QadObservation` every time our address changes, for as long as it stays
136/// open, and those belong to no cycle.
137pub(super) enum ProbeEvent {
138    /// A QAD probe finished. On success it carries the connection to keep
139    /// open.
140    #[cfg(not(wasm_browser))]
141    QadResult(AddrFamily, Result<(QadProbeReport, QadConn), QadProbeError>),
142    /// An open QAD connection reported an address, possibly a new one.
143    #[cfg(not(wasm_browser))]
144    QadObservation(AddrFamily, QadProbeReport),
145    /// An HTTPS latency probe finished.
146    Https(Result<HttpsProbeReport, HttpsProbeError>),
147    /// The captive portal check finished (`None` if cancelled or timed out).
148    #[cfg(not(wasm_browser))]
149    CaptivePortal(Option<bool>),
150}

152/// How much of the probe set a cycle runs.
153///
154/// The scope plays two roles. On the request that starts a cycle it says how
155/// urgent the trigger is: a `Full` request comes from a real network change,
156/// so it aborts any cycle in progress and starts over, while a `Refresh`
157/// request waits for the current cycle to finish. On the cycle itself it
158/// says how much to probe.
159///
160/// The open QAD connections keep our address current on their own (see the
161/// [module docs](self)), so neither scope has to probe QAD just to stay up
162/// to date. The difference is what else a cycle does:
163///
164/// - `Full` throws away the open QAD connections and starts from nothing. It
165///   opens a fresh QAD connection for every available family, measures
166///   latency to every relay over HTTPS, and runs the captive portal check.
167///   This is what a real network change calls for.
168/// - `Refresh` keeps the open QAD connections and only does the work they do
169///   not cover. It re-measures relay latency over HTTPS and re-picks the
170///   preferred relay, and it opens a QAD connection only for a family that
171///   has none, because its connection dropped or its interface just came up.
172///   It skips the captive portal check.
173///
174/// A `Refresh` request can still turn into a `Full` cycle: the actor forces
175/// one when the full-report interval has elapsed, or when the last report
176/// found a captive portal and no working UDP.
177///
178/// The variants order `Refresh < Full` so that merging two pending requests
179/// can just take the more urgent one with `max`.
180#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
181pub(crate) enum ProbeScope {
182    /// Keep the open QAD connections. Re-measure relay latency and reconnect
183    /// any family that has no open connection.
184    Refresh,
185    /// Close the open QAD connections and run every probe from scratch.
186    Full,
187}
188
189impl ProbeScope {
190    /// Maps a netmon "is major change" flag to a [`ProbeScope`].
191    pub(crate) fn from_major(is_major: bool) -> Self {
192        if is_major { Self::Full } else { Self::Refresh }
193    }
194}

196/// The set of QAD address families a cycle probes and has heard back from.
197#[derive(Debug, Clone, Copy, Default)]
198struct AddrFamilies {
199    v4: bool,
200    v6: bool,
201}
202
203impl AddrFamilies {
204    /// Returns `true` if every family set in `needed` is also set here.
205    fn covers(self, needed: AddrFamilies) -> bool {
206        (self.v4 || !needed.v4) && (self.v6 || !needed.v6)
207    }
208
209    #[cfg(not(wasm_browser))]
210    fn set(&mut self, family: AddrFamily) {
211        match family {
212            AddrFamily::V4 => self.v4 = true,
213            AddrFamily::V6 => self.v6 = true,
214        }
215    }

218/// Tracks recent reports for preferred-relay selection and full-cycle
219/// cadence.
220#[derive(Debug)]
221struct ReportHistory {
222    /// When true, the next cycle is forced to be `Full` rather than `Refresh`.
223    next_full: bool,
224    /// Reports from the last five minutes, keyed by completion time.
225    prev: std::collections::BTreeMap<Instant, Report>,
226    /// The most recent completed report.
227    last: Option<Report>,
228    /// Time of the last `Full` report.
229    last_full: Instant,
230}
231
232impl Default for ReportHistory {
233    fn default() -> Self {
234        Self {
235            next_full: true,
236            prev: Default::default(),
237            last: Default::default(),
238            last_full: Instant::now(),
239        }
240    }
241}
242
243impl ReportHistory {
244    /// Records `r` and sets `r.preferred_relay` to the best candidate
245    /// across the last five minutes of reports.
246    ///
247    /// Applies hysteresis: the preferred relay only changes when the new
248    /// candidate is at least 33% faster than the current one.
249    fn record(&mut self, r: &mut Report) {
250        let mut prev_relay = None;
251        if let Some(ref last) = self.last {
252            prev_relay.clone_from(&last.preferred_relay);
253
254            if r.mapping_varies_by_dest_ipv4.is_none() {
255                r.mapping_varies_by_dest_ipv4 = last.mapping_varies_by_dest_ipv4;
256            }
257            if r.mapping_varies_by_dest_ipv6.is_none() {
258                r.mapping_varies_by_dest_ipv6 = last.mapping_varies_by_dest_ipv6;
259            }
260        }
261
262        let now = Instant::now();
263        const MAX_AGE: Duration = Duration::from_secs(5 * 60);
264
265        let mut best_recent = RelayLatencies::default();
266
267        let mut to_remove = Vec::new();
268        for (t, pr) in self.prev.iter() {
269            if now.duration_since(*t) > MAX_AGE {
270                to_remove.push(*t);
271                continue;
272            }
273            best_recent.merge(&pr.relay_latency);
274        }
275        best_recent.merge(&r.relay_latency);
276
277        for t in to_remove {
278            self.prev.remove(&t);
279        }
280
281        let mut best_any = Duration::default();
282        let mut old_relay_cur_latency = Duration::default();
283        for (_, url, duration) in r.relay_latency.iter() {
284            if Some(url) == prev_relay.as_ref() {
285                old_relay_cur_latency = duration;
286            }
287            if let Some(best) = best_recent.get(url)
288                && (r.preferred_relay.is_none() || best < best_any)
289            {
290                best_any = best;
291                r.preferred_relay.replace(url.clone());
292            }
293        }
294
295        // Hysteresis: don't switch if the new relay isn't much better.
296        if prev_relay.is_some()
297            && r.preferred_relay != prev_relay
298            && !old_relay_cur_latency.is_zero()
299            && best_any > old_relay_cur_latency / 3 * 2
300        {
301            r.preferred_relay = prev_relay;
302        }
303
304        self.prev.insert(now, r.clone());
305        self.last = Some(r.clone());
306    }
307}

309/// State of the currently running probe cycle.
310struct ProbeCycle {
311    started: Instant,
312    /// Probe tasks that have not yet reported a terminal result.
313    pending: usize,
314    /// QAD families this cycle is probing. The first report is held until
315    /// each of these has produced a result (or [`FIRST_REPORT_TIMEOUT`] fires).
316    expected: AddrFamilies,
317    /// QAD families that have produced a result so far.
318    seen: AddrFamilies,

416    /// Runs the actor until the shutdown token is cancelled.
417    ///
418    /// On shutdown, dropping `self` drops the [`JoinSet`] and the QAD
419    /// connections, aborting every task the actor owns.
420    pub(super) async fn run(mut self) {
421        loop {
422            let report_deadline = match self.cycle.as_ref().and_then(|c| c.report_deadline) {
423                Some(t) => MaybeFuture::Some(time::sleep_until(t)),
424                None => MaybeFuture::None,
425            };
426            let abort_deadline = match self.cycle.as_ref().and_then(|c| c.abort_deadline) {
427                Some(t) => MaybeFuture::Some(time::sleep_until(t)),
428                None => MaybeFuture::None,
429            };
430            n0_future::pin!(report_deadline);
431            n0_future::pin!(abort_deadline);
432
433            tokio::select! {
434                biased;
435
436                _ = self.shutdown.cancelled() => break,
437
438                _ = self.probe_requests.notify.notified() => {
439                    if let Some(req) = self.probe_requests.take() {
440                        self.handle_request(req);
441                    }
442                }
443
444                Some(ev) = self.events.recv() => self.handle_event(ev),
445
446                Some(joined) = self.tasks.join_next() => {
447                    if let Err(err) = joined {
448                        // A task that panicked never sent its terminal
449                        // event, so account for it here and let the cycle
450                        // finalize. (Aborted tasks live in a dropped
451                        // JoinSet and are never yielded here.)
452                        if err.is_panic() {
453                            error!("probe task panicked: {err:#}");
454                        }
455                        self.probe_finished();
456                        self.advance();
457                    }
458                }
459
460                _ = &mut report_deadline => self.on_report_deadline(),
461                _ = &mut abort_deadline => self.on_abort_deadline(),
462            }
463        }
464    }

466    /// Handles a probe request: defer it, restart the cycle, or start one.
467    fn handle_request(&mut self, req: ProbeRequest) {
468        if self.relay_map.is_empty() {
469            debug!("skipping net_report, empty RelayMap");
470            return;
471        }
472
473        if let Some(cycle) = &mut self.cycle {
474            match req.scope {
475                ProbeScope::Refresh => {
476                    // Defer: run right after the current cycle finishes so
477                    // the trigger is not lost (the old DirectAddrUpdateState
478                    // remembered this via `want_update`). Only `Refresh`
479                    // requests are deferred, so the remembered scope stays
480                    // `Refresh`; just take the latest interface state.
481                    match &mut cycle.rerun {
482                        Some(pending) => pending.if_state = req.if_state,
483                        None => cycle.rerun = Some(req),
484                    }
485                    debug!("deferring probe request until current cycle finishes");
486                    return;
487                }
488                // Full: abort the current cycle and start fresh.
489                ProbeScope::Full => self.abort_cycle(),
490            }
491        }
492
493        self.start_cycle(req);
494    }
495
496    /// Aborts the current cycle's probe tasks and swaps in a fresh event
497    /// channel, so any results those tasks already queued are dropped instead
498    /// of leaking into the next cycle. Only a `Full` restart does this, and a
499    /// `Full` restart also closes the open QAD connections, so no observer
500    /// task is left running without an owner.
501    fn abort_cycle(&mut self) {
502        self.tasks = JoinSet::new();
503        let (tx, rx) = mpsc::channel(EVENT_CHANNEL_CAP);
504        self.events_tx = tx;
505        self.events = rx;
506        self.cycle = None;
507    }
508
509    /// Starts a new probe cycle.
510    fn start_cycle(&mut self, req: ProbeRequest) {
511        let ProbeRequest {
512            if_state,
513            scope: request_scope,
514        } = req;
515        let now = Instant::now();
516        // A `Refresh` request is promoted to a `Full` cycle when a full
517        // re-probe is due.
518        let major = request_scope == ProbeScope::Full
519            || self.reports.next_full
520            || now.duration_since(self.reports.last_full) > FULL_REPORT_INTERVAL
521            || self
522                .reports
523                .last
524                .as_ref()
525                .is_some_and(|r| !r.has_udp() && r.captive_portal == Some(true));
526        let scope = ProbeScope::from_major(major);
527
528        debug!(?request_scope, ?scope, "starting probe cycle");
529
530        if scope == ProbeScope::Full {
531            #[cfg(not(wasm_browser))]
532            self.qad_conns.clear();
533            self.reports.last = None;
534            self.reports.next_full = false;
535            self.reports.last_full = now;
536            self.metrics.reports_full.inc();
537        }
538        self.metrics.reports.inc();
539
540        // Start the report from scratch. spawn_qad_probes copies the last
541        // address from each still-open QAD connection back in.
542        self.report = Report::default();
543        self.cancel_https = CancellationToken::new();
544
545        #[cfg(not(wasm_browser))]
546        let (qad_pending, expected) = self.spawn_qad_probes(&if_state);
547        #[cfg(wasm_browser)]
548        let (qad_pending, expected) = (0usize, AddrFamilies::default());
549        let mut pending = qad_pending + self.spawn_https_probes();
550        #[cfg(not(wasm_browser))]
551        if scope == ProbeScope::Full && self.captive_portal_check {
552            self.spawn_captive_portal();
553            pending += 1;
554        }
555
556        self.cycle = Some(ProbeCycle {
557            started: now,
558            pending,
559            expected,
560            seen: AddrFamilies::default(),
561            published: false,
562            rerun: None,
563            report_deadline: Some(now + FIRST_REPORT_TIMEOUT),
564            abort_deadline: Some(now + ABORT_TIMEOUT),
565        });
566
567        // A cycle can start with no probes at all: every family already has
568        // an open connection and HTTPS is off. It is already complete, so
569        // finalize now to still update history and the preferred relay.
570        self.advance();
571    }
572
573    /// Applies one event to the report and drives the cycle forward.
574    fn handle_event(&mut self, ev: ProbeEvent) {
575        match ev {
576            #[cfg(not(wasm_browser))]
577            ProbeEvent::QadResult(family, result) => {
578                if let Some(c) = &mut self.cycle {
579                    c.seen.set(family);
580                }
581                self.probe_finished();
582                match result {
583                    Ok((report, conn)) => {
584                        debug!(?family, ?report, "QAD probe completed");
585                        // Accumulate: the first result sets the global
586                        // address; a second result from a different relay
587                        // decides mapping-varies-by-destination.
588                        self.report.apply_qad_result(family, &report);
589                        if self.qad_conns.slot(family).is_none() {
590                            // First result for this family: keep this
591                            // connection open, but let the other probes run
592                            // so a second result can decide mapping-varies.
593                            *self.qad_conns.slot_mut(family) = Some((report.relay_url, conn));
594                        } else {
595                            // Second result: mapping-varies is decided, so
596                            // stop the family's remaining probes and drop
597                            // this connection.
598                            conn.conn
599                                .close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
600                            self.qad_conns.cancel(family).cancel();
601                        }
602                        // UDP works, so skip captive portal detection.
603                        self.cancel_captive_portal.cancel();
604                    }
605                    Err(e) => debug!(?family, "QAD probe failed: {e:#}"),
606                }
607                self.publish();
608                self.advance();
609            }
610            #[cfg(not(wasm_browser))]
611            ProbeEvent::QadObservation(family, obs) => {
612                // Take observations only from the connection we kept for this
613                // family. A probe we dropped may still send one last
614                // observation before its task ends; ignore it.
615                let is_current = self
616                    .qad_conns
617                    .slot(family)
618                    .is_some_and(|(url, _)| *url == obs.relay_url);
619                if is_current {
620                    trace!(?family, ?obs, "QAD address observation");
621                    if let Some((_, conn)) = self.qad_conns.slot_mut(family) {
622                        conn.last = obs.clone();
623                    }
624                    self.report.apply_qad_observation(family, &obs);
625                    self.publish();
626                }
627            }
628            ProbeEvent::Https(result) => {
629                self.probe_finished();
630                match result {
631                    Ok(report) => {
632                        debug!(?report, "HTTPS probe completed");
633                        self.report.apply_https_result(&report);
634                        if self.have_all_relay_latencies() {
635                            self.cancel_https.cancel();
636                        }
637                    }
638                    Err(e) => debug!("HTTPS probe failed: {e:#}"),
639                }
640                self.publish();
641                self.advance();
642            }
643            #[cfg(not(wasm_browser))]
644            ProbeEvent::CaptivePortal(found) => {
645                self.probe_finished();
646                debug!(?found, "captive portal check completed");
647                self.report.captive_portal = found;
648                self.publish();
649                self.advance();
650            }
651        }
652    }

654    /// Records that one cycle probe has finished (however it exited).
655    fn probe_finished(&mut self) {
656        if let Some(c) = &mut self.cycle {
657            c.pending = c.pending.saturating_sub(1);
658        }
659    }

661    /// Publishes the current report, unless a guard holds it back.
662    ///
663    /// Two guards apply. An empty report is never published, since it would
664    /// overwrite addresses a caller has already seen. And the first report of
665    /// a cycle waits until every family the cycle probed has a result, so a
666    /// caller does not briefly see an IPv4-only report just before the IPv6
667    /// address arrives.
668    fn publish(&mut self) {
669        if !has_data(&self.report) {
670            return;
671        }
672        if let Some(c) = &self.cycle
673            && !c.published
674            && !c.seen.covers(c.expected)
675        {
676            return;
677        }
678        if let Some(c) = &mut self.cycle {
679            c.published = true;
680        }
681        self.report_out.set(Some(self.report.clone())).ok();
682    }
683
684    /// Finalizes the cycle if all its probes have finished.
685    fn advance(&mut self) {
686        if self.cycle.as_ref().is_some_and(|c| c.pending == 0) {
687            self.finish_cycle();
688        }
689    }
690
691    /// Commits the cycle to history, selects the preferred relay, and emits
692    /// the final report. Then applies any deferred rerun request.
693    fn finish_cycle(&mut self) {
694        let Some(cycle) = self.cycle.take() else {
695            return;
696        };
697        self.reports.record(&mut self.report);
698        // Keep the last good report rather than overwriting it with an empty
699        // one when a cycle discovered nothing.
700        if has_data(&self.report) {
701            self.report_out.set(Some(self.report.clone())).ok();
702        }
703        debug!(
704            report = ?self.report,
705            duration = ?cycle.started.elapsed(),
706            "net_report cycle complete",
707        );
708        if let Some(rerun) = cycle.rerun {
709            self.handle_request(rerun);
710        }
711    }

713    /// The first-report deadline fired. Publish what we have now, even if
714    /// some probed families have not answered yet.
715    fn on_report_deadline(&mut self) {
716        debug!("report deadline fired");
717        if let Some(c) = &mut self.cycle {
718            c.report_deadline = None;
719            c.published = true;
720        }
721        self.publish();
722    }
723
724    /// Abort deadline: stop remaining probes and finalize with what we have.
725    fn on_abort_deadline(&mut self) {
726        debug!("abort deadline fired, finalizing cycle");
727        // Aborting drops the tasks, so they send no further events.
728        self.tasks = JoinSet::new();
729        if let Some(c) = &mut self.cycle {
730            c.abort_deadline = None;
731            c.pending = 0;
732        }
733        self.finish_cycle();
734    }

736    /// Returns `true` when every relay has at least one latency sample.
737    /// Used to cancel remaining HTTPS probes once coverage is complete.
738    fn have_all_relay_latencies(&self) -> bool {
739        let num_relays = self.relay_map.len();
740        if num_relays == 0 {
741            return true;
742        }
743        let mut seen = BTreeSet::new();
744        for (_, url, _) in self.report.relay_latency.iter() {
745            seen.insert(url);
746        }
747        seen.len() >= num_relays
748    }

750    /// Starts a QAD probe for each family that needs one, and returns how
751    /// many it started together with the set of families being probed.
752    ///
753    /// A family needs a probe only when it has no open connection. Before
754    /// deciding, any connection that has since closed is dropped, and the
755    /// last address of each surviving connection is copied into the report.
756    #[cfg(not(wasm_browser))]
757    fn spawn_qad_probes(&mut self, if_state: &IfState) -> (usize, AddrFamilies) {
758        let Some(quic_client) = self.context.quic_client.clone() else {
759            return (0, AddrFamilies::default());
760        };
761
762        // Drop any connection that has closed, then copy the last address of
763        // each surviving one into the report.
764        for family in [AddrFamily::V4, AddrFamily::V6] {
765            if let Some((url, conn)) = self.qad_conns.slot(family)
766                && let Some(reason) = conn.conn.close_reason()
767            {
768                trace!(?family, ?url, "QAD conn closed: {reason}");
769                self.qad_conns.slot_mut(family).take();
770            }
771            if let Some(r) = self.qad_conns.current(family) {
772                self.report.apply_qad_observation(family, &r);
773            }
774        }
775
776        self.qad_conns.reset_cancels();
777
778        let families = [
779            (
780                AddrFamily::V4,
781                self.qad_conns.v4.is_none() && if_state.have_v4,
782            ),
783            (
784                AddrFamily::V6,
785                self.qad_conns.v6.is_none() && if_state.have_v6,
786            ),
787        ];
788
789        let mut expected = AddrFamilies::default();
790        for (family, needed) in families {
791            if needed {
792                expected.set(family);
793            }
794        }
795
796        const MAX_RELAYS: usize = 5;
797        let mut count = 0;
798        for relay in self
799            .relay_map
800            .relays::<Vec<_>>()
801            .into_iter()
802            .take(MAX_RELAYS)
803        {
804            for (family, needed) in families {
805                if needed {
806                    self.spawn_qad_probe(family, relay.clone(), quic_client.clone());
807                    count += 1;
808                }
809            }
810        }
811        (count, expected)
812    }

814    #[cfg(not(wasm_browser))]
815    fn spawn_qad_probe(
816        &mut self,
817        family: AddrFamily,
818        relay: Arc<iroh_relay::RelayConfig>,
819        quic_client: iroh_relay::quic::QuicClient,
820    ) {
821        use tracing::{Instrument, info_span};
822
823        let dns_resolver = self.context.dns_resolver.clone();
824        let relay_url = relay.url.clone();
825        let shutdown = self.shutdown.child_token();
826        let cancel = self.qad_conns.cancel(family).child_token();
827        let events = self.events_tx.clone();
828        let span = info_span!("QAD", ?family, %relay_url);
829        self.tasks.spawn(
830            async move {
831                let outcome = cancel
832                    .run_until_cancelled(time::timeout(
833                        QAD_PROBE_TIMEOUT,
834                        super::qad::run_probe(
835                            family,
836                            relay,
837                            quic_client,
838                            dns_resolver,
839                            shutdown,
840                            events.clone(),
841                        ),
842                    ))
843                    .await;
844                let result = match outcome {
845                    Some(Ok(Ok(x))) => Ok(x),
846                    Some(Ok(Err(e))) => Err(e),
847                    Some(Err(_)) => Err(n0_error::e!(QadProbeError::Timeout)),
848                    None => Err(n0_error::e!(QadProbeError::Cancelled)),
849                };
850                events
851                    .send(ProbeEvent::QadResult(family, result))
852                    .await
853                    .ok();
854            }
855            .instrument(span),
856        );
857    }

859    /// Spawns HTTPS latency probes according to the current [`ProbePlan`].
860    /// Returns the number spawned.
861    fn spawn_https_probes(&mut self) -> usize {
862        let plan = match self.reports.last {
863            Some(ref report) => {
864                ProbePlan::with_last_report(&self.relay_map, report, &self.protocols)
865            }
866            None => ProbePlan::initial(&self.relay_map, &self.protocols),
867        };
868        trace!(%plan, "HTTPS probe plan");
869
870        let mut count = 0;
871        for probe_set in plan.iter() {
872            for (delay, relay) in probe_set.params() {
873                self.spawn_https_probe(*delay, Arc::clone(relay));
874                count += 1;
875            }
876        }
877        count
878    }

880    fn spawn_https_probe(&mut self, delay: Duration, relay: Arc<iroh_relay::RelayConfig>) {
881        let cancel = self.cancel_https.child_token();
882        let events = self.events_tx.clone();
883        #[cfg(not(wasm_browser))]
884        let context = self.context.clone();
885        #[cfg(not(wasm_browser))]
886        let tls_config = self.tls_config.clone();
887        self.tasks.spawn(async move {
888            let outcome = cancel
889                .run_until_cancelled(time::timeout(HTTPS_PROBE_TIMEOUT, async move {
890                    if !delay.is_zero() {
891                        time::sleep(delay).await;
892                    }
893                    super::https::run_probe(
894                        #[cfg(not(wasm_browser))]
895                        &context.dns_resolver,
896                        relay.url.clone(),
897                        #[cfg(not(wasm_browser))]
898                        tls_config,
899                    )
900                    .await
901                }))
902                .await;
903            let result = match outcome {
904                Some(Ok(Ok(r))) => Ok(r),
905                Some(Ok(Err(e))) => Err(n0_error::e!(HttpsProbeError::ProbeFailure, e)),
906                Some(Err(_)) => Err(n0_error::e!(HttpsProbeError::Timeout)),
907                None => Err(n0_error::e!(HttpsProbeError::Cancelled)),
908            };
909            events.send(ProbeEvent::Https(result)).await.ok();
910        });
911    }

913    /// Spawns a captive portal detection check, delayed by
914    /// [`CAPTIVE_PORTAL_DELAY`] to give QAD probes time to succeed first,
915    /// and cancelled if QAD confirms UDP connectivity.
916    #[cfg(not(wasm_browser))]
917    fn spawn_captive_portal(&mut self) {
918        self.cancel_captive_portal = CancellationToken::new();
919        let cancel = self.cancel_captive_portal.clone();
920        let dns = self.context.dns_resolver.clone();
921        let relay_map = self.relay_map.clone();
922        let tls = self.tls_config.clone();
923        let preferred = self
924            .reports
925            .last
926            .as_ref()
927            .and_then(|r| r.preferred_relay.clone());
928        let events = self.events_tx.clone();
929
930        self.tasks.spawn(async move {
931            trace!("captive portal check scheduled");
932            let outcome = cancel
933                .run_until_cancelled(async move {
934                    time::sleep(CAPTIVE_PORTAL_DELAY).await;
935                    time::timeout(
936                        CAPTIVE_PORTAL_TIMEOUT,
937                        super::captive_portal::check(&dns, &relay_map, preferred, tls),
938                    )
939                    .await
940                })
941                .await;
942            let found = match outcome {
943                None => None,
944                Some(Ok(Ok(found))) => Some(found),
945                Some(Ok(Err(e))) => {
946                    debug!("captive portal check failed: {e:#}");
947                    None
948                }
949                Some(Err(_)) => {
950                    debug!("captive portal check timed out");
951                    None
952                }
953            };
954            events.send(ProbeEvent::CaptivePortal(found)).await.ok();
955        });
956    }
