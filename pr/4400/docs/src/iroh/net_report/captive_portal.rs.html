41/// Checks whether the host is behind a captive portal.
42///
43/// Returns `true` if we appear to be behind one. The check requests a
44/// relay's `/generate_204` endpoint: a portal intercepting HTTP answers with
45/// something other than the expected empty 204.
46#[cfg(not(wasm_browser))]
47pub(super) async fn check(
48    dns_resolver: DnsResolver,
49    dm: RelayMap,
50    preferred_relay: Option<RelayUrl>,
51    tls_config: rustls::ClientConfig,
52) -> Result<bool, CaptivePortalError> {
53    trace!("captive portal check starting");
54    // If we have a preferred relay and we can use it for non-QAD requests, try that;
55    // otherwise, pick a random one suitable for non-STUN requests.
56
57    use crate::util::reqwest_client_builder;
58
59    let preferred_relay = preferred_relay.and_then(|url| dm.get(&url).map(|_| url));
60
61    let url = match preferred_relay {
62        Some(url) => url,
63        None => {
64            let urls: Vec<_> = dm.urls();
65            if urls.is_empty() {
66                trace!("No suitable relay for captive portal check");
67                return Ok(false);
68            }
69
70            let i = (0..urls.len()).choose(&mut rand::rng()).unwrap_or_default();
71            urls[i].clone()
72        }
73    };
74
75    let mut builder = reqwest_client_builder(tls_config, dns_resolver.clone())
76        .redirect(reqwest::redirect::Policy::none());
77
78    if let Some(Host::Domain(domain)) = url.host() {
79        // Use our own resolver rather than getaddrinfo
80        //
81        // Be careful, a non-zero port will override the port in the URI.
82        //
83        // Ideally we would try to resolve **both** IPv4 and IPv6 rather than purely race
84        // them.  But our resolver doesn't support that yet.
85        let addrs: Vec<_> = dns_resolver
86            .lookup_ipv4_ipv6_staggered(domain, DNS_TIMEOUT, DNS_STAGGERING_MS)
87            .await?
88            .map(|ipaddr| SocketAddr::new(ipaddr, 0))
89            .collect();
90        builder = builder.resolve_to_addrs(domain, &addrs);
91    }
92    let client = builder
93        .build()
94        .map_err(|err| e!(CaptivePortalError::CreateReqwestClient, err))?;
95
96    // Note: the set of valid characters in a challenge and the total
97    // length is limited; see is_challenge_char in bin/iroh-relay for more
98    // details.
99
100    let host_name = url.host_str().unwrap_or_default();
101    let challenge = format!("ts_{host_name}");
102    let portal_url = format!("http://{host_name}/generate_204");
103    let res = client
104        .request(reqwest::Method::GET, portal_url)
105        .header("X-Iroh-Challenge", &challenge)
106        .send()
107        .await
108        .map_err(|err| e!(CaptivePortalError::HttpRequest, err))?;
109
110    let expected_response = format!("response {challenge}");
111    let is_valid_response = res
112        .headers()
113        .get("X-Iroh-Response")
114        .map(|s| s.to_str().unwrap_or_default())
115        == Some(&expected_response);
116
117    trace!(
118        "check_captive_portal url={} status_code={} valid_response={}",
119        res.url(),
120        res.status(),
121        is_valid_response,
122    );
123    let has_captive = res.status() != 204 || !is_valid_response;
124
125    Ok(has_captive)
126}
