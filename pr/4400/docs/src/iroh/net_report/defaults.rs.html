1//! Default values used in net_report.

3/// Contains all timeouts that we use in `iroh-net-report`.
4pub(in crate::net_report) mod timeouts {
5    use n0_future::time::Duration;
6
7    /// Maximum number of seconds before a net report is guaranteed to emit.
8    ///
9    /// Exposed as a public constant for use in documentation and caller
10    /// timeout calculations.
11    pub const FIRST_REPORT_TIMEOUT_SECS: u64 = 3;
12
13    /// Deadline after which the actor emits a report even if probes still run.
14    ///
15    /// Guarantees consumers see results within a bounded time. HTTPS and
16    /// captive portal probes continue past this point.
17    pub(in crate::net_report) const FIRST_REPORT_TIMEOUT: Duration =
18        Duration::from_secs(FIRST_REPORT_TIMEOUT_SECS);
19
20    /// Deadline after which remaining HTTPS probes are cancelled.
21    ///
22    /// QAD probes are not affected and keep running until
23    /// [`QAD_PROBE_TIMEOUT`]. This bounds total network activity per
24    /// cycle while giving degraded-link QAD probes time to finish.
25    pub(in crate::net_report) const ABORT_TIMEOUT: Duration = Duration::from_secs(30);
26
27    /// Per-probe timeout for individual HTTPS latency measurements.
28    ///
29    /// Matches [`QAD_PROBE_TIMEOUT`]: a probe on a degraded link should have
30    /// time to complete after the initial report deadline rather than being
31    /// cut short at [`FIRST_REPORT_TIMEOUT`].
32    pub(in crate::net_report) const HTTPS_PROBE_TIMEOUT: Duration = Duration::from_secs(15);
33
34    /// Per-probe timeout for individual QAD (QUIC Address Discovery) probes.
35    ///
36    /// QAD probes on degraded links should have time to complete after the
37    /// initial report deadline.
38    pub(in crate::net_report) const QAD_PROBE_TIMEOUT: Duration = Duration::from_secs(15);
39
40    /// Delay before starting the captive portal check.
41    ///
42    /// Gives fast QAD probes time to succeed first: if UDP works, the
43    /// captive portal check is cancelled before it starts. Short enough
44    /// that the check still runs promptly when UDP is blocked.
45    pub(in crate::net_report) const CAPTIVE_PORTAL_DELAY: Duration = Duration::from_millis(200);
46
47    /// Timeout for the captive portal HTTP request itself.
48    ///
49    /// Must be shorter than `ABORT_TIMEOUT - CAPTIVE_PORTAL_DELAY` so the
50    /// check finishes before the cycle's abort deadline.
51    pub(in crate::net_report) const CAPTIVE_PORTAL_TIMEOUT: Duration = Duration::from_secs(2);
52
53    /// Timeout for DNS resolution used by probe helpers.
54    pub(in crate::net_report) const DNS_TIMEOUT: Duration = Duration::from_secs(3);
55
56    /// Cadence after which the next cycle is forced to be `Full`.
57    ///
58    /// Not a timeout in the "cancellation" sense; it just promotes a
59    /// `Refresh` to a `Full` once this interval has elapsed.
60    pub(in crate::net_report) const FULL_REPORT_INTERVAL: Duration = Duration::from_secs(5 * 60);
61}
