72/// Executes an HTTPS probe.
73///
74/// Sends a GET request to the relay's probe endpoint and measures the
75/// round-trip time. DNS resolution goes through our own resolver (on
76/// native targets) so it follows the same path as relay client connections.
77#[allow(clippy::unused_async)]
78pub(super) async fn run_probe(
79    #[cfg(not(wasm_browser))] dns_resolver: &DnsResolver,
80    relay: RelayUrl,
81    #[cfg(not(wasm_browser))] tls_config: rustls::ClientConfig,
82) -> Result<HttpsProbeReport, MeasureHttpsLatencyError> {
83    trace!("HTTPS probe start");
84    let url = relay.join(RELAY_PROBE_PATH)?;
85
86    // This should also use same connection establishment as relay client itself, which
87    // needs to be more configurable so users can do more crazy things:
88    // https://github.com/n0-computer/iroh/issues/2901
89    #[cfg(not(wasm_browser))]
90    let mut builder = reqwest_client_builder(tls_config, dns_resolver.clone())
91        .redirect(reqwest::redirect::Policy::none());
92    #[cfg(wasm_browser)]
93    let mut builder = reqwest_client_builder();
94
95    #[cfg(not(wasm_browser))]
96    if let Some(Host::Domain(domain)) = url.host() {
97        // Use our own resolver rather than getaddrinfo
98        //
99        // Be careful, a non-zero port will override the port in the URI.
100        //
101        // The relay Client uses `.lookup_ipv4_ipv6` to connect, so use the same function
102        // but staggered for reliability.  Ideally this tries to resolve **both** IPv4 and
103        // IPv6 though.  But our resolver does not have a function for that yet.
104        let addrs: Vec<_> = dns_resolver
105            .lookup_ipv4_ipv6_staggered(domain, DNS_TIMEOUT, DNS_STAGGERING_MS)
106            .await?
107            .map(|ipaddr| SocketAddr::new(ipaddr, 0))
108            .collect();
109        trace!(?addrs, "resolved addrs");
110        builder = builder.resolve_to_addrs(domain, &addrs);
111    }
112
113    let client = builder
114        .build()
115        .map_err(|err| e!(MeasureHttpsLatencyError::CreateReqwestClient, err))?;
116
117    let start = Instant::now();
118    let response = client
119        .request(reqwest::Method::GET, url)
120        .send()
121        .await
122        .map_err(|err| e!(MeasureHttpsLatencyError::HttpRequest, err))?;
123    let latency = start.elapsed();
124    if response.status().is_success() {
125        // Drain the response body to be nice to the server, up to a limit.
126        const MAX_BODY_SIZE: usize = 8 << 10; // 8 KiB
127        let mut body_size = 0;
128        let mut stream = response.bytes_stream();
129        // ignore failing frames
130        while let Some(Ok(chunk)) = stream.next().await {
131            body_size += chunk.len();
132            if body_size >= MAX_BODY_SIZE {
133                break;
134            }
135        }
136
137        Ok(HttpsProbeReport {
138            relay_url: relay,
139            latency,
140        })
141    } else {
142        Err(e!(MeasureHttpsLatencyError::InvalidResponse {
143            status: response.status()
144        }))
145    }
146}

148#[cfg(all(test, with_crypto_provider))]
149mod tests {
150    use iroh_dns::dns::DnsResolver;
151    use iroh_relay::tls::{CaTlsConfig, default_provider};
152    use n0_error::Result;
153
154    use super::{super::test_utils, *};
155
156    #[tokio::test]
157    async fn test_measure_https_latency() -> Result {
158        let (_server, relay) = test_utils::relay().await;
159        let dns_resolver = DnsResolver::new();
160        tracing::info!(relay_url = ?relay.url , "RELAY_URL");
161        let report = run_probe(
162            &dns_resolver,
163            relay.url,
164            CaTlsConfig::insecure_skip_verify()
165                .client_config(default_provider())
166                .expect("infallible"),
167        )
168        .await?;
169
170        assert!(report.latency > Duration::ZERO);
171
172        Ok(())
173    }
