29/// The interface state the actor uses to pick address families to probe.
30#[derive(Debug, Clone, Default)]
31pub(crate) struct IfState {
32    /// Whether IPv4 is available on some interface.
33    pub(crate) have_v4: bool,
34    /// Whether IPv6 is available on some interface.
35    pub(crate) have_v6: bool,
36}
37
38impl IfState {
39    /// Returns an [`IfState`] with both families available, for tests.
40    #[cfg(all(test, with_crypto_provider))]
41    pub(super) fn fake() -> Self {
42        IfState {
43            have_v4: true,
44            have_v6: true,
45        }
46    }
47}
48
49impl From<netwatch::netmon::State> for IfState {
50    fn from(value: netwatch::netmon::State) -> Self {
51        IfState {
52            have_v4: value.have_v4,
53            have_v6: value.have_v6,
54        }
55    }
56}

58/// Configuration for the net report component.
59///
60/// Controls which probes and checks are performed when generating network
61/// reports, and how they are paced.
62#[derive(Debug, Clone)]
63#[non_exhaustive]
64pub struct NetReportConfig {
65    /// Whether to run HTTPS latency probes against relay servers.
66    ///
67    /// HTTPS latency probes perform an empty HTTPS GET request to each configured
68    /// relay server and measure latency.
69    ///
70    /// They are performed in addition to the QUIC address discovery (QAD) probes.
71    /// In networks that do not allow QUIC traffic, they are the only way to detect
72    /// relay latencies and thus the preferred relay.
73    ///
74    /// Disabling them is harmless on networks that do allow QUIC traffic, but will
75    /// completely prevent finding the home relay on networks that do block QUIC.
76    pub https_probes: bool,
77
78    /// Whether to check for captive portals when generating the first report.
79    ///
80    /// This is done by accessing a well-known URL that is available on each relay
81    /// server, `/generate_204`. If a GET request to this URL returns anything else
82    /// but a 204 No Content response, we assume we are behind a captive portal.
83    ///
84    /// When we have detected that we are behind a captive portal, we try to contact
85    /// the relay servers more frequently in case the captive portal status changes.
86    pub captive_portal_check: bool,
87
88    /// Delay inserted between successive QAD probes within a report cycle.
89    ///
90    /// A cycle runs QUIC address discovery against several relays. By default
91    /// those probes all start at once, a brief burst of QUIC handshakes. On
92    /// resource-constrained devices (an ESP32, say) that burst is disruptive,
93    /// so a non-zero value spreads the probes out: the first still starts
94    /// immediately, and each following one is delayed by a further multiple
95    /// of this duration. Setting it does not delay the first result.
96    ///
97    /// Defaults to [`Duration::ZERO`], which keeps the all-at-once behavior.
98    pub qad_stagger: Duration,
99}
100
101impl NetReportConfig {
102    /// Creates a minimal configuration that disables all optional probes and checks.
103    pub fn minimal() -> Self {
104        Self {
105            https_probes: false,
106            captive_portal_check: false,
107            qad_stagger: Duration::ZERO,
108        }
109    }
110}
111
112impl Default for NetReportConfig {
113    fn default() -> Self {
114        Self {
115            https_probes: true,
116            captive_portal_check: true,
117            qad_stagger: Duration::ZERO,
118        }
119    }
120}

137/// Handle to the net report actor.
138///
139/// The handle only triggers probe cycles; it does not carry the results.
140/// The [`NetReportActor`] runs in the background and writes each [`Report`]
141/// into the `report_out` [`Watchable`] that the caller passes to
142/// [`Client::new`] and watches directly.
143#[derive(Debug)]
144pub(crate) struct Client {
145    /// Shared slot the client writes probe triggers into.
146    probe_requests: Arc<RequestSlot>,
147    /// Aborts the background actor when the client is dropped.
148    _actor_handle: AbortOnDropHandle<()>,
149}
150
151impl Client {
152    /// Creates a new net report client, spawning the background actor.
153    ///
154    /// The actor writes reports into `report_out` as probe results arrive.
155    /// It is owned by the caller (the socket) so it can be read via the
156    /// public `Endpoint::net_report` accessor; the client only drives
157    /// probes into it.
158    pub(crate) fn new(
159        #[cfg(not(wasm_browser))] dns_resolver: DnsResolver,
160        relay_map: RelayMap,
161        opts: Options,
162        metrics: Arc<Metrics>,
163        shutdown: CancellationToken,
164        report_out: Watchable<Option<Report>>,
165    ) -> Self {
166        let probe_requests = Arc::new(RequestSlot::new());
167
168        let actor = NetReportActor::new(
169            Arc::clone(&probe_requests),
170            report_out,
171            relay_map,
172            opts,
173            #[cfg(not(wasm_browser))]
174            dns_resolver,
175            shutdown,
176            metrics,
177        );
178
179        let handle = n0_future::task::spawn(actor.run());
180
181        Self {
182            probe_requests,
183            _actor_handle: AbortOnDropHandle::new(handle),
184        }
185    }
186
187    /// Triggers a probe cycle.
188    ///
189    /// Non-blocking; returns immediately. Calls that arrive close together
190    /// coalesce into a single request, keeping the most urgent scope.
191    pub(crate) fn run_probes(&self, if_state: IfState, scope: ProbeScope) {
192        self.probe_requests.request(if_state, scope);
193    }
194}

196#[cfg(test)]
197mod test_utils {
198    //! Creates a relay server against which to perform tests
199
200    use iroh_relay::{RelayConfig, RelayQuicConfig, server};
201
202    pub(crate) async fn relay() -> (server::Server, RelayConfig) {
203        let server = server::Server::spawn(server::testing::server_config())
204            .await
205            .expect("should serve relay");
206        let quic = Some(RelayQuicConfig::new(
207            server.quic_addr().expect("server should run quic").port(),
208        ));
209        let endpoint_desc =
210            RelayConfig::new(server.https_url().expect("should work as relay"), quic);
211
212        (server, endpoint_desc)
213    }
214
215    /// Create a [`crate::RelayMap`] of the given size.
216    pub(crate) async fn relay_map(relays: usize) -> (Vec<server::Server>, crate::RelayMap) {
217        let mut servers = Vec::with_capacity(relays);
218        let mut endpoints = Vec::with_capacity(relays);
219        for _ in 0..relays {
220            let (relay_server, endpoint) = relay().await;
221            servers.push(relay_server);
222            endpoints.push(endpoint);
223        }
224        (servers, crate::RelayMap::from_iter(endpoints))
225    }
226}

230    use std::net::{Ipv4Addr, SocketAddr};

232    use iroh_dns::dns::DnsResolver;
233    use iroh_relay::tls::{CaTlsConfig, default_provider};
234    use n0_error::{Result, StdResultExt};
235    use n0_future::time::Duration;
236    use n0_tracing_test::traced_test;
237    use n0_watcher::Watcher;
238    use tokio_util::sync::CancellationToken;
239
240    use super::*;
241
242    #[tokio::test(flavor = "multi_thread")]
243    #[traced_test]
244    async fn test_basic() -> Result<()> {
245        let (server, relay) = test_utils::relay().await;
246        let client_config = iroh_relay::tls::make_dangerous_client_config();
247        let ep = noq::Endpoint::client(SocketAddr::new(Ipv4Addr::LOCALHOST.into(), 0)).anyerr()?;
248        let quic_addr_disc = QuicConfig {
249            ep: ep.clone(),
250            client_config,
251            ipv4: true,
252            ipv6: true,
253        };
254        let relay_map = RelayMap::from(relay);
255
256        let resolver = DnsResolver::new();
257        let tls_config = CaTlsConfig::insecure_skip_verify()
258            .client_config(default_provider())
259            .expect("infallible");
260        let opts = Options::new(tls_config).quic_config(Some(quic_addr_disc));
261        let cancel = CancellationToken::new();
262        let report_out = Watchable::new(None);
263        let client = Client::new(
264            resolver,
265            relay_map,
266            opts,
267            Default::default(),
268            cancel.child_token(),
269            report_out.clone(),
270        );
271        let if_state = IfState::fake();
272
273        // Trigger a probe cycle.
274        client.run_probes(if_state, ProbeScope::Full);
275
276        // Wait for the report to have QAD data.
277        let mut watcher = report_out.watch();
278        let report = tokio::time::timeout(Duration::from_secs(10), async {
279            loop {
280                let r = watcher.updated().await.expect("watcher closed");
281                if let Some(ref r) = r
282                    && r.global_v4.is_some()
283                    && r.preferred_relay.is_some()
284                {
285                    return r.clone();
286                }
287            }
288        })
289        .await
290        .expect("timed out waiting for report");
291
292        assert!(report.has_udp(), "want UDP");
293        assert!(
294            !report.relay_latency.is_empty(),
295            "expected at least 1 key in RelayLatency; got none",
296        );
297        assert!(report.global_v4.is_some(), "expected globalV4 set");
298        assert!(report.preferred_relay.is_some());
299
300        cancel.cancel();
301        drop(client);
302        ep.wait_idle().await;
303        server.shutdown().await?;
304
305        Ok(())
306    }
