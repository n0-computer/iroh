12impl Stats {
13    pub fn stream_finished(&mut self, stream_result: TransferResult) {
14        self.total_size += stream_result.size;
15        self.streams += 1;
16
17        self.stream_stats
18            .duration_hist
19            .record(stream_result.duration.as_millis() as u64)
20            .unwrap();
21        self.stream_stats
22            .throughput_hist
23            .record(stream_result.throughput as u64)
24            .unwrap();
25        self.stream_stats
26            .ttfb_hist
27            .record(stream_result.ttfb.as_nanos() as u64)
28            .unwrap();
29        self.stream_stats
30            .chunk_time
31            .record(
32                stream_result.duration.as_nanos() as u64 / std::cmp::max(stream_result.chunks, 1),
33            )
34            .unwrap();
35        self.stream_stats.chunks += stream_result.chunks;
36        self.stream_stats
37            .chunk_size
38            .record(stream_result.avg_chunk_size)
39            .unwrap();
40    }

86pub struct StreamStats {

88    pub throughput_hist: Histogram<u64>,
89    pub ttfb_hist: Histogram<u64>,
90    pub chunk_time: Histogram<u64>,
91    pub chunks: u64,
92    pub chunk_size: Histogram<u64>,
93}
94
95impl Default for StreamStats {

98            duration_hist: Histogram::<u64>::new(3).unwrap(),
99            throughput_hist: Histogram::<u64>::new(3).unwrap(),
100            ttfb_hist: Histogram::<u64>::new(3).unwrap(),
101            chunk_time: Histogram::<u64>::new(3).unwrap(),
102            chunks: 0,

108#[derive(Debug)]
109pub struct TransferResult {
110    pub duration: Duration,
111    pub size: u64,
112    pub throughput: f64,
113    pub ttfb: Duration,
114    pub chunks: u64,
115    pub avg_chunk_size: u64,
116}
117
118impl TransferResult {
119    pub fn new(duration: Duration, size: u64, ttfb: Duration, chunks: u64) -> Self {
120        let throughput = throughput_bps(duration, size);
121        TransferResult {
122            duration,
123            size,
124            throughput,
125            ttfb,
126            chunks,
127            avg_chunk_size: size / std::cmp::max(chunks, 1),
128        }
129    }
130}
131
132pub fn throughput_bps(duration: Duration, size: u64) -> f64 {
133    (size as f64) / (duration.as_secs_f64())
