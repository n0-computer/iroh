16/// DNS node discovery
17///
18/// When asked to resolve a [`NodeId`], this service performs a lookup in the Domain Name System (DNS).
19///
20/// It uses the [`Endpoint`]'s DNS resolver to query for `TXT` records under the domain
21/// `_iroh.<z32-node-id>.<origin-domain>`:
22///
23/// * `_iroh`: is the record name
24/// * `<z32-node-id>` is the [`NodeId`] encoded in [`z-base-32`] format
25/// * `<origin-domain>` is the node origin domain as set in [`DnsDiscovery::builder`].
26///
27/// Each TXT record returned from the query is expected to contain a string in the format `<name>=<value>`.
28/// If a TXT record contains multiple character strings, they are concatenated first.
29/// The supported attributes are:
30/// * `relay=<url>`: The URL of the home relay server of the node
31///
32/// The DNS resolver defaults to using the nameservers configured on the host system, but can be changed
33/// with [`crate::endpoint::Builder::dns_resolver`].
34///
35/// [z-base-32]: https://philzimmermann.com/docs/human-oriented-base-32-encoding.txt
36/// [`Endpoint`]: crate::Endpoint
37#[derive(Debug)]
38pub struct DnsDiscovery {
39    origin_domain: String,
40    dns_resolver: DnsResolver,
41}
42
43/// Builder for [`DnsDiscovery`].
44///
45/// See [`DnsDiscovery::builder`].
46#[derive(Debug)]
47pub struct DnsDiscoveryBuilder {
48    origin_domain: String,
49    dns_resolver: Option<DnsResolver>,
50}
51
52impl DnsDiscoveryBuilder {
53    /// Sets the DNS resolver to use.
54    pub fn dns_resolver(mut self, dns_resolver: DnsResolver) -> Self {
55        self.dns_resolver = Some(dns_resolver);
56        self
57    }
58
59    /// Builds a [`DnsDiscovery`] with the passed [`DnsResolver`].
60    pub fn build(self) -> DnsDiscovery {
61        DnsDiscovery {
62            dns_resolver: self.dns_resolver.unwrap_or_default(),
63            origin_domain: self.origin_domain,
64        }
65    }
66}
67
68impl DnsDiscovery {
69    /// Creates a [`DnsDiscoveryBuilder`] that implements [`IntoDiscovery`].
70    pub fn builder(origin_domain: String) -> DnsDiscoveryBuilder {
71        DnsDiscoveryBuilder {
72            origin_domain,
73            dns_resolver: None,
74        }
75    }
76
77    /// Creates a new DNS discovery using the `iroh.link` domain.
78    ///
79    /// This uses the [`N0_DNS_NODE_ORIGIN_PROD`] domain.
80    ///
81    /// # Usage during tests
82    ///
83    /// For testing it is possible to use the [`N0_DNS_NODE_ORIGIN_STAGING`] domain
84    /// with [`DnsDiscovery::builder`].  This would then use a hosted staging discovery
85    /// service for testing purposes.
86    pub fn n0_dns() -> DnsDiscoveryBuilder {
87        if force_staging_infra() {
88            Self::builder(N0_DNS_NODE_ORIGIN_STAGING.to_string())
89        } else {
90            Self::builder(N0_DNS_NODE_ORIGIN_PROD.to_string())
91        }
92    }
93}
94
95impl IntoDiscovery for DnsDiscoveryBuilder {
96    fn into_discovery(
97        mut self,
98        context: &DiscoveryContext,
99    ) -> Result<impl Discovery, IntoDiscoveryError> {
100        if self.dns_resolver.is_none() {
101            self.dns_resolver = Some(context.dns_resolver().clone());
102        }
103        Ok(self.build())
104    }
105}
106
107impl Discovery for DnsDiscovery {
108    fn resolve(&self, node_id: NodeId) -> Option<BoxStream<Result<DiscoveryItem, DiscoveryError>>> {
109        let resolver = self.dns_resolver.clone();
110        let origin_domain = self.origin_domain.clone();
111        let fut = async move {
112            let node_info = resolver
113                .lookup_node_by_id_staggered(&node_id, &origin_domain, DNS_STAGGERING_MS)
114                .await
115                .map_err(|e| DiscoveryError::from_err("dns", e))?;
116            Ok(DiscoveryItem::new(node_info, "dns", None))
117        };
118        let stream = n0_future::stream::once_future(fut);
119        Some(Box::pin(stream))
120    }
121}
