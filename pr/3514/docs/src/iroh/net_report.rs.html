94/// The maximum latency of all nodes, if none are found yet.
95///
96/// Normally the max latency of all nodes is computed, but if we don't yet know any nodes
97/// latencies we return this as default.  This is the value of the initial QAD probe
98/// delays.  It is only used as time to wait for further latencies to arrive, which *should*
99/// never happen unless there already is at least one latency.  Yet here we are, defining a
100/// default which will never be used.
101const DEFAULT_MAX_LATENCY: Duration = Duration::from_millis(100);

131#[cfg(not(wasm_browser))]
132impl QadConns {
133    fn clear(&mut self) {
134        if let Some((_, conn)) = self.v4.take() {
135            conn.conn
136                .close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
137        }
138        if let Some((_, conn)) = self.v6.take() {
139            conn.conn
140                .close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
141        }
142    }
143
144    fn current(&self) -> Vec<ProbeReport> {
145        let mut reports = Vec::new();
146        if let Some((_, ref conn)) = self.v4 {
147            if let Some(mut r) = conn.observer.get() {
148                // grab latest rtt
149                r.latency = conn.conn.rtt();
150                reports.push(ProbeReport::QadIpv4(r));
151            }
152        }
153
154        if let Some((_, ref conn)) = self.v6 {
155            if let Some(mut r) = conn.observer.get() {
156                // grab latest rtt
157                r.latency = conn.conn.rtt();
158                reports.push(ProbeReport::QadIpv6(r));
159            }
160        }
161
162        reports
163    }
164
165    fn watch_v4(&self) -> impl n0_future::Stream<Item = Option<QadProbeReport>> + Unpin + use<> {
166        let watcher = self.v4.as_ref().map(|(_url, conn)| conn.observer.watch());
167
168        if let Some(watcher) = watcher {
169            watcher.stream_updates_only().boxed()
170        } else {
171            n0_future::stream::empty().boxed()
172        }
173    }
174
175    fn watch_v6(&self) -> impl n0_future::Stream<Item = Option<QadProbeReport>> + Unpin + use<> {
176        let watcher = self.v6.as_ref().map(|(_url, conn)| conn.observer.watch());
177        if let Some(watcher) = watcher {
178            watcher.stream_updates_only().boxed()
179        } else {
180            n0_future::stream::empty().boxed()
181        }
182    }

255    /// Generates a [`Report`].
256    ///
257    /// Look at [`Options`] for the different configuration options.
258    pub(crate) async fn get_report(&mut self, if_state: IfStateDetails, is_major: bool) -> Report {
259        let now = Instant::now();
260
261        let mut do_full = is_major
262            || self.reports.next_full
263            || now.duration_since(self.reports.last_full) > FULL_REPORT_INTERVAL;
264
265        debug!(%do_full, "net_report starting");
266
267        // If the last report had a captive portal and reported no UDP access,
268        // it's possible that we didn't get a useful net_report due to the
269        // captive portal blocking us. If so, make this report a full (non-incremental) one.
270        if !do_full {
271            if let Some(ref last) = self.reports.last {
272                if !last.has_udp() && last.captive_portal == Some(true) {
273                    do_full = true;
274                }
275            }
276        }
277        if do_full {
278            self.reports.last = None; // causes ProbePlan::new below to do a full (initial) plan
279            self.reports.next_full = false;
280            self.reports.last_full = now;
281            self.metrics.reports_full.inc();
282        }
283        self.metrics.reports.inc();
284
285        let enough_relays = std::cmp::min(self.relay_map.len(), ENOUGH_NODES);
286        #[cfg(wasm_browser)]
287        let if_state = IfStateDetails::default();
288        #[cfg(not(wasm_browser))]
289        let if_state = IfStateDetails {
290            have_v4: if_state.have_v4,
291            have_v6: if_state.have_v6,
292        };
293
294        let mut report = Report::default();
295
296        // Start the reportgen client to start any needed probes
297        let (actor, mut probe_rx) = reportgen::Client::new(
298            self.reports.last.clone(),
299            self.relay_map.clone(),
300            self.probes.clone(),
301            if_state.clone(),
302            #[cfg(not(wasm_browser))]
303            self.socket_state.clone(),
304            #[cfg(any(test, feature = "test-utils"))]
305            self.insecure_skip_relay_cert_verify,
306        );
307
308        #[cfg(not(wasm_browser))]
309        let reports = self
310            .spawn_qad_probes(&if_state, enough_relays, do_full)
311            .await;
312
313        #[cfg(not(wasm_browser))]
314        for r in reports {
315            report.update(&r);
316        }
317
318        let mut timeout_fut = std::pin::pin!(MaybeFuture::default());
319
320        #[cfg(not(wasm_browser))]
321        let mut qad_v4_stream = self.qad_conns.watch_v4();
322        #[cfg(wasm_browser)]
323        let mut qad_v4_stream = n0_future::stream::empty::<Option<()>>();
324        #[cfg(not(wasm_browser))]
325        let mut qad_v6_stream = self.qad_conns.watch_v6();
326        #[cfg(wasm_browser)]
327        let mut qad_v6_stream = n0_future::stream::empty::<Option<()>>();
328
329        loop {
330            tokio::select! {
331                biased;
332
333                _ = &mut timeout_fut, if timeout_fut.is_some() => {
334                    trace!("timeout done, shutting down");
335                    drop(actor); // shuts down the probes
336                    break;
337                }
338
339                Some(Some(r)) = qad_v4_stream.next() => {
340                    #[cfg(not(wasm_browser))]
341                    {
342                        trace!(?r, "new report from QAD V4");
343                        report.update(&ProbeReport::QadIpv4(r));
344                    }
345                }
346
347                Some(Some(r)) = qad_v6_stream.next() => {
348                    #[cfg(not(wasm_browser))]
349                    {
350                        trace!(?r, "new report from QAD V6");
351                        report.update(&ProbeReport::QadIpv6(r));
352                    }
353                }
354
355                maybe_probe = probe_rx.recv() => {
356                    let Some(probe_res) = maybe_probe else {
357                        break;
358                    };
359                    trace!(?probe_res, "handling probe");
360                    match probe_res {
361                        ProbeFinished::Regular(probe) => match probe {
362                            Ok(probe) => {
363                                report.update(&probe);
364                                if timeout_fut.is_none() {
365                                    if let Some(timeout) = self.have_enough_reports(enough_relays, &report) {
366                                        timeout_fut.as_mut().set_future(time::sleep(timeout));
367                                    }
368                                }
369                            }
370                            Err(err) => {
371                                trace!("probe errored: {:?}", err);
372                            }
373                        },
374                        #[cfg(not(wasm_browser))]
375                        ProbeFinished::CaptivePortal(portal) => {
376                            report.captive_portal = portal;
377                        }
378                    }
379                }
380            }
381        }
382
383        self.add_report_history_and_set_preferred_relay(&mut report);
384        debug!(
385            ?report,
386            "generated report in {:02}ms",
387            now.elapsed().as_millis()
388        );
389
390        report
391    }

393    #[cfg(not(wasm_browser))]
394    async fn spawn_qad_probes(
395        &mut self,
396        if_state: &IfStateDetails,
397        enough_relays: usize,
398        do_full: bool,
399    ) -> Vec<ProbeReport> {
400        use tracing::{Instrument, warn_span};
401
402        debug!("spawning QAD probes");
403
404        let Some(ref quic_client) = self.socket_state.quic_client else {
405            return Vec::new();
406        };
407
408        if do_full {
409            // clear out existing connections if we are doing a full reset
410            self.qad_conns.clear();
411        }
412
413        if let Some((url, conn)) = &self.qad_conns.v4 {
414            // verify conn is still around
415            if let Some(reason) = conn.conn.close_reason() {
416                trace!(?url, "QAD v4 conn closed: {}", reason);
417                self.qad_conns.v4.take();
418            }
419        }
420        if let Some((url, conn)) = &self.qad_conns.v6 {
421            // verify conn is still around
422            if let Some(reason) = conn.conn.close_reason() {
423                trace!(?url, "QAD v6 conn closed: {}", reason);
424                self.qad_conns.v6.take();
425            }
426        }
427        if self.qad_conns.v4.is_some() && self.qad_conns.v6.is_some() == if_state.have_v6 {
428            trace!("not spawning QAD, already have probes");
429            return self.qad_conns.current();
430        }
431
432        // TODO: randomize choice?
433        const MAX_RELAYS: usize = 5;
434
435        let mut v4_buf = JoinSet::new();
436        let cancel_v4 = CancellationToken::new();
437        let mut v6_buf = JoinSet::new();
438        let cancel_v6 = CancellationToken::new();
439
440        for relay_node in self.relay_map.nodes().take(MAX_RELAYS) {
441            if if_state.have_v4 {
442                debug!(?relay_node.url, "v4 QAD probe");
443                let ip_mapped_addrs = self.socket_state.ip_mapped_addrs.clone();
444                let relay_node = relay_node.clone();
445                let dns_resolver = self.socket_state.dns_resolver.clone();
446                let quic_client = quic_client.clone();
447                let relay_url = relay_node.url.clone();
448                v4_buf.spawn(
449                    cancel_v4
450                        .child_token()
451                        .run_until_cancelled_owned(time::timeout(
452                            PROBES_TIMEOUT,
453                            run_probe_v4(ip_mapped_addrs, relay_node, quic_client, dns_resolver),
454                        ))
455                        .instrument(warn_span!("QAD-IPv4", %relay_url)),
456                );
457            }
458
459            if if_state.have_v6 {
460                debug!(?relay_node.url, "v6 QAD probe");
461                let ip_mapped_addrs = self.socket_state.ip_mapped_addrs.clone();
462                let relay_node = relay_node.clone();
463                let dns_resolver = self.socket_state.dns_resolver.clone();
464                let quic_client = quic_client.clone();
465                let relay_url = relay_node.url.clone();
466                v6_buf.spawn(
467                    cancel_v6
468                        .child_token()
469                        .run_until_cancelled_owned(time::timeout(
470                            PROBES_TIMEOUT,
471                            run_probe_v6(ip_mapped_addrs, relay_node, quic_client, dns_resolver),
472                        ))
473                        .instrument(warn_span!("QAD-IPv6", %relay_url)),
474                );
475            }
476        }
477
478        let mut reports = Vec::new();
479
480        // We set _pending to true if at least one report was started for each category.
481        // If we did not start any report for either category, _pending is set to false right away
482        // (it "completed" in the sense that nothing will ever run). If we did start at least one report,
483        // _pending is set to true, and will be set to false further down once the first task
484        // completed.
485        let mut ipv4_pending = !v4_buf.is_empty();
486        let mut ipv6_pending = !v6_buf.is_empty();
487        loop {
488            // We early-abort the tasks once we have at least `enough_relays` reports,
489            // and at least one ipv4 and one ipv6 report completed (if they were started, see comment above).
490            if reports.len() >= enough_relays && !ipv4_pending && !ipv6_pending {
491                debug!("enough probes: {}", reports.len());
492                cancel_v4.cancel();
493                cancel_v6.cancel();
494                break;
495            }
496
497            tokio::select! {
498                biased;
499
500                val = v4_buf.join_next(), if !v4_buf.is_empty() => {
501                    ipv4_pending = false;
502                    match val {
503                        Some(Ok(Some(Ok(res)))) => {
504                            match res {
505                                Ok((r, conn)) => {
506                                    debug!(?r, "got v4 QAD conn");
507                                    let url = r.node.clone();
508                                    reports.push(ProbeReport::QadIpv4(r));
509                                    if self.qad_conns.v4.is_none() {
510                                        self.qad_conns.v4.replace((url, conn));
511                                    } else {
512                                        conn.conn.close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
513                                    }
514                                }
515                                Err(err) => {
516                                    debug!("probe v4 failed: {err:?}");
517                                }
518                            }
519                        }
520                        Some(Err(err)) => {
521                            if err.is_panic() {
522                                panic!("probe v4 panicked: {err:?}");
523                            }
524                            warn!("probe v4 failed: {err:?}");
525                        }
526                        Some(Ok(None)) => {
527                            debug!("probe v4 canceled");
528                        }
529                        Some(Ok(Some(Err(time::Elapsed { .. })))) => {
530                            debug!("probe v4 timed out");
531                        }
532                        None => {}
533                    }
534                }
535                val = v6_buf.join_next(), if !v6_buf.is_empty() => {
536                    ipv6_pending = false;
537                    match val {
538                        Some(Ok(Some(Ok(res)))) => {
539                            match res {
540                                Ok((r, conn)) => {
541                                    debug!(?r, "got v6 QAD conn");
542                                    let url = r.node.clone();
543                                    reports.push(ProbeReport::QadIpv6(r));
544                                    if self.qad_conns.v6.is_none() {
545                                        self.qad_conns.v6.replace((url, conn));
546                                    } else {
547                                        conn.conn.close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
548                                    }
549                                }
550                                Err(err) => {
551                                    debug!("probe v6 failed: {err:?}");
552                                }
553                            }
554                        }
555                        Some(Err(err)) => {
556                            if err.is_panic() {
557                                panic!("probe v6 panicked: {err:?}");
558                            }
559                            warn!("probe v6 failed: {err:?}");
560                        }
561                        Some(Ok(None)) => {
562                            debug!("probe v6 canceled");
563                        }
564                        Some(Ok(Some(Err(time::Elapsed { .. })))) => {
565                            debug!("probe v6 timed out");
566                        }
567                        None => {}
568                    }
569                }
570                else => {
571                    break;
572                }
573            }
574        }
575
576        reports
577    }

579    fn have_enough_reports(&self, enough_relays: usize, report: &Report) -> Option<Duration> {
580        // Once we've heard from enough relay servers (3), start a timer to give up on the other
581        // probes. The timer's duration is a function of whether this is our initial full
582        // probe or an incremental one. For incremental ones, wait for the duration of the
583        // slowest relay. For initial ones, double that.
584        let latencies: Vec<Duration> = report.relay_latency.iter().map(|(_, l)| l).collect();
585        let have_enough_latencies = latencies.len() >= enough_relays;
586
587        if have_enough_latencies {
588            let timeout = match self.reports.last.is_some() {
589                true => Duration::from_secs(0),
590                false => latencies
591                    .iter()
592                    .max()
593                    .copied()
594                    .unwrap_or(DEFAULT_MAX_LATENCY),
595            };
596            debug!(
597                reports=latencies.len(),
598                delay=?timeout,
599                "Have enough probe reports, aborting further probes soon",
600            );
601
602            Some(timeout)
603        } else {
604            None
605        }
606    }

608    /// Adds `r` to the set of recent Reports and mutates `r.preferred_relay` to contain the best recent one.
609    fn add_report_history_and_set_preferred_relay(&mut self, r: &mut Report) {
610        let mut prev_relay = None;
611        if let Some(ref last) = self.reports.last {
612            prev_relay.clone_from(&last.preferred_relay);
613
614            // If we don't have new information, copy this from the last report
615            if r.mapping_varies_by_dest_ipv4.is_none() {
616                r.mapping_varies_by_dest_ipv4 = last.mapping_varies_by_dest_ipv4;
617            }
618            if r.mapping_varies_by_dest_ipv6.is_none() {
619                r.mapping_varies_by_dest_ipv6 = last.mapping_varies_by_dest_ipv6;
620            }
621        }
622
623        let now = Instant::now();
624        const MAX_AGE: Duration = Duration::from_secs(5 * 60);
625
626        // relay ID => its best recent latency in last MAX_AGE
627        let mut best_recent = RelayLatencies::default();
628
629        // chain the current report as we are still mutating it
630        let prevs_iter = self
631            .reports
632            .prev
633            .iter()
634            .map(|(a, b)| -> (&Instant, &Report) { (a, b) });
635
636        let mut to_remove = Vec::new();
637        for (t, pr) in prevs_iter {
638            if now.duration_since(*t) > MAX_AGE {
639                to_remove.push(*t);
640                continue;
641            }
642            best_recent.merge(&pr.relay_latency);
643        }
644        // merge in current run
645        best_recent.merge(&r.relay_latency);
646
647        for t in to_remove {
648            self.reports.prev.remove(&t);
649        }
650
651        // Then, pick which currently-alive relay server from the
652        // current report has the best latency over the past MAX_AGE.
653        let mut best_any = Duration::default();
654        let mut old_relay_cur_latency = Duration::default();
655        {
656            for (url, duration) in r.relay_latency.iter() {
657                if Some(url) == prev_relay.as_ref() {
658                    old_relay_cur_latency = duration;
659                }
660                if let Some(best) = best_recent.get(url) {
661                    if r.preferred_relay.is_none() || best < best_any {
662                        best_any = best;
663                        r.preferred_relay.replace(url.clone());
664                    }
665                }
666            }
667
668            // If we're changing our preferred relay but the old one's still
669            // accessible and the new one's not much better, just stick with
670            // where we are.
671            if prev_relay.is_some()
672                && r.preferred_relay != prev_relay
673                && !old_relay_cur_latency.is_zero()
674                && best_any > old_relay_cur_latency / 3 * 2
675            {
676                r.preferred_relay = prev_relay;
677            }
678        }
679
680        self.reports.prev.insert(now, r.clone());
681        self.reports.last = Some(r.clone());
682    }
683}

686async fn run_probe_v4(
687    ip_mapped_addrs: Option<IpMappedAddresses>,
688    relay_node: Arc<RelayNode>,
689    quic_client: QuicClient,
690    dns_resolver: DnsResolver,
691) -> n0_snafu::Result<(QadProbeReport, QadConn)> {
692    use n0_snafu::ResultExt;
693
694    let relay_addr_orig = reportgen::get_relay_addr_ipv4(&dns_resolver, &relay_node).await?;
695    let relay_addr =
696        reportgen::maybe_to_mapped_addr(ip_mapped_addrs.as_ref(), relay_addr_orig.into());
697
698    debug!(?relay_addr_orig, ?relay_addr, "relay addr v4");
699    let host = relay_node.url.host_str().context("missing host url")?;
700    let conn = quic_client.create_conn(relay_addr, host).await?;
701    let mut receiver = conn.observed_external_addr();
702
703    // wait for an addr
704    let addr = receiver
705        .wait_for(|addr| addr.is_some())
706        .await
707        .context("receiver dropped")?
708        .expect("known");
709    let report = QadProbeReport {
710        node: relay_node.url.clone(),
711        addr: SocketAddr::new(addr.ip().to_canonical(), addr.port()),
712        latency: conn.rtt(),
713    };
714
715    let observer = Watchable::new(None);
716    let node = relay_node.url.clone();
717    let handle = task::spawn({
718        let conn = conn.clone();
719        let observer = observer.clone();
720        async move {
721            loop {
722                let val = *receiver.borrow();
723                // if we've sent to an ipv4 address, but received an observed address
724                // that is ivp6 then the address is an [IPv4-Mapped IPv6 Addresses](https://doc.rust-lang.org/beta/std/net/struct.Ipv6Addr.html#ipv4-mapped-ipv6-addresses)
725                let val = val.map(|val| SocketAddr::new(val.ip().to_canonical(), val.port()));
726                let latency = conn.rtt();
727                trace!(?val, ?relay_addr, ?latency, "got addr V4");
728                observer
729                    .set(val.map(|addr| QadProbeReport {
730                        node: node.clone(),
731                        addr,
732                        latency,
733                    }))
734                    .ok();
735                if receiver.changed().await.is_err() {
736                    break;
737                }
738            }
739        }
740    });
741    let handle = AbortOnDropHandle::new(handle);
742
743    Ok((
744        report,
745        QadConn {
746            conn,
747            observer,
748            _handle: handle,
749        },
750    ))
751}

828            .await

839    }

841    /// Create a [`crate::RelayMap`] of the given size.
842    ///
843    /// This function uses [`relay`]. Note that the returned map uses internal order that will
844    /// often _not_ match the order of the servers.
845    pub(crate) async fn relay_map(relays: usize) -> (Vec<server::Server>, crate::RelayMap) {
846        let mut servers = Vec::with_capacity(relays);
847        let mut nodes = Vec::with_capacity(relays);
848        for _ in 0..relays {
849            let (relay_server, node) = relay().await;
850            servers.push(relay_server);
851            nodes.push(node);
852        }
853        (servers, crate::RelayMap::from_iter(nodes))
854    }
