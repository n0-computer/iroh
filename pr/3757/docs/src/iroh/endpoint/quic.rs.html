95/// Parameters governing the core QUIC state machine
96///
97/// Default values should be suitable for most internet applications. Applications protocols which
98/// forbid remotely-initiated streams should set `max_concurrent_bidi_streams` and
99/// `max_concurrent_uni_streams` to zero.
100///
101/// In some cases, performance or resource requirements can be improved by tuning these values to
102/// suit a particular application and/or network connection. In particular, data window sizes can be
103/// tuned for a particular expected round trip time, link capacity, and memory availability. Tuning
104/// for higher bandwidths and latencies increases worst-case memory consumption, but does not impair
105/// performance at lower bandwidths and latencies. The default configuration is tuned for a 100Mbps
106/// link with a 100ms round trip time.
107///
108/// Use the [`QuicTransportConfigBuilder`] to customize these tunable fields.
109///
110/// In iroh, the config has some specific default values that make iroh's holepunching work
111/// well with QUIC multipath. Adjusting those settings may cause suboptimal usage.
112///
113/// Look at the following methods for more details:
114/// - [`QuicTransportConfigBuilder::default_path_keep_alive_interval`]
115/// - [`QuicTransportConfigBuilder::default_path_max_idle_timeout`]
116/// - [`QuicTransportConfigBuilder::max_concurrent_multipath_paths`]
117/// - [`QuicTransportConfigBuilder::set_max_remote_nat_traversal_addresses`]
118///
119/// # Examples
120/// ```
121/// use std::time::Duration;
122///
123/// use iroh::endpoint::QuicTransportConfig;
124///
125/// let _cfg = QuicTransportConfig::builder()
126///     .max_idle_timeout(Duration::from_secs(35))
127///     .build();
128/// ```
129#[derive(Debug, Clone)]
130pub struct QuicTransportConfig(Arc<quinn::TransportConfig>);

153    /// Create a default [`QuicTransportConfigBuilder`].
154    fn new() -> Self {
155        let mut cfg = quinn::TransportConfig::default();
156        // Override some transport config settings.
157        cfg.keep_alive_interval(Some(HEARTBEAT_INTERVAL));
158        cfg.default_path_keep_alive_interval(Some(HEARTBEAT_INTERVAL));
159        cfg.default_path_max_idle_timeout(Some(PATH_MAX_IDLE_TIMEOUT));
160        cfg.max_concurrent_multipath_paths(MAX_MULTIPATH_PATHS + 1);
161        cfg.set_max_remote_nat_traversal_addresses(MAX_MULTIPATH_PATHS as u8);
162        Self(cfg)
163    }

187    /// Maximum duration of inactivity to accept before timing out the connection.
188    ///
189    /// The true idle timeout is the minimum of this and the peer's own max idle timeout. `None`
190    /// represents an infinite timeout. Defaults to 30 seconds.

297    /// The maximum UDP payload size guaranteed to be supported by the network.
298    ///
299    /// Must be at least 1200, which is the default, and lower than or equal to
300    /// [`QuicTransportConfigBuilder::initial_mtu`].
301    ///
302    /// Real-world MTUs can vary according to ISP, VPN, and properties of intermediate network links
303    /// outside of either endpoint's control. Extreme care should be used when raising this value
304    /// outside of private networks where these factors are fully controlled. If the provided value
305    /// is higher than what the network path actually supports, the result will be unpredictable and
306    /// catastrophic packet loss, without a possibility of repair. Prefer
307    /// [`QuicTransportConfigBuilder::initial_mtu`] together with
308    /// [`QuicTransportConfigBuilder::mtu_discovery_config`] to set a maximum UDP payload size that robustly
309    /// adapts to the network.

580/// A builder for a [`ServerConfig`].
581#[derive(Debug, Clone)]
582pub struct ServerConfigBuilder {
583    inner: quinn::ServerConfig,
584    transport: QuicTransportConfig,
585}
586
587/// Parameters governing incoming connections
588///
589/// Default values should be suitable for most internet applications.
590///
591/// Use a [`ServerConfigBuilder`] to adjust the default values.
592///
593/// To create a [`ServerConfig`] compatible with your [`Endpoint`] identity, use the [`Endpoint::create_server_config_builder`] method.
594///
595/// [`Endpoint`]: crate::Endpoint
596/// [`Endpoint::create_server_config_builder`]: crate::Endpoint::create_server_config_builder
597// Note: used in `iroh::endpoint::connection::Incoming::accept_with`
598// This is new-typed since `quinn::ServerConfig` takes a `TransportConfig`, which we new-type as a `QuicTransportConfig`
599#[derive(Debug, Clone)]
600pub struct ServerConfig(Arc<quinn::ServerConfig>);
601
602impl ServerConfig {
603    pub(crate) fn to_inner_arc(&self) -> Arc<quinn::ServerConfig> {
604        self.0.clone()
605    }
606
607    /// Transport configuration used for incoming connections.
608    pub fn transport_config(&self) -> QuicTransportConfig {
609        QuicTransportConfig(self.0.transport.clone())
610    }
611
612    /// TLS configuration used for incoming connections.
613    pub fn crypto(&self) -> Arc<dyn CryptoServerConfig> {
614        self.0.crypto.clone()
615    }
616
617    /// Configuration for sending and handling validation tokens.
618    pub fn validation_token(&self) -> ValidationTokenConfig {
619        self.0.validation_token.clone()
620    }
621}
622
623impl ServerConfigBuilder {
624    /// Build a [`ServerConfig`] from a [`ServerConfigBuilder`].
625    pub fn build(self) -> ServerConfig {
626        ServerConfig(Arc::new(self.inner))
627    }
628
629    pub(crate) fn new(inner: quinn::ServerConfig, transport: QuicTransportConfig) -> Self {
630        Self { inner, transport }
631    }
632
633    /// Sets a custom [`QuicTransportConfig`].
634    pub fn set_transport_config(mut self, transport: QuicTransportConfig) -> Self {
635        self.inner.transport_config(transport.to_inner_arc());
636        self.transport = transport;
637        self
638    }
639
640    /// Sets a custom [`ValidationTokenConfig`].
641    pub fn set_validation_token_config(mut self, validation_token: ValidationTokenConfig) -> Self {
642        self.inner.validation_token_config(validation_token);
643        self
644    }
645
646    /// Private key used to authenticate data included in handshake tokens
647    pub fn set_token_key(mut self, value: Arc<dyn HandshakeTokenKey>) -> Self {
648        self.inner.token_key(value);
649        self
650    }
651
652    /// Duration after a retry token was issued for which it's considered valid
653    ///
654    /// Defaults to 15 seconds.
655    pub fn set_retry_token_lifetime(mut self, value: Duration) -> Self {
656        self.inner.retry_token_lifetime(value);
657        self
658    }
659
660    /// Maximum number of [`Incoming`] to allow to exist at a time.
661    ///
662    /// An [`Incoming`] comes into existence when an incoming connection attempt
663    /// is received and stops existing when the application either accepts it or otherwise disposes
664    /// of it. While this limit is reached, new incoming connection attempts are immediately
665    /// refused. Larger values have greater worst-case memory consumption, but accommodate greater
666    /// application latency in handling incoming connection attempts.
667    ///
668    /// The default value is set to 65536. With a typical Ethernet MTU of 1500 bytes, this limits
669    /// memory consumption from this to under 100 MiB--a generous amount that still prevents memory
670    /// exhaustion in most contexts.
671    ///
672    /// [`Incoming`]: crate::endpoint::Incoming
673    pub fn set_max_incoming(mut self, max_incoming: usize) -> Self {
674        self.inner.max_incoming(max_incoming);
675        self
676    }
677
678    /// Maximum number of received bytes to buffer for each [`Incoming`].
679    ///
680    /// An [`Incoming`] comes into existence when an incoming connection attempt
681    /// is received and stops existing when the application either accepts it or otherwise disposes
682    /// of it. This limit governs only packets received within that period, and does not include
683    /// the first packet. Packets received in excess of this limit are dropped, which may cause
684    /// 0-RTT or handshake data to have to be retransmitted.
685    ///
686    /// The default value is set to 10 MiB--an amount such that in most situations a client would
687    /// not transmit that much 0-RTT data faster than the server handles the corresponding
688    /// [`Incoming`].
689    ///
690    /// [`Incoming`]: crate::endpoint::Incoming
691    pub fn set_incoming_buffer_size(mut self, incoming_buffer_size: u64) -> Self {
692        self.inner.incoming_buffer_size(incoming_buffer_size);
693        self
694    }
695
696    /// Maximum number of received bytes to buffer for all [`Incoming`]
697    /// collectively.
698    ///
699    /// An [`Incoming`] comes into existence when an incoming connection attempt
700    /// is received and stops existing when the application either accepts it or otherwise disposes
701    /// of it. This limit governs only packets received within that period, and does not include
702    /// the first packet. Packets received in excess of this limit are dropped, which may cause
703    /// 0-RTT or handshake data to have to be retransmitted.
704    ///
705    /// The default value is set to 100 MiB--a generous amount that still prevents memory
706    /// exhaustion in most contexts.
707    ///
708    /// [`Incoming`]: crate::endpoint::Incoming
709    pub fn set_incoming_buffer_size_total(mut self, incoming_buffer_size_total: u64) -> Self {
710        self.inner
711            .incoming_buffer_size_total(incoming_buffer_size_total);
712        self
713    }
714
715    /// Object to get current [`SystemTime`].
716    ///
717    /// This exists to allow system time to be mocked in tests, or wherever else desired.
718    ///
719    /// Defaults to [`quinn::StdSystemTime`], which simply calls [`SystemTime::now()`](std::time::SystemTime::now).
720    ///
721    /// [`SystemTime`]: std::time::SystemTime
722    pub fn set_time_source(mut self, time_source: Arc<dyn TimeSource>) -> Self {
723        self.inner.time_source(time_source);
724        self
725    }
