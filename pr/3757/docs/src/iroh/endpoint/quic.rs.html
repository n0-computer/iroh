92/// Parameters governing the core QUIC state machine
93///
94/// Default values should be suitable for most internet applications. Applications protocols which
95/// forbid remotely-initiated streams should set `max_concurrent_bidi_streams` and
96/// `max_concurrent_uni_streams` to zero.
97///
98/// In some cases, performance or resource requirements can be improved by tuning these values to
99/// suit a particular application and/or network connection. In particular, data window sizes can be
100/// tuned for a particular expected round trip time, link capacity, and memory availability. Tuning
101/// for higher bandwidths and latencies increases worst-case memory consumption, but does not impair
102/// performance at lower bandwidths and latencies. The default configuration is tuned for a 100Mbps
103/// link with a 100ms round trip time.
104///
105/// Use the [`QuicTransportConfigBuilder`] to customize these tunable fields.
106///
107/// In iroh, the config has some specific default values that make iroh's holepunching work
108/// well with QUIC multipath. Adjusting those settings may cause suboptimal usage.
109///
110/// Look at the following methods for more details:
111/// - [`QuicTransportConfigBuilder::default_path_keep_alive_interval`]
112/// - [`QuicTransportConfigBuilder::default_path_max_idle_timeout`]
113/// - [`QuicTransportConfigBuilder::max_concurrent_multipath_paths`]
114/// - [`QuicTransportConfigBuilder::set_max_remote_nat_traversal_addresses`]
115///
116/// # Examples
117/// ```
118/// use std::time::Duration;
119///
120/// use iroh::endpoint::QuicTransportConfig;
121///
122/// let _cfg = QuicTransportConfig::builder()
123///     .max_idle_timeout(Duration::from_secs(35))
124///     .build();
125/// ```
126#[derive(Debug, Clone)]
127pub struct QuicTransportConfig(Arc<quinn::TransportConfig>);

150    /// Create a default [`QuicTransportConfigBuilder`].
151    fn new() -> Self {
152        let mut cfg = quinn::TransportConfig::default();
153        // Override some transport config settings.
154        cfg.keep_alive_interval(Some(HEARTBEAT_INTERVAL));
155        cfg.default_path_keep_alive_interval(Some(HEARTBEAT_INTERVAL));
156        cfg.default_path_max_idle_timeout(Some(PATH_MAX_IDLE_TIMEOUT));

158        cfg.set_max_remote_nat_traversal_addresses(MAX_MULTIPATH_PATHS as u8);
159        Self(cfg)
160    }

184    /// Maximum duration of inactivity to accept before timing out the connection.
185    ///
186    /// The true idle timeout is the minimum of this and the peer's own max idle timeout. `None`
187    /// represents an infinite timeout. Defaults to 30 seconds.
188    ///

194    /// # use iroh::endpoint::{QuicTransportConfig, VarInt, VarIntBoundsExceeded};
195    /// # fn main() -> Result<(), VarIntBoundsExceeded> {
196    /// let mut config = QuicTransportConfig::default();
197    ///
198    /// // Set the idle timeout as `VarInt`-encoded milliseconds
199    /// config.max_idle_timeout(Some(VarInt::from_u32(10_000).into()));

281    /// The initial value to be used as the maximum UDP payload size before running MTU discovery

294    /// The maximum UDP payload size guaranteed to be supported by the network.
295    ///
296    /// Must be at least 1200, which is the default, and lower than or equal to
297    /// [`QuicTransportConfigBuilder::initial_mtu`].
298    ///
299    /// Real-world MTUs can vary according to ISP, VPN, and properties of intermediate network links
300    /// outside of either endpoint's control. Extreme care should be used when raising this value
301    /// outside of private networks where these factors are fully controlled. If the provided value
302    /// is higher than what the network path actually supports, the result will be unpredictable and
303    /// catastrophic packet loss, without a possibility of repair. Prefer
304    /// [`QuicTransportConfigBuilder::initial_mtu`] together with
305    /// [`QuicTransportConfigBuilder::mtu_discovery_config`] to set a maximum UDP payload size that robustly

456    /// Enables the Multipath Extension for QUIC.
457    ///
458    /// Setting this to any nonzero value will enable the Multipath Extension for QUIC,
459    /// <https://datatracker.ietf.org/doc/draft-ietf-quic-multipath/>.

461    /// The value provided specifies the number maximum number of paths this endpoint may open

467        if max_concurrent < MAX_MULTIPATH_PATHS + 1 {
468            warn!(
469                "QuicTransportConfig::max_concurrent_multipath_paths must be at minimum {}, ignoring user supplied value",
470                MAX_MULTIPATH_PATHS + 1

474        self.0.max_concurrent_multipath_paths(max_concurrent);
475        self

584/// Parameters governing incoming connections
585///
586/// Default values should be suitable for most internet applications.
587///
588/// Use a [`ServerConfigBuilder`] to adjust the default values.
589///
590/// To create a [`ServerConfig`] compatible with your [`Endpoint`] identity, use the [`Endpoint::create_server_config_builder`] method.
591///
592/// [`Endpoint`]: crate::Endpoint
593/// [`Endpoint::create_server_config_builder`]: crate::Endpoint::create_server_config_builder
594// Note: used in `iroh::endpoint::connection::Incoming::accept_with`
595// This is new-typed since `quinn::ServerConfig` takes a `TransportConfig`, which we new-type as a `QuicTransportConfig`
596#[derive(Debug, Clone)]
597pub struct ServerConfig(Arc<quinn::ServerConfig>);
598
599impl ServerConfig {
600    pub(crate) fn to_inner_arc(&self) -> Arc<quinn::ServerConfig> {
601        self.0.clone()
602    }
603}
604
605impl ServerConfigBuilder {
606    /// Build a [`ServerConfig`] from a [`ServerConfigBuilder`].
607    pub fn build(self) -> ServerConfig {
608        ServerConfig(Arc::new(self.inner))
609    }
610
611    pub(crate) fn new(inner: quinn::ServerConfig, transport: QuicTransportConfig) -> Self {
612        Self { inner, transport }
613    }
614
615    /// Sets a custom [`QuicTransportConfig`].
616    pub fn set_transport_config(mut self, transport: QuicTransportConfig) -> Self {
617        self.inner.transport_config(transport.to_inner_arc());
618        self.transport = transport;
619        self
620    }
621
622    /// Sets a custom [`ValidationTokenConfig`].
623    pub fn set_validation_token_config(mut self, validation_token: ValidationTokenConfig) -> Self {
624        self.inner.validation_token_config(validation_token);
625        self
626    }
627
628    /// Private key used to authenticate data included in handshake tokens
629    pub fn set_token_key(mut self, value: Arc<dyn HandshakeTokenKey>) -> Self {
630        self.inner.token_key(value);
631        self
632    }
633
634    /// Duration after a retry token was issued for which it's considered valid
635    ///
636    /// Defaults to 15 seconds.
637    pub fn set_retry_token_lifetime(mut self, value: Duration) -> Self {
638        self.inner.retry_token_lifetime(value);
639        self
640    }
641
642    /// Maximum number of [`Incoming`] to allow to exist at a time.
643    ///
644    /// An [`Incoming`] comes into existence when an incoming connection attempt
645    /// is received and stops existing when the application either accepts it or otherwise disposes
646    /// of it. While this limit is reached, new incoming connection attempts are immediately
647    /// refused. Larger values have greater worst-case memory consumption, but accommodate greater
648    /// application latency in handling incoming connection attempts.
649    ///
650    /// The default value is set to 65536. With a typical Ethernet MTU of 1500 bytes, this limits
651    /// memory consumption from this to under 100 MiB--a generous amount that still prevents memory
652    /// exhaustion in most contexts.
653    ///
654    /// [`Incoming`]: crate::endpoint::Incoming
655    pub fn set_max_incoming(mut self, max_incoming: usize) -> Self {
656        self.inner.max_incoming(max_incoming);
657        self
658    }
659
660    /// Maximum number of received bytes to buffer for each [`Incoming`].
661    ///
662    /// An [`Incoming`] comes into existence when an incoming connection attempt
663    /// is received and stops existing when the application either accepts it or otherwise disposes
664    /// of it. This limit governs only packets received within that period, and does not include
665    /// the first packet. Packets received in excess of this limit are dropped, which may cause
666    /// 0-RTT or handshake data to have to be retransmitted.
667    ///
668    /// The default value is set to 10 MiB--an amount such that in most situations a client would
669    /// not transmit that much 0-RTT data faster than the server handles the corresponding
670    /// [`Incoming`].
671    ///
672    /// [`Incoming`]: crate::endpoint::Incoming
673    pub fn set_incoming_buffer_size(mut self, incoming_buffer_size: u64) -> Self {
674        self.inner.incoming_buffer_size(incoming_buffer_size);
675        self
676    }
677
678    /// Maximum number of received bytes to buffer for all [`Incoming`]
679    /// collectively.
680    ///
681    /// An [`Incoming`] comes into existence when an incoming connection attempt
682    /// is received and stops existing when the application either accepts it or otherwise disposes
683    /// of it. This limit governs only packets received within that period, and does not include
684    /// the first packet. Packets received in excess of this limit are dropped, which may cause
685    /// 0-RTT or handshake data to have to be retransmitted.
686    ///
687    /// The default value is set to 100 MiB--a generous amount that still prevents memory
688    /// exhaustion in most contexts.
689    ///
690    /// [`Incoming`]: crate::endpoint::Incoming
691    pub fn set_incoming_buffer_size_total(mut self, incoming_buffer_size_total: u64) -> Self {
692        self.inner
693            .incoming_buffer_size_total(incoming_buffer_size_total);
694        self
695    }
696
697    /// Object to get current [`SystemTime`].
698    ///
699    /// This exists to allow system time to be mocked in tests, or wherever else desired.
700    ///
701    /// Defaults to [`quinn::StdSystemTime`], which simply calls [`SystemTime::now()`](std::time::SystemTime::now).
702    ///
703    /// [`SystemTime`]: std::time::SystemTime
704    pub fn set_time_source(mut self, time_source: Arc<dyn TimeSource>) -> Self {
705        self.inner.time_source(time_source);
706        self
707    }
