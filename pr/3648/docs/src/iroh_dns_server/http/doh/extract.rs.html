34/// A DNS packet encoding type
35#[derive(Debug)]
36pub enum DnsMimeType {
37    /// application/dns-message
38    Message,
39    /// application/dns-json
40    Json,
41}
42
43impl Display for DnsMimeType {
44    fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
45        match self {
46            DnsMimeType::Message => write!(f, "application/dns-message"),
47            DnsMimeType::Json => write!(f, "application/dns-json"),
48        }
49    }
50}
51
52impl DnsMimeType {
53    /// Turn this mime type to an `Accept` HTTP header value
54    pub fn to_header_value(&self) -> HeaderValue {
55        HeaderValue::from_static(match self {
56            Self::Message => "application/dns-message",
57            Self::Json => "application/dns-json",
58        })
59    }
60}

67// See: https://developers.google.com/speed/public-dns/docs/doh/json#supported_parameters
68#[derive(Debug, Deserialize)]
69pub struct DnsQuery {
70    /// Record name to look up, e.g. example.com
71    pub name: String,
72    /// Record type, e.g. A/AAAA/TXT, etc.
73    #[serde(rename = "type")]
74    pub record_type: Option<String>,
75    /// Used to disable DNSSEC validation
76    pub cd: Option<bool>,
77    /// Desired content type. E.g. "application/dns-message" or "application/dns-json"
78    #[allow(dead_code)]
79    pub ct: Option<String>,
80    /// Whether to return DNSSEC entries such as RRSIG, NSEC or NSEC3
81    #[serde(rename = "do")]
82    pub dnssec_ok: Option<bool>,
83    /// Privacy setting for how your IP address is forwarded to authoritative nameservers
84    #[allow(dead_code)]
85    pub edns_client_subnet: Option<String>,
86    /// Some url-safe random characters to pad your messages for privacy (to avoid being fingerprinted by encrypted message length)
87    #[allow(dead_code)]
88    pub random_padding: Option<String>,
89    /// Whether to provide answers for all records up to the root
90    #[serde(rename = "rd")]
91    pub recursion_desired: Option<bool>,
92}

94/// A DNS request encoded in the query string
95#[derive(Debug)]
96pub struct DnsRequestQuery(pub(crate) DNSRequest, pub(crate) DnsMimeType);

98/// A DNS request encoded in the body
99#[derive(Debug)]
100pub struct DnsRequestBody(pub(crate) DNSRequest);

102impl<S> FromRequestParts<S> for DnsRequestQuery
103where
104    S: Send + Sync,
105{
106    type Rejection = AppError;
107
108    #[allow(clippy::manual_async_fn)]
109    fn from_request_parts(
110        parts: &mut Parts,
111        state: &S,
112    ) -> impl Future<Output = Result<Self, Self::Rejection>> + Send {
113        async move {
114            let ConnectInfo(src_addr) = ConnectInfo::from_request_parts(parts, state).await?;
115
116            match parts.headers.get(header::ACCEPT) {
117                Some(content_type) if content_type == "application/dns-message" => {
118                    handle_dns_message_query(parts, state, src_addr).await
119                }
120                Some(content_type) if content_type == "application/dns-json" => {
121                    handle_dns_json_query(parts, state, src_addr).await
122                }
123                Some(content_type) if content_type == "application/x-javascript" => {
124                    handle_dns_json_query(parts, state, src_addr).await
125                }
126                None => handle_dns_message_query(parts, state, src_addr).await,
127                _ => Err(AppError::with_status(StatusCode::NOT_ACCEPTABLE)),
128            }
129        }
130    }
131}

133impl<S> FromRequest<S> for DnsRequestBody
134where
135    S: Send + Sync,
136{
137    type Rejection = AppError;
138
139    #[allow(clippy::manual_async_fn)]
140    fn from_request(
141        req: Request<Body>,
142        state: &S,
143    ) -> impl Future<Output = Result<Self, Self::Rejection>> + Send {
144        async move {
145            let (mut parts, body) = req.into_parts();
146
147            let ConnectInfo(src_addr) = ConnectInfo::from_request_parts(&mut parts, state).await?;
148
149            let req = Request::from_parts(parts, body);
150
151            let body = Bytes::from_request(req, state)
152                .await
153                .map_err(|_| AppError::with_status(StatusCode::INTERNAL_SERVER_ERROR))?;
154
155            let request = decode_request(&body, src_addr)?;
156
157            Ok(DnsRequestBody(request))
158        }
159    }
160}

162async fn handle_dns_message_query<S>(
163    parts: &mut Parts,
164    state: &S,
165    src_addr: SocketAddr,
166) -> Result<DnsRequestQuery, AppError>
167where
168    S: Send + Sync,
169{
170    let Query(params) = Query::<DnsMessageQuery>::from_request_parts(parts, state).await?;
171
172    let buf = base64_url::decode(params.dns.as_bytes())
173        .map_err(|err| AppError::new(StatusCode::BAD_REQUEST, Some(err)))?;
174
175    let request = decode_request(&buf, src_addr)?;
176
177    Ok(DnsRequestQuery(request, DnsMimeType::Message))
178}
179
180async fn handle_dns_json_query<S>(
181    parts: &mut Parts,
182    state: &S,
183    src_addr: SocketAddr,
184) -> Result<DnsRequestQuery, AppError>
185where
186    S: Send + Sync,
187{
188    let Query(dns_query) = Query::<DnsQuery>::from_request_parts(parts, state).await?;
189
190    let request = encode_query_as_request(dns_query, src_addr)?;
191
192    Ok(DnsRequestQuery(request, DnsMimeType::Json))
193}

195/// Exposed to make it usable internally...
196pub(crate) fn encode_query_as_request(
197    question: DnsQuery,
198    src_addr: SocketAddr,
199) -> Result<DNSRequest, AppError> {
200    let query_type = if let Some(record_type) = question.record_type {
201        record_type
202            .parse::<u16>()
203            .map(proto::rr::RecordType::from)
204            .or_else(|_| FromStr::from_str(&record_type.to_uppercase()))
205            .map_err(|err| AppError::new(StatusCode::BAD_REQUEST, Some(err)))?
206    } else {
207        proto::rr::RecordType::A
208    };
209
210    let name = proto::rr::Name::from_utf8(question.name)
211        .map_err(|err| AppError::new(StatusCode::BAD_REQUEST, Some(err)))?;
212
213    let query = proto::op::Query::query(name, query_type);
214
215    let mut message = proto::op::Message::new();
216
217    message
218        .add_query(query)
219        .set_message_type(proto::op::MessageType::Query)
220        .set_op_code(proto::op::OpCode::Query)
221        .set_checking_disabled(question.cd.unwrap_or(false))
222        .set_recursion_desired(question.recursion_desired.unwrap_or(true))
223        .set_recursion_available(true)
224        .set_authentic_data(question.dnssec_ok.unwrap_or(false));
225
226    // This is kind of a hack, but the only way I can find to
227    // create a MessageRequest is by decoding a buffer of bytes,
228    // so we encode the message into a buffer and then decode it
229    let mut buf = Vec::with_capacity(4096);
230    let mut encoder = BinEncoder::new(&mut buf);
231
232    message
233        .emit(&mut encoder)
234        .map_err(|err| AppError::new(StatusCode::BAD_REQUEST, Some(err)))?;
235
236    let request = decode_request(&buf, src_addr)?;
237
238    Ok(request)
239}
240
241fn decode_request(bytes: &[u8], src_addr: SocketAddr) -> Result<DNSRequest, AppError> {
242    let mut decoder = BinDecoder::new(bytes);
243
244    match MessageRequest::read(&mut decoder) {
245        Ok(message) => {
246            info!("received message {message:?}");
247            if message.message_type() != proto::op::MessageType::Query {
248                return Err(AppError::new(
249                    StatusCode::BAD_REQUEST,
250                    Some("Invalid message type: expected query"),
251                ));
252            }
253
254            let request = DNSRequest::new(message, src_addr, Protocol::Https);
255
256            Ok(request)
257        }
258        Err(err) => Err(AppError::new(
259            StatusCode::BAD_REQUEST,
260            Some(format!("Invalid DNS message: {err}")),
261        )),
262    }
