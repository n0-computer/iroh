26/// Errors when spawning the H3 relay server.
27#[allow(missing_docs)]
28#[stack_error(derive, add_meta)]
29#[non_exhaustive]
30pub enum H3SpawnError {
31    #[error(transparent)]
32    NoInitialCipherSuite {
33        #[error(std_err, from)]
34        source: NoInitialCipherSuite,
35    },
36    #[error("Unable to spawn QUIC endpoint for H3 relay")]
37    EndpointServer {
38        #[error(std_err)]
39        source: std::io::Error,
40    },
41    #[error("Unable to get local address from H3 endpoint")]
42    LocalAddr {
43        #[error(std_err)]
44        source: std::io::Error,
45    },
46}

48/// A running H3/WebTransport relay server.
49#[derive(Debug)]
50pub struct H3RelayServer {
51    bind_addr: SocketAddr,
52    cancel: CancellationToken,
53    handle: AbortOnDropHandle<()>,
54}
55
56impl H3RelayServer {
57    /// Returns the bound socket address.
58    pub fn bind_addr(&self) -> SocketAddr {
59        self.bind_addr
60    }
61
62    /// Returns a handle for controlling this server.
63    pub fn handle(&self) -> H3ServerHandle {
64        H3ServerHandle {
65            cancel: self.cancel.clone(),
66        }
67    }
68
69    /// Returns a mutable reference to the task handle.
70    pub fn task_handle(&mut self) -> &mut AbortOnDropHandle<()> {
71        &mut self.handle
72    }
73
74    /// Spawns the H3/WebTransport relay server.
75    ///
76    /// The `service` is the same [`RelayService`] that backs the WebSocket server,
77    /// so both transports register into one [`Clients`] registry and share the
78    /// access control, key cache, and metrics. `bind_addr` and `server_config` are
79    /// the QUIC-specific bind address and TLS configuration.
80    pub(crate) fn spawn(
81        bind_addr: SocketAddr,
82        server_config: rustls::ServerConfig,
83        service: RelayService,
84    ) -> Result<Self, H3SpawnError> {
85        let mut server_tls_config = server_config;
86        server_tls_config.alpn_protocols = vec![ALPN_RELAY_H3.to_vec()];
87
88        let quic_server_config = QuicServerConfig::try_from(server_tls_config)?;
89        let mut server_config = noq::ServerConfig::with_crypto(Arc::new(quic_server_config));
90        let transport_config = Arc::get_mut(&mut server_config.transport).expect("not used yet");
91        // Uni streams: high limit for per-message uni streams.
92        // Bidi streams: 1 for the CONNECT session.
93        transport_config
94            .max_concurrent_uni_streams(256u32.into())
95            .max_concurrent_bidi_streams(2_u8.into());
96
97        let endpoint = noq::Endpoint::server(server_config, bind_addr)
98            .map_err(|err| e!(H3SpawnError::EndpointServer, err))?;
99        let bind_addr = endpoint
100            .local_addr()
101            .map_err(|err| e!(H3SpawnError::LocalAddr, err))?;
102
103        info!(?bind_addr, "H3/WT relay server listening");
104
105        let cancel = CancellationToken::new();
106        let cancel_loop = cancel.clone();
107
108        let clients = service.clients().clone();
109        let key_cache = service.key_cache().clone();
110        let access = service.access().clone();
111        let metrics = service.metrics().clone();
112
113        let task = tokio::task::spawn(
114            async move {
115                let mut set = JoinSet::new();
116                debug!("waiting for connections");
117                loop {
118                    tokio::select! {
119                        biased;
120                        _ = cancel_loop.cancelled() => {
121                            break;
122                        }
123                        Some(res) = set.join_next() => {
124                            if let Err(err) = res
125                                && err.is_panic()
126                            {
127                                panic!("H3/WT relay task panicked: {err:#?}");
128                            }
129                        }
130                        res = endpoint.accept() => match res {
131                            Some(incoming) => {
132                                let remote_addr = incoming.remote_address();
133                                debug!(%remote_addr, "accepting QUIC connection");
134                                let clients = clients.clone();
135                                let key_cache = key_cache.clone();
136                                let metrics = metrics.clone();
137                                let access = access.clone();
138                                set.spawn(
139                                    async move {
140                                        if let Err(err) = handle_wt_connection(
141                                            incoming, clients, key_cache, access, metrics,
142                                        ).await {
143                                            warn!("WT connection error: {err:#}");
144                                        }
145                                    }
146                                    .instrument(info_span!("wt-relay-conn", %remote_addr))
147                                );
148                            }
149                            None => {
150                                debug!("endpoint closed");
151                                break;
152                            }
153                        }
154                    }
155                }
156                endpoint.close(0u32.into(), b"server shutdown");
157                endpoint.wait_idle().await;
158                set.abort_all();
159                while !set.is_empty() {
160                    _ = set.join_next().await;
161                }
162                debug!("H3/WT relay server shut down");
163            }
164            .instrument(info_span!("h3-wt-relay-serve")),
165        );
166
167        Ok(Self {
168            bind_addr,
169            cancel,
170            handle: AbortOnDropHandle::new(task),
171        })
172    }
173
174    /// Gracefully shuts down the server.
175    pub async fn shutdown(mut self) {
176        self.cancel.cancel();
177        if !self.task_handle().is_finished() {
178            _ = self.task_handle().await;
179        }
180    }
181}

196/// Connection handling errors.
197#[stack_error(derive, add_meta)]
198#[non_exhaustive]
199enum WtConnectionError {
200    #[error(transparent)]
201    Quic {
202        #[error(from, std_err)]
203        source: noq::ConnectionError,
204    },
205    /// Error decoding the WebTransport settings sent by the client.
206    ///
207    /// The concrete error type is `web_transport_proto::SettingsError`. Use
208    /// [`AnyError::downcast_ref`] to recover it. Note that the concrete downcast
209    /// type is not covered by any semver guarantees and may change between releases.
210    #[error("WebTransport settings error")]
211    Settings { source: AnyError },
212    /// Error encoding or decoding a WebTransport CONNECT message.
213    ///
214    /// The concrete error type is `web_transport_proto::ConnectError`. Use
215    /// [`AnyError::downcast_ref`] to recover it. Note that the concrete downcast
216    /// type is not covered by any semver guarantees and may change between releases.
217    #[error("WebTransport CONNECT error")]
218    Connect { source: AnyError },
219    #[error(transparent)]
220    Handshake {
221        #[error(from, std_err)]
222        source: handshake::Error,
223    },
224    #[error(transparent)]
225    StreamWrite {
226        #[error(from, std_err)]
227        source: noq::WriteError,
228    },
229    #[error("invalid WebTransport CONNECT request")]
230    Http {
231        #[error(from, std_err)]
232        source: http::Error,
233    },
234}

236/// Handle a single QUIC connection and serve WebTransport relay sessions.
237async fn handle_wt_connection(
238    incoming: noq::Incoming,
239    clients: Clients,
240    key_cache: KeyCache,
241    access: Arc<dyn DynAccessControl>,
242    metrics: Arc<Metrics>,
243) -> Result<(), WtConnectionError> {
244    let conn = incoming.await?;
245
246    trace!("WT srv: QUIC connection established");
247
248    let mut server_settings = wt::Settings::default();
249    server_settings.enable_webtransport(1);
250
251    let mut settings_buf = bytes::BytesMut::new();
252    server_settings.encode(&mut settings_buf);
253
254    let (send_result, recv_result) = tokio::join!(
255        async {
256            trace!("WT srv: sending settings");
257            let mut uni = conn.open_uni().await?;
258            uni.write_all(&settings_buf).await?;
259            trace!("WT srv: settings sent");
260            Ok::<_, WtConnectionError>(())
261        },
262        async {
263            trace!("WT srv: waiting for client settings");
264            let uni = conn.accept_uni().await?;
265            let mut uni = tokio::io::BufReader::new(uni);
266            let s = wt::Settings::read(&mut uni)
267                .await
268                .map_err(|err| e!(WtConnectionError::Settings, anyerr!(err)))?;
269            trace!("WT srv: client settings received");
270            Ok::<_, WtConnectionError>(s)
271        }
272    );
273    send_result?;
274    let client_settings = recv_result?;
275
276    if client_settings.supports_webtransport() == 0 {
277        warn!("client does not support WebTransport");
278        return Ok(());
279    }
280
281    trace!("WT srv: waiting for CONNECT");
282    let (mut send, mut recv) = conn.accept_bi().await?;
283    // Record the CONNECT stream's QUIC stream ID as the session ID.
284    let session_id: u64 = send.id().into();
285    let connect_req = wt::ConnectRequest::read(&mut recv)
286        .await
287        .map_err(|err| e!(WtConnectionError::Connect, anyerr!(err)))?;
288    trace!(url = %connect_req.url, session_id, "WT srv: CONNECT received");
289
290    if connect_req.url.path() != RELAY_PATH {
291        warn!(path = %connect_req.url.path(), "invalid path, expected {RELAY_PATH}");
292        let mut buf = bytes::BytesMut::new();
293        let _ = wt::ConnectResponse::new(http::StatusCode::NOT_FOUND).encode(&mut buf);
294        let _ = send.write_all(&buf).await;
295        return Ok(());
296    }
297
298    let protocol_version = connect_req
299        .protocols
300        .iter()
301        .filter_map(|s| ProtocolVersion::match_from_str(s.as_str()))
302        .max();
303    let Some(protocol_version) = protocol_version else {
304        warn!("unsupported or missing relay subprotocol");
305        let mut buf = bytes::BytesMut::new();
306        let _ = wt::ConnectResponse::new(http::StatusCode::BAD_REQUEST).encode(&mut buf);
307        let _ = send.write_all(&buf).await;
308        return Ok(());
309    };
310
311    let client_auth_header = connect_req
312        .headers
313        .get(CLIENT_AUTH_HEADER.as_str())
314        .cloned();
315
316    let resp = wt::ConnectResponse::OK.with_protocol(protocol_version.to_string());
317    let mut resp_buf = bytes::BytesMut::new();
318    resp.encode(&mut resp_buf)
319        .map_err(|err| e!(WtConnectionError::Connect, anyerr!(err)))?;
320    trace!("WT srv: sending CONNECT response");
321    send.write_all(&resp_buf).await?;
322
323    // Use uni streams for relay messages (one stream per message).
324    let mut io = WtBytesFramed::new(conn.clone(), session_id);
325
326    trace!("WT srv: starting relay handshake");
327    let authentication = handshake::serverside(&mut io, client_auth_header).await?;
328
329    trace!(?authentication.mechanism, "verified authentication");
330
331    // Build a `ClientRequest` from the WebTransport CONNECT request and authorize it
332    // against the configured `AccessControl`, mirroring the WebSocket accept path.
333    let request_parts = {
334        let mut request = http::Request::builder()
335            .uri(connect_req.url.as_str())
336            .body(())?;
337        *request.headers_mut() = connect_req.headers;
338        request.into_parts().0
339    };
340    let request = ClientRequest::new(authentication.client_key, protocol_version, request_parts);
341    let guard = authentication
342        .authorize_with(&request, &access, &mut io)
343        .await?;
344
345    trace!("verified authorization");
346
347    let io = RelayedStream::new(io, key_cache.clone());
348
349    let endpoint_id = request.endpoint_id();
350    let client_conn = Config::new(guard, io, protocol_version);
351
352    trace!(endpoint_id = %endpoint_id.fmt_short(), "registering client");
353
354    clients.register(client_conn, metrics.clone());
355
356    debug!(endpoint_id = %endpoint_id.fmt_short(), "client registered");
357
358    // Keep the connection alive until the client disconnects.
359    conn.closed().await;
360
361    Ok(())
362}
