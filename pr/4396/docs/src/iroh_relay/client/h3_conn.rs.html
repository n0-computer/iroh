1//! WebTransport client connection for the relay protocol.
2//!
3//! This is the QUIC/WebTransport counterpart to the WebSocket transport in the
4//! parent [`client`](super) module. Both speak the same relay protocol and share
5//! the same [`handshake`], so a WebTransport client can relay to a WebSocket peer
6//! and vice versa. WebTransport reaches the first relay byte in ~2 RTTs where
7//! WebSocket needs 3-4 (TCP + TLS + HTTP upgrade + relay auth).
8//!
9//! Connecting happens in two phases so the [`ClientBuilder`] can race the QUIC
10//! handshake against a WebSocket connect and keep whichever wins:
11//!
12//! 1. [`quic_connect`] performs the QUIC handshake (1 RTT, TLS 1.3 included). The
13//!    [`ClientBuilder`] runs this Happy Eyeballs style across the resolved relay
14//!    addresses, the same way the WebSocket path dials.
15//! 2. [`wt_handshake`] completes the WebTransport session on the established
16//!    connection. Per RFC 9114 section 7.2.4.2 it does not wait for the peer's
17//!    settings before sending: client `SETTINGS` (on a uni stream) and the
18//!    WebTransport `CONNECT` request (on a bidi stream) go out in the first
19//!    flight while the server's `SETTINGS` are read concurrently. The `CONNECT`
20//!    carries the relay subprotocol and, when the connection exports keying
21//!    material, an RFC 5705 auth header so the relay [`handshake`] adds no extra
22//!    RTT. Once the server accepts, relay messages flow over [`WtBytesFramed`],
23//!    one QUIC uni stream per message.
24//!
25//! If the server does not speak WebTransport (its settings disable it, or the
26//! `CONNECT` is rejected) the caller falls back to WebSocket.
27//!
28//! [`ClientBuilder`]: super::ClientBuilder
29//! [`handshake`]: crate::protos::handshake
30

61/// Errors during WebTransport relay connection establishment.
62#[stack_error(derive, add_meta)]
63#[allow(missing_docs)]
64#[non_exhaustive]
65pub enum H3ConnectError {
66    #[error(transparent)]
67    QuicConnect {
68        #[error(from, std_err)]
69        source: noq::ConnectError,
70    },
71    #[error(transparent)]
72    QuicConnection {
73        #[error(from, std_err)]
74        source: noq::ConnectionError,
75    },
76    /// Error decoding the WebTransport settings sent by the server.
77    ///
78    /// The concrete error type is `web_transport_proto::SettingsError`. Use
79    /// [`AnyError::downcast_ref`] to recover it. Note that the concrete downcast
80    /// type is not covered by any semver guarantees and may change between releases.
81    #[error("WebTransport settings error")]
82    Settings { source: AnyError },
83    /// Error encoding or decoding a WebTransport CONNECT message.
84    ///
85    /// The concrete error type is `web_transport_proto::ConnectError`. Use
86    /// [`AnyError::downcast_ref`] to recover it. Note that the concrete downcast
87    /// type is not covered by any semver guarantees and may change between releases.
88    #[error("WebTransport CONNECT error")]
89    Connect { source: AnyError },
90    /// The server does not support WebTransport.
91    #[error("Server does not support WebTransport")]
92    WebTransportUnsupported {},
93    #[error("Server rejected session: {status}")]
94    Rejected { status: http::StatusCode },
95    #[error(transparent)]
96    Handshake {
97        #[error(from, std_err)]
98        source: handshake::Error,
99    },
100    #[error(transparent)]
101    StreamWrite {
102        #[error(from, std_err)]
103        source: noq::WriteError,
104    },
105    #[error("No local address available")]
106    NoLocalAddr {},
107    #[error(transparent)]
108    EndpointCreate {
109        #[error(from, std_err)]
110        source: std::io::Error,
111    },
112    #[error("QUIC connect timed out")]
113    ConnectTimeout {},
114}

116/// Result of the QUIC connect phase. Passed to [`wt_handshake`] to complete
117/// the WebTransport session setup.
118pub(crate) struct QuicConnected {
119    pub conn: noq::Connection,
120    pub local_addr: SocketAddr,
121}
122
123/// Phase 1: establish a QUIC connection (first server response).
124///
125/// Returns as soon as the QUIC handshake completes, confirming the server
126/// speaks QUIC on this port. This is the earliest signal for the WS/WT race.
127pub(crate) async fn quic_connect(
128    server_addr: SocketAddr,
129    server_name: &str,
130    tls_config: rustls::ClientConfig,
131) -> Result<QuicConnected, H3ConnectError> {
132    let bind_addr = if server_addr.is_ipv6() {
133        SocketAddr::from((Ipv6Addr::UNSPECIFIED, 0))
134    } else {
135        SocketAddr::from((Ipv4Addr::UNSPECIFIED, 0))
136    };
137
138    let endpoint = noq::Endpoint::client(bind_addr)?;
139    let local_addr = endpoint
140        .local_addr()
141        .map_err(|_| e!(H3ConnectError::NoLocalAddr))?;
142
143    let mut tls_config = tls_config;
144    tls_config.alpn_protocols = vec![ALPN_RELAY_H3.to_vec()];
145    let quic_client_config = QuicClientConfig::try_from(tls_config).map_err(|_| {
146        e!(H3ConnectError::EndpointCreate {
147            source: std::io::Error::new(
148                std::io::ErrorKind::InvalidInput,
149                "TLS config does not support TLS 1.3 (required for QUIC)",
150            )
151        })
152    })?;
153    let mut client_config = noq::ClientConfig::new(Arc::new(quic_client_config));
154    let mut transport = noq_proto::TransportConfig::default();
155    transport.max_concurrent_uni_streams(256u32.into());
156    client_config.transport_config(Arc::new(transport));
157
158    trace!(%server_addr, %server_name, "WT: QUIC connecting");
159    let connecting = endpoint.connect_with(client_config, server_addr, server_name)?;
160    let conn = tokio::time::timeout(QUIC_CONNECT_TIMEOUT, connecting)
161        .await
162        .map_err(|_| e!(H3ConnectError::ConnectTimeout))??;
163
164    trace!("WT: QUIC handshake complete");
165    Ok(QuicConnected { conn, local_addr })
166}

168/// Phase 2: complete the WebTransport handshake on an established QUIC connection.
169///
170/// Sends settings + CONNECT, receives server response, sets up the data stream,
171/// and runs the relay handshake.
172pub(crate) async fn wt_handshake(
173    quic: QuicConnected,
174    server_name: &str,
175    secret_key: &SecretKey,
176) -> Result<(WtBytesFramed, WtConnState, SocketAddr), H3ConnectError> {
177    let QuicConnected { conn, local_addr } = quic;
178
179    // Per RFC 9114 section 7.2.4.2, endpoints must not wait for peer settings
180    // before sending. We pipeline settings + CONNECT in the first flight and
181    // accept server settings concurrently. If the server does not support
182    // WebTransport, the CONNECT will be rejected and we fall back to WS.
183    let mut client_settings = wt::Settings::default();
184    client_settings.enable_webtransport(1);
185
186    // Build CONNECT request with relay subprotocol and auth header.
187    let url = format!("https://{server_name}{RELAY_PATH}");
188    let mut connect_req = wt::ConnectRequest::new(url::Url::parse(&url).expect("valid URL"))
189        .with_protocol(ProtocolVersion::default().to_string());
190
191    // Attach TLS keying material auth to avoid the challenge-response RTT.
192    if let Some(client_auth) = KeyMaterialClientAuth::new(secret_key, &conn) {
193        debug!("using TLS keying material for relay authentication");
194        connect_req = connect_req.with_header(CLIENT_AUTH_HEADER, client_auth.into_header_value());
195    }
196
197    // Pre-encode settings and CONNECT into byte buffers.
198    let mut settings_buf = bytes::BytesMut::new();
199    client_settings.encode(&mut settings_buf);
200
201    let mut connect_buf = bytes::BytesMut::new();
202    connect_req
203        .encode(&mut connect_buf)
204        .map_err(|err| e!(H3ConnectError::Connect, anyerr!(err)))?;
205
206    // Send settings and CONNECT in the same flight.
207    let send_all = async {
208        trace!("WT: opening uni stream for settings");
209        let mut uni = conn.open_uni().await?;
210        uni.write_all(&settings_buf).await?;
211        trace!("WT: settings written");
212
213        trace!("WT: opening bidi stream for CONNECT");
214        let (mut connect_send, connect_recv) = conn.open_bi().await?;
215        let session_id: u64 = connect_send.id().into();
216        connect_send.write_all(&connect_buf).await?;
217        trace!("WT: CONNECT written");
218
219        Ok::<_, H3ConnectError>((connect_recv, session_id))
220    };
221
222    let settings_recv = async {
223        trace!("WT: waiting for server settings");
224        let uni = conn.accept_uni().await?;
225        let mut uni = tokio::io::BufReader::new(uni);
226        let settings = wt::Settings::read(&mut uni)
227            .await
228            .map_err(|err| e!(H3ConnectError::Settings, anyerr!(err)))?;
229        trace!("WT: server settings received");
230        Ok::<_, H3ConnectError>(settings)
231    };
232
233    let (send_result, server_settings) = tokio::join!(send_all, settings_recv);
234    let (mut recv, session_id) = send_result?;
235    let server_settings = server_settings?;
236
237    if server_settings.supports_webtransport() == 0 {
238        return Err(e!(H3ConnectError::WebTransportUnsupported));
239    }
240
241    trace!("WT: pipelined, waiting for CONNECT response");
242
243    let resp = wt::ConnectResponse::read(&mut recv)
244        .await
245        .map_err(|err| e!(H3ConnectError::Connect, anyerr!(err)))?;
246
247    trace!(status = %resp.status, "WT: CONNECT response received");
248
249    if !resp.status.is_success() {
250        return Err(e!(H3ConnectError::Rejected {
251            status: resp.status,
252        }));
253    }
254
255    // Use uni streams for relay messages (one stream per message).
256    let mut io = WtBytesFramed::new(conn.clone(), session_id);
257
258    trace!("WT: starting relay handshake");
259    handshake::clientside(&mut io, secret_key).await?;
260    trace!("WT: relay handshake complete");
261
262    let state = WtConnState { _conn: conn };
263
264    Ok((io, state, local_addr))
265}

267/// Convenience: run both phases in sequence. Used in tests.
268#[cfg(test)]
269pub(crate) async fn connect_h3(
270    server_addr: SocketAddr,
271    server_name: &str,
272    tls_config: rustls::ClientConfig,
273    secret_key: &SecretKey,
274) -> Result<(WtBytesFramed, WtConnState, SocketAddr), H3ConnectError> {
275    let quic = quic_connect(server_addr, server_name, tls_config).await?;
276    wt_handshake(quic, server_name, secret_key).await
277}
