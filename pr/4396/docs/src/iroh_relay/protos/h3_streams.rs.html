1//! WebTransport stream adapter for the relay protocol.
2//!
3//! [`WtBytesFramed`] uses a new unidirectional QUIC stream per relay message.
4//! Each stream carries `[Frame::WEBTRANSPORT][session_id][payload]`. The stream
5//! is finished after the payload, so the receiver reads to EOF. Successive send
6//! streams get increasing priority so the QUIC scheduler prefers newer messages
7//! over retransmissions of older ones.
8//!
9//! [`BytesStreamSink`]: super::streams::BytesStreamSink
10
11use std::{
12    pin::Pin,
13    task::{Context, Poll},
14};

17use n0_error::{StdResultExt, anyerr};

19use tokio_util::sync::ReusableBoxFuture;

22use super::streams::StreamError;
23use crate::{ExportKeyingMaterial, MAX_PACKET_SIZE};

25/// Maximum bytes to read from a single uni stream before rejecting.
26const MAX_UNI_STREAM_SIZE: usize = MAX_PACKET_SIZE + 64;
27
28/// Relay transport using one unidirectional QUIC stream per message.
29///
30/// Each message is sent on a fresh uni stream with a WT session header and
31/// the raw payload. The receiver accepts uni streams and reads each to EOF.
32/// This eliminates head-of-line blocking: retransmission on one stream does
33/// not delay delivery of later messages on other streams.
34pub struct WtBytesFramed {
35    conn: noq::Connection,
36    session_id: u64,
37    pending_send: Option<Bytes>,
38    send_fut: ReusableBoxFuture<'static, Result<(), StreamError>>,
39    recv_fut: ReusableBoxFuture<'static, Result<Bytes, StreamError>>,
40    send_busy: bool,
41    send_priority: i32,
42}
43
44impl std::fmt::Debug for WtBytesFramed {
45    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
46        f.debug_struct("WtBytesFramed").finish()
47    }
48}

50impl WtBytesFramed {
51    /// Create from a QUIC connection and the WebTransport session ID.
52    pub fn new(conn: noq::Connection, session_id: u64) -> Self {
53        let recv_conn = conn.clone();
54        let hdr_len = wt_header_len(session_id);
55        Self {
56            conn,
57            session_id,
58            pending_send: None,
59            send_fut: ReusableBoxFuture::new(std::future::pending()),
60            recv_fut: ReusableBoxFuture::new(recv_one_message(recv_conn, hdr_len)),
61            send_busy: false,
62            send_priority: 0,
63        }
64    }
65}
66
67/// Encode the WT uni stream header for the given session ID.
68fn encode_wt_header(session_id: u64) -> BytesMut {
69    let mut hdr = BytesMut::with_capacity(16);
70    wt::Frame::WEBTRANSPORT.encode(&mut hdr);
71    wt::VarInt::from_u64(session_id)
72        .expect("session ID fits in varint")
73        .encode(&mut hdr);
74    hdr
75}
76
77fn wt_header_len(session_id: u64) -> usize {
78    encode_wt_header(session_id).len()
79}

113/// Accept a uni stream, skip the WT header, read the payload to EOF.
114async fn recv_one_message(
115    conn: noq::Connection,
116    wt_header_len: usize,
117) -> Result<Bytes, StreamError> {
118    let mut recv = conn.accept_uni().await.anyerr()?;
119    let data = recv.read_to_end(MAX_UNI_STREAM_SIZE).await.anyerr()?;
120    if data.len() < wt_header_len {
121        return Err(anyerr!("uni stream shorter than WT header",));
122    }
123    Ok(Bytes::from(data).slice(wt_header_len..))
124}
125
126/// Open a uni stream, write WT header and payload, finish the stream.
127async fn send_one_message(
128    conn: noq::Connection,
129    session_id: u64,
130    priority: i32,
131    payload: Bytes,
132) -> Result<(), StreamError> {
133    let mut stream = conn.open_uni().await.anyerr()?;
134    let _ = stream.set_priority(priority);
135    stream
136        .write_chunk(encode_wt_header(session_id).freeze())
137        .await
138        .anyerr()?;
139    stream.write_chunk(payload).await.anyerr()?;
140    stream.finish().anyerr()?;
141    Ok(())
142}

144// -- Stream: accept uni streams, read each to EOF -----------------------------
145
146impl Stream for WtBytesFramed {
147    type Item = Result<Bytes, StreamError>;
148
149    fn poll_next(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
150        let this = self.get_mut();
151
152        match ready!(this.recv_fut.poll(cx)) {
153            Ok(payload) => {
154                // Immediately set up the next recv.
155                let conn = this.conn.clone();
156                let hdr_len = wt_header_len(this.session_id);
157                this.recv_fut.set(recv_one_message(conn, hdr_len));
158                // The relay protocol never frames an empty message, so an empty
159                // payload is malformed. Yield it and let the frame decoder reject
160                // it; returning `Pending` here would strand the stream, since the
161                // freshly armed `recv_fut` has not been polled to register a waker.
162                Poll::Ready(Some(Ok(payload)))
163            }
164            Err(e) => {
165                // Connection closed or error. Don't set up a new recv.
166                Poll::Ready(Some(Err(e)))
167            }
168        }
169    }
170}

172// -- Sink: new uni stream per message -----------------------------------------
173
174impl Sink<Bytes> for WtBytesFramed {
175    type Error = StreamError;
176
177    fn poll_ready(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
178        let this = self.get_mut();
179
180        // Flush in-progress send.
181        if this.send_busy {
182            match ready!(this.send_fut.poll(cx)) {
183                Ok(()) => {
184                    this.send_busy = false;
185                }
186                Err(e) => {
187                    this.send_busy = false;
188                    return Poll::Ready(Err(e));
189                }
190            }
191        }
192
193        // Start sending a pending message.
194        if let Some(msg) = this.pending_send.take() {
195            let conn = this.conn.clone();
196            let session_id = this.session_id;
197            let priority = this.send_priority;
198            this.send_priority = this.send_priority.saturating_add(1);
199            this.send_fut
200                .set(send_one_message(conn, session_id, priority, msg));
201            this.send_busy = true;
202            let pin = Pin::new(this);
203            return pin.poll_ready(cx);
204        }
205
206        Poll::Ready(Ok(()))
207    }
208
209    fn start_send(self: Pin<&mut Self>, item: Bytes) -> Result<(), Self::Error> {
210        self.get_mut().pending_send = Some(item);
211        Ok(())
212    }
213
214    fn poll_flush(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
215        self.poll_ready(cx)
216    }
217
218    fn poll_close(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
219        ready!(self.poll_ready(cx)?);
220        Poll::Ready(Ok(()))
221    }
222}
