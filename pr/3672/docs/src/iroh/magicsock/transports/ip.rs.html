25fn bind_with_fallback(mut addr: SocketAddr) -> io::Result<netwatch::UdpSocket> {
26    debug!(%addr, "binding");
27
28    // First try binding a preferred port, if specified
29    match netwatch::UdpSocket::bind_full(addr) {
30        Ok(socket) => {
31            let local_addr = socket.local_addr()?;
32            debug!(%addr, %local_addr, "successfully bound");
33            return Ok(socket);
34        }
35        Err(err) => {
36            debug!(%addr, "failed to bind: {err:#}");
37            // If that was already the fallback port, then error out
38            if addr.port() == 0 {
39                return Err(err);
40            }
41        }
42    }
43
44    // Otherwise, try binding with port 0
45    addr.set_port(0);
46    netwatch::UdpSocket::bind_full(addr)
47}

72    /// NOTE: Receiving on a closed socket will return [`Poll::Pending`] indefinitely.
73    pub(super) fn poll_recv(
74        &mut self,
75        cx: &mut Context,
76        bufs: &mut [io::IoSliceMut<'_>],
77        metas: &mut [quinn_udp::RecvMeta],
78        source_addrs: &mut [Addr],
79    ) -> Poll<io::Result<usize>> {
80        match self.socket.poll_recv_quinn(cx, bufs, metas) {
81            Poll::Pending => Poll::Pending,
82            Poll::Ready(Ok(n)) => {
83                for (source_addr, meta) in source_addrs.iter_mut().zip(metas.iter_mut()).take(n) {
84                    if meta.addr.is_ipv4() {
85                        // The AsyncUdpSocket is an AF_INET6 socket and needs to show this
86                        // as coming from an IPv4-mapped IPv6 addresses, since Quinn will
87                        // use those when sending on an INET6 socket.
88                        let v6_ip = match meta.addr.ip() {
89                            IpAddr::V4(ipv4_addr) => ipv4_addr.to_ipv6_mapped(),
90                            IpAddr::V6(ipv6_addr) => ipv6_addr,
91                        };
92                        meta.addr = SocketAddr::new(v6_ip.into(), meta.addr.port());
93                    }
94                    // The transport addresses are internal to iroh and we always want those
95                    // to remain the canonical address.
96                    *source_addr =
97                        SocketAddr::new(meta.addr.ip().to_canonical(), meta.addr.port()).into();
98                }
99                Poll::Ready(Ok(n))
100            }
101            Poll::Ready(Err(err)) => Poll::Ready(Err(err)),
102        }
103    }

149    pub(super) fn rebind(&self) -> io::Result<()> {
150        let old_addr = self.local_addr.get();
151        self.socket.rebind()?;
152        let addr = self.socket.local_addr()?;
153        self.local_addr.set(addr).ok();
154        trace!("rebound from {} to {}", old_addr, addr);
155
156        Ok(())
157    }

173impl IpSender {
174    pub(super) fn is_valid_send_addr(&self, dst: &SocketAddr) -> bool {
175        // Our net-tools crate binds sockets to their specific family.  This means an IPv6
176        // socket can not sent to IPv4, on any platform.  So we need to convert and
177        // IPv4-mapped IPv6 address back to it's canonical IPv4 address.
178        let dst_ip = dst.ip().to_canonical();
179
180        #[allow(clippy::match_like_matches_macro)]
181        match (self.bind_addr.ip(), dst_ip) {
182            (IpAddr::V4(_), IpAddr::V4(_)) => true,
183            (IpAddr::V6(_), IpAddr::V6(_)) => true,
184            _ => false,
185        }
186    }
187
188    /// Creates a canonical socket address.
189    ///
190    /// We may be asked to send IPv4-mapped IPv6 addresses.  But our sockets are configured
191    /// to only send their actual family.  So we need to map those back to the canonical
192    /// addresses.
193    #[inline]
194    fn canonical_addr(addr: SocketAddr) -> SocketAddr {
195        SocketAddr::new(addr.ip().to_canonical(), addr.port())
196    }
197
198    pub(super) async fn send(
199        &self,
200        dst: SocketAddr,
201        src: Option<IpAddr>,
202        transmit: &Transmit<'_>,
203    ) -> io::Result<()> {
204        let total_bytes = transmit.contents.len() as u64;
205        let res = self
206            .sender
207            .send(&quinn_udp::Transmit {
208                destination: Self::canonical_addr(dst),
209                ecn: transmit.ecn,
210                contents: transmit.contents,
211                segment_size: transmit.segment_size,
212                src_ip: src,
213            })
214            .await;
215
216        match res {
217            Ok(res) => {
218                match dst {
219                    SocketAddr::V4(_) => {
220                        self.metrics.send_ipv4.inc_by(total_bytes);
221                    }
222                    SocketAddr::V6(_) => {
223                        self.metrics.send_ipv6.inc_by(total_bytes);
224                    }
225                }
226                Ok(res)
227            }
228            Err(err) => Err(err),
229        }
230    }
231
232    pub(super) fn poll_send(
233        mut self: Pin<&mut Self>,
234        cx: &mut std::task::Context,
235        dst: SocketAddr,
236        src: Option<IpAddr>,
237        transmit: &Transmit<'_>,
238    ) -> Poll<io::Result<()>> {
239        let total_bytes = transmit.contents.len() as u64;
240        let res = Pin::new(&mut self.sender).poll_send(
241            &quinn_udp::Transmit {
242                destination: Self::canonical_addr(dst),
243                ecn: transmit.ecn,
244                contents: transmit.contents,
245                segment_size: transmit.segment_size,
246                src_ip: src,
247            },
248            cx,
249        );
250
251        match res {
252            Poll::Ready(Ok(res)) => {
253                match dst {
254                    SocketAddr::V4(_) => {
255                        self.metrics.send_ipv4.inc_by(total_bytes);
256                    }
257                    SocketAddr::V6(_) => {
258                        self.metrics.send_ipv6.inc_by(total_bytes);
259                    }
260                }
261                Poll::Ready(Ok(res))
262            }
263            Poll::Ready(Err(err)) => Poll::Ready(Err(err)),
264            Poll::Pending => Poll::Pending,
265        }
266    }
