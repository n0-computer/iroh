1//! The reportgen actor is responsible for generating a single net_report report.
2//!
3//! It is implemented as an actor with [`Client`] as handle.
4//!
5//! The actor starts generating the report as soon as it is created, it does not receive any
6//! messages from the client.  It follows roughly these steps:
7//!
8//! - Determines host IPv6 support.
9//! - Creates hairpin actor.
10//! - Creates portmapper future.
11//! - Creates captive portal detection future.
12//! - Creates Probe Set futures.
13//!   - These send messages to the reportgen actor.
14//! - Loops driving the futures and handling actor messages:
15//!   - Disables futures as they are completed or aborted.
16//!   - Stop if there are no outstanding tasks/futures, or on timeout.
17//! - Sends the completed report to the net_report actor.

256    async fn run(&mut self) {
257        match self.run_inner().await {
258            Ok(_) => debug!("reportgen actor finished"),
259            Err(err) => {
260                self.net_report

263                    .ok();
264            }
265        }

268    /// Runs the main reportgen actor logic.
269    ///
270    /// This actor runs by:
271    ///
272    /// - Creates a hairpin actor.
273    /// - Creates a captive portal future.
274    /// - Creates ProbeSet futures in a group of futures.
275    /// - Runs a main loop:
276    ///   - Drives all the above futures.
277    ///   - Receives actor messages (sent by those futures).
278    ///   - Updates the report, cancels unneeded futures.
279    /// - Sends the report to the net_report actor.
280    async fn run_inner(&mut self) -> Result<(), ActorRunError> {
281        #[cfg(not(wasm_browser))]
282        let port_mapper = self.socket_state.port_mapper.is_some();
283        #[cfg(wasm_browser)]
284        let port_mapper = false;
285        debug!(%port_mapper, "reportstate actor starting");
286
287        self.report.os_has_ipv6 = super::os_has_ipv6();
288
289        let mut port_mapping = self.prepare_portmapper_task();
290        let mut captive_task = self.prepare_captive_portal_task();
291        let mut probes = self.spawn_probes_task().await;
292
293        let total_timer = time::sleep(OVERALL_REPORT_TIMEOUT);
294        tokio::pin!(total_timer);
295        let probe_timer = time::sleep(PROBES_TIMEOUT);
296        tokio::pin!(probe_timer);
297
298        loop {
299            trace!(awaiting = ?self.outstanding_tasks, "tick; awaiting tasks");
300            if self.outstanding_tasks.all_done() {
301                debug!("all tasks done");
302                break;
303            }
304            tokio::select! {
305                biased;
306                _ = &mut total_timer => {
307                    trace!("tick: total_timer expired");
308                    return Err(TimeoutSnafu.build());
309                }
310
311                _ = &mut probe_timer => {
312                    warn!("tick: probes timed out");
313                    // Set new timeout to not go into this branch multiple times.  We need
314                    // the abort to finish all probes normally.  PROBES_TIMEOUT is
315                    // sufficiently far in the future.
316                    probe_timer.as_mut().reset(Instant::now() + PROBES_TIMEOUT);
317                    probes.abort_all();
318                    self.handle_abort_probes();
319                }
320
321                // Drive the portmapper.
322                pm = &mut port_mapping, if self.outstanding_tasks.port_mapper => {
323                    debug!(report=?pm, "tick: portmapper probe report");
324                    self.report.portmap_probe = pm;
325                    port_mapping.inner = None;
326                    self.outstanding_tasks.port_mapper = false;
327                }
328
329                // Check for probes finishing.
330                set_result = probes.join_next(), if self.outstanding_tasks.probes => {
331                    trace!("tick: probes done: {:?}", set_result);
332                    match set_result {
333                        Some(Ok(Ok(report))) => self.handle_probe_report(report),
334                        Some(Ok(Err(_))) => (),
335                        Some(Err(e)) => {
336                            warn!("probes task error: {:?}", e);
337                        }
338                        None => {
339                            self.handle_abort_probes();
340                        }
341                    }
342                    trace!("tick: probes handled");
343                }
344
345                // Drive the captive task.
346                found = &mut captive_task, if self.outstanding_tasks.captive_task => {
347                    trace!("tick: captive portal task done");
348                    self.report.captive_portal = found;
349                    captive_task.inner = None;
350                    self.outstanding_tasks.captive_task = false;
351                }
352
353                // Handle actor messages.
354                msg = self.msg_rx.recv() => {
355                    trace!("tick: msg recv: {:?}", msg);
356                    match msg {
357                        Some(msg) => self.handle_message(msg),
358                        None => {
359                            return Err(ClientGoneSnafu.build());
360                        }
361                    }
362                }
363            }
364        }
365
366        if !probes.is_empty() {
367            debug!(
368                "aborting {} probe sets, already have enough reports",
369                probes.len()
370            );
371            drop(probes);
372        }
373
374        debug!("Sending report to net_report actor");
375        self.net_report
376            .send(net_report::Message::ReportReady {
377                report: Box::new(self.report.clone()),
378            })
379            .await
380            .map_err(|_| ActorGoneSnafu.build())?;
381
382        Ok(())
383    }

385    /// Handles an actor message.
386    ///
387    /// Returns `true` if all the probes need to be aborted.
388    fn handle_message(&mut self, msg: Message) {
389        trace!(?msg, "handling message");
390        match msg {
391            Message::HairpinResult(works) => {
392                self.report.hair_pinning = Some(works);
393                self.outstanding_tasks.hairpin = false;
394            }
395            Message::ProbeWouldHelp(probe, relay_node, response_tx) => {
396                let res = self.probe_would_help(probe, relay_node);
397                if response_tx.send(res).is_err() {
398                    debug!("probe dropped before ProbeWouldHelp response sent");
399                }
400            }
401            Message::AbortProbes => {
402                self.handle_abort_probes();
403            }
404        }
405    }

407    fn handle_probe_report(&mut self, probe_report: ProbeReport) {
408        debug!(?probe_report, "finished probe");
409        update_report(&mut self.report, probe_report);
410
411        // When we discover the first IPv4 address we want to start the hairpin actor.
412        #[cfg(not(wasm_browser))]
413        if let Some(ref addr) = self.report.global_v4 {
414            if !self.hairpin_actor.has_started() {
415                self.hairpin_actor.start_check(*addr);
416                self.outstanding_tasks.hairpin = true;
417            }
418        }
419
420        // Once we've heard from enough relay servers (3), start a timer to give up on the other
421        // probes. The timer's duration is a function of whether this is our initial full
422        // probe or an incremental one. For incremental ones, wait for the duration of the
423        // slowest relay. For initial ones, double that.
424        let enough_relays = std::cmp::min(self.relay_map.len(), ENOUGH_NODES);
425        if self.report.relay_latency.len() == enough_relays {
426            let timeout = self.report.relay_latency.max_latency();
427            let timeout = match self.last_report.is_some() {
428                true => timeout,
429                false => timeout * 2,
430            };
431            let reportcheck = self.addr();
432            debug!(
433                reports=self.report.relay_latency.len(),
434                delay=?timeout,
435                "Have enough probe reports, aborting further probes soon",
436            );
437            task::spawn(
438                async move {
439                    time::sleep(timeout).await;
440                    // Because we do this after a timeout it is entirely normal that the
441                    // actor is no longer there by the time we send this message.
442                    reportcheck
443                        .send(Message::AbortProbes)
444                        .await
445                        .map_err(|err| trace!("Failed to abort all probes: {err:#}"))
446                        .ok();
447                }
448                .instrument(Span::current()),
449            );
450        }
451    }

453    /// Whether running this probe would still improve our report.
454    fn probe_would_help(&mut self, probe: Probe, relay_node: Arc<RelayNode>) -> bool {
455        // If the probe is for a relay we don't yet know about, that would help.
456        if self.report.relay_latency.get(&relay_node.url).is_none() {
457            return true;
458        }
459
460        // If the probe is for IPv6 and we don't yet have an IPv6 report, that would help.
461        #[cfg(not(wasm_browser))]
462        if probe.proto() == ProbeProto::StunIpv6 && self.report.relay_v6_latency.is_empty() {
463            return true;
464        }
465
466        // For IPv4, we need at least two IPv4 results overall to
467        // determine whether we're behind a NAT that shows us as
468        // different source IPs and/or ports depending on who we're
469        // talking to. If we don't yet have two results yet
470        // (`mapping_varies_by_dest_ip` is blank), then another IPv4 probe
471        // would be good.
472        #[cfg(not(wasm_browser))]
473        if probe.proto() == ProbeProto::StunIpv4 && self.report.mapping_varies_by_dest_ip.is_none()
474        {
475            return true;
476        }
477
478        // Otherwise not interesting.
479        false
480    }

483    ///
484    /// This makes sure that no further probes are run and also cancels the captive portal

530    /// Creates the future which will perform the captive portal check.
531    fn prepare_captive_portal_task(
532        &mut self,
533    ) -> MaybeFuture<Pin<Box<impl Future<Output = Option<bool>>>>> {
534        // In the browser case the compiler cannot infer the type of the future, because it's never set:
535        #[cfg(wasm_browser)]
536        let captive_task: MaybeFuture<Pin<Box<Pending<Option<bool>>>>> = MaybeFuture::default();
537
538        #[cfg(not(wasm_browser))]
539        let mut captive_task = MaybeFuture::default();
540
541        // If we're doing a full probe, also check for a captive portal. We
542        // delay by a bit to wait for UDP STUN to finish, to avoid the probe if
543        // it's unnecessary.
544        #[cfg(not(wasm_browser))]
545        if self.last_report.is_none() {
546            // Even if we're doing a non-incremental update, we may want to try our
547            // preferred relay for captive portal detection.
548            let preferred_relay = self
549                .last_report
550                .as_ref()
551                .and_then(|l| l.preferred_relay.clone());
552
553            let dns_resolver = self.socket_state.dns_resolver.clone();
554            let dm = self.relay_map.clone();
555            self.outstanding_tasks.captive_task = true;
556            captive_task.inner = Some(Box::pin(async move {
557                time::sleep(CAPTIVE_PORTAL_DELAY).await;
558                debug!("Captive portal check started after {CAPTIVE_PORTAL_DELAY:?}");
559                let captive_portal_check = time::timeout(
560                    CAPTIVE_PORTAL_TIMEOUT,
561                    check_captive_portal(&dns_resolver, &dm, preferred_relay)
562                        .instrument(debug_span!("captive-portal")),
563                );
564                match captive_portal_check.await {
565                    Ok(Ok(found)) => Some(found),
566                    Ok(Err(err)) => {
567                        match err {
568                            CaptivePortalError::CreateReqwestClient { ref source }
569                            | CaptivePortalError::HttpRequest { ref source } => {
570                                if source.is_connect() {
571                                    debug!("check_captive_portal failed: {err:#}");
572                                }
573                            }
574                            _ => warn!("check_captive_portal error: {err:#}"),
575                        }
576                        None
577                    }
578                    Err(_) => {
579                        warn!("check_captive_portal timed out");
580                        None
581                    }
582                }
583            }));
584        }
585
586        self.outstanding_tasks.captive_task = false;
587        captive_task
588    }

590    /// Prepares the future which will run all the probes as per generated ProbePlan.
591    ///
592    /// Probes operate like the following:
593    ///
594    /// - A future is created for each probe in all probe sets.
595    /// - All probes in a set are grouped in [`JoinSet`].
596    /// - All those probe sets are grouped in one overall [`JoinSet`].
597    ///   - This future is polled by the main actor loop to make progress.
598    /// - Once a probe future is polled:
599    ///   - Many probes start with a delay, they sleep during this time.
600    ///   - When a probe starts it first asks the reportgen [`Actor`] if it is still useful
601    ///     to run.  If not it aborts the entire probe set.
602    ///   - When a probe finishes, its [`ProbeReport`] is yielded to the reportgen actor.
603    /// - Probes get aborted in several ways:
604    ///   - A running it can fail and abort the entire probe set if it deems the
605    ///     failure permanent.  Probes in a probe set are essentially retries.
606    ///   - Once there are [`ProbeReport`]s from enough nodes, all remaining probes are
607    ///     aborted.  That is, the main actor loop stops polling them.
608    async fn spawn_probes_task(&mut self) -> JoinSet<Result<ProbeReport, ProbesError>> {
609        #[cfg(not(wasm_browser))]
610        let if_state = interfaces::State::new().await;
611        #[cfg(not(wasm_browser))]
612        debug!(%if_state, "Local interfaces");
613        let plan = match self.last_report {
614            Some(ref report) => ProbePlan::with_last_report(
615                &self.relay_map,
616                report,
617                &self.protocols,
618                #[cfg(not(wasm_browser))]
619                &if_state,
620            ),
621            None => ProbePlan::initial(
622                &self.relay_map,
623                &self.protocols,
624                #[cfg(not(wasm_browser))]
625                &if_state,
626            ),
627        };
628        trace!(%plan, "probe plan");
629
630        // The pinger is created here so that any sockets that might be bound for it are
631        // shared between the probes that use it.  It binds sockets lazily, so we can always
632        // create it.
633        #[cfg(not(wasm_browser))]
634        let pinger = Pinger::new();
635
636        // A collection of futures running probe sets.
637        let mut probes = JoinSet::default();
638        for probe_set in plan.iter() {
639            let mut set = JoinSet::default();
640            for probe in probe_set {
641                let reportstate = self.addr();
642                let relay_node = probe.node().clone();
643                let probe = probe.clone();
644                let net_report = self.net_report.clone();
645
646                #[cfg(not(wasm_browser))]
647                let pinger = pinger.clone();
648                #[cfg(not(wasm_browser))]
649                let socket_state = self.socket_state.clone();
650
651                let metrics = self.metrics.clone();
652                set.spawn(
653                    run_probe(
654                        reportstate,
655                        relay_node,
656                        probe.clone(),
657                        net_report,
658                        metrics,
659                        #[cfg(not(wasm_browser))]
660                        pinger,
661                        #[cfg(not(wasm_browser))]
662                        socket_state,
663                        #[cfg(any(test, feature = "test-utils"))]
664                        self.insecure_skip_relay_cert_verify,
665                    )
666                    .instrument(debug_span!("run_probe", %probe)),
667                );
668            }
669
670            // Add the probe set to all futures of probe sets.  Handle aborting a probe set
671            // if needed, only normal errors means the set continues.
672            probes.spawn(
673                async move {
674                    // Hack because ProbeSet is not it's own type yet.
675                    let mut probe_proto = None;
676                    while let Some(res) = set.join_next().await {
677                        match res {
678                            Ok(Ok(report)) => return Ok(report),
679                            Ok(Err(ProbeErrorWithProbe::Error(err, probe))) => {
680                                probe_proto = Some(probe.proto());
681                                warn!(?probe, "probe failed: {:#}", err);
682                                continue;
683                            }
684                            Ok(Err(ProbeErrorWithProbe::AbortSet(err, probe))) => {
685                                debug!(?probe, "probe set aborted: {:#}", err);
686                                set.abort_all();
687                                return Err(ProbeFailureSnafu.into_error(err));
688                            }
689                            Err(err) => {
690                                warn!("fatal probe set error, aborting: {:#}", err);
691                                continue;
692                            }
693                        }
694                    }
695                    warn!(?probe_proto, "no successful probes in ProbeSet");
696                    Err(AllProbesFailedSnafu.build())
697                }
698                .instrument(info_span!("probe")),
699            );
700        }
701        self.outstanding_tasks.probes = true;
702
703        probes
704    }
705}

758/// Errors for [`run_probe`].
759///
760/// The main purpose is to signal whether other probes in this probe set should still be
761/// run.  Recall that a probe set is normally a set of identical probes with delays,

767    /// Abort the current set.
768    AbortSet(ProbeError, Probe),
769    /// Continue the other probes in the set.

771}

838/// Executes a particular [`Probe`], including using a delayed start if needed.
839///
840/// If *stun_sock4* and *stun_sock6* are `None` the STUN probes are disabled.
841#[allow(clippy::too_many_arguments)]
842async fn run_probe(
843    reportstate: Addr,
844    relay_node: Arc<RelayNode>,
845    probe: Probe,
846    net_report: net_report::Addr,
847    metrics: Arc<Metrics>,
848    #[cfg(not(wasm_browser))] pinger: Pinger,
849    #[cfg(not(wasm_browser))] socket_state: SocketState,
850    #[cfg(any(test, feature = "test-utils"))] insecure_skip_relay_cert_verify: bool,
851) -> Result<ProbeReport, ProbeErrorWithProbe> {
852    if !probe.delay().is_zero() {
853        trace!("delaying probe");
854        time::sleep(probe.delay()).await;
855    }
856    debug!("starting probe");
857
858    let (would_help_tx, would_help_rx) = oneshot::channel();
859    if let Err(err) = reportstate
860        .send(Message::ProbeWouldHelp(
861            probe.clone(),
862            relay_node.clone(),
863            would_help_tx,
864        ))
865        .await
866    {
867        // this happens on shutdown or if the report is already finished
868        debug!("Failed to check if probe would help: {err:#}");
869        return Err(ProbeErrorWithProbe::AbortSet(
870            probe_error::ClientGoneSnafu.build(),
871            probe.clone(),
872        ));
873    }
874
875    if !would_help_rx.await.map_err(|_| {
876        ProbeErrorWithProbe::AbortSet(probe_error::ClientGoneSnafu.build(), probe.clone())
877    })? {
878        return Err(ProbeErrorWithProbe::AbortSet(
879            probe_error::NotUsefulSnafu.build(),
880            probe,
881        ));
882    }
883
884    #[cfg(not(wasm_browser))]
885    let relay_addr = get_relay_addr(&socket_state.dns_resolver, &relay_node, probe.proto())
886        .await
887        .map_err(|e| {
888            ProbeErrorWithProbe::AbortSet(
889                probe_error::GetRelayAddrSnafu.into_error(e),
890                probe.clone(),
891            )
892        })?;
893
894    let mut result = ProbeReport::new(probe.clone());
895    match probe {
896        #[cfg(not(wasm_browser))]
897        Probe::StunIpv4 { .. } | Probe::StunIpv6 { .. } => {
898            let maybe_sock = if matches!(probe, Probe::StunIpv4 { .. }) {
899                socket_state.stun_sock4.as_ref()
900            } else {
901                socket_state.stun_sock6.as_ref()
902            };
903            match maybe_sock {
904                Some(sock) => {
905                    result = run_stun_probe(sock, relay_addr, net_report, probe, &metrics).await?;
906                }
907                None => {
908                    return Err(ProbeErrorWithProbe::AbortSet(
909                        probe_error::StunSnafu.into_error(stun_error::NoSocketSnafu.build()),
910                        probe.clone(),
911                    ));
912                }
913            }
914        }
915        #[cfg(not(wasm_browser))]
916        Probe::IcmpV4 { .. } | Probe::IcmpV6 { .. } => {
917            result = run_icmp_probe(probe, relay_addr, pinger).await?
918        }
919        Probe::Https { ref node, .. } => {
920            debug!("sending probe HTTPS");
921            match measure_https_latency(
922                #[cfg(not(wasm_browser))]
923                &socket_state.dns_resolver,
924                node,
925                #[cfg(any(test, feature = "test-utils"))]
926                insecure_skip_relay_cert_verify,
927            )
928            .await
929            {
930                Ok((latency, ip)) => {
931                    debug!(?latency, "latency");
932                    result.latency = Some(latency);
933                    // We set these IPv4 and IPv6 but they're not really used
934                    // and we don't necessarily set them both. If UDP is blocked
935                    // and both IPv4 and IPv6 are available over TCP, it's basically
936                    // random which fields end up getting set here.
937                    // Since they're not needed, that's fine for now.
938                    match ip {
939                        IpAddr::V4(_) => result.ipv4_can_send = true,
940                        IpAddr::V6(_) => result.ipv6_can_send = true,
941                    }
942                }
943                Err(err) => {
944                    warn!("https latency measurement failed: {:?}", err);
945                }
946            }
947        }
948
949        #[cfg(not(wasm_browser))]
950        Probe::QuicIpv4 { ref node, .. } | Probe::QuicIpv6 { ref node, .. } => {
951            debug!("sending QUIC address discovery probe");
952            let url = node.url.clone();
953            match socket_state.quic_config {
954                Some(quic_config) => {
955                    result = run_quic_probe(
956                        quic_config,
957                        url,
958                        relay_addr,
959                        probe,
960                        socket_state.ip_mapped_addrs,
961                    )
962                    .await?;
963                }
964                None => {
965                    return Err(ProbeErrorWithProbe::AbortSet(
966                        probe_error::QuicSnafu.into_error(quic_error::NoEndpointSnafu.build()),
967                        probe.clone(),
968                    ));
969                }
970            }
971        }
972    }
973
974    trace!("probe successful");
975    Ok(result)
976}

978/// Run a STUN IPv4 or IPv6 probe.
979#[cfg(not(wasm_browser))]
980async fn run_stun_probe(
981    sock: &Arc<UdpSocket>,
982    relay_addr: SocketAddr,
983    net_report: net_report::Addr,
984    probe: Probe,
985    metrics: &Metrics,
986) -> Result<ProbeReport, ProbeErrorWithProbe> {
987    match probe.proto() {
988        ProbeProto::StunIpv4 => debug_assert!(relay_addr.is_ipv4()),
989        ProbeProto::StunIpv6 => debug_assert!(relay_addr.is_ipv6()),
990        _ => debug_assert!(false, "wrong probe"),
991    }
992    let txid = stun::TransactionId::default();
993    let req = stun::request(txid);
994
995    // Setup net_report to give us back the incoming STUN response.
996    let (stun_tx, stun_rx) = oneshot::channel();
997    let (inflight_ready_tx, inflight_ready_rx) = oneshot::channel();
998    net_report
999        .send(net_report::Message::InFlightStun(
1000            net_report::Inflight {
1001                txn: txid,
1002                start: Instant::now(),
1003                s: stun_tx,
1004            },
1005            inflight_ready_tx,
1006        ))
1007        .await
1008        .map_err(|_| {
1009            ProbeErrorWithProbe::Error(probe_error::ClientGoneSnafu.build(), probe.clone())
1010        })?;
1011    inflight_ready_rx.await.map_err(|_| {
1012        ProbeErrorWithProbe::Error(probe_error::ClientGoneSnafu.build(), probe.clone())
1013    })?;
1014
1015    // Send the probe.
1016    match sock.send_to(&req, relay_addr).await {
1017        Ok(n) if n == req.len() => {
1018            debug!(%relay_addr, %txid, "sending {} probe", probe.proto());
1019            let mut result = ProbeReport::new(probe.clone());
1020
1021            if matches!(probe, Probe::StunIpv4 { .. }) {
1022                result.ipv4_can_send = true;
1023                metrics.stun_packets_sent_ipv4.inc();
1024            } else {
1025                result.ipv6_can_send = true;
1026                metrics.stun_packets_sent_ipv6.inc();
1027            }
1028            let (delay, addr) = stun_rx.await.map_err(|_| {
1029                ProbeErrorWithProbe::Error(
1030                    probe_error::StunSnafu.into_error(stun_error::StunChannelGoneSnafu.build()),
1031                    probe.clone(),
1032                )
1033            })?;
1034            result.latency = Some(delay);
1035            result.addr = Some(addr);
1036            Ok(result)
1037        }
1038        Ok(n) => {
1039            let err = stun_error::SendFullSnafu.build();
1040            error!(%relay_addr, sent_len=n, req_len=req.len(), "{err:#}");
1041            Err(ProbeErrorWithProbe::Error(
1042                probe_error::StunSnafu.into_error(err),
1043                probe.clone(),
1044            ))
1045        }
1046        Err(err) => {
1047            let kind = err.kind();
1048            let err = stun_error::SendSnafu.into_error(err);
1049
1050            // It is entirely normal that we are on a dual-stack machine with no
1051            // routed IPv6 network.  So silence that case.
1052            // NetworkUnreachable and HostUnreachable are still experimental (io_error_more
1053            // #86442) but it is already emitted.  So hack around this.
1054            match format!("{kind:?}").as_str() {
1055                "NetworkUnreachable" | "HostUnreachable" => {
1056                    debug!(%relay_addr, "{err:#}");
1057                    Err(ProbeErrorWithProbe::AbortSet(
1058                        probe_error::StunSnafu.into_error(err),
1059                        probe.clone(),
1060                    ))
1061                }
1062                _ => {
1063                    // No need to log this, our caller does already log this.
1064                    Err(ProbeErrorWithProbe::Error(
1065                        probe_error::StunSnafu.into_error(err),
1066                        probe.clone(),
1067                    ))
1068                }
1069            }
1070        }
1071    }
1072}

1085/// Run a QUIC address discovery probe.
1086#[cfg(not(wasm_browser))]
1087async fn run_quic_probe(
1088    quic_config: QuicConfig,
1089    url: RelayUrl,
1090    relay_addr: SocketAddr,
1091    probe: Probe,
1092    ip_mapped_addrs: Option<IpMappedAddresses>,
1093) -> Result<ProbeReport, ProbeErrorWithProbe> {
1094    match probe.proto() {
1095        ProbeProto::QuicIpv4 => debug_assert!(relay_addr.is_ipv4()),
1096        ProbeProto::QuicIpv6 => debug_assert!(relay_addr.is_ipv6()),
1097        _ => debug_assert!(false, "wrong probe"),
1098    }
1099    let relay_addr = maybe_to_mapped_addr(ip_mapped_addrs, relay_addr);
1100    let host = match url.host_str() {
1101        Some(host) => host,
1102        None => {
1103            return Err(ProbeErrorWithProbe::Error(
1104                probe_error::QuicSnafu.into_error(quic_error::InvalidUrlSnafu.build()),
1105                probe.clone(),
1106            ));
1107        }
1108    };
1109    let quic_client = iroh_relay::quic::QuicClient::new(quic_config.ep, quic_config.client_config)
1110        .map_err(|e| {
1111            ProbeErrorWithProbe::Error(
1112                probe_error::QuicSnafu.into_error(quic_error::CreateClientSnafu.into_error(e)),
1113                probe.clone(),
1114            )
1115        })?;
1116    let (addr, latency) = quic_client
1117        .get_addr_and_latency(relay_addr, host)
1118        .await
1119        .map_err(|e| {
1120            ProbeErrorWithProbe::Error(
1121                probe_error::QuicSnafu.into_error(quic_error::GetAddrSnafu.into_error(e)),
1122                probe.clone(),
1123            )
1124        })?;
1125    let mut result = ProbeReport::new(probe.clone());
1126    if matches!(probe, Probe::QuicIpv4 { .. }) {
1127        result.ipv4_can_send = true;
1128    } else {
1129        result.ipv6_can_send = true;
1130    }
1131    result.addr = Some(addr);
1132    result.latency = Some(latency);
1133    Ok(result)
1134}

1149/// Reports whether or not we think the system is behind a
1150/// captive portal, detected by making a request to a URL that we know should
1151/// return a "204 No Content" response and checking if that's what we get.
1152///
1153/// The boolean return is whether we think we have a captive portal.
1154#[cfg(not(wasm_browser))]
1155async fn check_captive_portal(
1156    dns_resolver: &DnsResolver,
1157    dm: &RelayMap,
1158    preferred_relay: Option<RelayUrl>,
1159) -> Result<bool, CaptivePortalError> {
1160    // If we have a preferred relay node and we can use it for non-STUN requests, try that;
1161    // otherwise, pick a random one suitable for non-STUN requests.
1162    let preferred_relay = preferred_relay.and_then(|url| match dm.get_node(&url) {
1163        Some(node) if node.stun_only => Some(url),
1164        _ => None,
1165    });
1166
1167    let url = match preferred_relay {
1168        Some(url) => url,
1169        None => {
1170            let urls: Vec<_> = dm
1171                .nodes()
1172                .filter(|n| !n.stun_only)
1173                .map(|n| n.url.clone())
1174                .collect();
1175            if urls.is_empty() {
1176                debug!("No suitable relay node for captive portal check");
1177                return Ok(false);
1178            }
1179
1180            let i = (0..urls.len())
1181                .choose(&mut rand::thread_rng())
1182                .unwrap_or_default();
1183            urls[i].clone()
1184        }
1185    };
1186
1187    let mut builder = reqwest::ClientBuilder::new().redirect(reqwest::redirect::Policy::none());
1188
1189    if let Some(Host::Domain(domain)) = url.host() {
1190        // Use our own resolver rather than getaddrinfo
1191        //
1192        // Be careful, a non-zero port will override the port in the URI.
1193        //
1194        // Ideally we would try to resolve **both** IPv4 and IPv6 rather than purely race
1195        // them.  But our resolver doesn't support that yet.
1196        let addrs: Vec<_> = dns_resolver
1197            .lookup_ipv4_ipv6_staggered(domain, DNS_TIMEOUT, DNS_STAGGERING_MS)
1198            .await?
1199            .map(|ipaddr| SocketAddr::new(ipaddr, 0))
1200            .collect();
1201        builder = builder.resolve_to_addrs(domain, &addrs);
1202    }
1203    let client = builder
1204        .build()
1205        .context(captive_portal_error::CreateReqwestClientSnafu)?;
1206
1207    // Note: the set of valid characters in a challenge and the total
1208    // length is limited; see is_challenge_char in bin/iroh-relay for more
1209    // details.
1210
1211    let host_name = url.host_str().unwrap_or_default();
1212    let challenge = format!("ts_{}", host_name);
1213    let portal_url = format!("http://{}/generate_204", host_name);
1214    let res = client
1215        .request(reqwest::Method::GET, portal_url)
1216        .header("X-Tailscale-Challenge", &challenge)
1217        .send()
1218        .await
1219        .context(captive_portal_error::HttpRequestSnafu)?;
1220
1221    let expected_response = format!("response {challenge}");
1222    let is_valid_response = res
1223        .headers()
1224        .get("X-Tailscale-Response")
1225        .map(|s| s.to_str().unwrap_or_default())
1226        == Some(&expected_response);
1227
1228    debug!(
1229        "check_captive_portal url={} status_code={} valid_response={}",
1230        res.url(),
1231        res.status(),
1232        is_valid_response,
1233    );
1234    let has_captive = res.status() != 204 || !is_valid_response;
1235
1236    Ok(has_captive)
1237}

1283/// Returns the IP address to use to communicate to this relay node.
1284///
1285/// *proto* specifies the protocol of the probe.  Depending on the protocol we may return
1286/// different results.  Obviously IPv4 vs IPv6 but a [`RelayNode`] may also have disabled
1287/// some protocols.
1288///
1289/// If the protocol is `QuicIpv4` or `QuicIpv6`, and `IpMappedAddresses` is not `None`, we
1290/// assume that we are running this net report with `iroh`, and need to provide mapped
1291/// addresses to the probe in order for it to function in the specialize iroh-quinn
1292/// endpoint that expects mapped addresses.
1293#[cfg(not(wasm_browser))]
1294async fn get_relay_addr(
1295    dns_resolver: &DnsResolver,
1296    relay_node: &RelayNode,
1297    proto: ProbeProto,
1298) -> Result<SocketAddr, GetRelayAddrError> {
1299    use snafu::OptionExt;
1300
1301    if relay_node.stun_only && !matches!(proto, ProbeProto::StunIpv4 | ProbeProto::StunIpv6) {
1302        return Err(get_relay_addr_error::UnsupportedRelayNodeSnafu.build());
1303    }
1304    let port = get_port(relay_node, &proto).context(get_relay_addr_error::MissingPortSnafu)?;
1305
1306    match proto {
1307        ProbeProto::StunIpv4 | ProbeProto::IcmpV4 | ProbeProto::QuicIpv4 => {
1308            relay_lookup_ipv4_staggered(dns_resolver, relay_node, port).await
1309        }
1310
1311        ProbeProto::StunIpv6 | ProbeProto::IcmpV6 | ProbeProto::QuicIpv6 => {
1312            relay_lookup_ipv6_staggered(dns_resolver, relay_node, port).await
1313        }
1314
1315        ProbeProto::Https => Err(get_relay_addr_error::UnsupportedHttpsSnafu.build()),
1316    }
1317}

1319/// Do a staggared ipv4 DNS lookup based on [`RelayNode`]
1320///
1321/// `port` is combined with the resolved [`std::net::Ipv4Addr`] to return a [`SocketAddr`]
1322#[cfg(not(wasm_browser))]
1323async fn relay_lookup_ipv4_staggered(
1324    dns_resolver: &DnsResolver,
1325    relay: &RelayNode,
1326    port: u16,
1327) -> Result<SocketAddr, GetRelayAddrError> {
1328    match relay.url.host() {
1329        Some(url::Host::Domain(hostname)) => {
1330            debug!(%hostname, "Performing DNS A lookup for relay addr");
1331            match dns_resolver
1332                .lookup_ipv4_staggered(hostname, DNS_TIMEOUT, DNS_STAGGERING_MS)
1333                .await
1334            {
1335                Ok(mut addrs) => addrs
1336                    .next()
1337                    .map(|ip| ip.to_canonical())
1338                    .map(|addr| {
1339                        debug_assert!(addr.is_ipv4(), "bad DNS lookup: {:?}", addr);
1340                        SocketAddr::new(addr, port)
1341                    })
1342                    .ok_or(get_relay_addr_error::NoAddrFoundSnafu.build()),
1343                Err(err) => Err(get_relay_addr_error::DnsLookupSnafu.into_error(err)),
1344            }
1345        }
1346        Some(url::Host::Ipv4(addr)) => Ok(SocketAddr::new(addr.into(), port)),
1347        Some(url::Host::Ipv6(_addr)) => Err(get_relay_addr_error::NoAddrFoundSnafu.build()),
1348        None => Err(get_relay_addr_error::InvalidHostnameSnafu.build()),
1349    }
1350}

1384/// Runs an ICMP IPv4 or IPv6 probe.
1385///
1386/// The `pinger` is passed in so the ping sockets are only bound once
1387/// for the probe set.
1388#[cfg(not(wasm_browser))]
1389async fn run_icmp_probe(
1390    probe: Probe,
1391    relay_addr: SocketAddr,
1392    pinger: Pinger,
1393) -> Result<ProbeReport, ProbeErrorWithProbe> {
1394    match probe.proto() {
1395        ProbeProto::IcmpV4 => debug_assert!(relay_addr.is_ipv4()),
1396        ProbeProto::IcmpV6 => debug_assert!(relay_addr.is_ipv6()),
1397        _ => debug_assert!(false, "wrong probe"),
1398    }
1399    const DATA: &[u8; 15] = b"iroh icmp probe";
1400    debug!(dst = %relay_addr, len = DATA.len(), "ICMP Ping started");
1401    let latency = pinger
1402        .send(relay_addr.ip(), DATA)
1403        .await
1404        .map_err(|err| match err {
1405            PingError::CreateClientIpv4 { .. } | PingError::CreateClientIpv6 { .. } => {
1406                ProbeErrorWithProbe::AbortSet(probe_error::IcmpSnafu.into_error(err), probe.clone())
1407            }
1408            #[cfg(not(wasm_browser))]
1409            PingError::Ping { .. } => {
1410                ProbeErrorWithProbe::Error(probe_error::IcmpSnafu.into_error(err), probe.clone())
1411            }
1412        })?;
1413    debug!(dst = %relay_addr, len = DATA.len(), ?latency, "ICMP ping done");
1414    let mut report = ProbeReport::new(probe);
1415    report.latency = Some(latency);
1416    match relay_addr {
1417        SocketAddr::V4(_) => {
1418            report.ipv4_can_send = true;
1419            report.icmpv4 = Some(true);
1420        }
1421        SocketAddr::V6(_) => {
1422            report.ipv6_can_send = true;
1423            report.icmpv6 = Some(true);
1424        }
1425    }
1426    Ok(report)
1427}

1446/// Executes an HTTPS probe.
1447///
1448/// If `certs` is provided they will be added to the trusted root certificates, allowing the
1449/// use of self-signed certificates for servers.  Currently this is used for testing.
1450#[allow(clippy::unused_async)]
1451async fn measure_https_latency(
1452    #[cfg(not(wasm_browser))] dns_resolver: &DnsResolver,
1453    node: &RelayNode,
1454    #[cfg(any(test, feature = "test-utils"))] insecure_skip_relay_cert_verify: bool,
1455) -> Result<(Duration, IpAddr), MeasureHttpsLatencyError> {
1456    let url = node.url.join(RELAY_PROBE_PATH)?;
1457
1458    // This should also use same connection establishment as relay client itself, which
1459    // needs to be more configurable so users can do more crazy things:
1460    // https://github.com/n0-computer/iroh/issues/2901
1461    let mut builder = reqwest::ClientBuilder::new();
1462
1463    #[cfg(not(wasm_browser))]
1464    {
1465        builder = builder.redirect(reqwest::redirect::Policy::none());
1466    }
1467
1468    #[cfg(not(wasm_browser))]
1469    if let Some(Host::Domain(domain)) = url.host() {
1470        // Use our own resolver rather than getaddrinfo
1471        //
1472        // Be careful, a non-zero port will override the port in the URI.
1473        //
1474        // The relay Client uses `.lookup_ipv4_ipv6` to connect, so use the same function
1475        // but staggered for reliability.  Ideally this tries to resolve **both** IPv4 and
1476        // IPv6 though.  But our resolver does not have a function for that yet.
1477        let addrs: Vec<_> = dns_resolver
1478            .lookup_ipv4_ipv6_staggered(domain, DNS_TIMEOUT, DNS_STAGGERING_MS)
1479            .await?
1480            .map(|ipaddr| SocketAddr::new(ipaddr, 0))
1481            .collect();
1482        builder = builder.resolve_to_addrs(domain, &addrs);
1483    }
1484
1485    #[cfg(all(not(wasm_browser), any(test, feature = "test-utils")))]
1486    let builder = builder.danger_accept_invalid_certs(insecure_skip_relay_cert_verify);
1487
1488    let client = builder
1489        .build()
1490        .context(measure_https_latency_error::CreateReqwestClientSnafu)?;
1491
1492    let start = Instant::now();
1493    let response = client
1494        .request(reqwest::Method::GET, url)
1495        .send()
1496        .await
1497        .context(measure_https_latency_error::HttpRequestSnafu)?;
1498    let latency = start.elapsed();
1499    if response.status().is_success() {
1500        // Only `None` if a different hyper HttpConnector in the request.
1501        #[cfg(not(wasm_browser))]
1502        let remote_ip = response
1503            .remote_addr()
1504            .expect("missing HttpInfo from HttpConnector")
1505            .ip();
1506        #[cfg(wasm_browser)]
1507        let remote_ip = IpAddr::V4(std::net::Ipv4Addr::UNSPECIFIED);
1508
1509        // Drain the response body to be nice to the server, up to a limit.
1510        const MAX_BODY_SIZE: usize = 8 << 10; // 8 KiB
1511        let mut body_size = 0;
1512        let mut stream = response.bytes_stream();
1513        // ignore failing frames
1514        while let Some(Ok(chunk)) = stream.next().await {
1515            body_size += chunk.len();
1516            if body_size >= MAX_BODY_SIZE {
1517                break;
1518            }
1519        }
1520
1521        Ok((latency, remote_ip))
1522    } else {
1523        Err(measure_https_latency_error::InvalidResponseSnafu {
1524            status: response.status(),
1525        }
1526        .build())
1527    }
1528}

1530/// Updates a net_report [`Report`] with a new [`ProbeReport`].
1531fn update_report(report: &mut Report, probe_report: ProbeReport) {
1532    let relay_node = probe_report.probe.node();
1533    if let Some(latency) = probe_report.latency {
1534        report
1535            .relay_latency
1536            .update_relay(relay_node.url.clone(), latency);
1537
1538        #[cfg(not(wasm_browser))]
1539        if matches!(
1540            probe_report.probe.proto(),
1541            ProbeProto::StunIpv4
1542                | ProbeProto::StunIpv6
1543                | ProbeProto::QuicIpv4
1544                | ProbeProto::QuicIpv6
1545        ) {
1546            report.udp = true;
1547
1548            match probe_report.addr {
1549                Some(SocketAddr::V4(ipp)) => {
1550                    report.ipv4 = true;
1551                    report
1552                        .relay_v4_latency
1553                        .update_relay(relay_node.url.clone(), latency);
1554                    if report.global_v4.is_none() {
1555                        report.global_v4 = Some(ipp);
1556                    } else if report.global_v4 != Some(ipp) {
1557                        report.mapping_varies_by_dest_ip = Some(true);
1558                    } else if report.mapping_varies_by_dest_ip.is_none() {
1559                        report.mapping_varies_by_dest_ip = Some(false);
1560                    }
1561                }
1562                Some(SocketAddr::V6(ipp)) => {
1563                    report.ipv6 = true;
1564                    report
1565                        .relay_v6_latency
1566                        .update_relay(relay_node.url.clone(), latency);
1567                    if report.global_v6.is_none() {
1568                        report.global_v6 = Some(ipp);
1569                    } else if report.global_v6 != Some(ipp) {
1570                        report.mapping_varies_by_dest_ipv6 = Some(true);
1571                        warn!("IPv6 Address detected by STUN varies by destination");
1572                    } else if report.mapping_varies_by_dest_ipv6.is_none() {
1573                        report.mapping_varies_by_dest_ipv6 = Some(false);
1574                    }
1575                }
1576                None => {
1577                    // If we are here we had a relay server latency reported from a STUN probe.
1578                    // Thus we must have a reported address.
1579                    debug_assert!(probe_report.addr.is_some());
1580                }
1581            }
1582        }
1583    }
1584    report.ipv4_can_send |= probe_report.ipv4_can_send;
1585    report.ipv6_can_send |= probe_report.ipv6_can_send;
1586    report.icmpv4 = report
1587        .icmpv4
1588        .map(|val| val || probe_report.icmpv4.unwrap_or_default())
1589        .or(probe_report.icmpv4);
1590    report.icmpv6 = report
1591        .icmpv6
1592        .map(|val| val || probe_report.icmpv6.unwrap_or_default())
1593        .or(probe_report.icmpv6);
1594}

1599    /// Future to be polled.
1600    pub inner: Option<T>,
1601}

1603// NOTE: explicit implementation to bypass derive unnecessary bounds

1617        }

1803    //
1804    //    sudo setcap CAP_NET_RAW=eip target/debug/deps/iroh-abc123
1805    //
1806    // Finally run the test:
1807    //
1808    //    cargo nextest run -p iroh net_report::reportgen::tests
1809    //
1810    // This allows the pinger to create a SOCK_RAW socket for IPPROTO_ICMP.
1811    //
1812    //
1813    // ## Using sysctl
1814    //
1815    // Now you know the hard way, you can also get this permission a little easier, but
1816    // slightly less secure, by allowing any process running with your group ID to create a
1817    // SOCK_DGRAM for IPPROTO_ICMP.
1818    //
1819    // First find out your group ID:
1820    //

1838            delay: Duration::from_secs(0),
1839            node: Arc::new(node),
1840        };
1841
1842        // A single ICMP packet might get lost.  Try several and take the first.
1843        let (tx, mut rx) = tokio::sync::mpsc::unbounded_channel();
1844        let mut tasks = JoinSet::new();
1845        for i in 0..8 {
1846            let probe = probe.clone();
1847            let pinger = pinger.clone();
1848            let tx = tx.clone();
1849            tasks.spawn(async move {
1850                time::sleep(Duration::from_millis(i * 100)).await;
1851                let res = run_icmp_probe(probe, addr, pinger).await;
1852                tx.send(res).ok();
1853            });
1854        }
1855        let mut last_err = None;
1856        while let Some(res) = rx.recv().await {
1857            match res {
1858                Ok(report) => {
1859                    dbg!(&report);
1860                    assert_eq!(report.icmpv4, Some(true));
1861                    assert!(
1862                        report.latency.expect("should have a latency") > Duration::from_secs(0)
1863                    );
1864                    break;
1865                }
1866                Err(ProbeErrorWithProbe::Error(err, _probe)) => {
1867                    last_err = Some(err);
1868                }
1869                Err(ProbeErrorWithProbe::AbortSet(_err, _probe)) => {
1870                    // We don't have permission, too bad.
1871                    // panic!("no ping permission: {err:#}");
1872                    break;
1873                }
1874            }
1875        }
1876        if let Some(err) = last_err {
1877            panic!("Ping error: {err:#}");
1878        }
1879    }
