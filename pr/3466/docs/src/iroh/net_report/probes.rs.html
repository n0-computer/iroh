1//! The relay probes.
2//!
3//! All the probes try and establish the latency to the relay servers.  Preferably the QAD
4//! probes work and we also learn about our public IP addresses and ports.  But fallback
5//! probes for HTTPS exist as well.
6
7use std::{collections::BTreeSet, fmt, sync::Arc};
8
9use iroh_relay::{RelayMap, RelayNode};
10use n0_future::time::Duration;
11use snafu::Snafu;
12
13use crate::net_report::Report;

15/// The retransmit interval used.
16const DEFAULT_INITIAL_RETRANSMIT: Duration = Duration::from_millis(100);
17
18/// The delay before starting HTTPS probes.
19const HTTPS_OFFSET: Duration = Duration::from_millis(200);
20
21/// The protocol used to time a node's latency.
22#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, derive_more::Display)]
23#[repr(u8)]
24pub enum Probe {
25    /// HTTPS
26    Https,
27    /// QUIC Address Discovery Ipv4
28    #[cfg(not(wasm_browser))]
29    QadIpv4,
30    /// QUIC Address Discovery Ipv6
31    #[cfg(not(wasm_browser))]
32    QadIpv6,
33}
34
35/// A probe set is a sequence of similar [`Probe`]s with delays between them.
36///
37/// The probes are to the same Relayer and of the same [`Probe`] but will have different
38/// delays.  The delays are effectively retries, though they do not wait for the previous
39/// probe to be finished.  The first successful probe will cancel all other probes in the
40/// set.
41///
42/// This is a lot of type-safety by convention.  It would be so much nicer to have this
43/// compile-time checked but that introduces a giant mess of generics and traits and
44/// associated exploding types.
45///
46/// A [`ProbeSet`] implements [`IntoIterator`] similar to how [`Vec`] does.
47#[derive(Debug, PartialEq, Eq, PartialOrd, Ord)]
48pub(super) struct ProbeSet {
49    /// The [`Probe`] all the probes in this set have.
50    proto: Probe,
51    /// The data in the set.
52    probes: Vec<(Duration, Arc<RelayNode>)>,
53}
54
55#[derive(Debug, Snafu)]
56#[snafu(display("Mismatching probe"))]
57struct PushError;
58
59impl ProbeSet {
60    fn new(proto: Probe) -> Self {
61        Self {
62            probes: Vec::new(),
63            proto,
64        }
65    }
66
67    pub(super) fn proto(&self) -> Probe {
68        self.proto
69    }
70
71    fn push(&mut self, delay: Duration, node: Arc<RelayNode>) {
72        self.probes.push((delay, node));
73    }
74
75    fn is_empty(&self) -> bool {
76        self.probes.is_empty()
77    }
78
79    pub(super) fn params(&self) -> impl Iterator<Item = &(Duration, Arc<RelayNode>)> {
80        self.probes.iter()
81    }
82}

84/// A probe plan.
85///
86/// A probe plan contains a number of [`ProbeSet`]s containing probes to be executed.
87/// Generally the first probe of of a set which completes aborts the remaining probes of a
88/// set.  Sometimes a failing probe can also abort the remaining probes of a set.
89///
90/// The [`reportgen`] actor will also abort all the remaining [`ProbeSet`]s once it has
91/// sufficient information for a report.
92///
93/// [`reportgen`]: crate::net_report::reportgen
94#[derive(Debug, Default, PartialEq, Eq)]
95pub(super) struct ProbePlan {
96    set: BTreeSet<ProbeSet>,
97}
98
99impl ProbePlan {
100    /// Creates an initial probe plan
101    pub(super) fn initial(relay_map: &RelayMap, protocols: &BTreeSet<Probe>) -> Self {
102        let mut plan = Self::default();
103
104        for relay_node in relay_map.nodes() {
105            let mut https_probes = ProbeSet::new(Probe::Https);
106
107            for attempt in 0u32..3 {
108                let delay = HTTPS_OFFSET + DEFAULT_INITIAL_RETRANSMIT * attempt;
109                https_probes.push(delay, relay_node.clone());
110            }
111
112            plan.add_if_enabled(protocols, https_probes);
113        }
114        plan
115    }
116
117    /// Creates a follow up probe plan using a previous net_report report in browsers.
118    ///
119    /// This will only schedule HTTPS probes.
120    pub(super) fn with_last_report(
121        relay_map: &RelayMap,
122        last_report: &Report,
123        protocols: &BTreeSet<Probe>,
124    ) -> Self {
125        if last_report.relay_latency.is_empty() {
126            return Self::initial(relay_map, protocols);
127        }
128
129        // TODO: is this good?
130        Self::default()
131    }
132
133    /// Returns an iterator over the [`ProbeSet`]s in this plan.
134    pub(super) fn iter(&self) -> impl Iterator<Item = &ProbeSet> {
135        self.set.iter()
136    }
137
138    /// Adds a [`ProbeSet`] if it contains probes and the protocol indicated in
139    /// the [`ProbeSet] matches a protocol in our set of [`Probe`]s.
140    fn add_if_enabled(&mut self, protocols: &BTreeSet<Probe>, set: ProbeSet) {
141        if !set.is_empty() && protocols.contains(&set.proto) {
142            self.set.insert(set);
143        }
144    }
145}
146
147impl fmt::Display for ProbePlan {
148    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
149        writeln!(f, "ProbePlan {{")?;
150        for probe_set in self.set.iter() {
151            writeln!(f, r#"    ProbeSet("{}") {{"#, probe_set.proto)?;
152            for (delay, node) in probe_set.probes.iter() {
153                writeln!(f, "        {delay:?} to {node},")?;
154            }
155            writeln!(f, "    }}")?;
156        }
157        writeln!(f, "}}")
158    }
159}

161impl FromIterator<ProbeSet> for ProbePlan {
162    fn from_iter<T: IntoIterator<Item = ProbeSet>>(iter: T) -> Self {
163        Self {
164            set: iter.into_iter().collect(),
165        }
166    }
167}

176    /// Shorthand which declares a new ProbeSet.
177    ///
178    /// `$kind`: The `Probe`.
179    /// `$node`: Expression which will be an `Arc<RelayNode>`.
180    /// `$delays`: A `Vec` of the delays for this probe.
181    macro_rules! probeset {
182        (proto: Probe::$kind:ident, relay: $node:expr, delays: $delays:expr,) => {
183            ProbeSet {
184                proto: Probe::$kind,
185                probes: $delays.iter().map(|delay| (*delay, $node)).collect(),
186            }
187        };
188    }
189
190    fn default_protocols() -> BTreeSet<Probe> {
191        BTreeSet::from([Probe::QadIpv4, Probe::QadIpv6, Probe::Https])
192    }

194    #[tokio::test]
195    async fn test_initial_probeplan() {
196        let (_servers, relay_map) = test_utils::relay_map(2).await;
197        let relay_node_1 = relay_map.nodes().next().unwrap();
198        let relay_node_2 = relay_map.nodes().nth(1).unwrap();
199        let plan = ProbePlan::initial(&relay_map, &default_protocols());
200
201        let expected_plan: ProbePlan = [
202            probeset! {
203                proto: Probe::Https,
204                relay: relay_node_1.clone(),
205                delays: [
206                    Duration::from_millis(200),
207                    Duration::from_millis(300),
208                    Duration::from_millis(400)
209                ],
210            },
211            probeset! {
212                proto: Probe::Https,
213                relay: relay_node_2.clone(),
214                delays: [
215                    Duration::from_millis(200),
216                    Duration::from_millis(300),
217                    Duration::from_millis(400)
218                ],
219            },
220        ]
221        .into_iter()
222        .collect();
223
224        println!("expected:");
225        println!("{expected_plan}");
226        println!("actual:");
227        println!("{plan}");
228        // The readable error:
229        assert_eq!(plan.to_string(), expected_plan.to_string());
230        // Just in case there's a bug in the Display impl:
231        assert_eq!(plan, expected_plan);
232    }
