11#[derive(Debug, Serialize, Deserialize)]
12/// JSON representation of a DNS response
13/// See: <https://developers.google.com/speed/public-dns/docs/doh/json>
14pub struct DnsResponse {
15    /// Standard DNS response code
16    #[serde(rename = "Status")]
17    pub status: u32,
18    /// Whether the response was truncated
19    #[serde(rename = "TC")]
20    pub tc: bool,
21    /// Whether recursion was desired
22    #[serde(rename = "RD")]
23    pub rd: bool,
24    /// Whether recursion was available
25    #[serde(rename = "RA")]
26    pub ra: bool,
27    /// Whether the response was validated with DNSSEC
28    #[serde(rename = "AD")]
29    pub ad: bool,
30    /// Whether the client asked to disable DNSSEC validation
31    #[serde(rename = "CD")]
32    pub cd: bool,
33    /// The questions that this request answers
34    #[serde(rename = "Question")]
35    pub question: Vec<DohQuestionJson>,
36    /// The answers to the request
37    #[serde(rename = "Answer")]
38    #[serde(skip_serializing_if = "Vec::is_empty")]
39    pub answer: Vec<DohRecordJson>,
40    /// An optional comment
41    #[serde(rename = "Comment")]
42    pub comment: Option<String>,
43    /// IP Address / scope prefix-length of the client
44    /// See: <https://tools.ietf.org/html/rfc7871>
45    pub edns_client_subnet: Option<String>,
46}
47
48impl DnsResponse {
49    /// Create a new JSON response from a DNS message
50    pub fn from_message(message: proto::op::Message) -> Result<Self> {
51        ensure_whatever!(
52            message.message_type() == proto::op::MessageType::Response,
53            "Expected message type to be response"
54        );
55
56        ensure_whatever!(
57            message.query_count() == message.queries().len() as u16,
58            "Query count mismatch"
59        );
60
61        ensure_whatever!(
62            message.answer_count() == message.answers().len() as u16,
63            "Answer count mismatch"
64        );
65
66        let status: u32 =
67            <u16 as From<proto::op::ResponseCode>>::from(message.response_code()) as u32;
68
69        let question: Vec<_> = message
70            .queries()
71            .iter()
72            .map(DohQuestionJson::from_query)
73            .collect();
74
75        let answer: Vec<_> = message
76            .answers()
77            .iter()
78            .map(DohRecordJson::from_record)
79            .collect::<Result<_>>()?;
80
81        Ok(DnsResponse {
82            status,
83            tc: message.truncated(),
84            rd: message.recursion_desired(),
85            ra: message.recursion_available(),
86            ad: message.authentic_data(),
87            cd: message.checking_disabled(),
88            question,
89            answer,
90            comment: None,
91            edns_client_subnet: None,
92        })
93    }
94}

96#[derive(Debug, Serialize, Deserialize)]
97/// JSON representation of a DNS question
98pub struct DohQuestionJson {
99    /// FQDN with trailing dot
100    pub name: String,
101    /// Standard DNS RR type
102    #[serde(rename = "type")]
103    pub question_type: u16,
104}
105
106impl DohQuestionJson {
107    /// Create a new JSON question from a DNS query
108    pub fn from_query(query: &proto::op::Query) -> Self {
109        Self {
110            name: query.name().to_string(),
111            question_type: query.query_type().into(),
112        }
113    }
114}
115
116#[derive(Debug, Serialize, Deserialize)]
117/// JSON representation of a DNS record
118pub struct DohRecordJson {
119    /// FQDN with trailing dot
120    pub name: String,
121    /// Standard DNS RR type
122    #[serde(rename = "type")]
123    pub record_type: u16,
124    /// Time-to-live, in seconds
125    #[serde(rename = "TTL")]
126    pub ttl: u32,
127    /// Record data
128    pub data: String,
129}
130
131impl DohRecordJson {
132    /// Create a new JSON record from a DNS record
133    pub fn from_record(record: &proto::rr::Record) -> Result<Self> {
134        Ok(Self {
135            name: record.name().to_string(),
136            record_type: record.record_type().into(),
137            ttl: record.ttl(),
138            data: record.data().to_string(),
139        })
140    }
