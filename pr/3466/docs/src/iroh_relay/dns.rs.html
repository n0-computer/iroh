44pub enum DnsError {
45    #[snafu(transparent)]
46    Timeout { source: tokio::time::error::Elapsed },
47    #[snafu(display("No response"))]
48    NoResponse {},
49    #[snafu(display("Resolve failed ipv4: {ipv4}, ipv6 {ipv6}"))]

51        ipv4: Box<DnsError>,
52        ipv6: Box<DnsError>,

56    #[snafu(transparent)]

58        source: hickory_resolver::ResolveError,
59    },
60    #[snafu(display("invalid DNS response: not a query for _iroh.z32encodedpubkey"))]
61    InvalidResponse {},
62}

93impl DnsResolver {
94    /// Create a new DNS resolver with sensible cross-platform defaults.
95    ///
96    /// We first try to read the system's resolver from `/etc/resolv.conf`.
97    /// This does not work at least on some Androids, therefore we fallback
98    /// to the default `ResolverConfig` which uses eg. to google's `8.8.8.8` or `8.8.4.4`.
99    pub fn new() -> Self {
100        let resolver = Self::new_inner();
101        Self {
102            resolver: Arc::new(RwLock::new(resolver)),
103            nameserver: None,
104        }
105    }
106
107    fn new_inner() -> TokioResolver {
108        let (system_config, mut options) =
109            hickory_resolver::system_conf::read_system_conf().unwrap_or_default();
110
111        // Copy all of the system config, but strip the bad windows nameservers.  Unfortunately
112        // there is no easy way to do this.
113        let mut config = hickory_resolver::config::ResolverConfig::new();
114        if let Some(name) = system_config.domain() {
115            config.set_domain(name.clone());
116        }
117        for name in system_config.search() {
118            config.add_search(name.clone());
119        }
120        for nameserver_cfg in system_config.name_servers() {
121            if !WINDOWS_BAD_SITE_LOCAL_DNS_SERVERS.contains(&nameserver_cfg.socket_addr.ip()) {
122                config.add_name_server(nameserver_cfg.clone());
123            }
124        }
125
126        // see [`DnsResolver::lookup_ipv4_ipv6`] for info on why we avoid `LookupIpStrategy::Ipv4AndIpv6`
127        options.ip_strategy = hickory_resolver::config::LookupIpStrategy::Ipv4thenIpv6;
128
129        let mut builder =
130            TokioResolver::builder_with_config(config, TokioConnectionProvider::default());
131        *builder.options_mut() = options;
132        builder.build()
133    }

135    /// Create a new DNS resolver configured with a single UDP DNS nameserver.
136    pub fn with_nameserver(nameserver: SocketAddr) -> Self {
137        let resolver = Self::with_nameserver_inner(nameserver);
138        Self {
139            resolver: Arc::new(RwLock::new(resolver)),
140            nameserver: Some(nameserver),
141        }

144    fn with_nameserver_inner(nameserver: SocketAddr) -> TokioResolver {
145        let mut config = hickory_resolver::config::ResolverConfig::new();
146        let nameserver_config = hickory_resolver::config::NameServerConfig::new(
147            nameserver,
148            hickory_resolver::proto::xfer::Protocol::Udp,
149        );
150        config.add_name_server(nameserver_config);
151
152        let builder =
153            TokioResolver::builder_with_config(config, TokioConnectionProvider::default());
154        builder.build()
155    }

162    /// Recreate the inner resolver
163    pub async fn reset(&self) {
164        let mut this = self.resolver.write().await;
165        let resolver = if let Some(nameserver) = self.nameserver {
166            Self::with_nameserver_inner(nameserver)
167        } else {
168            Self::new_inner()
169        };

171        *this = resolver;
172    }

210    /// Resolve IPv4 and IPv6 in parallel with a timeout.
211    ///
212    /// `LookupIpStrategy::Ipv4AndIpv6` will wait for ipv6 resolution timeout, even if it is
213    /// not usable on the stack, so we manually query both lookups concurrently and time them out
214    /// individually.
215    pub async fn lookup_ipv4_ipv6<T: ToString>(
216        &self,
217        host: T,
218        timeout: Duration,
219    ) -> Result<impl Iterator<Item = IpAddr> + use<T>, DnsError> {
220        let host = host.to_string();
221        let res = tokio::join!(
222            self.lookup_ipv4(host.clone(), timeout),
223            self.lookup_ipv6(host, timeout)
224        );
225
226        match res {
227            (Ok(ipv4), Ok(ipv6)) => Ok(LookupIter::Both(ipv4.chain(ipv6))),
228            (Ok(ipv4), Err(_)) => Ok(LookupIter::Ipv4(ipv4)),
229            (Err(_), Ok(ipv6)) => Ok(LookupIter::Ipv6(ipv6)),
230            (Err(ipv4_err), Err(ipv6_err)) => Err(ResolveBothSnafu {
231                ipv4: Box::new(ipv4_err),
232                ipv6: Box::new(ipv6_err),
233            }
234            .build()),
235        }
236    }

238    /// Resolve a hostname from a URL to an IP address.
239    pub async fn resolve_host(
240        &self,
241        url: &Url,
242        prefer_ipv6: bool,
243        timeout: Duration,
244    ) -> Result<IpAddr, DnsError> {
245        let host = url.host().context(MissingHostSnafu)?;
246        match host {
247            url::Host::Domain(domain) => {
248                // Need to do a DNS lookup
249                let lookup = tokio::join!(
250                    self.lookup_ipv4(domain, timeout),
251                    self.lookup_ipv6(domain, timeout)
252                );
253                let (v4, v6) = match lookup {
254                    (Err(ipv4_err), Err(ipv6_err)) => {
255                        return Err(ResolveBothSnafu {
256                            ipv4: Box::new(ipv4_err),
257                            ipv6: Box::new(ipv6_err),
258                        }
259                        .build());
260                    }
261                    (Err(_), Ok(mut v6)) => (None, v6.next()),
262                    (Ok(mut v4), Err(_)) => (v4.next(), None),
263                    (Ok(mut v4), Ok(mut v6)) => (v4.next(), v6.next()),
264                };
265                if prefer_ipv6 {
266                    v6.or(v4).context(NoResponseSnafu)
267                } else {
268                    v4.or(v6).context(NoResponseSnafu)
269                }
270            }
271            url::Host::Ipv4(ip) => Ok(IpAddr::V4(ip)),
272            url::Host::Ipv6(ip) => Ok(IpAddr::V6(ip)),
273        }
274    }

277    ///
278    /// From the moment this function is called, each lookup is scheduled after the delays in
279    /// `delays_ms` with the first call being done immediately. `[200ms, 300ms]` results in calls

401impl reqwest::dns::Resolve for DnsResolver {
402    fn resolve(&self, name: reqwest::dns::Name) -> reqwest::dns::Resolving {
403        let this = self.clone();
404        let name = name.as_str().to_string();
405        Box::pin(async move {
406            let res = this.lookup_ipv4_ipv6(name, DNS_TIMEOUT).await;
407            match res {
408                Ok(addrs) => {
409                    let addrs: reqwest::dns::Addrs =
410                        Box::new(addrs.map(|addr| SocketAddr::new(addr, 0)));
411                    Ok(addrs)
412                }
413                Err(err) => {
414                    let err: Box<dyn std::error::Error + Send + Sync> = Box::new(err);
415                    Err(err)
416                }
417            }
418        })
419    }
420}

459/// Deprecated IPv6 site-local anycast addresses still configured by windows.
460///
461/// Windows still configures these site-local addresses as soon even as an IPv6 loopback
462/// interface is configured.  We do not want to use these DNS servers, the chances of them
463/// being usable are almost always close to zero, while the chance of DNS configuration
464/// **only** relying on these servers and not also being configured normally are also almost
465/// zero.  The chance of the DNS resolver accidentally trying one of these and taking a
466/// bunch of timeouts to figure out they're no good are on the other hand very high.
467const WINDOWS_BAD_SITE_LOCAL_DNS_SERVERS: [IpAddr; 3] = [
468    IpAddr::V6(Ipv6Addr::new(0xfec0, 0, 0, 0xffff, 0, 0, 0, 1)),
469    IpAddr::V6(Ipv6Addr::new(0xfec0, 0, 0, 0xffff, 0, 0, 0, 2)),
470    IpAddr::V6(Ipv6Addr::new(0xfec0, 0, 0, 0xffff, 0, 0, 0, 3)),
471];

473/// Helper enum to give a unified type to the iterators of [`DnsResolver::lookup_ipv4_ipv6`].
474enum LookupIter<A, B> {
475    Ipv4(A),
476    Ipv6(B),
477    Both(std::iter::Chain<A, B>),
478}
479
480impl<A: Iterator<Item = IpAddr>, B: Iterator<Item = IpAddr>> Iterator for LookupIter<A, B> {
481    type Item = IpAddr;
482
483    fn next(&mut self) -> Option<Self::Item> {
484        match self {
485            LookupIter::Ipv4(iter) => iter.next(),
486            LookupIter::Ipv6(iter) => iter.next(),
487            LookupIter::Both(iter) => iter.next(),
488        }
489    }
490}

492/// Staggers calls to the future F with the given delays.
493///
494/// The first call is performed immediately. The first call to succeed generates an Ok result
495/// ignoring any previous error. If all calls fail, an error summarizing all errors is returned.
496async fn stagger_call<
497    T,
498    E: std::fmt::Debug + std::fmt::Display,
499    F: Fn() -> Fut,
500    Fut: Future<Output = Result<T, E>>,
501>(
502    f: F,
503    delays_ms: &[u64],
504) -> Result<T, StaggeredError<E>> {
505    let mut calls = n0_future::FuturesUnorderedBounded::new(delays_ms.len() + 1);
506    // NOTE: we add the 0 delay here to have a uniform set of futures. This is more performant than
507    // using alternatives that allow futures of different types.
508    for delay in std::iter::once(&0u64).chain(delays_ms) {
509        let delay = add_jitter(delay);
510        let fut = f();
511        let staggered_fut = async move {
512            time::sleep(delay).await;
513            fut.await
514        };
515        calls.push(staggered_fut)
516    }
517
518    let mut errors = vec![];
519    while let Some(call_result) = calls.next().await {
520        match call_result {
521            Ok(t) => return Ok(t),
522            Err(e) => errors.push(e),
523        }
524    }
525
526    Err(StaggeredError::new(errors))
527}
528
529fn add_jitter(delay: &u64) -> Duration {
530    // If delay is 0, return 0 immediately.
531    if *delay == 0 {
532        return Duration::ZERO;
533    }
534
535    // Calculate jitter as a random value in the range of +/- MAX_JITTER_PERCENT of the delay.
536    let max_jitter = delay.saturating_mul(MAX_JITTER_PERCENT * 2) / 100;
537    let jitter = rand::random::<u64>() % max_jitter;
538
539    Duration::from_millis(delay.saturating_sub(max_jitter / 2).saturating_add(jitter))
540}
