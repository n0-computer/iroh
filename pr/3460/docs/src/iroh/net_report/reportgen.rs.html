1//! The reportgen actor is responsible for generating a single net_report report.
2//!
3//! It is implemented as an actor with [`Client`] as handle.
4//!
5//! The actor starts generating the report as soon as it is created, it does not receive any
6//! messages from the client.  It follows roughly these steps:
7//!
8//! - Determines host IPv6 support.
9//! - Creates portmapper future.
10//! - Creates captive portal detection future.
11//! - Creates Probe Set futures.
12//!   - These send messages to the reportgen actor.
13//! - Loops driving the futures and handling actor messages:
14//!   - Disables futures as they are completed or aborted.
15//!   - Stop if there are no outstanding tasks/futures, or on timeout.
16//! - Sends the completed report to the net_report actor.
17

113    /// Creates a new actor generating a single report.
114    ///
115    /// The actor starts running immediately and only generates a single report, after which
116    /// it shuts down.  Dropping this handle will abort the actor.
117    pub(super) fn new(
118        last_report: Option<Report>,
119        relay_map: RelayMap,
120        protocols: BTreeSet<Probe>,
121        if_state: IfStateDetails,
122        #[cfg(not(wasm_browser))] socket_state: SocketState,
123        #[cfg(any(test, feature = "test-utils"))] insecure_skip_relay_cert_verify: bool,
124    ) -> (Self, mpsc::Receiver<ProbeFinished>) {
125        let (msg_tx, msg_rx) = mpsc::channel(32);
126        let actor = Actor {
127            msg_tx,
128            last_report,
129            relay_map,
130            protocols,
131            #[cfg(not(wasm_browser))]
132            socket_state,
133            #[cfg(any(test, feature = "test-utils"))]
134            insecure_skip_relay_cert_verify,
135            if_state,
136        };
137        let task = task::spawn(actor.run().instrument(info_span!("reportgen.actor")));
138        (
139            Self {
140                _drop_guard: AbortOnDropHandle::new(task),
141            },
142            msg_rx,
143        )
144    }

188#[derive(Debug)]

195impl Actor {
196    async fn run(self) {
197        match time::timeout(OVERALL_REPORT_TIMEOUT, self.run_inner()).await {
198            Ok(()) => debug!("reportgen actor finished"),
199            Err(time::Elapsed { .. }) => {
200                warn!("reportgen timed out");
201            }
202        }
203    }
204
205    /// Runs the main reportgen actor logic.
206    ///
207    /// This actor runs by:
208    ///
209    /// - Creates a captive portal future.
210    /// - Creates ProbeSet futures in a group of futures.
211    /// - Runs a main loop:
212    ///   - Drives all the above futures.
213    ///   - Receives actor messages (sent by those futures).
214    ///   - Updates the report, cancels unneeded futures.
215    /// - Sends the report to the net_report actor.
216    async fn run_inner(self) {
217        debug!("reportstate actor starting");
218
219        let mut probes = JoinSet::default();
220
221        let _probes_token = self.spawn_probes_task(self.if_state.clone(), &mut probes);
222        let mut num_probes = probes.len();
223
224        let captive_token = self.prepare_captive_portal_task(&mut probes);
225
226        // any reports of working UDP/QUIC?
227        let mut have_udp = false;
228
229        // Check for probes finishing.
230        while let Some(probe_result) = probes.join_next().await {
231            trace!(?probe_result, num_probes, "processing finished probe");
232            match probe_result {
233                Ok(report) => {
234                    #[cfg_attr(wasm_browser, allow(irrefutable_let_patterns))]
235                    if let ProbeFinished::Regular(report) = &report {
236                        have_udp |= report.as_ref().map(|r| r.is_udp()).unwrap_or_default();
237                        num_probes -= 1;
238
239                        // If all probes are done & we have_udp cancel captive
240                        if num_probes == 0 {
241                            debug!("all regular probes done");
242                            debug_assert!(probes.len() <= 1, "{} probes", probes.len());
243
244                            if have_udp {
245                                captive_token.cancel();
246                            }
247                        }
248                    }
249                    self.msg_tx.send(report).await.ok();
250                }
251                Err(e) => {
252                    if e.is_panic() {
253                        error!("Task panicked {:?}", e);
254                        break;
255                    }
256                    warn!("probes task join error: {:?}", e);
257                }
258            }
259        }
260    }
261
262    /// Creates the future which will perform the captive portal check.
263    fn prepare_captive_portal_task(&self, tasks: &mut JoinSet<ProbeFinished>) -> CancellationToken {
264        let token = CancellationToken::new();
265
266        // If we're doing a full probe, also check for a captive portal. We
267        // delay by a bit to wait for UDP QAD to finish, to avoid the probe if
268        // it's unnecessary.
269        #[cfg(not(wasm_browser))]
270        if self.last_report.is_none() {
271            // Even if we're doing a non-incremental update, we may want to try our
272            // preferred relay for captive portal detection.
273            let preferred_relay = self
274                .last_report
275                .as_ref()
276                .and_then(|l| l.preferred_relay.clone());
277
278            let dns_resolver = self.socket_state.dns_resolver.clone();
279            let dm = self.relay_map.clone();
280            let token = token.clone();
281            tasks.spawn(
282                async move {
283                    let res = token
284                        .run_until_cancelled_owned(async move {
285                            time::sleep(CAPTIVE_PORTAL_DELAY).await;
286                            trace!("check started after {CAPTIVE_PORTAL_DELAY:?}");
287                            time::timeout(
288                                CAPTIVE_PORTAL_TIMEOUT,
289                                check_captive_portal(&dns_resolver, &dm, preferred_relay),
290                            )
291                            .await
292                        })
293                        .await;
294                    let res = match res {
295                        Some(Ok(Ok(found))) => Some(found),
296                        Some(Ok(Err(err))) => {
297                            match err {
298                                CaptivePortalError::CreateReqwestClient { source }
299                                | CaptivePortalError::HttpRequest { source }
300                                    if source.is_connect() =>
301                                {
302                                    debug!("check_captive_portal failed: {source:#}");
303                                }
304                                err => warn!("check_captive_portal error: {err:#}"),
305                            }
306                            None
307                        }
308                        Some(Err(time::Elapsed { .. })) => {
309                            warn!("probe timed out");
310                            None
311                        }
312                        None => {
313                            trace!("probe cancelled");
314                            None
315                        }
316                    };
317                    ProbeFinished::CaptivePortal(res)
318                }
319                .instrument(debug_span!("captive-portal")),
320            );
321        }
322        token
323    }
324
325    /// Prepares the future which will run all the probes as per generated ProbePlan.
326    ///
327    /// Probes operate like the following:
328    ///
329    /// - A future is created for each probe in all probe sets.
330    /// - All probes in a set are grouped in [`JoinSet`].
331    /// - All those probe sets are grouped in one overall [`JoinSet`].
332    ///   - This future is polled by the main actor loop to make progress.
333    /// - Once a probe future is polled:
334    ///   - Many probes start with a delay, they sleep during this time.
335    ///   - When a probe starts it first asks the reportgen [`Actor`] if it is still useful
336    ///     to run.  If not it aborts the entire probe set.
337    ///   - When a probe finishes, its [`ProbeReport`] is yielded to the reportgen actor.
338    /// - Probes get aborted in several ways:
339    ///   - A running it can fail and abort the entire probe set if it deems the
340    ///     failure permanent.  Probes in a probe set are essentially retries.
341    ///   - Once there are [`ProbeReport`]s from enough nodes, all remaining probes are
342    ///     aborted.  That is, the main actor loop stops polling them.
343    fn spawn_probes_task(
344        &self,
345        if_state: IfStateDetails,
346        probes: &mut JoinSet<ProbeFinished>,
347    ) -> CancellationToken {
348        debug!(?if_state, "local interface details");
349        let plan = match self.last_report {
350            Some(ref report) => {
351                ProbePlan::with_last_report(&self.relay_map, report, &self.protocols)
352            }
353            None => ProbePlan::initial(&self.relay_map, &self.protocols),
354        };
355        trace!(%plan, "probe plan");
356
357        let token = CancellationToken::new();
358
359        for probe_set in plan.iter() {
360            let set_token = token.child_token();
361            let proto = probe_set.proto();
362            for (delay, relay_node) in probe_set.params() {
363                let probe_token = set_token.child_token();
364
365                let fut = probe_token.run_until_cancelled_owned(time::timeout(
366                    PROBES_TIMEOUT,
367                    proto.run(
368                        *delay,
369                        relay_node.clone(),
370                        #[cfg(not(wasm_browser))]
371                        self.socket_state.clone(),
372                        #[cfg(any(test, feature = "test-utils"))]
373                        self.insecure_skip_relay_cert_verify,
374                    ),
375                ));
376                probes.spawn(
377                    async move {
378                        let res = fut.await;
379                        let res = match res {
380                            Some(Ok(Ok(report))) => Ok(report),
381                            Some(Ok(Err(err))) => {
382                                warn!("probe failed: {:#}", err);
383                                Err(probes_error::ProbeFailureSnafu {}.into_error(err))
384                            }
385                            Some(Err(time::Elapsed { .. })) => {
386                                Err(probes_error::TimeoutSnafu.build())
387                            }
388                            None => Err(probes_error::CancelledSnafu.build()),
389                        };
390                        ProbeFinished::Regular(res)
391                    }
392                    .instrument(debug_span!(
393                        "run-probe",
394                        ?proto,
395                        ?delay,
396                        ?relay_node
397                    )),
398                );
399            }
400        }
401
402        token
403    }
404}

485impl Probe {
486    /// Executes this particular [`Probe`], including using a delayed start if needed.
487    async fn run(
488        self,
489        delay: Duration,
490        relay_node: Arc<RelayNode>,
491        #[cfg(not(wasm_browser))] socket_state: SocketState,
492        #[cfg(any(test, feature = "test-utils"))] insecure_skip_relay_cert_verify: bool,
493    ) -> Result<ProbeReport, ProbeError> {
494        if !delay.is_zero() {
495            trace!("delaying probe");
496            time::sleep(delay).await;
497        }
498        debug!("starting probe");
499
500        match self {
501            Probe::Https => {
502                match run_https_probe(
503                    #[cfg(not(wasm_browser))]
504                    &socket_state.dns_resolver,
505                    relay_node.url.clone(),
506                    #[cfg(any(test, feature = "test-utils"))]
507                    insecure_skip_relay_cert_verify,
508                )
509                .await
510                {
511                    Ok(report) => Ok(ProbeReport::Https(report)),
512                    Err(err) => Err(probe_error::HttpsSnafu.into_error(err)),
513                }
514            }
515            #[cfg(not(wasm_browser))]
516            Probe::QadIpv4 | Probe::QadIpv6 => unreachable!("must not be used"),
517        }
518    }
519}

545/// Reports whether or not we think the system is behind a
546/// captive portal, detected by making a request to a URL that we know should
547/// return a "204 No Content" response and checking if that's what we get.
548///
549/// The boolean return is whether we think we have a captive portal.
550#[cfg(not(wasm_browser))]
551async fn check_captive_portal(
552    dns_resolver: &DnsResolver,
553    dm: &RelayMap,
554    preferred_relay: Option<RelayUrl>,
555) -> Result<bool, CaptivePortalError> {
556    // If we have a preferred relay node and we can use it for non-QAD requests, try that;
557    // otherwise, pick a random one suitable for non-STUN requests.
558
559    let preferred_relay = preferred_relay.and_then(|url| dm.get_node(&url).map(|_| url));
560
561    let url = match preferred_relay {
562        Some(url) => url,
563        None => {
564            let urls: Vec<_> = dm.nodes().map(|n| n.url.clone()).collect();
565            if urls.is_empty() {
566                debug!("No suitable relay node for captive portal check");
567                return Ok(false);
568            }
569
570            let i = (0..urls.len())
571                .choose(&mut rand::thread_rng())
572                .unwrap_or_default();
573            urls[i].clone()
574        }
575    };
576
577    let mut builder = reqwest::ClientBuilder::new().redirect(reqwest::redirect::Policy::none());
578
579    if let Some(Host::Domain(domain)) = url.host() {
580        // Use our own resolver rather than getaddrinfo
581        //
582        // Be careful, a non-zero port will override the port in the URI.
583        //
584        // Ideally we would try to resolve **both** IPv4 and IPv6 rather than purely race
585        // them.  But our resolver doesn't support that yet.
586        let addrs: Vec<_> = dns_resolver
587            .lookup_ipv4_ipv6_staggered(domain, DNS_TIMEOUT, DNS_STAGGERING_MS)
588            .await?
589            .map(|ipaddr| SocketAddr::new(ipaddr, 0))
590            .collect();
591        builder = builder.resolve_to_addrs(domain, &addrs);
592    }
593    let client = builder
594        .build()
595        .context(captive_portal_error::CreateReqwestClientSnafu)?;
596
597    // Note: the set of valid characters in a challenge and the total
598    // length is limited; see is_challenge_char in bin/iroh-relay for more
599    // details.
600
601    let host_name = url.host_str().unwrap_or_default();
602    let challenge = format!("ts_{host_name}");
603    let portal_url = format!("http://{host_name}/generate_204");
604    let res = client
605        .request(reqwest::Method::GET, portal_url)
606        .header("X-Tailscale-Challenge", &challenge)
607        .send()
608        .await
609        .context(captive_portal_error::HttpRequestSnafu)?;
610
611    let expected_response = format!("response {challenge}");
612    let is_valid_response = res
613        .headers()
614        .get("X-Tailscale-Response")
615        .map(|s| s.to_str().unwrap_or_default())
616        == Some(&expected_response);
617
618    debug!(
619        "check_captive_portal url={} status_code={} valid_response={}",
620        res.url(),
621        res.status(),
622        is_valid_response,
623    );
624    let has_captive = res.status() != 204 || !is_valid_response;
625
626    Ok(has_captive)
627}

664pub(super) async fn get_relay_addr_ipv4(
665    dns_resolver: &DnsResolver,
666    relay_node: &RelayNode,
667) -> Result<SocketAddrV4, GetRelayAddrError> {
668    let port = get_quic_port(relay_node).context(get_relay_addr_error::MissingPortSnafu)?;
669    relay_lookup_ipv4_staggered(dns_resolver, relay_node, port).await
670}

681/// Do a staggared ipv4 DNS lookup based on [`RelayNode`]
682///
683/// `port` is combined with the resolved [`std::net::Ipv4Addr`] to return a [`SocketAddr`]
684#[cfg(not(wasm_browser))]
685async fn relay_lookup_ipv4_staggered(
686    dns_resolver: &DnsResolver,
687    relay: &RelayNode,
688    port: u16,
689) -> Result<SocketAddrV4, GetRelayAddrError> {
690    match relay.url.host() {
691        Some(url::Host::Domain(hostname)) => {
692            trace!(%hostname, "Performing DNS A lookup for relay addr");
693            match dns_resolver
694                .lookup_ipv4_staggered(hostname, DNS_TIMEOUT, DNS_STAGGERING_MS)
695                .await
696            {
697                Ok(mut addrs) => addrs
698                    .next()
699                    .map(|ip| ip.to_canonical())
700                    .map(|addr| match addr {
701                        IpAddr::V4(ip) => SocketAddrV4::new(ip, port),
702                        IpAddr::V6(_) => unreachable!("bad DNS lookup: {:?}", addr),
703                    })
704                    .ok_or(get_relay_addr_error::NoAddrFoundSnafu.build()),
705                Err(err) => Err(get_relay_addr_error::DnsLookupSnafu.into_error(err)),
706            }
707        }
708        Some(url::Host::Ipv4(addr)) => Ok(SocketAddrV4::new(addr, port)),
709        Some(url::Host::Ipv6(_addr)) => Err(get_relay_addr_error::NoAddrFoundSnafu.build()),
710        None => Err(get_relay_addr_error::InvalidHostnameSnafu.build()),
711    }
712}

714/// Do a staggared ipv6 DNS lookup based on [`RelayNode`]
715///
716/// `port` is combined with the resolved [`std::net::Ipv6Addr`] to return a [`SocketAddr`]
717#[cfg(not(wasm_browser))]
718async fn relay_lookup_ipv6_staggered(
719    dns_resolver: &DnsResolver,
720    relay: &RelayNode,
721    port: u16,
722) -> Result<SocketAddrV6, GetRelayAddrError> {
723    match relay.url.host() {
724        Some(url::Host::Domain(hostname)) => {
725            trace!(%hostname, "Performing DNS AAAA lookup for relay addr");
726            match dns_resolver
727                .lookup_ipv6_staggered(hostname, DNS_TIMEOUT, DNS_STAGGERING_MS)
728                .await
729            {
730                Ok(mut addrs) => addrs
731                    .next()
732                    .map(|addr| match addr {
733                        IpAddr::V4(_) => unreachable!("bad DNS lookup: {:?}", addr),
734                        IpAddr::V6(ip) => SocketAddrV6::new(ip, port, 0, 0),
735                    })
736                    .ok_or(get_relay_addr_error::NoAddrFoundSnafu.build()),
737                Err(err) => Err(get_relay_addr_error::DnsLookupSnafu.into_error(err)),
738            }
739        }
740        Some(url::Host::Ipv4(_addr)) => Err(get_relay_addr_error::NoAddrFoundSnafu.build()),
741        Some(url::Host::Ipv6(addr)) => Ok(SocketAddrV6::new(addr, port, 0, 0)),
742        None => Err(get_relay_addr_error::InvalidHostnameSnafu.build()),
743    }
744}

747#[snafu(module)]
748#[non_exhaustive]
749pub enum MeasureHttpsLatencyError {
750    #[snafu(transparent)]
751    InvalidUrl { source: url::ParseError },
752    #[cfg(not(wasm_browser))]
753    #[snafu(transparent)]
754    DnsLookup { source: StaggeredError<DnsError> },
755    #[snafu(display("Creating HTTP client failed"))]
756    CreateReqwestClient { source: reqwest::Error },
757    #[snafu(display("HTTP request failed"))]
758    HttpRequest { source: reqwest::Error },
759    #[snafu(display("Error response from server {status}: {:?}", status.canonical_reason()))]
760    InvalidResponse { status: StatusCode },
761}

763/// Executes an HTTPS probe.
764///
765/// If `certs` is provided they will be added to the trusted root certificates, allowing the
766/// use of self-signed certificates for servers.  Currently this is used for testing.
767#[allow(clippy::unused_async)]
768async fn run_https_probe(
769    #[cfg(not(wasm_browser))] dns_resolver: &DnsResolver,
770    relay_node: RelayUrl,
771    #[cfg(any(test, feature = "test-utils"))] insecure_skip_relay_cert_verify: bool,
772) -> Result<HttpsProbeReport, MeasureHttpsLatencyError> {
773    trace!("HTTPS probe start");
774    let url = relay_node.join(RELAY_PROBE_PATH)?;
775
776    // This should also use same connection establishment as relay client itself, which
777    // needs to be more configurable so users can do more crazy things:
778    // https://github.com/n0-computer/iroh/issues/2901
779    let mut builder = reqwest::ClientBuilder::new();
780
781    #[cfg(not(wasm_browser))]
782    {
783        builder = builder.redirect(reqwest::redirect::Policy::none());
784    }
785
786    #[cfg(not(wasm_browser))]
787    if let Some(Host::Domain(domain)) = url.host() {
788        // Use our own resolver rather than getaddrinfo
789        //
790        // Be careful, a non-zero port will override the port in the URI.
791        //
792        // The relay Client uses `.lookup_ipv4_ipv6` to connect, so use the same function
793        // but staggered for reliability.  Ideally this tries to resolve **both** IPv4 and
794        // IPv6 though.  But our resolver does not have a function for that yet.
795        let addrs: Vec<_> = dns_resolver
796            .lookup_ipv4_ipv6_staggered(domain, DNS_TIMEOUT, DNS_STAGGERING_MS)
797            .await?
798            .map(|ipaddr| SocketAddr::new(ipaddr, 0))
799            .collect();
800        trace!(?addrs, "resolved addrs");
801        builder = builder.resolve_to_addrs(domain, &addrs);
802    }
803
804    #[cfg(all(not(wasm_browser), any(test, feature = "test-utils")))]
805    let builder = builder.danger_accept_invalid_certs(insecure_skip_relay_cert_verify);
806
807    let client = builder
808        .build()
809        .context(measure_https_latency_error::CreateReqwestClientSnafu)?;
810
811    let start = Instant::now();
812    let response = client
813        .request(reqwest::Method::GET, url)
814        .send()
815        .await
816        .context(measure_https_latency_error::HttpRequestSnafu)?;
817    let latency = start.elapsed();
818    if response.status().is_success() {
819        // Drain the response body to be nice to the server, up to a limit.
820        const MAX_BODY_SIZE: usize = 8 << 10; // 8 KiB
821        let mut body_size = 0;
822        let mut stream = response.bytes_stream();
823        // ignore failing frames
824        while let Some(Ok(chunk)) = stream.next().await {
825            body_size += chunk.len();
826            if body_size >= MAX_BODY_SIZE {
827                break;
828            }
829        }
830
831        Ok(HttpsProbeReport {
832            node: relay_node,
833            latency,
834        })
835    } else {
836        Err(measure_https_latency_error::InvalidResponseSnafu {
837            status: response.status(),
838        }
839        .build())
840    }
841}
