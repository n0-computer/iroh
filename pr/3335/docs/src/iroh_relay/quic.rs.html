66    impl QuicServer {
67        /// Returns a handle for this server.
68        ///
69        /// The server runs in the background as several async tasks.  This allows controlling
70        /// the server, in particular it allows gracefully shutting down the server.
71        pub fn handle(&self) -> ServerHandle {
72            ServerHandle {

74            }

77        /// Returns the [`AbortOnDropHandle`] for the supervisor task managing the endpoint.
78        ///
79        /// This is the root of all the tasks for the QUIC address discovery service.  Aborting it will abort all the
80        /// other tasks for the service.  Awaiting it will complete when all the service tasks are
81        /// completed.[]
82        pub fn task_handle(&mut self) -> &mut AbortOnDropHandle<()> {
83            &mut self.handle
84        }

87        pub fn bind_addr(&self) -> SocketAddr {
88            self.bind_addr

91        /// Spawns a QUIC server that creates and QUIC endpoint and listens
92        /// for QUIC connections for address discovery
93        ///
94        /// # Errors
95        /// If the given `quic_config` contains a [`rustls::ServerConfig`] that cannot
96        /// be converted to a [`QuicServerConfig`], usually because it does not support
97        /// TLS 1.3, a [`NoInitialCipherSuite`] will occur.
98        ///
99        /// # Panics
100        /// If there is a panic during a connection, it will be propagated
101        /// up here. Any other errors in a connection will be logged as a
102        ///  warning.
103        pub(crate) fn spawn(mut quic_config: QuicConfig) -> Result<Self, QuicSpawnError> {
104            quic_config.server_config.alpn_protocols =
105                vec![crate::quic::ALPN_QUIC_ADDR_DISC.to_vec()];
106            let server_config = QuicServerConfig::try_from(quic_config.server_config)?;
107            let mut server_config = quinn::ServerConfig::with_crypto(Arc::new(server_config));
108            let transport_config =
109                Arc::get_mut(&mut server_config.transport).expect("not used yet");
110            transport_config
111                .max_concurrent_uni_streams(0_u8.into())
112                .max_concurrent_bidi_streams(0_u8.into())
113                // enable sending quic address discovery frames
114                .send_observed_address_reports(true);
115
116            let endpoint = quinn::Endpoint::server(server_config, quic_config.bind_addr)
117                .context(EndpointServerSnafu)?;
118            let bind_addr = endpoint.local_addr().context(LocalAddrSnafu)?;
119
120            info!(?bind_addr, "QUIC server listening on");
121
122            let cancel = CancellationToken::new();
123            let cancel_accept_loop = cancel.clone();
124
125            let task = tokio::task::spawn(
126                async move {
127                    let mut set = JoinSet::new();
128                    debug!("waiting for connections...");
129                    loop {
130                        tokio::select! {
131                            biased;
132                            _ = cancel_accept_loop.cancelled() => {
133                                break;
134                            }
135                            Some(res) = set.join_next() => {
136                                if let Err(err) = res {
137                                    if err.is_panic() {
138                                        panic!("task panicked: {err:#?}");
139                                    } else {
140                                        debug!("error accepting incoming connection: {err:#?}");
141                                    }
142                                }
143                            }
144                            res = endpoint.accept() => match res {
145                                Some(conn) => {
146                                     debug!("accepting connection");
147                                     let remote_addr = conn.remote_address();
148                                     set.spawn(
149                                         handle_connection(conn).instrument(info_span!("qad-conn", %remote_addr))
150                                     );                                }
151                                None => {
152                                    debug!("endpoint closed");
153                                    break;
154                                }
155                            }
156                        }
157                    }
158                    // close all connections and wait until they have all grace
159                    // fully closed.
160                    endpoint.close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
161                    endpoint.wait_idle().await;
162
163                    // all tasks should be closed, since the endpoint has shutdown
164                    // all connections, but await to ensure they are finished.
165                    set.abort_all();
166                    while !set.is_empty() {
167                        _ = set.join_next().await;
168                    }
169
170                    debug!("quic endpoint has been shutdown.");
171                }
172                .instrument(info_span!("quic-endpoint")),
173            );
174            Ok(Self {
175                bind_addr,
176                cancel,
177                handle: AbortOnDropHandle::new(task),
178            })
179        }
180
181        /// Closes the underlying QUIC endpoint and the tasks running the
182        /// QUIC connections.
183        pub async fn shutdown(mut self) {
184            self.cancel.cancel();
185            if !self.task_handle().is_finished() {
186                // only possible error is a `JoinError`, no errors about what might
187                // have happened during a connection are propagated.
188                _ = self.task_handle().await;
189            }
190        }
191    }

208    /// Handle the connection from the client.
209    async fn handle_connection(incoming: quinn::Incoming) -> Result<(), ConnectionError> {
210        let connection = match incoming.await {
211            Ok(conn) => conn,
212            Err(e) => {
213                return Err(e);
214            }
215        };
216        debug!("established");
217        // wait for the client to close the connection
218        let connection_err = connection.closed().await;
219        match connection_err {
220            quinn::ConnectionError::ApplicationClosed(ApplicationClose { error_code, .. })
221                if error_code == QUIC_ADDR_DISC_CLOSE_CODE =>
222            {
223                Ok(())
224            }
225            _ => Err(connection_err),
226        }
227    }

250/// Handles the client side of QUIC address discovery.
251#[derive(Debug)]
252pub struct QuicClient {
253    /// A QUIC Endpoint.
254    ep: quinn::Endpoint,
255    /// A client config.
256    client_config: quinn::ClientConfig,
257}
258
259impl QuicClient {
260    /// Create a new QuicClient to handle the client side of QUIC
261    /// address discovery.
262    pub fn new(
263        ep: quinn::Endpoint,
264        mut client_config: rustls::ClientConfig,
265    ) -> Result<Self, Error> {
266        // add QAD alpn
267        client_config.alpn_protocols = vec![ALPN_QUIC_ADDR_DISC.into()];
268        // go from rustls client config to rustls QUIC specific client config to
269        // a quinn client config
270        let mut client_config =
271            quinn::ClientConfig::new(Arc::new(QuicClientConfig::try_from(client_config)?));
272
273        // enable the receive side of address discovery
274        let mut transport = quinn_proto::TransportConfig::default();
275        // Setting the initial RTT estimate to a low value means
276        // we're sacrificing initial throughput, which is fine for
277        // QAD, which doesn't require us to have good initial throughput.
278        // It also implies a 999ms probe timeout, which means that
279        // if the packet gets lots (e.g. because we're probing ipv6, but
280        // ipv6 packets always get lost in our network configuration) we
281        // time out *closing the connection* after only 999ms.
282        // Even if the round trip time is bigger than 999ms, this doesn't
283        // prevent us from connecting, since that's dependent on the idle
284        // timeout (set to 30s by default).
285        transport.initial_rtt(Duration::from_millis(111));
286        transport.receive_observed_address_reports(true);
287        client_config.transport_config(Arc::new(transport));
288
289        Ok(Self { ep, client_config })
290    }

292    /// Client side of QUIC address discovery.
293    ///
294    /// Creates a connection and returns the observed address
295    /// and estimated latency of the connection.
296    ///
297    /// Consumes and gracefully closes the connection.
298    pub async fn get_addr_and_latency(
299        &self,
300        server_addr: SocketAddr,
301        host: &str,
302    ) -> Result<(SocketAddr, std::time::Duration), Error> {
303        let connecting = self
304            .ep
305            .connect_with(self.client_config.clone(), server_addr, host);
306        let conn = connecting?.await?;
307        let mut external_addresses = conn.observed_external_addr();
308        // TODO(ramfox): I'd like to be able to cancel this so we can close cleanly
309        // if there the task that runs this function gets aborted.
310        // tokio::select! {
311        //     _ = cancel.cancelled() => {
312        //         conn.close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
313        //         bail!("QUIC address discovery canceled early");
314        //     },
315        //     res = external_addresses.wait_for(|addr| addr.is_some()) => {
316        //         let addr = res?.expect("checked");
317        //         let latency = conn.rtt() / 2;
318        //         // gracefully close the connections
319        //         conn.close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
320        //         Ok((addr, latency))
321        //     }
322
323        let res = match external_addresses.wait_for(|addr| addr.is_some()).await {
324            Ok(res) => res,
325            Err(err) => {
326                // attempt to gracefully close the connections
327                conn.close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
328                return Err(err.into());
329            }
330        };
331        let mut observed_addr = res.expect("checked");
332        // if we've sent to an ipv4 address, but received an observed address
333        // that is ivp6 then the address is an [IPv4-Mapped IPv6 Addresses](https://doc.rust-lang.org/beta/std/net/struct.Ipv6Addr.html#ipv4-mapped-ipv6-addresses)
334        observed_addr = SocketAddr::new(observed_addr.ip().to_canonical(), observed_addr.port());
335        let latency = conn.rtt();
336        // gracefully close the connections
337        conn.close(QUIC_ADDR_DISC_CLOSE_CODE, QUIC_ADDR_DISC_CLOSE_REASON);
338        Ok((observed_addr, latency))
339    }

362        let host: Ipv4Addr = "127.0.0.1".parse().unwrap();
363        // create a server config with self signed certificates
364        let (_, server_config) = super::super::server::testing::self_signed_tls_certs_and_config();
365        let bind_addr = SocketAddr::new(host.into(), 0);
366        let quic_server = QuicServer::spawn(QuicConfig {
367            server_config,
368            bind_addr,
369        })?;
370
371        // create a client-side endpoint
372        let client_endpoint =
373            quinn::Endpoint::client(SocketAddr::new(host.into(), 0)).context("client")?;
374        let client_addr = client_endpoint.local_addr().context("local addr")?;
375
376        // create the client configuration used for the client endpoint when they
377        // initiate a connection with the server
378        let client_config = crate::client::make_dangerous_client_config();
379        let quic_client = QuicClient::new(client_endpoint.clone(), client_config)?;
380

382            .get_addr_and_latency(quic_server.bind_addr(), &host.to_string())

384
385        // wait until the endpoint delivers the closing message to the server
386        client_endpoint.wait_idle().await;
387        // shut down the quic server

389
390        assert_eq!(client_addr, addr);
391        Ok(())

440    /// Makes sure that, even though the RTT was set to some fairly low value,
441    /// we *do* try to connect for longer than what the time out would be after closing
442    /// the connection, when we *don't* close the connection.
443    ///
444    /// In this case we don't simulate it via synthetically high RTT, but by dropping
445    /// all packets on the server-side for 2 seconds.
446    #[tokio::test]
447    #[traced_test]
448    async fn test_qad_connect_delayed() -> Result {
449        // Create a socket for our QAD server.  We need the socket separately because we
450        // need to pop off messages before we attach it to the Quinn Endpoint.
451        let socket = tokio::net::UdpSocket::bind(SocketAddr::new(Ipv4Addr::LOCALHOST.into(), 0))
452            .await
453            .context("bind")?;
454        let server_addr = socket.local_addr().context("local addr")?;
455        info!(addr = ?server_addr, "server socket bound");
456
457        // Create a QAD server with a self-signed cert, all manually.
458        let cert =
459            rcgen::generate_simple_self_signed(vec!["localhost".into()]).context("self signed")?;
460        let key = PrivatePkcs8KeyDer::from(cert.key_pair.serialize_der());
461        let mut server_crypto = rustls::ServerConfig::builder()
462            .with_no_client_auth()
463            .with_single_cert(vec![cert.cert.into()], key.into())
464            .context("tls")?;
465        server_crypto.key_log = Arc::new(rustls::KeyLogFile::new());
466        server_crypto.alpn_protocols = vec![ALPN_QUIC_ADDR_DISC.to_vec()];
467        let mut server_config = quinn::ServerConfig::with_crypto(Arc::new(
468            QuicServerConfig::try_from(server_crypto).context("config")?,
469        ));
470        let transport_config = Arc::get_mut(&mut server_config.transport).unwrap();
471        transport_config.send_observed_address_reports(true);
472
473        let start = Instant::now();
474        let server_task = tokio::spawn(
475            async move {
476                info!("Dropping all packets");
477                time::timeout(Duration::from_secs(2), async {
478                    let mut buf = [0u8; 1500];
479                    loop {
480                        let (len, src) = socket.recv_from(&mut buf).await.unwrap();
481                        debug!(%len, ?src, "Dropped a packet");
482                    }
483                })
484                .await
485                .ok();
486                info!("starting server");
487                let server = quinn::Endpoint::new(
488                    Default::default(),
489                    Some(server_config),
490                    socket.into_std().unwrap(),
491                    Arc::new(quinn::TokioRuntime),
492                )
493                .context("endpoint new")?;
494                info!("accepting conn");
495                let incoming = server.accept().await.expect("missing conn");
496                info!("incoming!");
497                let conn = incoming.await.context("incoming")?;
498                conn.closed().await;
499                server.wait_idle().await;
500                Ok::<_, Error>(())
501            }
502            .instrument(info_span!("server")),
503        );
504        let server_task = AbortOnDropHandle::new(server_task);
505
506        info!("starting client");
507        let client_endpoint =
508            quinn::Endpoint::client(SocketAddr::new(Ipv4Addr::LOCALHOST.into(), 0))
509                .context("client")?;
510
511        // create the client configuration used for the client endpoint when they
512        // initiate a connection with the server
513        let client_config = crate::client::make_dangerous_client_config();
514        let quic_client = QuicClient::new(client_endpoint.clone(), client_config)?;
515
516        // Now we should still connect, but it should take more than 1s.
517        info!("making QAD request");
518        let (addr, latency) = time::timeout(
519            Duration::from_secs(10),
520            quic_client.get_addr_and_latency(server_addr, "localhost"),
521        )
522        .await
523        .context("timeout")??;
524        let duration = start.elapsed();
525        info!(?duration, ?addr, ?latency, "QAD succeeded");
526        assert!(duration >= Duration::from_secs(1));
527
528        time::timeout(Duration::from_secs(10), server_task)
529            .await
530            .context("timeout")?
531            .context("server task")??;
532
533        Ok(())
534    }
