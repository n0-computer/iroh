37impl NodeAuthority {
38    pub fn new(
39        zones: ZoneStore,
40        static_authority: InMemoryAuthority,
41        origins: Vec<Name>,
42        serial: u32,
43    ) -> Result<Self> {
44        if origins.is_empty() {
45            whatever!("at least one origin is required");
46        }
47        let first_origin = LowerName::from(&origins[0]);
48        Ok(Self {
49            static_authority,
50            origins,
51            serial,
52            zones,
53            first_origin,
54        })
55    }
56
57    pub fn origins(&self) -> impl Iterator<Item = &Name> {
58        self.origins.iter()
59    }
60
61    pub fn serial(&self) -> u32 {
62        self.serial
63    }
64
65    async fn resolve_pkarr(
66        &self,
67        name: Name,
68        pubkey: PublicKeyBytes,
69        origin: Name,
70        record_type: RecordType,
71        lookup_options: LookupOptions,
72    ) -> Result<AuthLookup, LookupError> {
73        debug!(%origin, %pubkey, %name, "resolve in pkarr zones");
74        match self
75            .zones
76            .resolve(&pubkey, &name, record_type)
77            .await
78            .map_err(err_refused)?
79        {
80            Some(pkarr_set) => {
81                debug!(%origin, %pubkey, %name, "found {} records in pkarr zone", pkarr_set.records_without_rrsigs().count());
82                let new_origin =
83                    Name::parse(&pubkey.to_z32(), Some(&origin)).map_err(err_refused)?;
84                let record_set = record_set_append_origin(&pkarr_set, &new_origin, self.serial())
85                    .map_err(err_refused)?;
86                let records = LookupRecords::new(lookup_options, Arc::new(record_set));
87                let answers = AuthLookup::answers(records, None);
88                Ok(answers)
89            }
90            None => Err(err_nx_domain("not found")),
91        }
92    }
93}

112        &self.first_origin

114
115    async fn lookup(
116        &self,
117        name: &LowerName,
118        record_type: RecordType,
119        lookup_options: LookupOptions,
120    ) -> LookupControlFlow<Self::Lookup> {
121        debug!(name=%name, "lookup in node authority");
122        match record_type {
123            RecordType::SOA | RecordType::NS => {
124                self.static_authority
125                    .lookup(name, record_type, lookup_options)
126                    .await
127            }
128            _ => match parse_name_as_pkarr_with_origin(name, &self.origins) {
129                Ok((name, pubkey, origin)) => {
130                    let res = self
131                        .resolve_pkarr(name, pubkey, origin, record_type, lookup_options)
132                        .await;
133                    LookupControlFlow::Continue(res)
134                }
135                Err(err) => {
136                    debug!(%name, failed_with=%err, "not a pkarr name, resolve in static authority");
137                    self.static_authority
138                        .lookup(name, record_type, lookup_options)
139                        .await
140                }
141            },
142        }
143    }
144
145    async fn search(
146        &self,
147        request_info: RequestInfo<'_>,
148        lookup_options: LookupOptions,
149    ) -> LookupControlFlow<Self::Lookup> {
150        debug!("search in node authority for {}", request_info.query);
151        let lookup_name = request_info.query.name();
152        let record_type: RecordType = request_info.query.query_type();
153        match record_type {
154            RecordType::SOA => {
155                self.static_authority
156                    .lookup(self.origin(), record_type, lookup_options)
157                    .await
158            }
159            RecordType::AXFR => {
160                LookupControlFlow::Continue(Err(LookupError::from(ResponseCode::Refused)))
161            }
162            _ => self.lookup(lookup_name, record_type, lookup_options).await,
163        }
164    }
165
166    async fn get_nsec_records(
167        &self,
168        _name: &LowerName,
169        _lookup_options: LookupOptions,
170    ) -> LookupControlFlow<Self::Lookup> {
171        LookupControlFlow::Skip
172    }
173}

175fn parse_name_as_pkarr_with_origin(
176    name: impl Into<Name>,
177    allowed_origins: &[Name],
178) -> Result<(Name, PublicKeyBytes, Name)> {
179    let name = name.into();
180    trace!("resolve {name}");
181    for origin in allowed_origins.iter() {
182        trace!("try {origin}");
183        if !origin.zone_of(&name) {
184            continue;
185        }
186        if name.num_labels() < origin.num_labels() + 1 {
187            whatever!("not a valid pkarr name: missing pubkey");
188        }
189        trace!("parse {origin}");
190        let labels = name.iter().rev();
191        let mut labels_without_origin = labels.skip(origin.num_labels() as usize);
192        let pkey_label = labels_without_origin.next().expect("length checked above");
193        let pkey_str = std::str::from_utf8(pkey_label).e()?;
194        let pkey =
195            PublicKeyBytes::from_z32(pkey_str).context("not a valid pkarr name: invalid pubkey")?;
196        let remaining_name = Name::from_labels(labels_without_origin.rev()).e()?;
197        return Ok((remaining_name, pkey, origin.clone()));
198    }
199    whatever!("name does not match any allowed origin");
200}

202fn err_refused(e: impl fmt::Debug) -> LookupError {
203    trace!("lookup failed (refused): {e:?}");
204    LookupError::from(ResponseCode::Refused)
205}
206fn err_nx_domain(e: impl fmt::Debug) -> LookupError {
207    trace!("lookup failed (nxdomain): {e:?}");
208    LookupError::from(ResponseCode::NXDomain)
