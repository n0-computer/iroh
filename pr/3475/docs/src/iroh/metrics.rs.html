13/// Metrics collected by an [`crate::endpoint::Endpoint`].
14///
15/// See [`crate::endpoint::Endpoint::metrics`] for details.
16#[derive(Default, Debug, Clone, Serialize, Deserialize, MetricsGroupSet)]
17#[metrics(name = "endpoint")]
18#[non_exhaustive]
19pub struct EndpointMetrics {
20    /// Metrics collected by the endpoint's socket.
21    pub magicsock: Arc<MagicsockMetrics>,
22    /// Metrics collected by net reports.
23    pub net_report: Arc<NetReportMetrics>,
24    /// Metrics collected by the portmapper service.
25    #[cfg(not(wasm_browser))]
26    pub portmapper: Arc<PortmapMetrics>,
27}

29#[cfg(test)]
30mod tests {
31    use super::EndpointMetrics;
32    #[test]
33    fn test_serde() {
34        let metrics = EndpointMetrics::default();
35        metrics.magicsock.actor_link_change.inc();
36        metrics.net_report.reports.inc_by(10);
37        let encoded = postcard::to_stdvec(&metrics).unwrap();
38        let decoded: EndpointMetrics = postcard::from_bytes(&encoded).unwrap();
39        assert_eq!(decoded.magicsock.actor_link_change.get(), 1);
40        assert_eq!(decoded.net_report.reports.get(), 10);
41    }
