97/// An iroh node, which we can have connections with.
98///
99/// The whole point of the magicsock is that we can have multiple **paths** to a particular
100/// node.  One of these paths is via the endpoint's home relay node but as we establish a
101/// connection we'll hopefully discover more direct paths.

103pub(super) struct NodeState {

105    ///
106    /// [`NodeMap`]: super::NodeMap

108    /// The UDP address used on the QUIC-layer to address this node.
109    quic_mapped_addr: NodeIdMappedAddr,
110    /// The global identifier for this endpoint.
111    node_id: NodeId,
112    /// The last time we pinged all endpoints.
113    last_full_ping: Option<Instant>,
114    /// The url of relay node that we can relay over to communicate.
115    ///
116    /// The fallback/bootstrap path, if non-zero (non-zero for well-behaved clients).
117    relay_url: Option<(RelayUrl, PathState)>,

121    ///

211    /// Returns info about this node.
212    pub(super) fn info(&self, now: Instant) -> RemoteInfo {
213        let conn_type = self.conn_type.get();
214        let latency = match conn_type {
215            ConnectionType::Direct(addr) => self
216                .udp_paths
217                .paths()
218                .get(&addr.into())
219                .and_then(|state| state.latency()),
220            ConnectionType::Relay(ref url) => self
221                .relay_url
222                .as_ref()
223                .filter(|(relay_url, _)| relay_url == url)
224                .and_then(|(_, state)| state.latency()),
225            ConnectionType::Mixed(addr, ref url) => {
226                let addr_latency = self
227                    .udp_paths
228                    .paths()
229                    .get(&addr.into())
230                    .and_then(|state| state.latency());
231                let relay_latency = self
232                    .relay_url
233                    .as_ref()
234                    .filter(|(relay_url, _)| relay_url == url)
235                    .and_then(|(_, state)| state.latency());
236                addr_latency.min(relay_latency)
237            }
238            ConnectionType::None => None,
239        };
240
241        let addrs = self
242            .udp_paths
243            .paths()
244            .iter()
245            .map(|(addr, path_state)| DirectAddrInfo {
246                addr: SocketAddr::from(*addr),
247                latency: path_state.validity.latency(),
248                last_control: path_state.last_control_msg(now),
249                last_payload: path_state
250                    .last_payload_msg
251                    .as_ref()
252                    .map(|instant| now.duration_since(*instant)),
253                last_alive: path_state
254                    .last_alive()
255                    .map(|instant| now.duration_since(instant)),
256                sources: path_state
257                    .sources
258                    .iter()
259                    .map(|(source, instant)| (source.clone(), now.duration_since(*instant)))
260                    .collect(),
261            })
262            .collect();
263
264        RemoteInfo {
265            node_id: self.node_id,
266            relay_url: self.relay_url.clone().map(|r| r.into()),
267            addrs,
268            conn_type,
269            latency,
270            last_used: self.last_used.map(|instant| now.duration_since(instant)),
271        }

279    /// Returns the address(es) that should be used for sending the next packet.
280    ///
281    /// This may return to send on one, both or no paths.
282    fn addr_for_send(
283        &self,
284        have_ipv6: bool,
285        metrics: &MagicsockMetrics,
286    ) -> (Option<SocketAddr>, Option<RelayUrl>) {
287        #[cfg(any(test, feature = "test-utils"))]
288        if self.path_selection == PathSelection::RelayOnly {
289            debug!(
290                "in `RelayOnly` mode, giving the relay address as the only viable address for this endpoint"
291            );
292            return (None, self.relay_url());
293        }
294        let (best_addr, relay_url) = match self.udp_paths.send_addr(have_ipv6) {
295            UdpSendAddr::Valid(addr) => {
296                // If we have a valid address we use it.
297                trace!(%addr, ?have_ipv6, "UdpSendAddr is valid, use it");
298                (Some(*addr), None)
299            }
300            UdpSendAddr::Outdated(addr) => {
301                // If the address is outdated we use it, but send via relay at the same time.
302                // We also send disco pings so that it will become valid again if it still
303                // works (i.e. we don't need to holepunch again).
304                trace!(%addr, ?have_ipv6, "UdpSendAddr is outdated, use it together with relay");
305                (Some(*addr), self.relay_url())
306            }
307            UdpSendAddr::Unconfirmed(addr) => {
308                trace!(%addr, ?have_ipv6, "UdpSendAddr is unconfirmed, use it together with relay");
309                (Some(*addr), self.relay_url())
310            }
311            UdpSendAddr::None => {
312                trace!(?have_ipv6, "No UdpSendAddr, use relay");
313                (None, self.relay_url())
314            }
315        };
316        let typ = match (best_addr, relay_url.clone()) {
317            (Some(best_addr), Some(relay_url)) => ConnectionType::Mixed(best_addr, relay_url),
318            (Some(best_addr), None) => ConnectionType::Direct(best_addr),
319            (None, Some(relay_url)) => ConnectionType::Relay(relay_url),
320            (None, None) => ConnectionType::None,
321        };
322        if matches!(&typ, ConnectionType::Direct(_)) {
323            let before = self
324                .has_been_direct
325                .swap(true, std::sync::atomic::Ordering::Relaxed);
326            if !before {
327                metrics.nodes_contacted_directly.inc();
328            }
329        }
330        if let Ok(prev_typ) = self.conn_type.set(typ.clone()) {
331            // The connection type has changed.
332            event!(
333                target: "iroh::_events::conn_type::changed",
334                Level::DEBUG,
335                remote_node = %self.node_id.fmt_short(),
336                conn_type = ?typ,
337            );
338            info!(%typ, "new connection type");
339
340            // Update some metrics
341            match (prev_typ, typ) {
342                (ConnectionType::Relay(_), ConnectionType::Direct(_))
343                | (ConnectionType::Mixed(_, _), ConnectionType::Direct(_)) => {
344                    metrics.num_direct_conns_added.inc();
345                    metrics.num_relay_conns_removed.inc();
346                }
347                (ConnectionType::Direct(_), ConnectionType::Relay(_))
348                | (ConnectionType::Direct(_), ConnectionType::Mixed(_, _)) => {
349                    metrics.num_direct_conns_removed.inc();
350                    metrics.num_relay_conns_added.inc();
351                }
352                (ConnectionType::None, ConnectionType::Direct(_)) => {
353                    metrics.num_direct_conns_added.inc();
354                }
355                (ConnectionType::Direct(_), ConnectionType::None) => {
356                    metrics.num_direct_conns_removed.inc();
357                }
358                (ConnectionType::None, ConnectionType::Relay(_))
359                | (ConnectionType::None, ConnectionType::Mixed(_, _)) => {
360                    metrics.num_relay_conns_added.inc();
361                }
362                (ConnectionType::Relay(_), ConnectionType::None)
363                | (ConnectionType::Mixed(_, _), ConnectionType::None) => {
364                    metrics.num_relay_conns_removed.inc();
365                }
366                _ => (),
367            }
368        }
369        (best_addr, relay_url)
370    }

386    /// Whether we need to send another call-me-maybe to the endpoint.
387    ///
388    /// Basically we need to send a call-me-maybe if we need to find a better path.  Maybe
389    /// we only have a relay path, or our path is expired.
390    ///
391    /// When a call-me-maybe message is sent we also need to send pings to all known paths
392    /// of the endpoint.  The [`NodeState::send_call_me_maybe`] function takes care of this.
393    #[cfg(not(wasm_browser))]
394    #[instrument("want_call_me_maybe", skip_all)]
395    fn want_call_me_maybe(&self, now: &Instant, have_ipv6: bool) -> bool {
396        trace!("full ping: wanted?");
397        let Some(last_full_ping) = self.last_full_ping else {
398            debug!("no previous full ping: need full ping");
399            return true;
400        };
401        match &self.udp_paths.send_addr(have_ipv6) {
402            UdpSendAddr::None | UdpSendAddr::Unconfirmed(_) => {
403                debug!("best addr not set: need full ping");
404                true
405            }
406            UdpSendAddr::Outdated(_) => {
407                debug!("best addr expired: need full ping");
408                true
409            }
410            UdpSendAddr::Valid(addr) => {
411                let latency = self
412                    .udp_paths
413                    .paths()
414                    .get(&(*addr).into())
415                    .expect("send path not tracked?")
416                    .latency()
417                    .expect("send_addr marked valid incorrectly");
418                if latency > GOOD_ENOUGH_LATENCY && *now - last_full_ping >= UPGRADE_INTERVAL {
419                    debug!(
420                        "full ping interval expired and latency is only {}ms: need full ping",
421                        latency.as_millis()
422                    );
423                    true
424                } else {
425                    trace!(?now, "best_addr valid: not needed");
426                    false
427                }
428            }
429        }
430    }

438    /// Cleanup the expired ping for the passed in txid.
439    #[instrument("disco", skip_all, fields(node = %self.node_id.fmt_short()))]
440    pub(super) fn ping_timeout(&mut self, txid: stun_rs::TransactionId, now: Instant) {
441        if let Some(sp) = self.sent_pings.remove(&txid) {
442            debug!(tx = %HEXLOWER.encode(&txid), addr = %sp.to, "pong not received in timeout");
443            match sp.to {
444                SendAddr::Udp(addr) => {
445                    if let Some(path_state) =
446                        self.udp_paths.access_mut(now).paths().get_mut(&addr.into())
447                    {
448                        path_state.last_ping = None;
449                        let consider_alive = path_state
450                            .last_alive()
451                            .map(|last_alive| last_alive.elapsed() <= PING_TIMEOUT_DURATION)
452                            .unwrap_or(false);
453                        if !consider_alive {
454                            // If there was no sign of life from this path during the time
455                            // which we should have received the pong, clear best addr and
456                            // pong.  Both are used to select this path again, but we know
457                            // it's not a usable path now.
458                            path_state.validity = PathValidity::empty();
459                        }
460                    }
461                }
462                SendAddr::Relay(ref url) => {
463                    if let Some((home_relay, relay_state)) = self.relay_url.as_mut() {
464                        if home_relay == url {
465                            // lost connectivity via relay
466                            relay_state.last_ping = None;
467                        }
468                    }
469                }
470            }
471        }
472    }

507    /// Record the fact that a ping has been sent out.
508    pub(super) fn ping_sent(
509        &mut self,
510        to: SendAddr,
511        tx_id: stun_rs::TransactionId,
512        purpose: DiscoPingPurpose,
513        sender: mpsc::Sender<ActorMessage>,
514    ) {
515        trace!(%to, tx = %HEXLOWER.encode(&tx_id), ?purpose, "record ping sent");
516
517        let now = Instant::now();
518        let mut path_found = false;
519        match to {
520            SendAddr::Udp(addr) => {
521                if let Some(st) = self.udp_paths.access_mut(now).paths().get_mut(&addr.into()) {
522                    st.last_ping.replace(now);
523                    path_found = true
524                }
525            }
526            SendAddr::Relay(ref url) => {
527                if let Some((home_relay, relay_state)) = self.relay_url.as_mut() {
528                    if home_relay == url {
529                        relay_state.last_ping.replace(now);
530                        path_found = true
531                    }
532                }
533            }
534        }
535        if !path_found {
536            // Shouldn't happen. But don't ping an endpoint that's not active for us.
537            warn!(%to, ?purpose, "unexpected attempt to ping no longer live path");
538            return;
539        }
540
541        let id = self.id;
542        let _expiry_task = AbortOnDropHandle::new(task::spawn(async move {
543            time::sleep(PING_TIMEOUT_DURATION).await;
544            sender
545                .send(ActorMessage::EndpointPingExpired(id, tx_id))
546                .await
547                .ok();
548        }));
549        self.sent_pings.insert(
550            tx_id,
551            SentPing {
552                to,
553                at: now,
554                purpose,
555                _expiry_task,
556            },
557        );
558    }

560    /// Send a DISCO call-me-maybe message to the peer.
561    ///
562    /// This takes care of sending the needed pings beforehand.  This ensures that we open
563    /// our firewall's port so that when the receiver sends us DISCO pings in response to
564    /// our call-me-maybe they will reach us and the other side establishes a direct
565    /// connection upon our subsequent pong response.
566    ///
567    /// For [`SendCallMeMaybe::IfNoRecent`], **no** paths will be pinged if there already
568    /// was a recent call-me-maybe sent.
569    ///
570    /// The caller is responsible for sending the messages.
571    #[must_use = "actions must be handled"]
572    fn send_call_me_maybe(&mut self, now: Instant, always: SendCallMeMaybe) -> Vec<PingAction> {
573        match always {
574            SendCallMeMaybe::Always => (),
575            SendCallMeMaybe::IfNoRecent => {
576                let had_recent_call_me_maybe = self
577                    .last_call_me_maybe
578                    .map(|when| when.elapsed() < HEARTBEAT_INTERVAL)
579                    .unwrap_or(false);
580                if had_recent_call_me_maybe {
581                    trace!("skipping call-me-maybe, still recent");
582                    return Vec::new();
583                }
584            }
585        }
586        // We send pings regardless of whether we have a RelayUrl.  If we were given any
587        // direct address paths to contact but no RelayUrl, we still need to send a DISCO
588        // ping to the direct address paths so that the other node will learn about us and
589        // accepts the connection.
590        let mut msgs = self.send_pings(now);
591
592        if let Some(url) = self.relay_url() {
593            debug!(%url, "queue call-me-maybe");
594            msgs.push(PingAction::SendCallMeMaybe {
595                relay_url: url,
596                dst_node: self.node_id,
597            });
598            self.last_call_me_maybe = Some(now);
599        } else {
600            debug!("can not send call-me-maybe, no relay URL");
601        }
602
603        msgs
604    }

606    /// Send DISCO Pings to all the paths of this node.
607    ///
608    /// Any paths to the node which have not been recently pinged will be sent a disco
609    /// ping.
610    ///
611    /// The caller is responsible for sending the messages.
612    #[must_use = "actions must be handled"]
613    fn send_pings(&mut self, now: Instant) -> Vec<PingAction> {
614        // We allocate +1 in case the caller wants to add a call-me-maybe message.
615        let mut ping_msgs = Vec::with_capacity(self.udp_paths.paths().len() + 1);
616
617        if let Some((url, state)) = self.relay_url.as_ref() {
618            if state.needs_ping(&now) {
619                debug!(%url, "relay path needs ping");
620                if let Some(msg) =
621                    self.start_ping(SendAddr::Relay(url.clone()), DiscoPingPurpose::Discovery)
622                {
623                    ping_msgs.push(PingAction::SendPing(msg))
624                }
625            }
626        }
627
628        #[cfg(any(test, feature = "test-utils"))]
629        if self.path_selection == PathSelection::RelayOnly {
630            warn!("in `RelayOnly` mode, ignoring request to respond to a hole punching attempt.");
631            return ping_msgs;
632        }
633
634        self.prune_direct_addresses(now);
635        let mut ping_dsts = String::from("[");
636        self.udp_paths
637            .paths()
638            .iter()
639            .filter_map(|(ipp, state)| state.needs_ping(&now).then_some(*ipp))
640            .filter_map(|ipp| {
641                self.start_ping(SendAddr::Udp(ipp.into()), DiscoPingPurpose::Discovery)
642            })
643            .for_each(|msg| {
644                use std::fmt::Write;
645                write!(&mut ping_dsts, " {} ", msg.dst).ok();
646                ping_msgs.push(PingAction::SendPing(msg));
647            });
648        ping_dsts.push(']');
649        debug!(
650            %ping_dsts,
651            dst = %self.node_id.fmt_short(),
652            paths = %summarize_node_paths(self.udp_paths.paths()),
653            "sending pings to node",
654        );
655        self.last_full_ping.replace(now);
656        ping_msgs
657    }

659    pub(super) fn update_from_node_addr(
660        &mut self,
661        new_relay_url: Option<&RelayUrl>,
662        new_addrs: &BTreeSet<SocketAddr>,
663        source: super::Source,
664        have_ipv6: bool,
665        metrics: &MagicsockMetrics,
666    ) {
667        if matches!(
668            self.udp_paths.send_addr(have_ipv6),
669            UdpSendAddr::None | UdpSendAddr::Unconfirmed(_)
670        ) {
671            // we do not have a direct connection, so changing the relay information may
672            // have an effect on our connection status
673            if self.relay_url.is_none() && new_relay_url.is_some() {
674                // we did not have a relay connection before, but now we do
675                metrics.num_relay_conns_added.inc();
676            } else if self.relay_url.is_some() && new_relay_url.is_none() {
677                // we had a relay connection before but do not have one now
678                metrics.num_relay_conns_removed.inc();
679            }
680        }
681
682        let now = Instant::now();
683
684        if new_relay_url.is_some() && new_relay_url != self.relay_url().as_ref() {
685            debug!(
686                "Changing relay node from {:?} to {:?}",
687                self.relay_url, new_relay_url
688            );
689            self.relay_url = new_relay_url.map(|url| {
690                (
691                    url.clone(),
692                    PathState::new(self.node_id, url.clone().into(), source.clone(), now),
693                )
694            });
695        }
696
697        let mut access = self.udp_paths.access_mut(now);
698        for &addr in new_addrs.iter() {
699            access
700                .paths()
701                .entry(addr.into())
702                .and_modify(|path_state| {
703                    path_state.add_source(source.clone(), now);
704                })
705                .or_insert_with(|| {
706                    PathState::new(self.node_id, SendAddr::from(addr), source.clone(), now)
707                });
708        }
709        drop(access);
710        let paths = summarize_node_paths(self.udp_paths.paths());
711        debug!(new = ?new_addrs , %paths, "added new direct paths for endpoint");
712    }

714    /// Handle a received Disco Ping.
715    ///
716    /// - Ensures the paths the ping was received on is a known path for this endpoint.
717    ///
718    /// - If there is no best_addr for this endpoint yet, sends a ping itself to try and
719    ///   establish one.
720    ///
721    /// This is called once we've already verified that we got a valid discovery message
722    /// from `self` via ep.
723    pub(super) fn handle_ping(
724        &mut self,
725        path: SendAddr,
726        tx_id: stun_rs::TransactionId,
727    ) -> PingHandled {
728        let now = Instant::now();
729
730        let role = match path {
731            SendAddr::Udp(addr) => {
732                match self.udp_paths.access_mut(now).paths().entry(addr.into()) {
733                    Entry::Occupied(mut occupied) => occupied.get_mut().handle_ping(tx_id, now),
734                    Entry::Vacant(vacant) => {
735                        info!(%addr, "new direct addr for node");
736                        vacant.insert(PathState::with_ping(
737                            self.node_id,
738                            path.clone(),
739                            tx_id,
740                            Source::Udp,
741                            now,
742                        ));
743                        PingRole::NewPath
744                    }
745                }
746            }
747            SendAddr::Relay(ref url) => {
748                match self.relay_url.as_mut() {
749                    Some((home_url, _state)) if home_url != url => {
750                        // either the node changed relays or we didn't have a relay address for the
751                        // node. In both cases, trust the new confirmed url
752                        info!(%url, "new relay addr for node");
753                        self.relay_url = Some((
754                            url.clone(),
755                            PathState::with_ping(
756                                self.node_id,
757                                path.clone(),
758                                tx_id,
759                                Source::Relay,
760                                now,
761                            ),
762                        ));
763                        PingRole::NewPath
764                    }
765                    Some((_home_url, state)) => state.handle_ping(tx_id, now),
766                    None => {
767                        info!(%url, "new relay addr for node");
768                        self.relay_url = Some((
769                            url.clone(),
770                            PathState::with_ping(
771                                self.node_id,
772                                path.clone(),
773                                tx_id,
774                                Source::Relay,
775                                now,
776                            ),
777                        ));
778                        PingRole::NewPath
779                    }
780                }
781            }
782        };
783        event!(
784            target: "iroh::_events::ping::recv",
785            Level::DEBUG,
786            remote_node = %self.node_id.fmt_short(),
787            src = ?path,
788            txn = ?tx_id,
789            ?role,
790        );
791
792        if matches!(path, SendAddr::Udp(_)) && matches!(role, PingRole::NewPath) {
793            self.prune_direct_addresses(now);
794        }
795
796        // if the endpoint does not yet have a best_addr
797        let needs_ping_back = if matches!(path, SendAddr::Udp(_))
798            && matches!(
799                self.udp_paths.send_addr(true),
800                UdpSendAddr::None | UdpSendAddr::Unconfirmed(_) | UdpSendAddr::Outdated(_)
801            ) {
802            // We also need to send a ping to make this path available to us as well.  This
803            // is always sent together with a pong.  So in the worst case the pong gets lost
804            // and this ping does not.  In that case we ping-pong until both sides have
805            // received at least one pong.  Once both sides have received one pong they both
806            // have a best_addr and this ping will stop being sent.
807            self.start_ping(path, DiscoPingPurpose::PingBack)
808        } else {
809            None
810        };
811
812        debug!(
813            ?role,
814            needs_ping_back = ?needs_ping_back.is_some(),
815            paths = %summarize_node_paths(self.udp_paths.paths()),
816            "endpoint handled ping",
817        );
818        PingHandled {
819            role,
820            needs_ping_back,
821        }
822    }

824    /// Prune inactive paths.
825    ///
826    /// This trims the list of inactive paths for an endpoint.  At most
827    /// [`MAX_INACTIVE_DIRECT_ADDRESSES`] are kept.
828    pub(super) fn prune_direct_addresses(&mut self, now: Instant) {
829        // prune candidates are addresses that are not active
830        let mut prune_candidates: Vec<_> = self
831            .udp_paths
832            .paths()
833            .iter()
834            .filter(|(_ip_port, state)| !state.is_active())
835            .map(|(ip_port, state)| (*ip_port, state.last_alive()))
836            .filter(|(_ipp, last_alive)| match last_alive {
837                Some(last_seen) => last_seen.elapsed() > LAST_ALIVE_PRUNE_DURATION,
838                None => true,
839            })
840            .collect();
841        let prune_count = prune_candidates
842            .len()
843            .saturating_sub(MAX_INACTIVE_DIRECT_ADDRESSES);
844        if prune_count == 0 {
845            // nothing to do, within limits
846            debug!(
847                paths = %summarize_node_paths(self.udp_paths.paths()),
848                "prune addresses: {prune_count} pruned",
849            );
850            return;
851        }
852
853        // sort leaving the worst addresses first (never contacted) and better ones (most recently
854        // used ones) last
855        prune_candidates.sort_unstable_by_key(|(_ip_port, last_alive)| *last_alive);
856        prune_candidates.truncate(prune_count);
857        for (ip_port, _last_alive) in prune_candidates.into_iter() {
858            self.remove_direct_addr(&ip_port, now, "inactive");
859        }
860        debug!(
861            paths = %summarize_node_paths(self.udp_paths.paths()),
862            "prune addresses: {prune_count} pruned",
863        );
864    }

876    /// Handles a Pong message (a reply to an earlier ping).
877    ///
878    /// It reports the address and key that should be inserted for the endpoint if any.
879    #[instrument(skip(self))]
880    pub(super) fn handle_pong(
881        &mut self,
882        m: &disco::Pong,
883        src: SendAddr,
884    ) -> Option<(SocketAddr, PublicKey)> {
885        event!(
886            target: "iroh::_events::pong::recv",
887            Level::DEBUG,
888            remote_node = %self.node_id.fmt_short(),
889            ?src,
890            txn = ?m.tx_id,
891        );
892        let is_relay = src.is_relay();
893        match self.sent_pings.remove(&m.tx_id) {
894            None => {
895                // This is not a pong for a ping we sent.  In reality however we probably
896                // did send this ping but it has timed-out by the time we receive this pong
897                // so we removed the state already.
898                debug!(tx = %HEXLOWER.encode(&m.tx_id), "received unknown pong (did it timeout?)");
899                None
900            }
901            Some(sp) => {
902                let mut node_map_insert = None;
903
904                let now = Instant::now();
905                let latency = now - sp.at;
906
907                debug!(
908                    tx = %HEXLOWER.encode(&m.tx_id),
909                    src = %src,
910                    reported_ping_src = %m.ping_observed_addr,
911                    ping_dst = %sp.to,
912                    is_relay = %src.is_relay(),
913                    latency = %latency.as_millis(),
914                    "received pong",
915                );
916
917                match src {
918                    SendAddr::Udp(addr) => {
919                        match self.udp_paths.access_mut(now).paths().get_mut(&addr.into()) {
920                            None => {
921                                warn!("ignoring pong: no state for src addr");
922                                // This is no longer an endpoint we care about.
923                                return node_map_insert;
924                            }
925                            Some(st) => {
926                                node_map_insert = Some((addr, self.node_id));
927                                st.add_pong_reply(PongReply {
928                                    latency,
929                                    pong_at: now,
930                                    from: src,
931                                    pong_src: m.ping_observed_addr.clone(),
932                                });
933                            }
934                        }
935                        debug!(
936                            paths = %summarize_node_paths(self.udp_paths.paths()),
937                            "handled pong",
938                        );
939                    }
940                    SendAddr::Relay(ref url) => match self.relay_url.as_mut() {
941                        Some((home_url, state)) if home_url == url => {
942                            state.add_pong_reply(PongReply {
943                                latency,
944                                pong_at: now,
945                                from: src,
946                                pong_src: m.ping_observed_addr.clone(),
947                            });
948                        }
949                        other => {
950                            // if we are here then we sent this ping, but the url changed
951                            // waiting for the response. It was either set to None or changed to
952                            // another relay. This should either never happen or be extremely
953                            // unlikely. Log and ignore for now
954                            warn!(
955                                stored=?other,
956                                received=?url,
957                                "ignoring pong via relay for different relay from last one",
958                            );
959                        }
960                    },
961                }
962
963                // Promote this pong response to our current best address if it's lower latency.
964                // TODO(bradfitz): decide how latency vs. preference order affects decision
965                if let SendAddr::Udp(_to) = sp.to {
966                    debug_assert!(!is_relay, "mismatching relay & udp");
967                }
968
969                node_map_insert
970            }
971        }
972    }

974    /// Handles a DISCO CallMeMaybe discovery message.
975    ///
976    /// The contract for use of this message is that the node has already pinged to us via
977    /// UDP, so their stateful firewall should be open. Now we can Ping back and make it
978    /// through.
979    ///
980    /// However if the remote side has no direct path information to us, they would not have
981    /// had any [`IpPort`]s to send pings to and our pings might end up blocked.  But at
982    /// least open the firewalls on our side, giving the other side another change of making
983    /// it through when it pings in response.
984    pub(super) fn handle_call_me_maybe(&mut self, m: disco::CallMeMaybe) -> Vec<PingAction> {
985        let now = Instant::now();
986        let mut call_me_maybe_ipps = BTreeSet::new();
987
988        let mut guard = self.udp_paths.access_mut(now);
989
990        for peer_sockaddr in &m.my_numbers {
991            if let IpAddr::V6(ip) = peer_sockaddr.ip() {
992                if netwatch::ip::is_unicast_link_local(ip) {
993                    // We send these out, but ignore them for now.
994                    // TODO: teach the ping code to ping on all interfaces for these.
995                    continue;
996                }
997            }
998            let ipp = IpPort::from(*peer_sockaddr);
999            call_me_maybe_ipps.insert(ipp);
1000            guard
1001                .paths()
1002                .entry(ipp)
1003                .or_insert_with(|| {
1004                    PathState::new(
1005                        self.node_id,
1006                        SendAddr::from(*peer_sockaddr),
1007                        Source::Relay,
1008                        now,
1009                    )
1010                })
1011                .call_me_maybe_time
1012                .replace(now);
1013        }
1014
1015        // Zero out all the last_ping times to force send_pings to send new ones, even if
1016        // it's been less than 5 seconds ago.  Also clear pongs for direct addresses not
1017        // included in the updated set.
1018        for (ipp, st) in guard.paths().iter_mut() {
1019            st.last_ping = None;
1020            if !call_me_maybe_ipps.contains(ipp) {
1021                // TODO: This seems like a weird way to signal that the endpoint no longer
1022                // thinks it has this IpPort as an available path.
1023                if !st.validity.is_empty() {
1024                    debug!(path=?ipp ,"clearing recent pong");
1025                    st.validity = PathValidity::empty();
1026                }
1027            }
1028        }
1029        if guard.has_best_addr_changed() {
1030            // Clear the last call-me-maybe send time so we will send one again.
1031            self.last_call_me_maybe = None;
1032        }
1033        debug!(
1034            paths = %summarize_node_paths(self.udp_paths.paths()),
1035            "updated endpoint paths from call-me-maybe",
1036        );
1037        self.send_pings(now)
1038    }

1099    /// Send a heartbeat to the node to keep the connection alive, or trigger a full ping
1100    /// if necessary.
1101    #[instrument("stayin_alive", skip_all, fields(node = %self.node_id.fmt_short()))]
1102    pub(super) fn stayin_alive(&mut self, have_ipv6: bool) -> Vec<PingAction> {
1103        trace!("stayin_alive");
1104        let now = Instant::now();
1105        if !self.is_active(&now) {
1106            trace!("skipping stayin alive: session is inactive");
1107            return Vec::new();
1108        }
1109
1110        // If we do not have an optimal addr, send pings to all known places.
1111        if self.want_call_me_maybe(&now, have_ipv6) {
1112            debug!("sending a call-me-maybe");
1113            return self.send_call_me_maybe(now, SendCallMeMaybe::Always);
1114        }
1115
1116        // Send heartbeat ping to keep the current addr going as long as we need it.
1117        if let Some(udp_addr) = self.udp_paths.send_addr(have_ipv6).get_addr() {
1118            let elapsed = self.last_ping(&SendAddr::Udp(udp_addr)).map(|l| now - l);
1119            // Send a ping if the last ping is older than 2 seconds.
1120            let needs_ping = match elapsed {
1121                Some(e) => e >= STAYIN_ALIVE_MIN_ELAPSED,
1122                None => false,
1123            };
1124
1125            if needs_ping {
1126                debug!(
1127                    dst = %udp_addr,
1128                    since_last_ping=?elapsed,
1129                    "send stayin alive ping",
1130                );
1131                if let Some(msg) =
1132                    self.start_ping(SendAddr::Udp(udp_addr), DiscoPingPurpose::StayinAlive)
1133                {
1134                    return vec![PingAction::SendPing(msg)];
1135                }
1136            }
1137        }
1138
1139        Vec::new()
1140    }

1264/// Information about a *direct address*.
1265///
1266/// The *direct addresses* of an iroh node are those that could be used by other nodes to
1267/// establish direct connectivity, depending on the network situation. Due to NAT configurations,
1268/// for example, not all direct addresses of a node are usable by all peers.
1269#[derive(Debug, Clone, Eq, PartialEq, Serialize, Deserialize)]
1270pub struct DirectAddrInfo {
1271    /// The UDP address reported by the remote node.
1272    pub addr: SocketAddr,
1273    /// The latency to the remote node over this network path.
1274    ///
1275    /// If there has never been any connectivity via this address no latency will be known.
1276    pub latency: Option<Duration>,
1277    /// Last control message received by this node about this address.
1278    ///
1279    /// This contains the elapsed duration since the control message was received and the
1280    /// kind of control message received at that time.  Only the most recent control message
1281    /// is returned.
1282    ///
1283    /// Note that [`ControlMsg::CallMeMaybe`] is received via a relay path, while
1284    /// [`ControlMsg::Ping`] and [`ControlMsg::Pong`] are received on the path to
1285    /// [`DirectAddrInfo::addr`] itself and thus convey very different information.
1286    pub last_control: Option<(Duration, ControlMsg)>,
1287    /// Elapsed time since the last payload message was received on this network path.
1288    ///
1289    /// This indicates how long ago a QUIC datagram was received from the remote node sent
1290    /// from this [`DirectAddrInfo::addr`].  It indicates the network path was in use to
1291    /// transport payload data.
1292    pub last_payload: Option<Duration>,
1293    /// Elapsed time since this network path was known to exist.
1294    ///
1295    /// A network path is considered to exist only because the remote node advertised it.
1296    /// It may not mean the path is usable.  However, if there was any communication with
1297    /// the remote node over this network path it also means the path exists.
1298    ///
1299    /// The elapsed time since *any* confirmation of the path's existence was received is
1300    /// returned.  If the remote node moved networks and no longer has this path, this could
1301    /// be a long duration.  If the path was added via [`Endpoint::add_node_addr`] or some
1302    /// node discovery the path may never have been known to exist.
1303    ///
1304    /// [`Endpoint::add_node_addr`]: crate::endpoint::Endpoint::add_node_addr
1305    pub last_alive: Option<Duration>,
1306    /// A [`HashMap`] of [`Source`]s to [`Duration`]s.
1307    ///
1308    /// The [`Duration`] indicates the elapsed time since this source last
1309    /// recorded this address.
1310    ///
1311    /// The [`Duration`] will always indicate the most recent time the source
1312    /// recorded this address.
1313    pub sources: HashMap<Source, Duration>,
1314}

1343/// Details about a remote iroh node which is known to this node.
1344///
1345/// Having details of a node does not mean it can be connected to, nor that it has ever been
1346/// connected to in the past. There are various reasons a node might be known: it could have
1347/// been manually added via [`Endpoint::add_node_addr`], it could have been added by some
1348/// discovery mechanism, the node could have contacted this node, etc.
1349///
1350/// [`Endpoint::add_node_addr`]: crate::endpoint::Endpoint::add_node_addr
1351#[derive(Debug, Clone, Eq, PartialEq, Serialize, Deserialize)]
1352pub struct RemoteInfo {
1353    /// The globally unique identifier for this node.
1354    pub node_id: NodeId,
1355    /// Relay server information, if available.
1356    pub relay_url: Option<RelayUrlInfo>,
1357    /// The addresses at which this node might be reachable.
1358    ///
1359    /// Some of these addresses might only be valid for networks we are not part of, but the remote
1360    /// node might be a part of.
1361    pub addrs: Vec<DirectAddrInfo>,
1362    /// The type of connection we have to the node, either direct or over relay.
1363    pub conn_type: ConnectionType,
1364    /// The latency of the current network path to the remote node.
1365    pub latency: Option<Duration>,
1366    /// Time elapsed time since last we have sent to or received from the node.
1367    ///
1368    /// This is the duration since *any* data (payload or control messages) was sent or receive
1369    /// from the remote node. Note that sending to the remote node does not imply
1370    /// the remote node received anything.
1371    pub last_used: Option<Duration>,
1372}
1373
1374impl RemoteInfo {
1375    /// Get the duration since the last activity we received from this endpoint
1376    /// on any of its direct addresses.
1377    pub fn last_received(&self) -> Option<Duration> {
1378        self.addrs
1379            .iter()
1380            .filter_map(|addr| addr.last_control.map(|x| x.0).min(addr.last_payload))
1381            .min()
1382    }
1383
1384    /// Whether there is a possible known network path to the remote node.
1385    ///
1386    /// Note that this does not provide any guarantees of whether any network path is
1387    /// usable.
1388    pub fn has_send_address(&self) -> bool {
1389        self.relay_url.is_some() || !self.addrs.is_empty()
1390    }
1391
1392    /// Returns a deduplicated list of [`Source`]s merged from all address in the [`RemoteInfo`].
1393    ///
1394    /// Deduplication is on the (`Source`, `Duration`) tuple, so you will get multiple [`Source`]s
1395    /// for each `Source` variant, if different addresses were discovered from the same [`Source`]
1396    /// at different times.
1397    ///
1398    /// The list is sorted from least to most recent [`Source`].
1399    pub fn sources(&self) -> Vec<(Source, Duration)> {
1400        let mut sources = vec![];
1401        for addr in &self.addrs {
1402            for source in &addr.sources {
1403                let source = (source.0.clone(), *source.1);
1404                if !sources.contains(&source) {
1405                    sources.push(source)
1406                }
1407            }
1408        }
1409        sources.sort_by(|a, b| b.1.cmp(&a.1));
1410        sources
1411    }
1412}

1414/// The type of connection we have to the endpoint.
1415#[derive(derive_more::Display, Default, Debug, Clone, Eq, PartialEq, Serialize, Deserialize)]
1416pub enum ConnectionType {
1417    /// Direct UDP connection
1418    #[display("direct({_0})")]
1419    Direct(SocketAddr),
1420    /// Relay connection over relay
1421    #[display("relay({_0})")]
1422    Relay(RelayUrl),
1423    /// Both a UDP and a relay connection are used.
1424    ///
1425    /// This is the case if we do have a UDP address, but are missing a recent confirmation that
1426    /// the address works.
1427    #[display("mixed(udp: {_0}, relay: {_1})")]
1428    Mixed(SocketAddr, RelayUrl),
1429    /// We have no verified connection to this PublicKey
1430    #[default]
1431    #[display("none")]
1432    None,
1433}
