40pub trait Resolver: fmt::Debug + Send + Sync + 'static {
41    /// Looks up an IPv4 address.

49
50    /// Clears the internal cache.
51    fn clear_cache(&self);

53    /// Completely resets the DNS resolver.
54    ///

279    /// Resolves IPv4 and IPv6 in parallel with a timeout.
280    ///
281    /// `LookupIpStrategy::Ipv4AndIpv6` will wait for ipv6 resolution timeout, even if it is
282    /// not usable on the stack, so we manually query both lookups concurrently and time them out
283    /// individually.
284    pub async fn lookup_ipv4_ipv6<T: ToString>(
285        &self,
286        host: T,
287        timeout: Duration,
288    ) -> Result<impl Iterator<Item = IpAddr> + use<T>, DnsError> {
289        let host = host.to_string();
290        let res = tokio::join!(
291            self.lookup_ipv4(host.clone(), timeout),
292            self.lookup_ipv6(host, timeout)
293        );
294
295        match res {
296            (Ok(ipv4), Ok(ipv6)) => Ok(LookupIter::Both(ipv4.chain(ipv6))),
297            (Ok(ipv4), Err(_)) => Ok(LookupIter::Ipv4(ipv4)),
298            (Err(_), Ok(ipv6)) => Ok(LookupIter::Ipv6(ipv6)),
299            (Err(ipv4_err), Err(ipv6_err)) => Err(ResolveBothSnafu {
300                ipv4: Box::new(ipv4_err),
301                ipv6: Box::new(ipv6_err),
302            }
303            .build()),
304        }
305    }

307    /// Resolves a hostname from a URL to an IP address.
308    pub async fn resolve_host(
309        &self,
310        url: &Url,
311        prefer_ipv6: bool,
312        timeout: Duration,
313    ) -> Result<IpAddr, DnsError> {
314        let host = url.host().context(MissingHostSnafu)?;
315        match host {
316            url::Host::Domain(domain) => {
317                // Need to do a DNS lookup
318                let lookup = tokio::join!(
319                    self.lookup_ipv4(domain, timeout),
320                    self.lookup_ipv6(domain, timeout)
321                );
322                let (v4, v6) = match lookup {
323                    (Err(ipv4_err), Err(ipv6_err)) => {
324                        return Err(ResolveBothSnafu {
325                            ipv4: Box::new(ipv4_err),
326                            ipv6: Box::new(ipv6_err),
327                        }
328                        .build());
329                    }
330                    (Err(_), Ok(mut v6)) => (None, v6.next()),
331                    (Ok(mut v4), Err(_)) => (v4.next(), None),
332                    (Ok(mut v4), Ok(mut v6)) => (v4.next(), v6.next()),
333                };
334                if prefer_ipv6 {
335                    v6.or(v4).context(NoResponseSnafu)
336                } else {
337                    v4.or(v6).context(NoResponseSnafu)
338                }
339            }
340            url::Host::Ipv4(ip) => Ok(IpAddr::V4(ip)),
341            url::Host::Ipv6(ip) => Ok(IpAddr::V6(ip)),
342        }
343    }

465impl reqwest::dns::Resolve for DnsResolver {
466    fn resolve(&self, name: reqwest::dns::Name) -> reqwest::dns::Resolving {
467        let this = self.clone();
468        let name = name.as_str().to_string();
469        Box::pin(async move {
470            let res = this.lookup_ipv4_ipv6(name, DNS_TIMEOUT).await;
471            match res {
472                Ok(addrs) => {
473                    let addrs: reqwest::dns::Addrs =
474                        Box::new(addrs.map(|addr| SocketAddr::new(addr, 0)));
475                    Ok(addrs)
476                }
477                Err(err) => {
478                    let err: Box<dyn std::error::Error + Send + Sync> = Box::new(err);
479                    Err(err)
480                }
481            }
482        })
483    }

486/// Wrapper enum that contains either a hickory resolver or a custom resolver.
487///
488/// We do this to save the cost of boxing the futures and iterators when using
489/// default hickory resolver.
490#[derive(Debug, Clone)]
491enum DnsResolverInner {
492    Hickory(Arc<RwLock<HickoryResolver>>),
493    Custom(Arc<RwLock<dyn Resolver>>),
494}
495
496impl DnsResolverInner {
497    async fn lookup_ipv4(
498        &self,
499        host: String,
500    ) -> Result<impl Iterator<Item = Ipv4Addr> + use<>, DnsError> {
501        Ok(match self {
502            Self::Hickory(resolver) => Either::Left(resolver.read().await.lookup_ipv4(host).await?),
503            Self::Custom(resolver) => Either::Right(resolver.read().await.lookup_ipv4(host).await?),
504        })
505    }
506
507    async fn lookup_ipv6(
508        &self,
509        host: String,
510    ) -> Result<impl Iterator<Item = Ipv6Addr> + use<>, DnsError> {
511        Ok(match self {
512            Self::Hickory(resolver) => Either::Left(resolver.read().await.lookup_ipv6(host).await?),
513            Self::Custom(resolver) => Either::Right(resolver.read().await.lookup_ipv6(host).await?),
514        })
515    }
516
517    async fn lookup_txt(
518        &self,
519        host: String,
520    ) -> Result<impl Iterator<Item = TxtRecordData> + use<>, DnsError> {
521        Ok(match self {
522            Self::Hickory(resolver) => Either::Left(resolver.read().await.lookup_txt(host).await?),
523            Self::Custom(resolver) => Either::Right(resolver.read().await.lookup_txt(host).await?),
524        })
525    }
526
527    async fn clear_cache(&self) {
528        match self {
529            Self::Hickory(resolver) => resolver.read().await.clear_cache(),
530            Self::Custom(resolver) => resolver.read().await.clear_cache(),
531        }
532    }
533
534    async fn reset(&self) {
535        match self {
536            Self::Hickory(resolver) => resolver.write().await.reset(),
537            Self::Custom(resolver) => resolver.write().await.reset(),
538        }
539    }

548impl HickoryResolver {
549    fn new(builder: Builder) -> Self {
550        let (mut config, mut options) = if builder.use_system_defaults {
551            Self::system_defaults()
552        } else {
553            (Default::default(), Default::default())
554        };
555
556        for (addr, proto) in builder.nameservers.iter() {
557            let nameserver =
558                hickory_resolver::config::NameServerConfig::new(*addr, proto.to_hickory());
559            config.add_name_server(nameserver);
560        }
561
562        // see [`DnsResolver::lookup_ipv4_ipv6`] for info on why we avoid `LookupIpStrategy::Ipv4AndIpv6`
563        options.ip_strategy = hickory_resolver::config::LookupIpStrategy::Ipv4thenIpv6;
564
565        let mut hickory_builder =
566            TokioResolver::builder_with_config(config, TokioConnectionProvider::default());
567        *hickory_builder.options_mut() = options;
568        let resolver = hickory_builder.build();
569        Self { resolver, builder }
570    }
571
572    fn system_defaults() -> (ResolverConfig, ResolverOpts) {
573        let (system_config, options) =
574            hickory_resolver::system_conf::read_system_conf().unwrap_or_default();
575
576        // Copy all of the system config, but strip the bad windows nameservers.  Unfortunately
577        // there is no easy way to do this.
578        let mut config = hickory_resolver::config::ResolverConfig::new();
579        if let Some(name) = system_config.domain() {
580            config.set_domain(name.clone());
581        }
582        for name in system_config.search() {
583            config.add_search(name.clone());
584        }
585        for nameserver_cfg in system_config.name_servers() {
586            if !WINDOWS_BAD_SITE_LOCAL_DNS_SERVERS.contains(&nameserver_cfg.socket_addr.ip()) {
587                config.add_name_server(nameserver_cfg.clone());
588            }
589        }
590        (config, options)
591    }

629    }

634    }

641/// Record data for a TXT record.
642///
643/// This contains a list of character strings, as defined in [RFC 1035 Section 3.3.14].
644///
645/// [`TxtRecordData`] implements [`fmt::Display`], so you can call [`ToString::to_string`] to
646/// convert the record data into a string. This will parse each character string with
647/// [`String::from_utf8_lossy`] and then concatenate all strings without a separator.
648///
649/// If you want to process each character string individually, use [`Self::iter`].
650///
651/// [RFC 1035 Section 3.3.14]: https://datatracker.ietf.org/doc/html/rfc1035#section-3.3.14
652#[derive(Debug, Clone)]
653pub struct TxtRecordData(Box<[Box<[u8]>]>);
654
655impl TxtRecordData {
656    /// Returns an iterator over the character strings contained in this TXT record.
657    pub fn iter(&self) -> impl Iterator<Item = &[u8]> {
658        self.0.iter().map(|x| x.as_ref())
659    }
660}
661
662impl fmt::Display for TxtRecordData {
663    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
664        for s in self.iter() {
665            write!(f, "{}", &String::from_utf8_lossy(s))?
666        }
667        Ok(())
668    }
669}
670
671impl FromIterator<Box<[u8]>> for TxtRecordData {
672    fn from_iter<T: IntoIterator<Item = Box<[u8]>>>(iter: T) -> Self {
673        Self(iter.into_iter().collect())
674    }
675}
676
677impl From<Vec<Box<[u8]>>> for TxtRecordData {
678    fn from(value: Vec<Box<[u8]>>) -> Self {
679        Self(value.into_boxed_slice())
680    }
681}

702/// Windows still configures these site-local addresses as soon even as an IPv6 loopback
703/// interface is configured.  We do not want to use these DNS servers, the chances of them

706/// zero.  The chance of the DNS resolver accidentally trying one of these and taking a

710    IpAddr::V6(Ipv6Addr::new(0xfec0, 0, 0, 0xffff, 0, 0, 0, 2)),

714/// Helper enum to give a unified type to the iterators of [`DnsResolver::lookup_ipv4_ipv6`].
715enum LookupIter<A, B> {
716    Ipv4(A),
717    Ipv6(B),
718    Both(std::iter::Chain<A, B>),
719}
720
721impl<A: Iterator<Item = IpAddr>, B: Iterator<Item = IpAddr>> Iterator for LookupIter<A, B> {
722    type Item = IpAddr;
723
724    fn next(&mut self) -> Option<Self::Item> {
725        match self {
726            LookupIter::Ipv4(iter) => iter.next(),

733/// Staggers calls to the future F with the given delays.
734///
735/// The first call is performed immediately. The first call to succeed generates an Ok result
736/// ignoring any previous error. If all calls fail, an error summarizing all errors is returned.
737async fn stagger_call<
738    T,
739    E: std::fmt::Debug + std::fmt::Display,
740    F: Fn() -> Fut,
741    Fut: Future<Output = Result<T, E>>,
742>(
743    f: F,
744    delays_ms: &[u64],
745) -> Result<T, StaggeredError<E>> {
746    let mut calls = n0_future::FuturesUnorderedBounded::new(delays_ms.len() + 1);
747    // NOTE: we add the 0 delay here to have a uniform set of futures. This is more performant than
748    // using alternatives that allow futures of different types.
749    for delay in std::iter::once(&0u64).chain(delays_ms) {
750        let delay = add_jitter(delay);
751        let fut = f();
752        let staggered_fut = async move {
753            time::sleep(delay).await;
754            fut.await
755        };
756        calls.push(staggered_fut)
757    }
758
759    let mut errors = vec![];
760    while let Some(call_result) = calls.next().await {
761        match call_result {
762            Ok(t) => return Ok(t),
763            Err(e) => errors.push(e),
764        }
765    }
766
767    Err(StaggeredError::new(errors))
768}
769
770fn add_jitter(delay: &u64) -> Duration {
771    // If delay is 0, return 0 immediately.
772    if *delay == 0 {
773        return Duration::ZERO;
774    }
775
776    // Calculate jitter as a random value in the range of +/- MAX_JITTER_PERCENT of the delay.
777    let max_jitter = delay.saturating_mul(MAX_JITTER_PERCENT * 2) / 100;
778    let jitter = rand::random::<u64>() % max_jitter;
779
780    Duration::from_millis(delay.saturating_sub(max_jitter / 2).saturating_add(jitter))
781}
