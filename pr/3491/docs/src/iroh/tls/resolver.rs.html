9#[derive(Debug)]
10pub(super) struct AlwaysResolvesCert {
11    key: Arc<rustls::sign::CertifiedKey>,
12}

14/// Error for generating TLS configs.
15#[common_fields({
16    backtrace: Option<snafu::Backtrace>,
17    #[snafu(implicit)]
18    span_trace: n0_snafu::SpanTrace,
19})]
20#[derive(Debug, Snafu)]
21#[non_exhaustive]
22pub(super) enum CreateConfigError {
23    /// Rustls configuration error
24    #[snafu(display("rustls error"), context(false))]
25    Rustls { source: rustls::Error },
26}
27
28impl AlwaysResolvesCert {
29    pub(super) fn new(secret_key: &SecretKey) -> Result<Self, CreateConfigError> {
30        // Directly use the key
31        let client_private_key = secret_key
32            .secret()
33            .to_pkcs8_pem(LineEnding::default())
34            .expect("key is valid");
35
36        let client_private_key = PrivatePkcs8KeyDer::from_pem_slice(client_private_key.as_bytes())
37            .expect("cannot open private key file");
38        let client_private_key = rustls::crypto::ring::sign::any_eddsa_type(&client_private_key)?;
39
40        let client_public_key = client_private_key
41            .public_key()
42            .ok_or(rustls::Error::InconsistentKeys(
43                rustls::InconsistentKeys::Unknown,
44            ))
45            .expect("cannot load public key");
46        let client_public_key_as_cert = CertificateDer::from(client_public_key.to_vec());
47
48        let certified_key =
49            rustls::sign::CertifiedKey::new(vec![client_public_key_as_cert], client_private_key);
50
51        let key = Arc::new(certified_key);
52
53        Ok(Self { key })
54    }
55}
