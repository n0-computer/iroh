1//! Presets allow configuring an endpoint quickly with a chosen set of defaults.
2//!
3//! # Example
4//!
5//! ```no_run
6//! # async fn wrapper() -> n0_error::Result {
7//! use iroh::{Endpoint, RelayMode, Watcher, endpoint::presets};
8//!
9//! let endpoint = Endpoint::builder(presets::N0).bind().await?;
10//! # let _ = endpoint;
11//! # Ok(())
12//! # }
13//! ```
14
15use crate::endpoint::Builder;
16
17/// Defines a preset
18pub trait Preset {
19    /// Applies the configuration to the passed in [`Builder`].
20    fn apply(self, builder: Builder) -> Builder;
21}
22
23/// Configures the endpoint to use the n0 defaults
24///
25/// Currently this consists of
26/// - the DNS Address Lookup service.
27/// - the default relay servers provided by Number 0.
28/// - setting the [`rustls::crypto::CryptoProvider`] to [ring] or [aws-lc-rs], depending
29///   on which feature is enabled in iroh (preferring ring if both are enabled).
30///
31/// Due to the last point, this preset is only available with the `ring` or
32/// `aws-lc-rs` preset installed.
33/// If you want to set your own crypto provider, we recommend copying the
34/// implementation of this preset into your own and setting the appropriate crypto
35/// provider there.
36///
37/// The default address lookup service publishes to and resolves from the
38/// n0.computer dns server `iroh.link`.
39///
40/// This is equivalent to adding both a [`crate::address_lookup::PkarrPublisher`]
41/// and a [`crate::address_lookup::DnsAddressLookup`], both configured to use the
42/// n0.computer dns server.
43///
44/// This will by default use [`N0_DNS_PKARR_RELAY_PROD`].
45/// When in tests, or when the `test-utils` feature is enabled, this will use the
46/// [`N0_DNS_PKARR_RELAY_STAGING`].
47///
48/// [ring]: rustls::crypto::ring::default_provider
49/// [aws-lc-rs]: rustls::crypto::aws_lc_rs::default_provider
50/// [`N0_DNS_PKARR_RELAY_PROD`]: crate::address_lookup::N0_DNS_PKARR_RELAY_PROD
51/// [`N0_DNS_PKARR_RELAY_STAGING`]: crate::address_lookup::N0_DNS_PKARR_RELAY_STAGING
52#[cfg(any(feature = "ring", feature = "aws-lc-rs"))]
53#[derive(Debug, Copy, Clone, Default)]
54pub struct N0;
55
56#[cfg(any(feature = "ring", feature = "aws-lc-rs"))]
57impl Preset for N0 {
58    fn apply(self, mut builder: Builder) -> Builder {
59        use crate::{address_lookup::PkarrPublisher, endpoint::default_relay_mode};
60        use std::sync::Arc;
61
62        builder = builder.address_lookup(PkarrPublisher::n0_dns());
63
64        // Resolve using HTTPS requests to our DNS server's /pkarr path in browsers
65        #[cfg(wasm_browser)]
66        {
67            use crate::address_lookup::PkarrResolver;
68
69            builder = builder.address_lookup(PkarrResolver::n0_dns());
70        }
71        // Resolve using DNS queries outside browsers.
72        #[cfg(not(wasm_browser))]
73        {
74            builder = builder.address_lookup(crate::address_lookup::DnsAddressLookup::n0_dns());
75        }
76
77        #[cfg(feature = "ring")]
78        {
79            builder = builder.crypto_provider(Arc::new(rustls::crypto::ring::default_provider()));
80        }
81
82        #[cfg(all(feature = "aws-lc-rs", not(feature = "ring")))]
83        {
84            builder =
85                builder.crypto_provider(Arc::new(rustls::crypto::aws_lc_rs::default_provider()));
86        }
87
88        builder = builder.relay_mode(default_relay_mode());
89
90        builder
91    }
92}

94/// Configures the endpoint to use the n0 defaults, but with relay mode disabled.
95///
96/// Currently this consists of
97/// - setting `RelayMode::Disabled`
98/// - the DNS Address Lookup service, that publishes IP addresses rather than
99///   relay urls.
100/// - setting the [`rustls::crypto::CryptoProvider`] to [ring] or [aws-lc-rs], depending
101///   on which feature is enabled in iroh (preferring ring if both are enabled).
102///
103/// Due to the last point, this preset is only available with the `ring` or
104/// `aws-lc-rs` preset installed.
105/// If you want to set your own crypto provider, we recommend copying the
106/// implementation of this preset into your own and setting the appropriate crypto
107/// provider there.
108///
109/// The default address lookup service publishes to and resolves from the
110/// n0.computer dns server `iroh.link`.
111///
112/// This is equivalent to adding both a [`crate::address_lookup::PkarrPublisher`]
113/// and a [`crate::address_lookup::DnsAddressLookup`], both configured to use the
114/// n0.computer dns server.
115///
116/// This will by default use [`N0_DNS_PKARR_RELAY_PROD`].
117/// When in tests, or when the `test-utils` feature is enabled, this will use the
118/// [`N0_DNS_PKARR_RELAY_STAGING`].
119///
120/// [ring]: rustls::crypto::ring::default_provider
121/// [aws-lc-rs]: rustls::crypto::aws_lc_rs::default_provider
122/// [`N0_DNS_PKARR_RELAY_PROD`]: crate::address_lookup::N0_DNS_PKARR_RELAY_PROD
123/// [`N0_DNS_PKARR_RELAY_STAGING`]: crate::address_lookup::N0_DNS_PKARR_RELAY_STAGING
124#[cfg(any(feature = "ring", feature = "aws-lc-rs"))]
125#[derive(Debug, Copy, Clone, Default)]
126pub struct N0DisableRelay;
127
128#[cfg(any(feature = "ring", feature = "aws-lc-rs"))]
129impl Preset for N0DisableRelay {
130    fn apply(self, builder: Builder) -> Builder {
131        use crate::RelayMode;
132
133        N0.apply(builder).relay_mode(RelayMode::Disabled)
134    }
135}
