32/// Map of the [`EndpointState`] information for all the known endpoints.
33///
34/// The endpoints can be looked up by:
35///
36/// - The endpoint's ID in this map, only useful if you know the ID from an insert or lookup.
37///   This is static and never changes.
38///
39/// - The [`EndpointIdMappedAddr`] which internally identifies the endpoint to the QUIC stack.  This
40///   is static and never changes.
41///
42/// - The endpoints's public key, aka `PublicKey` or "endpoint_key".  This is static and never changes,
43///   however an endpoint could be added when this is not yet known.
44///
45/// - A public socket address on which they are reachable on the internet, known as ip-port.
46///   These come and go as the endpoint moves around on the internet
47///
48/// An index of endpointInfos by endpoint key, EndpointIdMappedAddr, and discovered ip:port endpoints.
49#[derive(Debug, Default)]
50pub(super) struct EndpointMap {
51    inner: Mutex<EndpointMapInner>,
52}

77/// The origin or *source* through which an address associated with a remote endpoint
78/// was discovered.
79///
80/// An aggregate of the [`Source`]s of all the addresses of an endpoint describe the
81/// [`Source`]s of the endpoint itself.
82///
83/// A [`Source`] helps track how and where an address was learned. Multiple
84/// sources can be associated with a single address, if we have discovered this
85/// address through multiple means.
86///
87/// Each time a [`EndpointAddr`] is added to the endpoint map a [`Source`] must be supplied to indicate
88/// how the address was obtained.
89///
90/// A [`Source`] can describe a variety of places that an address or endpoint was
91/// discovered, such as a configured discovery service, the network itself
92/// (if another endpoint has reached out to us), or as a user supplied [`EndpointAddr`].
93
94#[derive(Serialize, Deserialize, strum::Display, Debug, Clone, Eq, PartialEq, Hash)]
95#[strum(serialize_all = "kebab-case")]
96pub enum Source {
97    /// Address was loaded from the fs.
98    Saved,
99    /// An endpoint communicated with us first via UDP.
100    Udp,
101    /// An endpoint communicated with us first via relay.
102    Relay,
103    /// Application layer added the address directly.
104    App,
105    /// The address was discovered by a discovery service.
106    #[strum(serialize = "{name}")]
107    Discovery {
108        /// The name of the discovery service that discovered the address.
109        name: String,
110    },
111    /// Application layer with a specific name added the endpoint directly.
112    #[strum(serialize = "{name}")]
113    NamedApp {
114        /// The name of the application that added the endpoint
115        name: String,
116    },
117}

369    /// Add the contact information for an endpoint.
370    #[instrument(skip_all, fields(endpoint = %endpoint_addr.id.fmt_short()))]
371    fn add_endpoint_addr(
372        &mut self,
373        endpoint_addr: EndpointAddr,
374        source: Source,
375        have_ipv6: bool,
376        metrics: &Metrics,
377    ) {
378        let source0 = source.clone();
379        let endpoint_id = endpoint_addr.id;
380        let relay_url = endpoint_addr.relay_urls().next().cloned();
381        #[cfg(any(test, feature = "test-utils"))]
382        let path_selection = self.path_selection;
383        let endpoint_state =
384            self.get_or_insert_with(EndpointStateKey::EndpointId(endpoint_id), || Options {
385                endpoint_id,
386                relay_url,
387                active: false,
388                source,
389                #[cfg(any(test, feature = "test-utils"))]
390                path_selection,
391            });
392        endpoint_state.update_from_endpoint_addr(
393            endpoint_addr.relay_urls().next(),
394            endpoint_addr.ip_addrs().copied(),
395            source0,
396            have_ipv6,
397            metrics,
398        );
399        let id = endpoint_state.id();
400        for addr in endpoint_addr.ip_addrs() {
401            self.set_endpoint_state_for_ip_port(*addr, id);
402        }
403    }

416    /// Removes a direct address from an endpoint.
417    fn remove_by_ipp(&mut self, ipp: IpPort, now: Instant, why: &'static str) {
418        if let Some(id) = self.by_ip_port.remove(&ipp) {
419            if let Entry::Occupied(mut entry) = self.by_id.entry(id) {
420                let endpoint = entry.get_mut();
421                endpoint.remove_direct_addr(&ipp, now, why);
422                if endpoint.ip_addrs().count() == 0 {
423                    let endpoint_id = endpoint.public_key();
424                    let mapped_addr = endpoint.quic_mapped_addr();
425                    self.by_endpoint_key.remove(endpoint_id);
426                    self.by_quic_mapped_addr.remove(mapped_addr);
427                    debug!(endpoint_id=%endpoint_id.fmt_short(), why, "removing endpoint");
428                    entry.remove();
429                }
430            }
431        }
432    }

592    fn handle_ping(
593        &mut self,
594        sender: EndpointId,
595        src: SendAddr,
596        tx_id: TransactionId,
597    ) -> PingHandled {
598        #[cfg(any(test, feature = "test-utils"))]
599        let path_selection = self.path_selection;
600        let endpoint_state = self.get_or_insert_with(EndpointStateKey::EndpointId(sender), || {
601            debug!("received ping: endpoint unknown, add to endpoint map");
602            let source = if src.is_relay() {
603                Source::Relay
604            } else {
605                Source::Udp
606            };
607            Options {
608                endpoint_id: sender,
609                relay_url: src.relay_url(),
610                active: true,
611                source,
612                #[cfg(any(test, feature = "test-utils"))]
613                path_selection,
614            }
615        });
616
617        let handled = endpoint_state.handle_ping(src.clone(), tx_id);
618        if let SendAddr::Udp(ref addr) = src {
619            if matches!(handled.role, PingRole::NewPath) {
620                self.set_endpoint_key_for_ip_port(*addr, &sender);
621            }
622        }
623        handled
624    }

626    /// Inserts a new endpoint into the [`EndpointMap`].
627    fn insert_endpoint(&mut self, options: Options) -> &mut EndpointState {
628        info!(
629            endpoint = %options.endpoint_id.fmt_short(),
630            relay_url = ?options.relay_url,
631            source = %options.source,
632            "inserting new endpoint in EndpointMap",
633        );
634        let id = self.next_id;
635        self.next_id = self.next_id.wrapping_add(1);
636        let endpoint_state = EndpointState::new(id, options);
637
638        // update indices
639        self.by_quic_mapped_addr
640            .insert(*endpoint_state.quic_mapped_addr(), id);
641        self.by_endpoint_key
642            .insert(*endpoint_state.public_key(), id);
643
644        self.by_id.insert(id, endpoint_state);
645        self.by_id.get_mut(&id).expect("just inserted")
646    }

648    /// Makes future endpoint lookups by ipp return the same endpoint as a lookup by nk.
649    ///
650    /// This should only be called with a fully verified mapping of ipp to
651    /// nk, because calling this function defines the endpoint we hand to
652    /// WireGuard for packets received from ipp.
653    fn set_endpoint_key_for_ip_port(&mut self, ipp: impl Into<IpPort>, nk: &PublicKey) {
654        let ipp = ipp.into();
655        if let Some(id) = self.by_ip_port.get(&ipp) {
656            if !self.by_endpoint_key.contains_key(nk) {
657                self.by_endpoint_key.insert(*nk, *id);
658            }
659            self.by_ip_port.remove(&ipp);
660        }
661        if let Some(id) = self.by_endpoint_key.get(nk) {
662            trace!("insert ip -> id: {:?} -> {}", ipp, id);
663            self.by_ip_port.insert(ipp, *id);
664        }
665    }
666
667    fn set_endpoint_state_for_ip_port(&mut self, ipp: impl Into<IpPort>, id: usize) {
668        let ipp = ipp.into();
669        trace!(?ipp, ?id, "set endpoint for ip:port");
670        self.by_ip_port.insert(ipp, id);
671    }
672
673    /// Prunes endpoints without recent activity so that at most [`MAX_INACTIVE_ENDPOINTS`] are kept.
674    fn prune_inactive(&mut self) {
675        let now = Instant::now();
676        let mut prune_candidates: Vec<_> = self
677            .by_id
678            .values()
679            .filter(|endpoint| !endpoint.is_active(&now))
680            .map(|endpoint| (*endpoint.public_key(), endpoint.last_used()))
681            .collect();
682
683        let prune_count = prune_candidates
684            .len()
685            .saturating_sub(MAX_INACTIVE_ENDPOINTS);
686        if prune_count == 0 {
687            // within limits
688            return;
689        }
690
691        prune_candidates.sort_unstable_by_key(|(_pk, last_used)| *last_used);
692        prune_candidates.truncate(prune_count);
693        for (public_key, last_used) in prune_candidates.into_iter() {
694            let endpoint = public_key.fmt_short();
695            match last_used.map(|instant| instant.elapsed()) {
696                Some(last_used) => trace!(%endpoint, ?last_used, "pruning inactive"),
697                None => trace!(%endpoint, last_used=%"never", "pruning inactive"),
698            }
699
700            let Some(id) = self.by_endpoint_key.remove(&public_key) else {
701                debug_assert!(false, "missing by_endpoint_key entry for pk in by_id");
702                continue;
703            };
704
705            let Some(ep) = self.by_id.remove(&id) else {
706                debug_assert!(false, "missing by_id entry for id in by_endpoint_key");
707                continue;
708            };
709
710            for ip_port in ep.ip_addrs() {
711                self.by_ip_port.remove(&ip_port);
712            }
713
714            self.by_quic_mapped_addr.remove(ep.quic_mapped_addr());
715        }
716    }
717}

719/// An (Ip, Port) pair.
720///
721/// NOTE: storing an [`IpPort`] is safer than storing a [`SocketAddr`] because for IPv6 socket
722/// addresses include fields that can't be assumed consistent even within a single connection.
723#[derive(Debug, derive_more::Display, Clone, Copy, Hash, PartialEq, Eq, PartialOrd, Ord)]
724#[display("{}", SocketAddr::from(*self))]
725pub struct IpPort {
726    ip: IpAddr,
727    port: u16,
728}
729
730impl From<SocketAddr> for IpPort {
731    fn from(socket_addr: SocketAddr) -> Self {
732        Self {
733            ip: socket_addr.ip(),
734            port: socket_addr.port(),
735        }
736    }
737}
738
739impl From<IpPort> for SocketAddr {
740    fn from(ip_port: IpPort) -> Self {
741        let IpPort { ip, port } = ip_port;
742        (ip, port).into()
743    }
744}
745
746impl IpPort {
747    pub fn ip(&self) -> &IpAddr {
748        &self.ip
749    }
750
751    pub fn port(&self) -> u16 {
752        self.port
753    }
754}

851    #[traced_test]
852    fn test_prune_direct_addresses() {
853        let endpoint_map = EndpointMap::default();
854        let mut rng = rand_chacha::ChaCha8Rng::seed_from_u64(0u64);
855        let public_key = SecretKey::generate(&mut rng).public();
856        let id = endpoint_map
857            .inner
858            .lock()
859            .unwrap()
860            .insert_endpoint(Options {
861                endpoint_id: public_key,
862                relay_url: None,
863                active: false,
864                source: Source::NamedApp {
865                    name: "test".into(),
866                },
867                path_selection: PathSelection::default(),
868            })
869            .id();
870
871        const LOCALHOST: IpAddr = IpAddr::V4(std::net::Ipv4Addr::LOCALHOST);
872
873        // add [`MAX_INACTIVE_DIRECT_ADDRESSES`] active direct addresses and double
874        // [`MAX_INACTIVE_DIRECT_ADDRESSES`] that are inactive
875
876        info!("Adding active addresses");
877        for i in 0..MAX_INACTIVE_DIRECT_ADDRESSES {
878            let addr = SocketAddr::new(LOCALHOST, 5000 + i as u16);
879            let endpoint_addr = EndpointAddr::new(public_key).with_ip_addr(addr);
880            // add address
881            endpoint_map.add_test_addr(endpoint_addr);
882            // make it active
883            endpoint_map.inner.lock().unwrap().receive_udp(addr);
884        }
885
886        info!("Adding offline/inactive addresses");
887        for i in 0..MAX_INACTIVE_DIRECT_ADDRESSES * 2 {
888            let addr = SocketAddr::new(LOCALHOST, 6000 + i as u16);
889            let endpoint_addr = EndpointAddr::new(public_key).with_ip_addr(addr);
890            endpoint_map.add_test_addr(endpoint_addr);
891        }
892
893        let mut endpoint_map_inner = endpoint_map.inner.lock().unwrap();
894        let endpoint = endpoint_map_inner.by_id.get_mut(&id).unwrap();
895
896        info!("Adding alive addresses");
897        for i in 0..MAX_INACTIVE_DIRECT_ADDRESSES {
898            let addr = SendAddr::Udp(SocketAddr::new(LOCALHOST, 7000 + i as u16));
899            let txid = TransactionId::from([i as u8; 12]);
900            // Note that this already invokes .prune_direct_addresses() because these are
901            // new UDP paths.
902            endpoint.handle_ping(addr, txid);
903        }
904
905        info!("Pruning addresses");
906        endpoint.prune_direct_addresses(Instant::now());
907
908        // Half the offline addresses should have been pruned.  All the active and alive
909        // addresses should have been kept.
910        assert_eq!(
911            endpoint.ip_addrs().count(),
912            MAX_INACTIVE_DIRECT_ADDRESSES * 3
913        );
914
915        // We should have both offline and alive addresses which are not active.
916        assert_eq!(
917            endpoint
918                .ip_addr_states()
919                .filter(|(_addr, state)| !state.is_active())
920                .count(),
921            MAX_INACTIVE_DIRECT_ADDRESSES * 2
922        )
923    }
