1//! Addressing for iroh endpoints.
2//!
3//! This module contains some common addressing types for iroh.  An endpoint is uniquely
4//! identified by the [`EndpointId`] but that does not make it addressable on the network layer.
5//! For this the addition of a [`RelayUrl`] and/or direct addresses are required.
6//!
7//! The primary way of addressing an endpoint is by using the [`EndpointAddr`].
8
9use std::{collections::BTreeSet, net::SocketAddr};
10
11use serde::{Deserialize, Serialize};
12
13use crate::{EndpointId, PublicKey, RelayUrl};
14
15/// Network-level addressing information for an iroh endpoint.
16///
17/// This combines an endpoint's identifier with network-level addressing information of how to
18/// contact the endpoint.
19///
20/// To establish a network connection to an endpoint both the [`EndpointId`] and one or more network
21/// paths are needed.  The network paths can come from various sources:
22///
23/// - A [discovery] service which can provide routing information for a given [`EndpointId`].
24///
25/// - A [`RelayUrl`] of the endpoint's [home relay], this allows establishing the connection via
26///   the Relay server and is very reliable.
27///
28/// - One or more *direct addresses* on which the endpoint might be reachable.  Depending on the
29///   network location of both endpoints it might not be possible to establish a direct
30///   connection without the help of a [Relay server].
31///
32/// This structure will always contain the required [`EndpointId`] and will contain an optional
33/// number of network-level addressing information.  It is a generic addressing type used
34/// whenever a connection to other endpoints needs to be established.
35///
36/// [discovery]: https://docs.rs/iroh/*/iroh/index.html#endpoint-discovery
37/// [home relay]: https://docs.rs/iroh/*/iroh/relay/index.html
38/// [Relay server]: https://docs.rs/iroh/*/iroh/index.html#relay-servers
39#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, PartialOrd, Ord, Hash)]
40pub struct EndpointAddr {
41    /// The endpoint's identifier.
42    pub id: EndpointId,
43    /// The endpoint's addresses
44    pub addrs: BTreeSet<TransportAddr>,
45}
46
47/// Available address types.
48#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, PartialOrd, Ord, Hash)]
49#[non_exhaustive]
50pub enum TransportAddr {
51    /// Relays
52    Relay(RelayUrl),
53    /// IP based addresses
54    Ip(SocketAddr),
55}
56
57impl EndpointAddr {
58    /// Creates a new [`EndpointAddr`] with no addresses.
59    pub fn new(id: PublicKey) -> Self {
60        EndpointAddr {
61            id,
62            addrs: Default::default(),
63        }
64    }
65
66    /// Adds a relay url.
67    pub fn with_relay_url(mut self, relay_url: RelayUrl) -> Self {
68        self.addrs.insert(TransportAddr::Relay(relay_url));
69        self
70    }
71
72    /// Adds the given IP addresses.
73    pub fn with_ip_addresses(mut self, addresses: impl IntoIterator<Item = SocketAddr>) -> Self {
74        for addr in addresses.into_iter() {
75            self.addrs.insert(TransportAddr::Ip(addr));
76        }
77        self
78    }
79
80    /// Creates a new [`EndpointAddr`] from its parts.
81    pub fn from_parts(id: PublicKey, addrs: impl IntoIterator<Item = TransportAddr>) -> Self {
82        Self {
83            id,
84            addrs: addrs.into_iter().collect(),
85        }
86    }
87
88    /// Returns true, if only a [`EndpointId`] is present.
89    pub fn is_empty(&self) -> bool {
90        self.addrs.is_empty()
91    }
92
93    /// Returns the IP addresses of this peer.
94    pub fn ip_addresses(&self) -> impl Iterator<Item = &SocketAddr> {
95        self.addrs.iter().filter_map(|addr| match addr {
96            TransportAddr::Ip(addr) => Some(addr),
97            _ => None,
98        })
99    }
100
101    /// Returns the relay url of this peer.
102    pub fn relay_urls(&self) -> impl Iterator<Item = &RelayUrl> {
103        self.addrs.iter().filter_map(|addr| match addr {
104            TransportAddr::Relay(url) => Some(url),
105            _ => None,
106        })
107    }
108}
109
110impl From<(PublicKey, Option<RelayUrl>, &[SocketAddr])> for EndpointAddr {
111    fn from(value: (PublicKey, Option<RelayUrl>, &[SocketAddr])) -> Self {
112        let (id, relay_url, ip_addrs_iter) = value;
113        let mut addrs = BTreeSet::new();
114        if let Some(url) = relay_url {
115            addrs.insert(TransportAddr::Relay(url));
116        }
117        for addr in ip_addrs_iter {
118            addrs.insert(TransportAddr::Ip(*addr));
119        }
120        EndpointAddr { id, addrs }
121    }
122}
123
124impl From<EndpointId> for EndpointAddr {
125    fn from(endpoint_id: EndpointId) -> Self {
126        EndpointAddr::new(endpoint_id)
127    }
128}

130#[cfg(test)]
131mod tests {
132    use super::*;
133
134    #[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, PartialOrd, Ord, Hash)]
135    #[non_exhaustive]
136    enum NewAddrType {
137        /// Relays
138        Relay(RelayUrl),
139        /// IP based addresses
140        Ip(SocketAddr),
141        /// New addr type for testing
142        Cool(u16),
143    }
144
145    #[test]
146    fn test_roundtrip_new_addr_type() {
147        let old = vec![
148            TransportAddr::Ip("127.0.0.1:9".parse().unwrap()),
149            TransportAddr::Relay("https://example.com".parse().unwrap()),
150        ];
151        let old_ser = postcard::to_stdvec(&old).unwrap();
152        let old_back: Vec<TransportAddr> = postcard::from_bytes(&old_ser).unwrap();
153        assert_eq!(old, old_back);
154
155        let new = vec![
156            NewAddrType::Ip("127.0.0.1:9".parse().unwrap()),
157            NewAddrType::Relay("https://example.com".parse().unwrap()),
158            NewAddrType::Cool(4),
159        ];
160        let new_ser = postcard::to_stdvec(&new).unwrap();
161        let new_back: Vec<NewAddrType> = postcard::from_bytes(&new_ser).unwrap();
162
163        assert_eq!(new, new_back);
164
165        // serialize old into new
166        let old_new_back: Vec<NewAddrType> = postcard::from_bytes(&old_ser).unwrap();
167
168        assert_eq!(
169            old_new_back,
170            vec![
171                NewAddrType::Ip("127.0.0.1:9".parse().unwrap()),
172                NewAddrType::Relay("https://example.com".parse().unwrap()),
173            ]
174        );
175    }
