26/// IP transport configuration
27#[derive(Debug, Copy, Clone, PartialEq, Eq)]
28pub enum Config {
29    /// Default IPv4 binding
30    V4Default {
31        /// The IP address to bind on
32        ip_addr: Ipv4Addr,
33        /// The port to bind on
34        port: u16,
35    },
36    /// Default IPv6 binding
37    V6Default {
38        /// The IP address to bind on
39        ip_addr: Ipv6Addr,
40        /// The scope_id
41        scope_id: u32,
42        /// The port to bind on
43        port: u16,
44    },
45    /// General IPv4 binding
46    V4 {
47        /// The IP address to bind on
48        ip_addr: Ipv4Net,
49        /// The port to bind on
50        port: u16,
51    },
52    /// General IPv6 binding
53    V6 {
54        /// The IP address to bind on
55        ip_addr: Ipv6Net,
56        /// The scope id.
57        scope_id: u32,
58        /// The port to bind on
59        port: u16,
60    },

63impl Config {

75    pub fn is_default(&self) -> bool {
76        matches!(self, Self::V4Default { .. } | Self::V6Default { .. })
77    }

79    /// Does this configuration match to send to the given `dst` address.
80    pub(crate) fn is_valid_send_addr(&self, dst: SocketAddr) -> bool {
81        match self {
82            Self::V4Default { .. } => matches!(dst, SocketAddr::V4(_)),
83            Self::V6Default { .. } => matches!(dst, SocketAddr::V6(_)),
84            Self::V4 { ip_addr, .. } => match dst {
85                SocketAddr::V4(dst_v4) => ip_addr.contains(dst_v4.ip()),
86                SocketAddr::V6(_) => false,
87            },
88            Self::V6 {
89                ip_addr, scope_id, ..
90            } => match dst {
91                SocketAddr::V6(dst_v6) => {
92                    if ip_addr.contains(dst_v6.ip()) {
93                        return true;
94                    }
95                    if dst_v6.ip().is_unicast_link_local() {
96                        // If we have a link local interface, use the scope id
97                        if *scope_id == dst_v6.scope_id() {
98                            return true;
99                        }
100                    }
101                    false
102                }
103                SocketAddr::V4(_) => false,
104            },
105        }
106    }
107}

109impl From<Config> for SocketAddr {
110    fn from(value: Config) -> Self {
111        match value {
112            Config::V4Default { ip_addr, port } => SocketAddr::V4(SocketAddrV4::new(ip_addr, port)),
113            Config::V6Default {
114                ip_addr,
115                scope_id,
116                port,
117            } => SocketAddr::V6(SocketAddrV6::new(ip_addr, port, 0, scope_id)),
118            Config::V4 { ip_addr, port } => SocketAddr::V4(SocketAddrV4::new(ip_addr.addr(), port)),
119            Config::V6 {
120                ip_addr,
121                scope_id,
122                port,
123            } => SocketAddr::V6(SocketAddrV6::new(ip_addr.addr(), port, 0, scope_id)),
124        }
125    }
126}

128fn bind_with_fallback(mut addr: SocketAddr) -> io::Result<netwatch::UdpSocket> {
129    debug!(?addr, "binding");
130    // First try binding a preferred port, if specified
131    match netwatch::UdpSocket::bind_full(addr) {
132        Ok(socket) => {
133            let local_addr = socket.local_addr()?;
134            debug!(%addr, %local_addr, "successfully bound");
135            return Ok(socket);
136        }
137        Err(err) => {
138            debug!(%addr, "failed to bind: {err:#}");
139            // If that was already the fallback port, then error out
140            if addr.port() == 0 {
141                return Err(err);
142            }
143        }
144    }
145
146    // Otherwise, try binding with port 0
147    addr.set_port(0);
148    netwatch::UdpSocket::bind_full(addr)
149}

174    /// NOTE: Receiving on a closed socket will return [`Poll::Pending`] indefinitely.
175    pub(super) fn poll_recv(
176        &mut self,
177        cx: &mut Context,
178        bufs: &mut [io::IoSliceMut<'_>],
179        metas: &mut [quinn_udp::RecvMeta],
180        source_addrs: &mut [Addr],
181    ) -> Poll<io::Result<usize>> {
182        match self.socket.poll_recv_quinn(cx, bufs, metas) {
183            Poll::Pending => Poll::Pending,
184            Poll::Ready(Ok(n)) => {
185                for (source_addr, meta) in source_addrs.iter_mut().zip(metas.iter_mut()).take(n) {
186                    if meta.addr.is_ipv4() {
187                        // The AsyncUdpSocket is an AF_INET6 socket and needs to show this
188                        // as coming from an IPv4-mapped IPv6 addresses, since Quinn will
189                        // use those when sending on an INET6 socket.
190                        let v6_ip = match meta.addr.ip() {
191                            IpAddr::V4(ipv4_addr) => ipv4_addr.to_ipv6_mapped(),
192                            IpAddr::V6(ipv6_addr) => ipv6_addr,
193                        };
194                        meta.addr = SocketAddr::new(v6_ip.into(), meta.addr.port());
195                    }
196                    // The transport addresses are internal to iroh and we always want those
197                    // to remain the canonical address.
198                    *source_addr =
199                        SocketAddr::new(meta.addr.ip().to_canonical(), meta.addr.port()).into();
200                }
201                Poll::Ready(Ok(n))
202            }
203            Poll::Ready(Err(err)) => Poll::Ready(Err(err)),
204        }
205    }

251    pub(super) fn rebind(&self) -> io::Result<()> {
252        let old_addr = self.local_addr.get();
253        self.socket.rebind()?;
254        let addr = self.socket.local_addr()?;
255        self.local_addr.set(addr).ok();
256        trace!("rebound from {} to {}", old_addr, addr);
257
258        Ok(())
259    }

280    /// Creates a canonical socket address.
281    ///
282    /// We may be asked to send IPv4-mapped IPv6 addresses.  But our sockets are configured
283    /// to only send their actual family.  So we need to map those back to the canonical
284    /// addresses.
285    #[inline]
286    fn canonical_addr(addr: SocketAddr) -> SocketAddr {
287        SocketAddr::new(addr.ip().to_canonical(), addr.port())
288    }
289
290    pub(super) fn poll_send(
291        mut self: Pin<&mut Self>,
292        cx: &mut std::task::Context,
293        dst: SocketAddr,
294        src: Option<IpAddr>,
295        transmit: &Transmit<'_>,
296    ) -> Poll<io::Result<()>> {
297        let total_bytes = transmit.contents.len() as u64;
298        let res = Pin::new(&mut self.sender).poll_send(
299            &quinn_udp::Transmit {
300                destination: Self::canonical_addr(dst),
301                ecn: transmit.ecn,
302                contents: transmit.contents,
303                segment_size: transmit.segment_size,
304                src_ip: src,
305            },
306            cx,
307        );
308
309        match res {
310            Poll::Ready(Ok(res)) => {
311                match dst {
312                    SocketAddr::V4(_) => {
313                        self.metrics.send_ipv4.inc_by(total_bytes);
314                    }
315                    SocketAddr::V6(_) => {
316                        self.metrics.send_ipv6.inc_by(total_bytes);
317                    }
318                }
319                Poll::Ready(Ok(res))
320            }
321            Poll::Ready(Err(err)) => Poll::Ready(Err(err)),
322            Poll::Pending => Poll::Pending,
323        }
324    }
325}

355impl IpTransports {
356    pub(super) fn create_sender(&self) -> IpTransportsSender {
357        let ip_v4_default = self.v4_default.as_ref().map(|t| t.create_sender());
358        let ip_v4 = self.v4.iter().map(|t| t.create_sender()).collect();
359        let ip_v6_default = self.v6_default.as_ref().map(|t| t.create_sender());
360        let ip_v6 = self.v6.iter().map(|t| t.create_sender()).collect();
361
362        IpTransportsSender {
363            v4_default: ip_v4_default,
364            v4: ip_v4,
365            v6_default: ip_v6_default,
366            v6: ip_v6,
367        }
368    }
369
370    pub(super) fn iter(&self) -> impl Iterator<Item = &IpTransport> {
371        self.v4_default
372            .iter()
373            .chain(self.v4.iter())
374            .chain(self.v6_default.iter())
375            .chain(self.v6.iter())
376    }
377
378    pub(super) fn bind(
379        configs: impl Iterator<Item = Config>,
380        metrics: &EndpointMetrics,
381    ) -> io::Result<Self> {
382        let mut ip_v4_default = None;
383        let mut ip_v4 = Vec::new();
384        let mut ip_v6_default = None;
385        let mut ip_v6 = Vec::new();
386
387        for config in configs {
388            match IpTransport::bind(config, metrics.magicsock.clone()) {
389                Ok(transport) => {
390                    if config.is_ipv4() {
391                        if config.is_default() {
392                            if ip_v4_default.is_some() {
393                                return Err(io::Error::other(
394                                    "can only have a single IPv4 default transport",
395                                ));
396                            }
397                            ip_v4_default = Some(transport);
398                        } else {
399                            ip_v4.push(transport);
400                        }
401                    } else if config.is_ipv6() {
402                        if config.is_default() {
403                            if ip_v6_default.is_some() {
404                                return Err(io::Error::other(
405                                    "can only have a single IPv6 default transport",
406                                ));
407                            }
408                            ip_v6_default = Some(transport);
409                        } else {
410                            ip_v6.push(transport);
411                        }
412                    }
413                }
414                Err(err) => {
415                    if config.is_ipv6() {
416                        tracing::info!("bind ignoring IPv6 bind failure: {:?}", err);
417                    } else {
418                        return Err(err);
419                    }
420                }
421            }
422        }
423
424        Ok(Self {
425            v4_default: ip_v4_default,
426            v4: ip_v4,
427            v6_default: ip_v6_default,
428            v6: ip_v6,
429        })
430    }
431
432    pub(super) fn poll_recv(
433        &mut self,
434        cx: &mut Context,
435        bufs: &mut [io::IoSliceMut<'_>],
436        metas: &mut [quinn_udp::RecvMeta],
437        source_addrs: &mut [Addr],
438    ) -> Poll<io::Result<usize>> {
439        macro_rules! poll_transport {
440            ($socket:expr) => {
441                match $socket.poll_recv(cx, bufs, metas, source_addrs)? {
442                    Poll::Pending | Poll::Ready(0) => {}
443                    Poll::Ready(n) => {
444                        return Poll::Ready(Ok(n));
445                    }
446                }
447            };
448        }
449
450        if let Some(ref mut transport) = self.v4_default {
451            poll_transport!(transport);
452        }
453
454        for transport in &mut self.v4 {
455            poll_transport!(transport);
456        }
457
458        if let Some(ref mut transport) = self.v6_default {
459            poll_transport!(transport);
460        }
461
462        for transport in &mut self.v6 {
463            poll_transport!(transport);
464        }
465
466        Poll::Pending
467    }
