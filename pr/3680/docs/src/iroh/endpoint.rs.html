162    /// Binds the magic endpoint.
163    pub async fn bind(mut self) -> Result<Endpoint, BindError> {
164        let mut rng = rand::rng();
165        let secret_key = self
166            .secret_key
167            .unwrap_or_else(move || SecretKey::generate(&mut rng));
168
169        // Override some transport config settings.
170        self.transport_config
171            .keep_alive_interval(Some(HEARTBEAT_INTERVAL));
172        self.transport_config
173            .default_path_keep_alive_interval(Some(HEARTBEAT_INTERVAL));
174        self.transport_config
175            .default_path_max_idle_timeout(Some(PATH_MAX_IDLE_TIMEOUT));
176        self.transport_config
177            .max_concurrent_multipath_paths(MAX_MULTIPATH_PATHS);
178
179        let static_config = StaticConfig {
180            transport_config: Arc::new(self.transport_config),
181            tls_config: tls::TlsConfig::new(secret_key.clone(), self.max_tls_tickets),
182            keylog: self.keylog,
183        };
184        let server_config = static_config.create_server_config(self.alpn_protocols);
185
186        #[cfg(not(wasm_browser))]
187        let dns_resolver = self.dns_resolver.unwrap_or_default();
188
189        let metrics = EndpointMetrics::default();
190
191        let msock_opts = magicsock::Options {
192            transports: self.transports,
193            secret_key,
194            discovery_user_data: self.discovery_user_data,
195            proxy_url: self.proxy_url,
196            #[cfg(not(wasm_browser))]
197            dns_resolver,
198            server_config,
199            #[cfg(any(test, feature = "test-utils"))]
200            insecure_skip_relay_cert_verify: self.insecure_skip_relay_cert_verify,
201            metrics,
202        };
203
204        let msock = magicsock::MagicSock::spawn(msock_opts).await?;
205        trace!("created magicsock");
206        debug!(version = env!("CARGO_PKG_VERSION"), "iroh Endpoint created");
207
208        let ep = Endpoint {
209            msock,
210            static_config: Arc::new(static_config),
211        };
212
213        // Add discovery mechanisms
214        for create_service in self.discovery {
215            let service = create_service.into_discovery(&ep)?;
216            ep.discovery().add_boxed(service);
217        }
218
219        Ok(ep)
220    }

293    /// Sets the relay servers to assist in establishing connectivity.
294    ///
295    /// Relay servers are used to establish initial connection with another iroh endpoint.
296    /// They also perform various functions related to hole punching, see the [crate docs]
297    /// for more details.
298    ///
299    /// By default the [number 0] relay servers are used, see [`RelayMode::Default`].
300    ///
301    /// When using [RelayMode::Custom], the provided `relay_map` must contain at least one
302    /// configured relay endpoint.  If an invalid RelayMap is provided [`bind`]
303    /// will result in an error.
304    ///
305    /// [`bind`]: Builder::bind
306    /// [crate docs]: crate
307    /// [number 0]: https://n0.computer
308    pub fn relay_mode(mut self, relay_mode: RelayMode) -> Self {
309        let transport: Option<_> = relay_mode.into();
310        match transport {
311            Some(transport) => {
312                if let Some(og) = self
313                    .transports
314                    .iter_mut()
315                    .find(|t| matches!(t, TransportConfig::Relay { .. }))
316                {
317                    *og = transport;
318                } else {
319                    self.transports.push(transport);
320                }
321            }
322            None => {
323                self.transports
324                    .retain(|t| !matches!(t, TransportConfig::Relay { .. }));
325            }
326        }
327        self
328    }

479/// Controls an iroh endpoint, establishing connections with other endpoints.
480///
481/// This is the main API interface to create connections to, and accept connections from
482/// other iroh endpoints.  The connections are peer-to-peer and encrypted, a Relay server is
483/// used to make the connections reliable.  See the [crate docs] for a more detailed
484/// overview of iroh.
485///
486/// It is recommended to only create a single instance per application.  This ensures all
487/// the connections made share the same peer-to-peer connections to other iroh endpoints,
488/// while still remaining independent connections.  This will result in more optimal network
489/// behaviour.
490///
491/// The endpoint is created using the [`Builder`], which can be created using
492/// [`Endpoint::builder`].
493///
494/// Once an endpoint exists, new connections are typically created using the
495/// [`Endpoint::connect`] and [`Endpoint::accept`] methods.  Once established, the
496/// [`Connection`] gives access to most [QUIC] features.  Individual streams to send data to
497/// the peer are created using the [`Connection::open_bi`], [`Connection::accept_bi`],
498/// [`Connection::open_uni`] and [`Connection::open_bi`] functions.
499///
500/// Note that due to the light-weight properties of streams a stream will only be accepted
501/// once the initiating peer has sent some data on it.
502///
503/// [QUIC]: https://quicwg.org
504#[derive(Clone, Debug)]
505pub struct Endpoint {
506    /// Handle to the magicsocket/actor
507    pub(crate) msock: Handle,
508    /// Configuration structs for quinn, holds the transport config, certificate setup, secret key etc.
509    static_config: Arc<StaticConfig>,
510}

619    /// Connects to a remote [`Endpoint`].
620    ///
621    /// A value that can be converted into an [`EndpointAddr`] is required. This can be either an
622    /// [`EndpointAddr`] or an [`EndpointId`].
623    ///
624    /// The [`EndpointAddr`] must contain the [`EndpointId`] to dial and may also contain a [`RelayUrl`]
625    /// and direct addresses. If direct addresses are provided, they will be used to try and
626    /// establish a direct connection without involving a relay server.
627    ///
628    /// If neither a [`RelayUrl`] or direct addresses are configured in the [`EndpointAddr`] it
629    /// may still be possible a connection can be established.  This depends on which, if any,
630    /// [`crate::discovery::Discovery`] services were configured using [`Builder::discovery`].  The discovery
631    /// service will also be used if the remote endpoint is not reachable on the provided direct
632    /// addresses and there is no [`RelayUrl`].
633    ///
634    /// If addresses or relay servers are neither provided nor can be discovered, the
635    /// connection attempt will fail with an error.
636    ///
637    /// The `alpn`, or application-level protocol identifier, is also required. The remote
638    /// endpoint must support this `alpn`, otherwise the connection attempt will fail with
639    /// an error.
640    ///
641    /// [`RelayUrl`]: crate::RelayUrl
642    pub async fn connect(
643        &self,
644        endpoint_addr: impl Into<EndpointAddr>,
645        alpn: &[u8],
646    ) -> Result<Connection, ConnectError> {
647        let endpoint_addr = endpoint_addr.into();
648        let remote = endpoint_addr.id;
649        let connecting = self
650            .connect_with_opts(endpoint_addr, alpn, Default::default())
651            .await?;
652        let conn = connecting.await?;

655            me = %self.id().fmt_short(),
656            remote = %remote.fmt_short(),
657            alpn = %String::from_utf8_lossy(alpn),
658            "Connection established."
659        );
660        Ok(conn)
661    }

663    /// Starts a connection attempt with a remote [`Endpoint`].
664    ///
665    /// Like [`Endpoint::connect`] (see also its docs for general details), but allows for a more
666    /// advanced connection setup with more customization in two aspects:
667    /// 1. The returned future resolves to a [`Connecting`], which can be further processed into
668    ///    a [`Connection`] by awaiting, or alternatively allows connecting with 0-RTT via
669    ///    [`Connecting::into_0rtt`].
670    ///    **Note:** Please read the documentation for `into_0rtt` carefully to assess
671    ///    security concerns.
672    /// 2. The [`TransportConfig`] for the connection can be modified via the provided
673    ///    [`ConnectOptions`].
674    ///    **Note:** Please be aware that changing transport config settings may have adverse effects on
675    ///    establishing and maintaining direct connections.  Carefully test settings you use and
676    ///    consider this currently as still rather experimental.
677    #[instrument(name = "connect", skip_all, fields(
678        me = %self.id().fmt_short(),
679        remote = tracing::field::Empty,
680        alpn = String::from_utf8_lossy(alpn).to_string(),
681    ))]
682    pub async fn connect_with_opts(
683        &self,
684        endpoint_addr: impl Into<EndpointAddr>,
685        alpn: &[u8],
686        options: ConnectOptions,
687    ) -> Result<Connecting, ConnectWithOptsError> {
688        let endpoint_addr: EndpointAddr = endpoint_addr.into();
689        let endpoint_id = endpoint_addr.id;
690
691        tracing::Span::current().record("remote", tracing::field::display(endpoint_id.fmt_short()));
692
693        // Connecting to ourselves is not supported.
694        ensure!(endpoint_id != self.id(), ConnectWithOptsError::SelfConnect);
695
696        trace!(
697            dst_endpoint_id = %endpoint_id.fmt_short(),
698            relay_url = ?endpoint_addr.relay_urls().next().cloned(),
699            ip_addresses = ?endpoint_addr.ip_addrs().cloned().collect::<Vec<_>>(),
700            "connecting",
701        );
702
703        let mapped_addr = self.msock.resolve_remote(endpoint_addr).await??;
704
705        let transport_config = options
706            .transport_config
707            .unwrap_or(self.static_config.transport_config.clone());
708
709        // Start connecting via quinn. This will time out after 10 seconds if no reachable
710        // address is available.
711
712        let client_config = {
713            let mut alpn_protocols = vec![alpn.to_vec()];
714            alpn_protocols.extend(options.additional_alpns);
715            let quic_client_config = self
716                .static_config
717                .tls_config
718                .make_client_config(alpn_protocols, self.static_config.keylog);
719            let mut client_config = quinn::ClientConfig::new(Arc::new(quic_client_config));
720            client_config.transport_config(transport_config);
721            client_config
722        };
723
724        let dest_addr = mapped_addr.private_socket_addr();
725        let server_name = &tls::name::encode(endpoint_id);
726        let connect = self
727            .msock
728            .endpoint()
729            .connect_with(client_config, dest_addr, server_name)?;
730
731        Ok(Connecting::new(connect, self.clone(), endpoint_id))
732    }

781    ///
782    /// The observed [`EndpointAddr`] will have the current [`RelayUrl`] and direct addresses.
783    ///
784    /// ```no_run
785    /// # async fn wrapper() -> n0_error::Result<()> {
786    /// use iroh::{Endpoint, Watcher};

792    /// let endpoint_addr = endpoint.watch_addr().get();
793    /// # let _ = endpoint_addr;
794    /// # Ok(())
795    /// # }
796    /// ```
797    ///
798    /// The [`Endpoint::online`] method can be used as a convenience method to
799    /// understand if the endpoint has ever been considered "online". But after
800    /// that initial call to [`Endpoint::online`], to understand if your
801    /// endpoint is no longer able to be connected to by endpoints outside
802    /// of the private or local network, watch for changes in it's [`EndpointAddr`].
803    /// If there are no `addrs`in the [`EndpointAddr`], you may not be dialable by other endpoints
804    /// on the internet.
805    ///
806    ///
807    /// The `EndpointAddr` will change as:
808    /// - network conditions change
809    /// - the endpoint connects to a relay server
810    /// - the endpoint changes its preferred relay server
811    /// - more addresses are discovered for this endpoint
812    ///
813    /// [`RelayUrl`]: crate::RelayUrl
814    #[cfg(not(wasm_browser))]
815    pub fn watch_addr(&self) -> impl n0_watcher::Watcher<Value = EndpointAddr> + use<> {

817        let watch_relay = self.msock.home_relay();
818        let endpoint_id = self.id();
819
820        watch_addrs.or(watch_relay).map(move |(addrs, relays)| {
821            EndpointAddr::from_parts(
822                endpoint_id,
823                relays
824                    .into_iter()
825                    .map(TransportAddr::Relay)
826                    .chain(addrs.into_iter().map(|x| TransportAddr::Ip(x.addr))),
827            )
828        })
829    }

939    /// Returns metrics collected for this endpoint.
940    ///
941    /// The endpoint internally collects various metrics about its operation.
942    /// The returned [`EndpointMetrics`] struct contains all of these metrics.
943    ///
944    /// You can access individual metrics directly by using the public fields:
945    /// ```rust
946    /// # use std::collections::BTreeMap;
947    /// # use iroh::endpoint::Endpoint;
948    /// # async fn wrapper() -> n0_error::Result<()> {
949    /// let endpoint = Endpoint::bind().await?;
950    /// assert_eq!(endpoint.metrics().magicsock.recv_datagrams.get(), 0);
951    /// # Ok(())
952    /// # }
953    /// ```
954    ///
955    /// [`EndpointMetrics`] implements [`MetricsGroupSet`], and each field
956    /// implements [`MetricsGroup`]. These traits provide methods to iterate over
957    /// the groups in the set, and over the individual metrics in each group, without having
958    /// to access each field manually. With these methods, it is straightforward to collect
959    /// all metrics into a map or push their values to a metrics collector.
960    ///
961    /// For example, the following snippet collects all metrics into a map:
962    /// ```rust
963    /// # use std::collections::BTreeMap;
964    /// # use iroh_metrics::{Metric, MetricsGroup, MetricValue, MetricsGroupSet};
965    /// # use iroh::endpoint::Endpoint;
966    /// # async fn wrapper() -> n0_error::Result<()> {
967    /// let endpoint = Endpoint::bind().await?;
968    /// let metrics: BTreeMap<String, MetricValue> = endpoint
969    ///     .metrics()
970    ///     .iter()
971    ///     .map(|(group, metric)| {
972    ///         let name = [group, metric.name()].join(":");
973    ///         (name, metric.value())
974    ///     })
975    ///     .collect();
976    ///
977    /// assert_eq!(metrics["magicsock:recv_datagrams"], MetricValue::Counter(0));
978    /// # Ok(())
979    /// # }
980    /// ```
981    ///
982    /// The metrics can also be encoded into the OpenMetrics text format, as used by Prometheus.
983    /// To do so, use the [`iroh_metrics::Registry`], add the endpoint metrics to the
984    /// registry with [`Registry::register_all`], and encode the metrics to a string with
985    /// [`encode_openmetrics_to_string`]:
986    /// ```rust
987    /// # use iroh_metrics::{Registry, MetricsSource};

989    /// # async fn wrapper() -> n0_error::Result<()> {
990    /// let endpoint = Endpoint::bind().await?;
991    /// let mut registry = Registry::default();
992    /// registry.register_all(endpoint.metrics());

995    /// assert!(s.contains(r#"magicsock_recv_datagrams_total 0"#));
996    /// # Ok(())
997    /// # }
998    /// ```

1000    /// Through a registry, you can also add labels or prefixes to metrics with
1001    /// [`Registry::sub_registry_with_label`] or [`Registry::sub_registry_with_prefix`].
1002    /// Furthermore, [`iroh_metrics::service`] provides functions to easily start services
1003    /// to serve the metrics with a HTTP server, dump them to a file, or push them

1046    /// [`Registry`]: iroh_metrics::Registry
1047    /// [`Registry::register_all`]: iroh_metrics::Registry::register_all

1089    /// Closes the QUIC endpoint and the magic socket.
1090    ///
1091    /// This will close any remaining open [`Connection`]s with an error code
1092    /// of `0` and an empty reason.  Though it is best practice to close those
1093    /// explicitly before with a custom error code and reason.
1094    ///
1095    /// It will then make a best effort to wait for all close notifications to be
1096    /// acknowledged by the peers, re-transmitting them if needed. This ensures the
1097    /// peers are aware of the closed connections instead of having to wait for a timeout
1098    /// on the connection. Once all connections are closed or timed out, the future
1099    /// finishes.
1100    ///
1101    /// The maximum time-out that this future will wait for depends on QUIC transport
1102    /// configurations of non-drained connections at the time of calling, and their current
1103    /// estimates of round trip time. With default parameters and a conservative estimate
1104    /// of round trip time, this call's future should take 3 seconds to resolve in cases of
1105    /// bad connectivity or failed connections. In the usual case, this call's future should
1106    /// return much more quickly.
1107    ///
1108    /// It is highly recommended you *do* wait for this close call to finish, if possible.
1109    /// Not doing so will make connections that were still open while closing the endpoint
1110    /// time out on the remote end. Thus remote ends will assume connections to have failed
1111    /// even if all application data was transmitted successfully.
1112    ///
1113    /// Note: Someone used to closing TCP sockets might wonder why it is necessary to wait
1114    /// for timeouts when closing QUIC endpoints, while they don't have to do this for TCP
1115    /// sockets. This is due to QUIC and its acknowledgments being implemented in user-land,
1116    /// while TCP sockets usually get closed and drained by the operating system in the
1117    /// kernel during the "Time-Wait" period of the TCP socket.
1118    ///
1119    /// Be aware however that the underlying UDP sockets are only closed once all clones of
1120    /// the the respective [`Endpoint`] are dropped.
1121    pub async fn close(&self) {
1122        if self.is_closed() {
1123            return;
1124        }
1125
1126        tracing::debug!("Connections closed");
1127        self.msock.close().await;
1128    }

1195/// Read a proxy url from the environment, in this order
1196///
1197/// - `HTTP_PROXY`
1198/// - `http_proxy`
1199/// - `HTTPS_PROXY`
1200/// - `https_proxy`
1201fn proxy_url_from_env() -> Option<Url> {
1202    if let Some(url) = std::env::var("HTTP_PROXY")
1203        .ok()
1204        .and_then(|s| s.parse::<Url>().ok())
1205    {
1206        if is_cgi() {
1207            warn!("HTTP_PROXY environment variable ignored in CGI");
1208        } else {
1209            return Some(url);
1210        }
1211    }
1212    if let Some(url) = std::env::var("http_proxy")
1213        .ok()
1214        .and_then(|s| s.parse::<Url>().ok())
1215    {
1216        return Some(url);
1217    }
1218    if let Some(url) = std::env::var("HTTPS_PROXY")
1219        .ok()
1220        .and_then(|s| s.parse::<Url>().ok())
1221    {

2378    /// Tests that initial connection establishment isn't extremely slow compared
2379    /// to subsequent connections.
2380    ///
2381    /// This is a time based test, but uses a very large ratio to reduce flakiness.
2382    /// It also does a number of connections to average out any anomalies.
2383    #[tokio::test]
2384    #[traced_test]
2385    async fn connect_multi_time() -> Result {
2386        let n = 32;
2387
2388        const NOOP_ALPN: &[u8] = b"noop";
2389
2390        #[derive(Debug, Clone)]
2391        struct Noop;
2392
2393        impl ProtocolHandler for Noop {
2394            async fn accept(&self, connection: Connection) -> Result<(), AcceptError> {
2395                connection.closed().await;
2396                Ok(())
2397            }
2398        }
2399
2400        async fn noop_server() -> Result<(Router, EndpointAddr)> {
2401            let endpoint = Endpoint::empty_builder(RelayMode::Disabled)
2402                .bind()
2403                .await
2404                .anyerr()?;
2405            let addr = endpoint.addr();
2406            let router = Router::builder(endpoint).accept(NOOP_ALPN, Noop).spawn();
2407            Ok((router, addr))
2408        }
2409
2410        let routers = stream::iter(0..n)
2411            .map(|_| noop_server())
2412            .buffered_unordered(32)
2413            .collect::<Vec<_>>()
2414            .await
2415            .into_iter()
2416            .collect::<Result<Vec<_>, _>>()
2417            .anyerr()?;
2418
2419        let addrs = routers
2420            .iter()
2421            .map(|(_, addr)| addr.clone())
2422            .collect::<Vec<_>>();
2423        let ids = addrs.iter().map(|addr| addr.id).collect::<Vec<_>>();
2424        let discovery = StaticProvider::from_endpoint_info(addrs);
2425        let endpoint = Endpoint::empty_builder(RelayMode::Disabled)
2426            .discovery(discovery)
2427            .bind()
2428            .await
2429            .anyerr()?;
2430        // wait for the endpoint to be initialized. This should not be needed,
2431        // but we don't want to measure endpoint init time but connection time
2432        // from a fully initialized endpoint.
2433        endpoint.addr();
2434        let t0 = Instant::now();
2435        for id in &ids {
2436            let conn = endpoint.connect(*id, NOOP_ALPN).await?;
2437            conn.close(0u32.into(), b"done");
2438        }
2439        let dt0 = t0.elapsed().as_secs_f64();
2440        let t1 = Instant::now();
2441        for id in &ids {
2442            let conn = endpoint.connect(*id, NOOP_ALPN).await?;
2443            conn.close(0u32.into(), b"done");
2444        }
2445        let dt1 = t1.elapsed().as_secs_f64();
2446
2447        assert!(dt0 / dt1 < 20.0, "First round: {dt0}s, second round {dt1}s");
2448        Ok(())
2449    }
