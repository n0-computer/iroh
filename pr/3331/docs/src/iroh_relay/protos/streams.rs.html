74#[cfg(not(wasm_browser))]
75impl<T: AsyncRead + AsyncWrite + Unpin> Stream for WsBytesFramed<T> {
76    type Item = Result<Bytes, StreamError>;
77
78    fn poll_next(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
79        loop {
80            match ready!(Pin::new(&mut self.io).poll_next(cx)) {
81                None => return Poll::Ready(None),
82                Some(Err(e)) => return Poll::Ready(Some(Err(e))),
83                Some(Ok(msg)) => {
84                    if msg.is_close() {
85                        // Indicate the stream is done when we receive a close message.
86                        // Note: We don't have to poll the stream to completion for it to close gracefully.
87                        return Poll::Ready(None);
88                    }
89                    if msg.is_ping() || msg.is_pong() {
90                        continue; // Responding appropriately to these is done inside of tokio_websockets/browser impls
91                    }
92                    if !msg.is_binary() {
93                        tracing::warn!(
94                            ?msg,
95                            "Got websocket message of unsupported type, skipping."
96                        );
97                        continue;
98                    }
99                    return Poll::Ready(Some(Ok(msg.into_payload().into())));
100                }
101            }
102        }
103    }
104}

106#[cfg(wasm_browser)]
107impl Stream for WsBytesFramed {
108    type Item = Result<Bytes, StreamError>;
109
110    fn poll_next(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
111        loop {
112            match ready!(Pin::new(&mut self.io).poll_next(cx)) {
113                None => return Poll::Ready(None),
114                Some(ws_stream_wasm::WsMessage::Binary(msg)) => {
115                    return Poll::Ready(Some(Ok(msg.into())))
116                }
117                Some(msg) => {
118                    tracing::warn!(?msg, "Got websocket message of unsupported type, skipping.");
119                    continue;
120                }
121            }
122        }
123    }
124}

126#[cfg(not(wasm_browser))]
127impl<T: AsyncRead + AsyncWrite + Unpin> Sink<Bytes> for WsBytesFramed<T> {
128    type Error = StreamError;
129
130    fn start_send(mut self: Pin<&mut Self>, bytes: Bytes) -> Result<(), Self::Error> {
131        let msg = tokio_websockets::Message::binary(tokio_websockets::Payload::from(bytes));
132        Pin::new(&mut self.io).start_send(msg)
133    }
134
135    fn poll_ready(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
136        Pin::new(&mut self.io).poll_ready(cx).map_err(Into::into)
137    }
138
139    fn poll_flush(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
140        Pin::new(&mut self.io).poll_flush(cx).map_err(Into::into)
141    }
142
143    fn poll_close(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
144        Pin::new(&mut self.io).poll_close(cx).map_err(Into::into)
145    }
146}
