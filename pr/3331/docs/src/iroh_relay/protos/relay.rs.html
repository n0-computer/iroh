1//! This module implements the relaying protocol used by the `server` and `client`.
2//!
3//! Protocol flow:
4//!
5//! Login:
6//!  * client connects
7//!  * -> client sends `FrameType::ClientInfo`
8//!
9//!  Steady state:
10//!  * server occasionally sends `FrameType::KeepAlive` (or `FrameType::Ping`)
11//!  * client responds to any `FrameType::Ping` with a `FrameType::Pong`
12//!  * clients sends `FrameType::SendPacket`
13//!  * server then sends `FrameType::RecvPacket` to recipient
14

29/// The maximum size of a packet sent over relay.
30/// (This only includes the data bytes visible to magicsock, not
31/// including its on-wire framing overhead)
32pub const MAX_PACKET_SIZE: usize = 64 * 1024;

54/// ProtocolVersion is bumped whenever there's a wire-incompatible change.
55///  - version 1 (zero on wire): consistent box headers, in use by employee dev nodes a bit
56///  - version 2: received packets have src addrs in FrameType::RecvPacket at beginning.
57///
58/// NOTE: we are technically running a modified version of the protocol.
59/// `FrameType::PeerPresent`, `FrameType::WatchConn`, `FrameType::ClosePeer`, have been removed.
60/// The server will error on that connection if a client sends one of these frames.
61/// We have split with the DERP protocol significantly starting with our relay protocol 3
62/// `FrameType::PeerPresent`, `FrameType::WatchConn`, `FrameType::ClosePeer`, `FrameType::ServerKey`, and `FrameType::ServerInfo` have been removed.
63/// The server will error on that connection if a client sends one of these frames.
64/// This materially affects the handshake protocol, and so relay nodes on version 3 will be unable to communicate
65/// with nodes running earlier protocol versions.
66pub(crate) const PROTOCOL_VERSION: usize = 3;
67
68/// Indicates this IS the client's home node
69const PREFERRED: u8 = 1u8;
70/// Indicates this IS NOT the client's home node
71const NOT_PREFERRED: u8 = 0u8;
72
73/// The one byte frame type at the beginning of the frame
74/// header. The second field is a big-endian u32 describing the
75/// length of the remaining frame (not including the initial 5 bytes)
76#[derive(Debug, PartialEq, Eq, num_enum::IntoPrimitive, num_enum::FromPrimitive, Clone, Copy)]
77#[repr(u8)]
78pub enum FrameType {
79    /// magic + 32b pub key + 24B nonce + bytes
80    ClientInfo = 2,
81    /// 32B dest pub key + packet bytes
82    SendPacket = 4,
83    /// v0/1 packet bytes, v2: 32B src pub key + packet bytes
84    RecvPacket = 5,
85    /// no payload, no-op (to be replaced with ping/pong)
86    KeepAlive = 6,
87    /// 1 byte payload: 0x01 or 0x00 for whether this is client's home node
88    NotePreferred = 7,
89    /// Sent from server to client to signal that a previous sender is no longer connected.
90    ///
91    /// That is, if A sent to B, and then if A disconnects, the server sends `FrameType::PeerGone`
92    /// to B so B can forget that a reverse path exists on that connection to get back to A
93    ///
94    /// 32B pub key of peer that's gone
95    PeerGone = 8,
96    /// Frames 9-11 concern meshing, which we have eliminated from our version of the protocol.
97    /// Messages with these frames will be ignored.
98    /// 8 byte ping payload, to be echoed back in FrameType::Pong
99    Ping = 12,
100    /// 8 byte payload, the contents of ping being replied to
101    Pong = 13,
102    /// Sent from server to client to tell the client if their connection is
103    /// unhealthy somehow.
104    ///
105    /// Currently this is used to indicate that the connection was closed because of authentication issues.
106    Health = 14,
107
108    /// Sent from server to client for the server to declare that it's restarting.
109    /// Payload is two big endian u32 durations in milliseconds: when to reconnect,
110    /// and how long to try total.
111    ///
112    /// Handled on the `[relay::Client]`, but currently never sent on the `[relay::Server]`
113    Restarting = 15,
114    /// Unknown frame type
115    #[num_enum(default)]
116    Unknown = 255,
117}
118
119impl std::fmt::Display for FrameType {
120    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
121        write!(f, "{self:?}")
122    }
123}

181/// Writes complete frame, errors if it is unable to write within the given `timeout`.
182/// Ignores the timeout if `None`
183///
184/// Does not flush.
185#[cfg(feature = "server")]
186pub(crate) async fn write_frame<S: Sink<Frame, Error = std::io::Error> + Unpin>(
187    mut writer: S,
188    frame: Frame,
189    timeout: Option<Duration>,
190) -> Result<(), SendError> {
191    if let Some(duration) = timeout {
192        tokio::time::timeout(duration, writer.send(frame)).await??;
193    } else {
194        writer.send(frame).await?;
195    }
196
197    Ok(())
198}

200/// Writes a `FrameType::ClientInfo`, including the client's [`PublicKey`],
201/// and the client's [`ClientInfo`], sealed using the server's [`PublicKey`].
202///
203/// Flushes after writing.
204#[deprecated = "switch to proper handshake"]
205pub(crate) async fn legacy_send_client_key<S: Sink<Frame, Error = ConnSendError> + Unpin>(
206    mut writer: S,
207    client_secret_key: &SecretKey,
208    client_info: &ClientInfo,
209) -> Result<(), SendError> {
210    let msg = postcard::to_stdvec(client_info)?;
211    let signature = client_secret_key.sign(&msg);
212
213    writer
214        .send(Frame::ClientInfo {
215            client_public_key: client_secret_key.public(),
216            message: msg.into(),
217            signature,
218        })
219        .await?;
220    writer.flush().await?;
221    Ok(())
222}

224/// Reads the `FrameType::ClientInfo` frame from the client (its proof of identity)
225/// upon it's initial connection.
226#[cfg(any(test, feature = "server"))]
227#[deprecated = "switch to proper handshake"]
228pub(crate) async fn legacy_recv_client_key<E, S: Stream<Item = Result<Frame, E>> + Unpin>(
229    stream: S,
230) -> Result<(PublicKey, ClientInfo), E>
231where
232    E: From<RecvError>,
233{
234    // the client is untrusted at this point, limit the input size even smaller than our usual
235    // maximum frame size, and give a timeout
236
237    // TODO: variable recv size: 256 * 1024
238    let buf = tokio::time::timeout(
239        std::time::Duration::from_secs(10),
240        recv_frame(FrameType::ClientInfo, stream),
241    )
242    .await
243    .map_err(RecvError::from)??;
244
245    if let Frame::ClientInfo {
246        client_public_key,
247        message,
248        signature,
249    } = buf
250    {
251        client_public_key
252            .verify(&message, &signature)
253            .map_err(RecvError::from)?;
254
255        let info: ClientInfo = postcard::from_bytes(&message).map_err(RecvError::from)?;
256        Ok((client_public_key, info))
257    } else {
258        Err(UnexpectedFrameSnafu {
259            got: buf.typ(),
260            expected: FrameType::ClientInfo,
261        }
262        .build()
263        .into())
264    }
265}

291/// The frames in the [`RelayCodec`].
292#[derive(Debug, Clone, PartialEq, Eq)]
293pub(crate) enum Frame {
294    ClientInfo {
295        client_public_key: PublicKey,
296        message: Bytes,
297        signature: Signature,
298    },
299    SendPacket {
300        dst_key: PublicKey,
301        packet: Bytes,
302    },
303    RecvPacket {
304        src_key: PublicKey,
305        content: Bytes,
306    },
307    KeepAlive,
308    NotePreferred {
309        preferred: bool,
310    },
311    NodeGone {
312        node_id: PublicKey,
313    },
314    Ping {
315        data: [u8; 8],
316    },
317    Pong {
318        data: [u8; 8],
319    },
320    Health {
321        problem: Bytes,
322    },
323    Restarting {
324        reconnect_in: u32,
325        try_for: u32,
326    },
327}

329impl Frame {
330    pub(crate) fn typ(&self) -> FrameType {
331        match self {
332            Frame::ClientInfo { .. } => FrameType::ClientInfo,
333            Frame::SendPacket { .. } => FrameType::SendPacket,
334            Frame::RecvPacket { .. } => FrameType::RecvPacket,
335            Frame::KeepAlive => FrameType::KeepAlive,
336            Frame::NotePreferred { .. } => FrameType::NotePreferred,
337            Frame::NodeGone { .. } => FrameType::PeerGone,
338            Frame::Ping { .. } => FrameType::Ping,
339            Frame::Pong { .. } => FrameType::Pong,
340            Frame::Health { .. } => FrameType::Health,
341            Frame::Restarting { .. } => FrameType::Restarting,
342        }
343    }
344
345    /// Serialized length (without the frame header)
346    #[cfg(not(wasm_browser))] // Not needed with websocket framing - thus not needed in browsers
347    pub(crate) fn len(&self) -> usize {
348        match self {
349            Frame::ClientInfo {
350                client_public_key: _,
351                message,
352                signature: _,
353            } => MAGIC.len() + PublicKey::LENGTH + message.len() + Signature::BYTE_SIZE,
354            Frame::SendPacket { dst_key: _, packet } => PublicKey::LENGTH + packet.len(),
355            Frame::RecvPacket {
356                src_key: _,
357                content,
358            } => PublicKey::LENGTH + content.len(),
359            Frame::KeepAlive => 0,
360            Frame::NotePreferred { .. } => 1,
361            Frame::NodeGone { .. } => PublicKey::LENGTH,
362            Frame::Ping { .. } => 8,
363            Frame::Pong { .. } => 8,
364            Frame::Health { problem } => problem.len(),
365            Frame::Restarting { .. } => 4 + 4,
366        }
367    }
368
369    /// Serialized length with frame header.
370    #[cfg(feature = "server")]
371    pub(crate) fn len_with_header(&self) -> usize {
372        self.len() + HEADER_LEN
373    }
374
375    /// Tries to decode a frame received over websockets.
376    ///
377    /// Specifically, bytes received from a binary websocket message frame.
378    #[allow(clippy::result_large_err)]
379    pub(crate) fn decode_from_ws_msg(bytes: Bytes, cache: &KeyCache) -> Result<Self, RecvError> {
380        if bytes.is_empty() {
381            return Err(TooSmallSnafu.build());
382        }
383        let typ = FrameType::from(bytes[0]);
384        let frame = Self::from_bytes(typ, bytes.slice(1..), cache)?;
385        Ok(frame)
386    }
387
388    /// Encodes this frame for sending over websockets.
389    ///
390    /// Specifically meant for being put into a binary websocket message frame.
391    pub(crate) fn encode_for_ws_msg(self, dst: &mut impl BufMut) {
392        dst.put_u8(self.typ().into());
393        self.write_to(dst);
394    }
395
396    /// Writes it self to the given buffer.
397    fn write_to(&self, dst: &mut impl BufMut) {
398        match self {
399            Frame::ClientInfo {
400                client_public_key,
401                message,
402                signature,
403            } => {
404                dst.put(MAGIC.as_bytes());
405                dst.put(client_public_key.as_ref());
406                dst.put(&signature.to_bytes()[..]);
407                dst.put(&message[..]);
408            }
409            Frame::SendPacket { dst_key, packet } => {
410                dst.put(dst_key.as_ref());
411                dst.put(packet.as_ref());
412            }
413            Frame::RecvPacket { src_key, content } => {
414                dst.put(src_key.as_ref());
415                dst.put(content.as_ref());
416            }
417            Frame::KeepAlive => {}
418            Frame::NotePreferred { preferred } => {
419                if *preferred {
420                    dst.put_u8(PREFERRED);
421                } else {
422                    dst.put_u8(NOT_PREFERRED);
423                }
424            }
425            Frame::NodeGone { node_id: peer } => {
426                dst.put(peer.as_ref());
427            }
428            Frame::Ping { data } => {
429                dst.put(&data[..]);
430            }
431            Frame::Pong { data } => {
432                dst.put(&data[..]);
433            }
434            Frame::Health { problem } => {
435                dst.put(problem.as_ref());
436            }
437            Frame::Restarting {
438                reconnect_in,
439                try_for,
440            } => {
441                dst.put_u32(*reconnect_in);
442                dst.put_u32(*try_for);
443            }
444        }
445    }

447    #[allow(clippy::result_large_err)]
448    fn from_bytes(
449        frame_type: FrameType,
450        content: Bytes,
451        cache: &KeyCache,
452    ) -> Result<Self, RecvError> {
453        let res = match frame_type {
454            FrameType::ClientInfo => {
455                if content.len() < PublicKey::LENGTH + Signature::BYTE_SIZE + MAGIC.len() {
456                    return Err(InvalidFrameSnafu.build());
457                }
458                if &content[..MAGIC.len()] != MAGIC.as_bytes() {
459                    return Err(InvalidFrameSnafu.build());
460                }
461
462                let start = MAGIC.len();
463                let client_public_key =
464                    cache.key_from_slice(&content[start..start + PublicKey::LENGTH])?;
465                let start = start + PublicKey::LENGTH;
466                let signature =
467                    Signature::from_slice(&content[start..start + Signature::BYTE_SIZE])?;
468                let start = start + Signature::BYTE_SIZE;
469                let message = content.slice(start..);
470                Self::ClientInfo {
471                    client_public_key,
472                    message,
473                    signature,
474                }
475            }
476            FrameType::SendPacket => {
477                if content.len() < PublicKey::LENGTH {
478                    return Err(InvalidFrameSnafu.build());
479                }
480                let frame_len = content.len() - PublicKey::LENGTH;
481                if frame_len > MAX_PACKET_SIZE {
482                    return Err(FrameTooLargeSnafu { frame_len }.build());
483                }
484
485                let dst_key = cache.key_from_slice(&content[..PublicKey::LENGTH])?;
486                let packet = content.slice(PublicKey::LENGTH..);
487                Self::SendPacket { dst_key, packet }
488            }
489            FrameType::RecvPacket => {
490                if content.len() < PublicKey::LENGTH {
491                    return Err(InvalidFrameSnafu.build());
492                }
493
494                let frame_len = content.len() - PublicKey::LENGTH;
495                if frame_len > MAX_PACKET_SIZE {
496                    return Err(FrameTooLargeSnafu { frame_len }.build());
497                }
498
499                let src_key = cache.key_from_slice(&content[..PublicKey::LENGTH])?;
500                let content = content.slice(PublicKey::LENGTH..);
501                Self::RecvPacket { src_key, content }
502            }
503            FrameType::KeepAlive => {
504                if !content.is_empty() {
505                    return Err(InvalidFrameSnafu.build());
506                }
507                Self::KeepAlive
508            }
509            FrameType::NotePreferred => {
510                if content.len() != 1 {
511                    return Err(InvalidFrameSnafu.build());
512                }
513                let preferred = match content[0] {
514                    PREFERRED => true,
515                    NOT_PREFERRED => false,
516                    _ => return Err(InvalidFrameSnafu.build()),
517                };
518                Self::NotePreferred { preferred }
519            }
520            FrameType::PeerGone => {
521                if content.len() != PublicKey::LENGTH {
522                    return Err(InvalidFrameSnafu.build());
523                }
524                let peer = cache.key_from_slice(&content[..32])?;
525                Self::NodeGone { node_id: peer }
526            }
527            FrameType::Ping => {
528                if content.len() != 8 {
529                    return Err(InvalidFrameSnafu.build());
530                }
531                let mut data = [0u8; 8];
532                data.copy_from_slice(&content[..8]);
533                Self::Ping { data }
534            }
535            FrameType::Pong => {
536                if content.len() != 8 {
537                    return Err(InvalidFrameSnafu.build());
538                }
539                let mut data = [0u8; 8];
540                data.copy_from_slice(&content[..8]);
541                Self::Pong { data }
542            }
543            FrameType::Health => Self::Health { problem: content },
544            FrameType::Restarting => {
545                if content.len() != 4 + 4 {
546                    return Err(InvalidFrameSnafu.build());
547                }
548                let reconnect_in = u32::from_be_bytes(
549                    content[..4]
550                        .try_into()
551                        .map_err(|_| InvalidFrameSnafu.build())?,
552                );
553                let try_for = u32::from_be_bytes(
554                    content[4..]
555                        .try_into()
556                        .map_err(|_| InvalidFrameSnafu.build())?,
557                );
558                Self::Restarting {
559                    reconnect_in,
560                    try_for,
561                }
562            }
563            _ => {
564                return Err(InvalidFrameTypeSnafu { frame_type }.build());
565            }
566        };
567        Ok(res)
568    }
569}

577#[cfg(not(wasm_browser))]
578mod framing {
579    use bytes::{Buf, BytesMut};
580    use tokio_util::codec::{Decoder, Encoder};
581
582    use super::*;
583
584    pub(super) const HEADER_LEN: usize = 5;
585
586    impl Decoder for RelayCodec {
587        type Item = Frame;
588        type Error = RecvError;
589
590        fn decode(&mut self, src: &mut BytesMut) -> Result<Option<Self::Item>, Self::Error> {
591            // Need at least 5 bytes
592            if src.len() < HEADER_LEN {
593                return Ok(None);
594            }
595
596            // Can't use the `Buf::get_*` APIs, as that advances the buffer.
597            let Some(frame_type) = src.first().map(|b| FrameType::from(*b)) else {
598                return Ok(None); // Not enough bytes
599            };
600            let Some(frame_len) = src
601                .get(1..5)
602                .and_then(|s| TryInto::<[u8; 4]>::try_into(s).ok())
603                .map(u32::from_be_bytes)
604                .map(|l| l as usize)
605            else {
606                return Ok(None); // Not enough bytes
607            };
608
609            if frame_len > MAX_FRAME_SIZE {
610                return Err(FrameTooLargeSnafu { frame_len }.build());
611            }
612
613            if src.len() < HEADER_LEN + frame_len {
614                // Optimization: prereserve the buffer space
615                src.reserve(HEADER_LEN + frame_len - src.len());
616
617                return Ok(None);
618            }
619
620            // advance the header
621            src.advance(HEADER_LEN);
622
623            let content = src.split_to(frame_len).freeze();
624            let frame = Frame::from_bytes(frame_type, content, &self.cache)?;
625
626            Ok(Some(frame))
627        }
628    }
629
630    impl Encoder<Frame> for RelayCodec {
631        type Error = std::io::Error;
632
633        fn encode(&mut self, frame: Frame, dst: &mut BytesMut) -> Result<(), Self::Error> {
634            let frame_len: usize = frame.len();
635            if frame_len > MAX_FRAME_SIZE {
636                return Err(std::io::Error::new(
637                    std::io::ErrorKind::InvalidData,
638                    format!("Frame of length {} is too large.", frame_len),
639                ));
640            }
641
642            let frame_len_u32 = u32::try_from(frame_len).expect("just checked");
643
644            dst.reserve(HEADER_LEN + frame_len);
645            dst.put_u8(frame.typ().into());
646            dst.put_u32(frame_len_u32);
647            frame.write_to(dst);
648
649            Ok(())
650        }
651    }
652}

654/// Receives the next frame and matches the frame type. If the correct type is found returns the content,
655/// otherwise an error.
656#[cfg(any(test, feature = "server"))]
657pub(crate) async fn recv_frame<E, S: Stream<Item = Result<Frame, E>> + Unpin>(
658    frame_type: FrameType,
659    mut stream: S,
660) -> Result<Frame, E>
661where
662    RecvError: Into<E>,
663{
664    match stream.next().await {
665        Some(Ok(frame)) => {
666            if frame_type != frame.typ() {
667                return Err(UnexpectedFrameSnafu {
668                    got: frame.typ(),
669                    expected: frame_type,
670                }
671                .build()
672                .into());
673            }
674            Ok(frame)
675        }
676        Some(Err(err)) => Err(err),
677        None => Err(RecvError::from(std::io::Error::new(
678            std::io::ErrorKind::UnexpectedEof,
679            "expected frame".to_string(),
680        ))
681        .into()),
682    }

839    fn secret_key() -> impl Strategy<Value = SecretKey> {
840        prop::array::uniform32(any::<u8>()).prop_map(SecretKey::from)
841    }
842
843    fn key() -> impl Strategy<Value = PublicKey> {
844        secret_key().prop_map(|key| key.public())
845    }
846
847    /// Generates random data, up to the maximum packet size minus the given number of bytes
848    fn data(consumed: usize) -> impl Strategy<Value = Bytes> {
849        let len = MAX_PACKET_SIZE - consumed;
850        prop::collection::vec(any::<u8>(), 0..len).prop_map(Bytes::from)
851    }
852
853    /// Generates a random valid frame
854    fn frame() -> impl Strategy<Value = Frame> {
855        let client_info = (secret_key()).prop_map(|secret_key| {
856            let info = ClientInfo {
857                version: PROTOCOL_VERSION,
858            };
859            let msg = postcard::to_stdvec(&info).expect("using default ClientInfo");
860            let signature = secret_key.sign(&msg);
861            Frame::ClientInfo {
862                client_public_key: secret_key.public(),
863                message: msg.into(),
864                signature,
865            }
866        });
867        let send_packet =
868            (key(), data(32)).prop_map(|(dst_key, packet)| Frame::SendPacket { dst_key, packet });
869        let recv_packet =
870            (key(), data(32)).prop_map(|(src_key, content)| Frame::RecvPacket { src_key, content });
871        let keep_alive = Just(Frame::KeepAlive);
872        let note_preferred = any::<bool>().prop_map(|preferred| Frame::NotePreferred { preferred });
873        let peer_gone = key().prop_map(|peer| Frame::NodeGone { node_id: peer });
874        let ping = prop::array::uniform8(any::<u8>()).prop_map(|data| Frame::Ping { data });
875        let pong = prop::array::uniform8(any::<u8>()).prop_map(|data| Frame::Pong { data });
876        let health = data(0).prop_map(|problem| Frame::Health { problem });
877        let restarting =
878            (any::<u32>(), any::<u32>()).prop_map(|(reconnect_in, try_for)| Frame::Restarting {
879                reconnect_in,
880                try_for,
881            });
882        prop_oneof![
883            client_info,
884            send_packet,
885            recv_packet,
886            keep_alive,
887            note_preferred,
888            peer_gone,
889            ping,
890            pong,
891            health,
892            restarting,
893        ]
894    }

896    fn inject_error(buf: &mut BytesMut) {
897        fn is_fixed_size(tpe: FrameType) -> bool {
898            match tpe {
899                FrameType::KeepAlive
900                | FrameType::NotePreferred
901                | FrameType::Ping
902                | FrameType::Pong
903                | FrameType::Restarting
904                | FrameType::PeerGone => true,
905                FrameType::ClientInfo
906                | FrameType::Health
907                | FrameType::SendPacket
908                | FrameType::RecvPacket
909                | FrameType::Unknown => false,
910            }
911        }
912        let tpe: FrameType = buf[0].into();
913        let mut len = u32::from_be_bytes(buf[1..5].try_into().unwrap()) as usize;
914        if is_fixed_size(tpe) {
915            buf.put_u8(0);
916            len += 1;
917        } else {
918            buf.resize(MAX_FRAME_SIZE + 1, 0);
919            len = MAX_FRAME_SIZE + 1;
920        }
921        buf[1..5].copy_from_slice(&u32::to_be_bytes(len as u32));
922    }
