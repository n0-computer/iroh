91/// A connection to a relay server.
92///
93/// This holds a connection to a relay server.  It is:
94///
95/// - A [`Stream`] for [`ReceivedMessage`] to receive from the server.
96/// - A [`Sink`] for [`SendMessage`] to send to the server.
97/// - A [`Sink`] for [`Frame`] to send to the server.
98///
99/// The [`Frame`] sink is a more internal interface, it allows performing the handshake.
100/// The [`SendMessage`] and [`ReceivedMessage`] are safer wrappers enforcing some protocol
101/// invariants.
102#[derive(derive_more::Debug)]
103pub(crate) enum Conn {
104    #[cfg(not(wasm_browser))]
105    Relay {
106        #[debug("Framed<MaybeTlsStreamChained, RelayCodec>")]
107        conn: Framed<MaybeTlsStreamChained, RelayCodec>,
108    },
109    #[cfg(not(wasm_browser))]
110    Ws {
111        #[debug("WebSocketStream<MaybeTlsStream<ProxyStream>>")]
112        conn: tokio_websockets::WebSocketStream<MaybeTlsStream<ProxyStream>>,
113        key_cache: KeyCache,
114    },
115    #[cfg(wasm_browser)]
116    WsBrowser {
117        #[debug("WebSocketStream")]
118        conn: ws_stream_wasm::WsStream,
119        key_cache: KeyCache,
120    },
121}

123impl Conn {
124    /// Constructs a new websocket connection, including the initial server handshake.
125    #[cfg(wasm_browser)]
126    pub(crate) async fn new_ws_browser(
127        conn: ws_stream_wasm::WsStream,
128        key_cache: KeyCache,
129        secret_key: &SecretKey,
130    ) -> Result<Self, handshake::Error> {
131        let mut io = HandshakeIo { io: conn };
132
133        // exchange information with the server
134        debug!("server_handshake: started");
135        handshake::clientside(&mut io, secret_key).await?;
136        debug!("server_handshake: done");
137
138        Ok(Self::WsBrowser {
139            conn: io.io,
140            key_cache,
141        })
142    }
143
144    /// Constructs a new websocket connection, including the initial server handshake.
145    #[cfg(not(wasm_browser))]
146    pub(crate) async fn new_relay(
147        conn: MaybeTlsStreamChained,
148        key_cache: KeyCache,
149        secret_key: &SecretKey,
150    ) -> Result<Self, handshake::Error> {
151        use n0_future::SinkExt;
152
153        let conn = Framed::new(conn, RelayCodec::new(key_cache));
154
155        let mut conn = conn.sink_err_into();
156
157        // exchange information with the server
158        debug!("server_handshake: started");
159        let client_info = ClientInfo {
160            version: PROTOCOL_VERSION,
161        };
162        debug!("server_handshake: sending client_key: {:?}", &client_info);
163        #[allow(deprecated)]
164        crate::protos::relay::legacy_send_client_key(&mut conn, secret_key, &client_info).await?;
165        debug!("server_handshake: done");
166
167        Ok(Self::Relay {
168            conn: conn.into_inner(),
169        })
170    }
171
172    #[cfg(not(wasm_browser))]
173    pub(crate) async fn new_ws(
174        conn: tokio_websockets::WebSocketStream<MaybeTlsStream<ProxyStream>>,
175        key_cache: KeyCache,
176        secret_key: &SecretKey,
177    ) -> Result<Self, handshake::Error> {
178        let mut io = HandshakeIo { io: conn };
179
180        // exchange information with the server
181        debug!("server_handshake: started");
182        handshake::clientside(&mut io, secret_key).await?;
183        debug!("server_handshake: done");
184
185        Ok(Self::Ws {
186            conn: io.io,
187            key_cache,
188        })
189    }
190}

192impl Stream for Conn {
193    type Item = Result<ReceivedMessage, RecvError>;
194
195    fn poll_next(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
196        match *self {
197            #[cfg(not(wasm_browser))]
198            Self::Relay { ref mut conn } => match ready!(Pin::new(conn).poll_next(cx)) {
199                Some(Ok(frame)) => {
200                    let message = ReceivedMessage::try_from(frame);
201                    Poll::Ready(Some(message))
202                }
203                Some(Err(err)) => Poll::Ready(Some(Err(err.into()))),
204                None => Poll::Ready(None),
205            },
206            #[cfg(not(wasm_browser))]
207            Self::Ws {
208                ref mut conn,
209                ref key_cache,
210            } => match ready!(Pin::new(conn).poll_next(cx)) {
211                Some(Ok(msg)) => {
212                    if msg.is_close() {
213                        // Indicate the stream is done when we receive a close message.
214                        // Note: We don't have to poll the stream to completion for it to close gracefully.
215                        return Poll::Ready(None);
216                    }
217                    if !msg.is_binary() {
218                        tracing::warn!(
219                            ?msg,
220                            "Got websocket message of unsupported type, skipping."
221                        );
222                        return Poll::Pending;
223                    }
224                    let frame = Frame::decode_from_ws_msg(msg.into_payload().into(), key_cache)?;
225                    let message = ReceivedMessage::try_from(frame);
226                    Poll::Ready(Some(message))
227                }
228                Some(Err(e)) => Poll::Ready(Some(Err(e.into()))),
229                None => Poll::Ready(None),
230            },
231            #[cfg(wasm_browser)]
232            Self::WsBrowser {
233                ref mut conn,
234                ref key_cache,
235            } => match ready!(Pin::new(conn).poll_next(cx)) {
236                Some(ws_stream_wasm::WsMessage::Binary(vec)) => {
237                    let frame = Frame::decode_from_ws_msg(Bytes::from(vec), key_cache)?;
238                    Poll::Ready(Some(ReceivedMessage::try_from(frame)))
239                }
240                Some(msg) => {
241                    tracing::warn!(?msg, "Got websocket message of unsupported type, skipping.");
242                    Poll::Pending
243                }
244                None => Poll::Ready(None),
245            },
246        }
247    }
248}

250// TODO(matheus23): Remove this impl, make `new_relay` work on the `Framed` directly, make the impl not rely on `ConnSendError`.
251impl Sink<Frame> for Conn {
252    type Error = SendError;
253
254    fn poll_ready(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
255        match *self {
256            #[cfg(not(wasm_browser))]
257            Self::Relay { ref mut conn } => Pin::new(conn).poll_ready(cx).map_err(Into::into),
258            #[cfg(not(wasm_browser))]
259            Self::Ws { ref mut conn, .. } => Pin::new(conn).poll_ready(cx).map_err(Into::into),
260            #[cfg(wasm_browser)]
261            Self::WsBrowser { ref mut conn, .. } => {
262                Pin::new(conn).poll_ready(cx).map_err(Into::into)
263            }
264        }
265    }
266
267    fn start_send(mut self: Pin<&mut Self>, frame: Frame) -> Result<(), Self::Error> {
268        if let Frame::SendPacket { dst_key: _, packet } = &frame {
269            if packet.len() > MAX_PACKET_SIZE {
270                return Err(ExceedsMaxPacketSizeSnafu { size: packet.len() }.build());
271            }
272        }
273        match *self {
274            #[cfg(not(wasm_browser))]
275            Self::Relay { ref mut conn } => Pin::new(conn).start_send(frame).map_err(Into::into),
276            #[cfg(not(wasm_browser))]
277            Self::Ws { ref mut conn, .. } => Pin::new(conn)
278                .start_send(tokio_websockets::Message::binary({
279                    let mut buf = BytesMut::new();
280                    frame.encode_for_ws_msg(&mut buf);
281                    tokio_websockets::Payload::from(buf.freeze())
282                }))
283                .map_err(Into::into),
284            #[cfg(wasm_browser)]
285            Self::WsBrowser { ref mut conn, .. } => Pin::new(conn)
286                .start_send(ws_stream_wasm::WsMessage::Binary(frame.encode_for_ws_msg()))
287                .map_err(Into::into),
288        }
289    }
290
291    fn poll_flush(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
292        match *self {
293            #[cfg(not(wasm_browser))]
294            Self::Relay { ref mut conn } => Pin::new(conn).poll_flush(cx).map_err(Into::into),
295            #[cfg(not(wasm_browser))]
296            Self::Ws { ref mut conn, .. } => Pin::new(conn).poll_flush(cx).map_err(Into::into),
297            #[cfg(wasm_browser)]
298            Self::WsBrowser { ref mut conn, .. } => {
299                Pin::new(conn).poll_flush(cx).map_err(Into::into)
300            }
301        }
302    }
303
304    fn poll_close(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
305        match *self {
306            #[cfg(not(wasm_browser))]
307            Self::Relay { ref mut conn } => Pin::new(conn).poll_close(cx).map_err(Into::into),
308            #[cfg(not(wasm_browser))]
309            Self::Ws { ref mut conn, .. } => Pin::new(conn).poll_flush(cx).map_err(Into::into),
310            #[cfg(wasm_browser)]
311            Self::WsBrowser { ref mut conn, .. } => {
312                Pin::new(conn).poll_close(cx).map_err(Into::into)
313            }
314        }
315    }

342            Self::Relay { ref mut conn } => Pin::new(conn).start_send(frame).map_err(Into::into),

385/// The messages received from a framed relay stream.
386///
387/// This is a type-validated version of the `Frame`s on the `RelayCodec`.
388#[derive(derive_more::Debug, Clone)]
389pub enum ReceivedMessage {
390    /// Represents an incoming packet.
391    ReceivedPacket {
392        /// The [`NodeId`] of the packet sender.
393        remote_node_id: NodeId,
394        /// The received packet bytes.
395        #[debug(skip)]
396        data: Bytes, // TODO: ref
397    },
398    /// Indicates that the client identified by the underlying public key had previously sent you a
399    /// packet but has now disconnected from the server.
400    NodeGone(NodeId),
401    /// Request from a client or server to reply to the
402    /// other side with a [`ReceivedMessage::Pong`] with the given payload.
403    Ping([u8; 8]),
404    /// Reply to a [`ReceivedMessage::Ping`] from a client or server
405    /// with the payload sent previously in the ping.
406    Pong([u8; 8]),
407    /// A one-way empty message from server to client, just to
408    /// keep the connection alive. It's like a [`ReceivedMessage::Ping`], but doesn't solicit
409    /// a reply from the client.
410    KeepAlive,
411    /// A one-way message from server to client, declaring the connection health state.
412    Health {
413        /// If set, is a description of why the connection is unhealthy.
414        ///
415        /// If `None` means the connection is healthy again.
416        ///
417        /// The default condition is healthy, so the server doesn't broadcast a [`ReceivedMessage::Health`]
418        /// until a problem exists.
419        problem: Option<String>,
420    },
421    /// A one-way message from server to client, advertising that the server is restarting.
422    ServerRestarting {
423        /// An advisory duration that the client should wait before attempting to reconnect.
424        /// It might be zero. It exists for the server to smear out the reconnects.
425        reconnect_in: Duration,
426        /// An advisory duration for how long the client should attempt to reconnect
427        /// before giving up and proceeding with its normal connection failure logic. The interval
428        /// between retries is undefined for now. A server should not send a TryFor duration more
429        /// than a few seconds.
430        try_for: Duration,
431    },
432}
433
434impl TryFrom<Frame> for ReceivedMessage {
435    type Error = RecvError;
436
437    fn try_from(frame: Frame) -> std::result::Result<Self, Self::Error> {
438        match frame {
439            Frame::KeepAlive => {
440                // A one-way keep-alive message that doesn't require an ack.
441                // This predated FrameType::Ping/FrameType::Pong.
442                Ok(ReceivedMessage::KeepAlive)
443            }
444            Frame::NodeGone { node_id } => Ok(ReceivedMessage::NodeGone(node_id)),
445            Frame::RecvPacket { src_key, content } => {
446                let packet = ReceivedMessage::ReceivedPacket {
447                    remote_node_id: src_key,
448                    data: content,
449                };
450                Ok(packet)
451            }
452            Frame::Ping { data } => Ok(ReceivedMessage::Ping(data)),
453            Frame::Pong { data } => Ok(ReceivedMessage::Pong(data)),
454            Frame::Health { problem } => {
455                let problem = std::str::from_utf8(&problem)
456                    .context(InvalidProtocolMessageEncodingSnafu)?
457                    .to_owned();
458                let problem = Some(problem);
459                Ok(ReceivedMessage::Health { problem })
460            }
461            Frame::Restarting {
462                reconnect_in,
463                try_for,
464            } => {
465                let reconnect_in = Duration::from_millis(reconnect_in as u64);
466                let try_for = Duration::from_millis(try_for as u64);
467                Ok(ReceivedMessage::ServerRestarting {
468                    reconnect_in,
469                    try_for,
470                })
471            }
472            _ => Err(UnexpectedFrameSnafu {
473                frame_type: frame.typ(),
474            }
475            .build()),
476        }
477    }
478}

480/// Messages we can send to a relay server.
481#[derive(Debug)]
482pub enum SendMessage {
483    /// Send a packet of data to the [`NodeId`].
484    SendPacket(NodeId, Bytes),
485    /// Sends a ping message to the connected relay server.
486    Ping([u8; 8]),
487    /// Sends a pong message to the connected relay server.
488    Pong([u8; 8]),
489}
490
491impl From<SendMessage> for Frame {
492    fn from(source: SendMessage) -> Self {
493        match source {
494            SendMessage::SendPacket(dst_key, packet) => Frame::SendPacket { dst_key, packet },
495            SendMessage::Ping(data) => Frame::Ping { data },
496            SendMessage::Pong(data) => Frame::Pong { data },
497        }
498    }
