20/// A Stream and Sink for [`Frame`]s connected to a single relay client.
21///
22/// The stream receives message from the client while the sink sends them to the client.
23#[derive(Debug)]
24pub(crate) enum RelayedStream {
25    Relay(Framed<MaybeTlsStream, RelayCodec>),
26    Ws(WebSocketStream<MaybeTlsStream>, KeyCache),
27}
28
29fn ws_to_io_err(e: tokio_websockets::Error) -> std::io::Error {
30    match e {
31        tokio_websockets::Error::Io(io_err) => io_err,
32        _ => std::io::Error::other(e.to_string()),
33    }
34}
35
36impl Sink<Frame> for RelayedStream {
37    type Error = std::io::Error;
38
39    fn poll_ready(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
40        match *self {
41            Self::Relay(ref mut framed) => Pin::new(framed).poll_ready(cx),
42            Self::Ws(ref mut ws, _) => Pin::new(ws).poll_ready(cx).map_err(ws_to_io_err),
43        }
44    }
45
46    fn start_send(mut self: Pin<&mut Self>, item: Frame) -> Result<(), Self::Error> {
47        match *self {
48            Self::Relay(ref mut framed) => Pin::new(framed).start_send(item),
49            Self::Ws(ref mut ws, _) => Pin::new(ws)
50                .start_send(tokio_websockets::Message::binary({
51                    let mut buf = BytesMut::new();
52                    item.encode_for_ws_msg(&mut buf);
53                    tokio_websockets::Payload::from(buf.freeze())
54                }))
55                .map_err(ws_to_io_err),
56        }
57    }
58
59    fn poll_flush(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
60        match *self {
61            Self::Relay(ref mut framed) => Pin::new(framed).poll_flush(cx),
62            Self::Ws(ref mut ws, _) => Pin::new(ws).poll_flush(cx).map_err(ws_to_io_err),
63        }
64    }
65
66    fn poll_close(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Result<(), Self::Error>> {
67        match *self {
68            Self::Relay(ref mut framed) => Pin::new(framed).poll_close(cx),
69            Self::Ws(ref mut ws, _) => Pin::new(ws).poll_close(cx).map_err(ws_to_io_err),
70        }
71    }
72}

74/// Relay stream errors
75#[derive(Debug, Snafu)]
76#[non_exhaustive]
77pub enum StreamError {
78    #[snafu(transparent)]
79    Proto { source: RecvError },
80    #[snafu(transparent)]
81    Ws { source: tokio_websockets::Error },
82}
83
84impl Stream for RelayedStream {
85    type Item = Result<Frame, StreamError>;
86
87    fn poll_next(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
88        match *self {
89            Self::Relay(ref mut framed) => Pin::new(framed).poll_next(cx).map_err(Into::into),
90            Self::Ws(ref mut ws, ref cache) => match Pin::new(ws).poll_next(cx) {
91                Poll::Ready(Some(Ok(msg))) => {
92                    if msg.is_close() {
93                        // Indicate the stream is done when we receive a close message.
94                        // Note: We don't have to poll the stream to completion for it to close gracefully.
95                        return Poll::Ready(None);
96                    }
97                    if !msg.is_binary() {
98                        tracing::warn!(
99                            ?msg,
100                            "Got websocket message of unsupported type, skipping."
101                        );
102                        return Poll::Pending;
103                    }
104                    let frame = Frame::decode_from_ws_msg(msg.into_payload().into(), cache)
105                        .map_err(Into::into);
106                    Poll::Ready(Some(frame))
107                }
108                Poll::Ready(Some(Err(e))) => Poll::Ready(Some(Err(e.into()))),
109                Poll::Ready(None) => Poll::Ready(None),
110                Poll::Pending => Poll::Pending,
111            },
112        }
113    }
114}

116/// The main underlying IO stream type used for the relay server.
117///
118/// Allows choosing whether or not the underlying [`tokio::net::TcpStream`] is served over Tls
119#[derive(Debug)]
120#[allow(clippy::large_enum_variant)]
121pub enum MaybeTlsStream {
122    /// A plain non-Tls [`tokio::net::TcpStream`]
123    Plain(tokio::net::TcpStream),
124    /// A Tls wrapped [`tokio::net::TcpStream`]
125    Tls(tokio_rustls::server::TlsStream<tokio::net::TcpStream>),
126    /// An in-memory bidirectional pipe.
127    #[cfg(test)]
128    Test(tokio::io::DuplexStream),
129}
130
131impl ExportKeyingMaterial for MaybeTlsStream {
132    fn export_keying_material<T: AsMut<[u8]>>(
133        &self,
134        output: T,
135        label: &[u8],
136        context: Option<&[u8]>,
137    ) -> Option<T> {
138        let Self::Tls(ref tls) = self else {
139            return None;
140        };
141
142        tls.get_ref()
143            .1
144            .export_keying_material(output, label, context)
145            .ok()
146    }
147}

149impl AsyncRead for MaybeTlsStream {
150    fn poll_read(
151        mut self: Pin<&mut Self>,
152        cx: &mut Context<'_>,
153        buf: &mut tokio::io::ReadBuf<'_>,
154    ) -> Poll<std::io::Result<()>> {
155        match &mut *self {
156            MaybeTlsStream::Plain(ref mut s) => Pin::new(s).poll_read(cx, buf),
157            MaybeTlsStream::Tls(ref mut s) => Pin::new(s).poll_read(cx, buf),
158            #[cfg(test)]
159            MaybeTlsStream::Test(ref mut s) => Pin::new(s).poll_read(cx, buf),
160        }
161    }
162}
