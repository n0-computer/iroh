29/// GET handler for resolving DoH queries
30pub async fn get(
31    State(state): State<AppState>,
32    DnsRequestQuery(request, accept_type): DnsRequestQuery,
33) -> AppResult<Response> {
34    let message_bytes = state.dns_handler.answer_request(request).await?;
35    let message = proto::op::Message::from_bytes(&message_bytes).e()?;
36
37    let min_ttl = message.answers().iter().map(|rec| rec.ttl()).min();
38
39    let mut response = match accept_type {
40        DnsMimeType::Message => (StatusCode::OK, message_bytes).into_response(),
41        DnsMimeType::Json => {
42            let response = self::response::DnsResponse::from_message(message)?;
43            (StatusCode::OK, Json(response)).into_response()
44        }
45    };
46
47    response
48        .headers_mut()
49        .insert(CONTENT_TYPE, accept_type.to_header_value());
50
51    if let Some(min_ttl) = min_ttl {
52        let maxage = HeaderValue::from_str(&format!("s-maxage={min_ttl}")).e()?;
53        response.headers_mut().insert(CACHE_CONTROL, maxage);
54    }
55
56    Ok(response)
57}

59/// POST handler for resolvng DoH queries
60pub async fn post(
61    State(state): State<AppState>,
62    DnsRequestBody(request): DnsRequestBody,
63) -> Response {
64    let response = match state.dns_handler.answer_request(request).await {
65        Ok(response) => response,
66        Err(err) => return (StatusCode::INTERNAL_SERVER_ERROR, err.to_string()).into_response(),
67    };
68
69    (
70        StatusCode::OK,
71        [(CONTENT_TYPE, DnsMimeType::Message.to_string())],
72        response,
73    )
74        .into_response()
75}
