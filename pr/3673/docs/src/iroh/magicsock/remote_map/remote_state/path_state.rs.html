10/// The state of a single path to the remote endpoint.
11///
12/// Each path is identified by the destination [`transports::Addr`] and they are stored in
13/// the [`RemoteStateActor::paths`] map.
14///
15/// [`transports::Addr`]: super::transports::Addr
16/// [`RemoteStateActor::paths`]: super::remote_state::RemoteStateActor
17#[derive(Debug, Default)]
18pub(super) struct PathState {
19    /// How we learned about this path, and when.
20    ///
21    /// We keep track of only the latest [`Instant`] for each [`Source`], keeping the size
22    /// of the map of sources down to one entry per type of source.
23    pub(super) sources: HashMap<Source, Instant>,
24    /// The last ping sent on this path.
25    pub(super) ping_sent: Option<TransactionId>,
26}
