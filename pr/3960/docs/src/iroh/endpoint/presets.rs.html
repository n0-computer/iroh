1//! Presets allow configuring an endpoint quickly with a chosen set of defaults.
2//!
3//! # Example
4//!
5//! ```no_run
6//! # async fn wrapper() -> n0_error::Result {
7//! use iroh::{Endpoint, RelayMode, Watcher, endpoint::presets};
8//!
9//! let endpoint = Endpoint::empty_builder(RelayMode::Disabled)
10//!     .preset(presets::N0)
11//!     .bind()
12//!     .await?;
13//! # let _ = endpoint;
14//! # Ok(())
15//! # }
16//! ```
17
18use crate::{
19    RelayMode,
20    address_lookup::{AddrFilter, IntoAddressLookup, PkarrPublisher},
21    endpoint::{Builder, default_relay_mode},
22};
23
24/// Defines a preset
25pub trait Preset {
26    /// Applies the configuration to the passed in [`Builder`].
27    fn apply(self, builder: Builder) -> Builder;
28}

30/// Configures the endpoint to use the n0 defaults
31///
32/// Currently this consists of
33/// - the DNS Address Lookup service.
34/// - the default relay servers provided by Number 0.
35///
36/// The default address lookup service publishes to and resolves from the
37/// n0.computer dns server `iroh.link`.
38///
39/// This is equivalent to adding both a [`crate::address_lookup::PkarrPublisher`]
40/// and a [`crate::address_lookup::DnsAddressLookup`], both configured to use the
41/// n0.computer dns server.
42///
43/// This will by default use [`N0_DNS_PKARR_RELAY_PROD`].
44/// When in tests, or when the `test-utils` feature is enabled, this will use the
45/// [`N0_DNS_PKARR_RELAY_STAGING`].
46///
47/// This preset includes an [`AddrFilter`] for the address lookup services that only publishes
48/// the endpoint's relay address, by default.
49///
50/// [`N0_DNS_PKARR_RELAY_PROD`]: crate::address_lookup::N0_DNS_PKARR_RELAY_PROD
51/// [`N0_DNS_PKARR_RELAY_STAGING`]: crate::address_lookup::N0_DNS_PKARR_RELAY_STAGING
52#[derive(Debug, Copy, Clone, Default)]
53pub struct N0;
54
55impl Preset for N0 {
56    fn apply(self, mut builder: Builder) -> Builder {
57        let filter = AddrFilter::new(|addrs| {
58            addrs
59                .into_iter()
60                .filter(|addr| matches!(addr, iroh_base::TransportAddr::Relay(_)))
61                .collect()
62        });
63        builder = builder.address_lookup(PkarrPublisher::n0_dns().with_addr_filter(filter.clone()));
64
65        // Resolve using HTTPS requests to our DNS server's /pkarr path in browsers
66        #[cfg(wasm_browser)]
67        {
68            use crate::address_lookup::PkarrResolver;
69
70            builder =
71                builder.address_lookup(PkarrResolver::n0_dns().with_addr_filter(filter.clone()));
72        }
73        // Resolve using DNS queries outside browsers.
74        #[cfg(not(wasm_browser))]
75        {
76            builder = builder.address_lookup(
77                crate::address_lookup::DnsAddressLookup::n0_dns().with_addr_filter(filter.clone()),
78            );
79        }
80
81        builder = builder.relay_mode(default_relay_mode());
82
83        builder
84    }
85}

87/// Configures the endpoint to use the n0 defaults, but with relay mode disabled.
88///
89/// Currently this consists of
90/// - setting `RelayMode::Disabled`
91/// - the DNS Address Lookup service, that publishes IP addresses rather than
92///   relay urls.
93///
94/// The default address lookup service publishes to and resolves from the
95/// n0.computer dns server `iroh.link`.
96///
97/// This is equivalent to adding both a [`crate::address_lookup::PkarrPublisher`]
98/// and a [`crate::address_lookup::DnsAddressLookup`], both configured to use the
99/// n0.computer dns server.

101/// This will by default use [`N0_DNS_PKARR_RELAY_PROD`].
102/// When in tests, or when the `test-utils` feature is enabled, this will use the
103/// [`N0_DNS_PKARR_RELAY_STAGING`].
104///
105/// This preset includes an [`AddrFilter`] for the address lookup services that only publishes
106/// the endpoint's IP addresses.
107///
108/// [`N0_DNS_PKARR_RELAY_PROD`]: crate::address_lookup::N0_DNS_PKARR_RELAY_PROD
109/// [`N0_DNS_PKARR_RELAY_STAGING`]: crate::address_lookup::N0_DNS_PKARR_RELAY_STAGING
110#[derive(Debug, Copy, Clone, Default)]
111pub struct N0DisableRelay;
