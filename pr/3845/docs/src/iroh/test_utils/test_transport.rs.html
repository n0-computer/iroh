43/// A simulated network for testing custom transports.
44///
45/// This allows creating multiple [`TestTransport`] instances that can communicate
46/// with each other through in-memory channels.
47///
48/// # Example
49///
50/// ```ignore
51/// use iroh::test_utils::custom_transport::TestNetwork;
52///
53/// let network = TestNetwork::new();
54/// let transport1 = network.create_transport(endpoint_id1)?;
55/// let transport2 = network.create_transport(endpoint_id2)?;
56/// // transport1 and transport2 can now communicate via the network
57/// ```
58#[derive(Debug, Clone, Default)]
59pub struct TestNetwork {
60    inner: Arc<Mutex<TestNetworkInner>>,
61}
62
63impl TestNetwork {
64    /// Creates a new empty test network.
65    pub fn new() -> Self {
66        Self::default()
67    }
68
69    /// Creates a discovery service for this network.
70    pub fn discovery(&self) -> impl AddressLookup {
71        TestAddrLookup {
72            network: self.clone(),
73        }
74    }
75
76    /// Creates a new test transport for the given endpoint ID.
77    ///
78    /// Returns an error if the ID already exists in the network.
79    pub fn create_transport(&self, id: EndpointId) -> io::Result<Arc<TestTransport>> {
80        let id_custom = to_custom_addr(id);
81        let mut guard = self.inner.lock().expect("poisoned");
82        if guard.channels.contains_key(&id) {
83            return Err(io::Error::other("endpoint ID already exists in network"));
84        }
85        guard.channels.insert(id, mpsc::channel(256));
86        drop(guard);
87        Ok(Arc::new(TestTransport {
88            id_watchable: n0_watcher::Watchable::new(vec![id_custom]),
89            network: self.clone(),
90            id,
91        }))
92    }
93}

105impl AddressLookup for TestAddrLookup {
106    fn publish(&self, _data: &EndpointData) {}
107
108    fn resolve(
109        &self,
110        endpoint_id: EndpointId,
111    ) -> Option<n0_future::stream::Boxed<Result<Item, crate::address_lookup::Error>>> {
112        if self
113            .network
114            .inner
115            .lock()
116            .expect("poisoned")
117            .channels
118            .contains_key(&endpoint_id)
119        {
120            Some(Box::pin(n0_future::stream::once(Ok(Item::new(
121                EndpointInfo {
122                    endpoint_id,
123                    data: EndpointData::new([TransportAddr::Custom(CustomAddr::from_parts(
124                        TEST_TRANSPORT_ID,
125                        endpoint_id.as_bytes(),
126                    ))]),
127                },
128                "test discovery",
129                None,
130            )))))
131        } else {
132            None
133        }
134    }
135}

143/// Converts an endpoint ID to a custom address for this test transport.
144pub fn to_custom_addr(endpoint: EndpointId) -> CustomAddr {
145    CustomAddr::from((TEST_TRANSPORT_ID, &endpoint.as_bytes()[..]))
146}
147
148fn try_parse_custom_addr(addr: &CustomAddr) -> io::Result<EndpointId> {
149    if addr.id() != TEST_TRANSPORT_ID {
150        return Err(io::Error::other("unexpected transport id"));
151    }
152    let key_bytes: &[u8; 32] = addr
153        .data()
154        .try_into()
155        .map_err(|_| io::Error::other("wrong key length"))?;
156    EndpointId::from_bytes(key_bytes).map_err(|_| io::Error::other("KeyParseError"))
157}
158
159impl TestSender {
160    fn send_sync(&self, dst: &CustomAddr, packets: Vec<Packet>) -> io::Result<()> {
161        let to_id = try_parse_custom_addr(dst)?;
162        let guard = self.network.inner.lock().expect("poisoned");
163        let (s, _) = guard
164            .channels
165            .get(&to_id)
166            .ok_or_else(|| io::Error::other("Unknown endpoint"))?;
167        for packet in packets {
168            let len = packet.data.len();
169            match s.try_send(packet) {
170                Ok(_) => info!(
171                    "send {} -> {}: sent {} bytes",
172                    self.id.fmt_short(),
173                    to_id.fmt_short(),
174                    len
175                ),
176                Err(TrySendError::Full(_)) => info!(
177                    "send {} -> {}: dropped {} bytes",
178                    self.id.fmt_short(),
179                    to_id.fmt_short(),
180                    len
181                ),
182                Err(TrySendError::Closed(_)) => return Err(io::Error::other("channel closed")),
183            }
184        }
185        Ok(())
186    }
187
188    fn split(&self, transmit: &Transmit) -> impl Iterator<Item = Packet> {
189        let from = to_custom_addr(self.id);
190        let segment_size = transmit.segment_size.unwrap_or(transmit.contents.len());
191        transmit
192            .contents
193            .chunks(segment_size)
194            .map(move |slice| Packet {
195                from: from.clone(),
196                data: Bytes::copy_from_slice(slice),
197            })
198    }
199}
200
201impl CustomSender for TestSender {
202    fn is_valid_send_addr(&self, addr: &CustomAddr) -> bool {
203        addr.id() == TEST_TRANSPORT_ID
204    }
205
206    fn poll_send(
207        &self,
208        _cx: &mut std::task::Context,
209        dst: &CustomAddr,
210        transmit: &Transmit<'_>,
211    ) -> Poll<io::Result<()>> {
212        let packets = self.split(transmit).collect();
213        Poll::Ready(self.send_sync(dst, packets))
214    }
215}

223impl CustomEndpoint for TestTransport {

228    fn create_sender(&self) -> Arc<dyn CustomSender> {
229        Arc::new(TestSender {
230            id: self.id,
231            network: self.network.clone(),
232        })
233    }
234
235    fn poll_recv(
236        &mut self,
237        cx: &mut std::task::Context,
238        bufs: &mut [io::IoSliceMut<'_>],
239        metas: &mut [quinn_udp::RecvMeta],
240        source_addrs: &mut [Addr],
241    ) -> Poll<io::Result<usize>> {
242        let n = bufs.len();
243        debug_assert_eq!(n, metas.len());
244        debug_assert_eq!(n, source_addrs.len());
245        if n == 0 {
246            return Poll::Ready(Ok(0));
247        }
248        let mut guard = self.network.inner.lock().expect("poisoned");
249        let Some((_, r)) = guard.channels.get_mut(&self.id) else {
250            info!("me: {} not found in channels", self.id.fmt_short());
251            return Poll::Ready(Ok(0));
252        };
253        let mut packets = Vec::new();
254        match r.poll_recv_many(cx, &mut packets, n) {
255            Poll::Pending => return Poll::Pending,
256            Poll::Ready(0) => return Poll::Ready(Err(io::Error::other("channel closed"))),
257            Poll::Ready(n) => n,
258        };
259        let mut count = 0;
260        for (((packet, meta), buf), source_addr) in
261            packets.into_iter().zip(metas).zip(bufs).zip(source_addrs)
262        {
263            if buf.len() < packet.data.len() {
264                break;
265            }
266            let from = try_parse_custom_addr(&packet.from).expect("valid custom addr");
267            info!(
268                "recv {} -> {}: copying {} bytes",
269                from.fmt_short(),
270                self.id.fmt_short(),
271                packet.data.len()
272            );
273            buf[..packet.data.len()].copy_from_slice(&packet.data);
274            *source_addr = packet.from.into();
275            meta.len = packet.data.len();
276            meta.stride = packet.data.len();
277            count += 1;
278        }
279        if count > 0 {
280            info!("recv {}: filled {count} slots", self.id.fmt_short());
281            Poll::Ready(Ok(count))
282        } else {
283            Poll::Pending
284        }
285    }
