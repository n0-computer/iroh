11/// Custom transport.
12///
13/// Usually a transport will only deal with a single custom address type, but
14/// the signature allows for dealing with multiple custom address types.
15///
16/// A transport is a factory for custom endpoints. Whenever an iroh endpoint is
17/// created using [crate::endpoint::Builder::bind], a new custom endpoint will
18/// be created using [CustomTransport::bind].
19pub trait CustomTransport: std::fmt::Debug + Send + Sync + 'static {
20    /// Create a custom endpoint
21    fn bind(&self) -> io::Result<Box<dyn CustomEndpoint>>;
22}
23
24/// Custom endpoint created by a [CustomTransport].
25///
26/// An endpoint has a local address (or multiple local addresses), can receive
27/// packets, and can create senders to send packets.
28pub trait CustomEndpoint: std::fmt::Debug + Send + Sync + 'static {
29    /// Watch local addrs
30    fn watch_local_addrs(&self) -> n0_watcher::Direct<Vec<CustomAddr>>;
31    /// Create a sender
32    fn create_sender(&self) -> Arc<dyn CustomSender>;
33    /// Poll recv
34    fn poll_recv(
35        &mut self,
36        cx: &mut Context,
37        bufs: &mut [io::IoSliceMut<'_>],
38        metas: &mut [quinn_udp::RecvMeta],
39        source_addrs: &mut [Addr],
40    ) -> Poll<io::Result<usize>>;
41}
42
43/// Custom sender
44///
45/// A sender provides a poll based interface to send packets to custom addresses.
46/// It can decide whether it wants to send to a given custom address type.
47///
48/// This is not enforced at type level, but [CustomSender::poll_send] should
49/// only be called with addresses for which [CustomSender::is_valid_send_addr]
50/// returns true.
51pub trait CustomSender: std::fmt::Debug + Send + Sync + 'static {
52    /// is addr valid for this transport?
53    fn is_valid_send_addr(&self, addr: &CustomAddr) -> bool;
54    /// poll_send
55    fn poll_send(
56        &self,
57        cx: &mut std::task::Context,
58        dst: &CustomAddr,
59        transmit: &Transmit<'_>,
60    ) -> Poll<io::Result<()>>;
61}
