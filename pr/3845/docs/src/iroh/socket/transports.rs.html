171    /// Binds the  transports.
172    pub(crate) fn bind(
173        configs: &[TransportConfig],
174        relay_actor_config: RelayActorConfig,
175        metrics: &EndpointMetrics,
176        shutdown_token: CancellationToken,
177    ) -> io::Result<Self> {
178        #[cfg(not(wasm_browser))]
179        let ip_configs = {
180            let mut ip_configs = Vec::new();
181
182            // user defined overrides defaults
183            let has_ipv4_default = configs
184                .iter()
185                .any(|t| t.is_ipv4_default() && t.is_user_defined());
186            let has_ipv6_default = configs
187                .iter()
188                .any(|t| t.is_ipv6_default() && t.is_user_defined());
189            for config in configs {
190                if let TransportConfig::Ip {
191                    config,
192                    is_user_defined,
193                } = config
194                {
195                    if !is_user_defined
196                        && (config.is_ipv4() && has_ipv4_default
197                            || config.is_ipv6() && has_ipv6_default)
198                    {
199                        continue;
200                    }
201                    ip_configs.push(*config);
202                }
203            }
204            ip_configs
205        };
206        #[cfg(not(wasm_browser))]
207        let ip = IpTransports::bind(ip_configs.into_iter(), metrics)?;
208
209        let relay = configs
210            .iter()
211            .filter(|t| matches!(t, TransportConfig::Relay { .. }))
212            .map(|_c| RelayTransport::new(relay_actor_config.clone(), shutdown_token.child_token()))
213            .collect();
214
215        let mut custom = Vec::new();
216        for config in configs.iter().filter_map(|t| {
217            if let TransportConfig::Custom(config) = t {
218                Some(config)
219            } else {
220                None
221            }
222        }) {
223            let transport = config.bind()?;
224            custom.push(transport);
225        }
226
227        Ok(Self {
228            #[cfg(not(wasm_browser))]
229            ip,
230            relay,
231            custom,
232            poll_recv_counter: Default::default(),
233            source_addrs: Default::default(),
234        })
235    }

237    pub(crate) fn poll_recv(
238        &mut self,
239        cx: &mut Context,
240        bufs: &mut [io::IoSliceMut<'_>],
241        metas: &mut [quinn_udp::RecvMeta],
242        sock: &Socket,
243    ) -> Poll<io::Result<usize>> {
244        debug_assert_eq!(bufs.len(), metas.len(), "non matching bufs & metas");
245        debug_assert!(bufs.len() <= quinn_udp::BATCH_SIZE, "too many buffers");
246        if sock.is_closing() {
247            return Poll::Pending;
248        }
249
250        match self.inner_poll_recv(cx, bufs, metas)? {
251            Poll::Pending | Poll::Ready(0) => Poll::Pending,
252            Poll::Ready(n) => {
253                sock.process_datagrams(&mut bufs[..n], &mut metas[..n], &self.source_addrs[..n]);
254                Poll::Ready(Ok(n))
255            }
256        }
257    }
258
259    /// Tries to recv data, on all available transports.
260    fn inner_poll_recv(
261        &mut self,
262        cx: &mut Context,
263        bufs: &mut [IoSliceMut<'_>],
264        metas: &mut [quinn_udp::RecvMeta],
265    ) -> Poll<io::Result<usize>> {
266        debug_assert_eq!(bufs.len(), metas.len(), "non matching bufs & metas");
267
268        macro_rules! poll_transport {
269            ($socket:expr) => {
270                match $socket.poll_recv(cx, bufs, metas, &mut self.source_addrs)? {
271                    Poll::Pending | Poll::Ready(0) => {}
272                    Poll::Ready(n) => {
273                        return Poll::Ready(Ok(n));
274                    }
275                }
276            };
277        }
278
279        // To improve fairness, every other call reverses the ordering of polling.
280
281        let counter = self.poll_recv_counter.wrapping_add(1);
282
283        if counter.is_multiple_of(2) {
284            #[cfg(not(wasm_browser))]
285            poll_transport!(&mut self.ip);
286
287            for transport in self.relay.iter_mut() {
288                poll_transport!(transport);
289            }
290            for transport in self.custom.iter_mut() {
291                poll_transport!(transport);
292            }
293        } else {
294            for transport in self.custom.iter_mut().rev() {
295                poll_transport!(transport);
296            }
297            for transport in self.relay.iter_mut().rev() {
298                poll_transport!(transport);
299            }
300            #[cfg(not(wasm_browser))]
301            poll_transport!(&mut self.ip);
302        }
303
304        Poll::Pending
305    }

316    /// Watch for all currently known local addresses, including IP based transports.
317    pub(crate) fn local_addrs_watch(&self) -> LocalAddrsWatch {
318        let ips = n0_watcher::Join::new(self.ip.iter().map(|t| t.local_addr_watch()));
319        let relays = n0_watcher::Join::new(self.relay.iter().map(|t| t.local_addr_watch()));
320        let custom = n0_watcher::Join::new(self.custom.iter().map(|t| t.watch_local_addrs()));
321
322        ips.or(custom).or(relays).map(|((ips, custom), relays)| {
323            let ips = ips.into_iter().map(Addr::from);
324            let custom = custom.into_iter().flatten().map(Addr::from);
325            let relays = relays
326                .into_iter()
327                .flatten()
328                .map(|(relay_url, endpoint_id)| Addr::Relay(relay_url, endpoint_id));
329            ips.chain(custom).chain(relays).collect()
330        })
331    }

370    #[cfg(not(wasm_browser))]
371    pub(crate) fn max_receive_segments(&self) -> NonZeroUsize {
372        // `max_receive_segments` controls the size of the `RecvMeta` buffer
373        // that quinn creates. Having buffers slightly bigger than necessary
374        // isn't terrible, and makes sure a single socket can read the maximum
375        // amount with a single poll. We considered adding these numbers instead,
376        // but we never get data from both sockets at the same time in `poll_recv`
377        // and it's impossible and unnecessary to be refactored that way.
378
379        let res = self.ip.iter().map(|t| t.max_receive_segments()).max();
380        res.unwrap_or(NonZeroUsize::MIN)
381    }

622/// The type of transport, either primary or backup.
623///
624/// Primary transports compete with each other based on biased RTT measurements.
625/// Backup transports are only used when no primary transport is available.
626#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
627pub(crate) enum TransportType {
628    /// A transport that has the potential to be the primary transport.
629    ///
630    /// It will compete with other Primary transports such as IP based
631    /// transports based on biased RTT measurements.
632    Primary,
633    /// A transport that is only used as a backup transport.
634    ///
635    /// It will only compete with other backup transports such as the relay
636    /// transport based on biased RTT measurements.
637    Backup,
638}

640impl TransportType {
641    /// Converts to the corresponding QUIC path status.
642    pub(super) fn to_path_status(self) -> PathStatus {
643        match self {
644            Self::Primary => PathStatus::Available,
645            Self::Backup => PathStatus::Backup,
646        }
647    }

650/// Bias configuration for a transport type.
651///
652/// This controls how a transport is prioritized during path selection.
653///
654/// # Examples
655///
656/// ```
657/// use std::time::Duration;
658///
659/// use iroh::endpoint::transports::TransportBias;
660///
661/// // A primary transport with 100ms RTT advantage (will be preferred)
662/// let bias = TransportBias::primary().with_rtt_advantage(Duration::from_millis(100));
663///
664/// // A primary transport with 50ms RTT disadvantage (will be less preferred)
665/// let bias = TransportBias::primary().with_rtt_disadvantage(Duration::from_millis(50));
666/// ```
667#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
668pub struct TransportBias {
669    /// Whether this is a primary or backup transport.
670    pub(crate) transport_type: TransportType,
671    /// RTT bias in nanoseconds. Negative values make this transport more preferred.
672    pub(crate) rtt_bias: i128,
673}
674
675impl TransportBias {
676    /// Creates a primary transport bias with no RTT advantage.
677    ///
678    /// Primary transports compete with each other based on biased RTT measurements.
679    pub fn primary() -> Self {
680        Self {
681            transport_type: TransportType::Primary,
682            rtt_bias: 0,
683        }
684    }
685
686    /// Creates a backup transport bias with no RTT advantage.
687    ///
688    /// Backup transports are only used when no primary transport is available.
689    pub(crate) fn backup() -> Self {
690        Self {
691            transport_type: TransportType::Backup,
692            rtt_bias: 0,
693        }
694    }
695
696    /// Adds an RTT advantage to this transport, making it more preferred.
697    ///
698    /// The advantage is subtracted from the measured RTT during path selection,
699    /// so a transport with a 100ms advantage will be preferred over one with
700    /// the same measured RTT but no advantage.
701    pub fn with_rtt_advantage(mut self, advantage: Duration) -> Self {
702        self.rtt_bias -= advantage.as_nanos() as i128;
703        self
704    }
705
706    /// Adds an RTT disadvantage to this transport, making it less preferred.
707    ///
708    /// The disadvantage is added to the measured RTT during path selection,
709    /// so a transport with a 100ms disadvantage will be avoided in favor of
710    /// one with the same measured RTT but no disadvantage.
711    pub fn with_rtt_disadvantage(mut self, disadvantage: Duration) -> Self {
712        self.rtt_bias += disadvantage.as_nanos() as i128;
713        self
714    }
715}

717/// A map from address kinds to their transport bias configuration.
718///
719/// This controls how different transport types are prioritized during path selection.
720/// By default:
721/// - IPv4 and IPv6 are primary transports (IPv6 has a small RTT advantage)
722/// - Relay is a backup transport (only used when no primary transport is available)
723#[derive(Debug, Clone)]
724pub struct TransportBiasMap {
725    map: Arc<FxHashMap<AddrKind, TransportBias>>,
726}
727
728/// How much do we prefer IPv6 over IPv4.
729pub(super) const IPV6_RTT_ADVANTAGE: Duration = Duration::from_millis(3);
730
731impl Default for TransportBiasMap {
732    fn default() -> Self {
733        let mut map = FxHashMap::default();
734        map.insert(AddrKind::IpV4, TransportBias::primary());
735        map.insert(
736            AddrKind::IpV6,
737            TransportBias::primary().with_rtt_advantage(IPV6_RTT_ADVANTAGE),
738        );
739        map.insert(AddrKind::Relay, TransportBias::backup());
740        Self { map: Arc::new(map) }
741    }
742}
743
744impl TransportBiasMap {
745    /// Returns a new map with the given bias added or updated.
746    pub fn with_bias(self, kind: AddrKind, bias: TransportBias) -> Self {
747        let mut map = (*self.map).clone();
748        map.insert(kind, bias);
749        Self { map: Arc::new(map) }
750    }
751
752    /// Gets the bias for the given address.
753    ///
754    /// Returns a primary transport with no RTT bias if no specific bias is configured.
755    pub fn get(&self, addr: &Addr) -> TransportBias {
756        self.map
757            .get(&addr.addr_kind())
758            .cloned()
759            .unwrap_or_else(TransportBias::primary)
760    }
761
762    /// Computes path selection data for a given address and RTT.
763    pub fn path_selection_data(&self, addr: &Addr, rtt: Duration) -> PathSelectionData {
764        let bias = self.get(addr);
765        let tpe = bias.transport_type;
766        let biased_rtt = rtt.as_nanos() as i128 + bias.rtt_bias;
767        PathSelectionData {
768            transport_type: tpe,
769            rtt,
770            biased_rtt,
771        }
772    }
773}
774
775/// Data used during path selection.
776#[derive(Debug)]
777pub struct PathSelectionData {
778    /// Type of the transport.
779    pub transport_type: TransportType,
780    /// Measured RTT for path selection.
781    pub rtt: Duration,
782    /// Biased RTT for path selection.
783    ///
784    /// This is an i128 so we can subtract an advantage for e.g. IPv6 without underflowing.
785    pub biased_rtt: i128,
786}
787
788impl PathSelectionData {
789    /// Key for sorting paths. Lower is better.
790    ///
791    /// First part is the status, 0 for Available, 1 for Backup.
792    /// Second part is the biased RTT.
793    pub fn sort_key(&self) -> (u8, i128) {
794        (self.transport_type as u8, self.biased_rtt)
795    }
796}

814/// A sender that sends to all our transports.
815#[derive(Debug, Clone)]
816pub(crate) struct TransportsSender {
817    #[cfg(not(wasm_browser))]
818    ip: IpTransportsSender,
819    relay: Vec<RelaySender>,
820    custom: Vec<Arc<dyn CustomSender>>,
821    max_transmit_segments: NonZeroUsize,
822}
823
824impl TransportsSender {
825    #[instrument(name = "poll_send", skip(self, cx, transmit), fields(len = transmit.contents.len()))]
826    pub(crate) fn poll_send(
827        mut self: Pin<&mut Self>,
828        cx: &mut std::task::Context,
829        dst: &Addr,
830        src: Option<IpAddr>,
831        transmit: &Transmit<'_>,
832    ) -> Poll<io::Result<()>> {
833        match dst {
834            #[cfg(wasm_browser)]
835            Addr::Ip(..) => {
836                return Poll::Ready(Err(io::Error::other("IP is unsupported in browser")));
837            }
838            #[cfg(not(wasm_browser))]
839            Addr::Ip(dst_addr) => match dst_addr {
840                SocketAddr::V4(_) => {
841                    if let Some(sender) = self
842                        .ip
843                        .v4_iter_mut()
844                        .find(|s| s.is_valid_send_addr(src, dst_addr))
845                    {
846                        return Pin::new(sender).poll_send(cx, *dst_addr, src, transmit);
847                    }
848                    if let Some(sender) = self.ip.v4_default_mut()
849                        && sender.is_valid_default_addr(src, dst_addr)
850                    {
851                        return Pin::new(sender).poll_send(cx, *dst_addr, src, transmit);
852                    }
853                }
854                SocketAddr::V6(_) => {
855                    if let Some(sender) = self
856                        .ip
857                        .v6_iter_mut()
858                        .find(|s| s.is_valid_send_addr(src, dst_addr))
859                    {
860                        return Pin::new(sender).poll_send(cx, *dst_addr, src, transmit);
861                    }
862                    if let Some(sender) = self.ip.v6_default_mut()
863                        && sender.is_valid_default_addr(src, dst_addr)
864                    {
865                        return Pin::new(sender).poll_send(cx, *dst_addr, src, transmit);
866                    }
867                }
868            },
869            Addr::Relay(url, endpoint_id) => {
870                let mut has_valid_sender = false;
871                for sender in self
872                    .relay
873                    .iter_mut()
874                    .filter(|s| s.is_valid_send_addr(url, endpoint_id))
875                {
876                    has_valid_sender = true;
877                    match sender.poll_send(cx, url.clone(), *endpoint_id, transmit) {
878                        Poll::Pending => {}
879                        Poll::Ready(res) => return Poll::Ready(res),
880                    }
881                }
882                if has_valid_sender {
883                    return Poll::Pending;
884                }
885            }
886            Addr::Custom(addr) => {
887                for sender in &mut self.custom {
888                    if sender.is_valid_send_addr(addr) {
889                        match sender.poll_send(cx, addr, transmit) {
890                            Poll::Pending => {}
891                            Poll::Ready(res) => return Poll::Ready(res),
892                        }
893                    }
894                }
895            }
896        }
897
898        // We "blackhole" data that we have not found any usable transport for on
899        // to make sure the QUIC stack picks up that currently this data does not arrive.
900        trace!(?src, ?dst, "no valid transport available");
901        Poll::Ready(Ok(()))
902    }
903}

923    fn create_sender(&self) -> Pin<Box<dyn quinn::UdpSender>> {
924        Box::pin(Sender {
925            sock: self.sock.clone(),
926            sender: self.transports.create_sender(),
927        })
928    }

930    fn poll_recv(
931        &mut self,
932        cx: &mut Context,
933        bufs: &mut [IoSliceMut<'_>],
934        meta: &mut [quinn_udp::RecvMeta],
935    ) -> Poll<io::Result<usize>> {

937    }

939    #[cfg(not(wasm_browser))]
940    fn local_addr(&self) -> io::Result<SocketAddr> {
941        let local_addrs = self.transports.local_addrs();
942        let addrs: Vec<_> = local_addrs
943            .into_iter()
944            .map(|addr| {
945                use crate::socket::mapped_addrs::DEFAULT_FAKE_ADDR;
946
947                match addr {
948                    Addr::Ip(addr) => addr,
949                    Addr::Relay(..) => DEFAULT_FAKE_ADDR.into(),
950                    Addr::Custom(_) => DEFAULT_FAKE_ADDR.into(),
951                }
952            })
953            .collect();
954
955        if let Some(addr) = addrs.iter().find(|addr| addr.is_ipv6()) {
956            return Ok(*addr);
957        }
958        if let Some(SocketAddr::V4(addr)) = addrs.first() {
959            // Pretend to be IPv6, because our `MappedAddr`s need to be IPv6.
960            let ip = addr.ip().to_ipv6_mapped().into();
961            return Ok(SocketAddr::new(ip, addr.port()));
962        }
963
964        if !self.transports.relay.is_empty() {
965            // pretend we have an address to make sure things are not too sad during startup
966            use crate::socket::mapped_addrs::DEFAULT_FAKE_ADDR;
967
968            return Ok(DEFAULT_FAKE_ADDR.into());
969        }
970        if !self.transports.custom.is_empty() {
971            // pretend we have an address to make sure things are not too sad during startup
972            use crate::socket::mapped_addrs::DEFAULT_FAKE_ADDR;
973
974            return Ok(DEFAULT_FAKE_ADDR.into());
975        }
976        Err(io::Error::other("no valid address available"))
977    }
978
979    #[cfg(wasm_browser)]
980    fn local_addr(&self) -> io::Result<SocketAddr> {
981        // Again, we need to pretend we're IPv6, because of our `MappedAddr`s.
982        Ok(SocketAddr::new(std::net::Ipv6Addr::LOCALHOST.into(), 0))
983    }
984
985    fn max_receive_segments(&self) -> NonZeroUsize {
986        self.transports.max_receive_segments()
987    }
988
989    fn may_fragment(&self) -> bool {
990        self.transports.may_fragment()
991    }

994/// A sender for [`Transport`].
995///
996/// This is special in that it handles [`MultipathMappedAddr::Mixed`] by delegating to the
997/// [`Socket`] which expands it back to one or more [`Addr`]s and sends it
998/// using the underlying [`Transports`].
999#[derive(Debug)]
1000#[pin_project::pin_project]
1001pub(crate) struct Sender {
1002    sock: Arc<Socket>,
1003    #[pin]
1004    sender: TransportsSender,
1005}
1006
1007impl Sender {
1008    /// Extracts the right [`Addr`] from the [`quinn_udp::Transmit`].
1009    ///
1010    /// Because Quinn does only know about IP transports we map other transports to private
1011    /// IPv6 Unique Local Address ranges.  This extracts the transport addresses out of the
1012    /// transmit's destination.
1013    fn mapped_addr(&self, transmit: &quinn_udp::Transmit) -> io::Result<MultipathMappedAddr> {
1014        if self.sock.is_closed() {
1015            return Err(io::Error::new(
1016                io::ErrorKind::NotConnected,
1017                "connection closed",
1018            ));
1019        }
1020
1021        Ok(MultipathMappedAddr::from(transmit.destination))
1022    }
1023}
1024
1025impl quinn::UdpSender for Sender {
1026    fn poll_send(
1027        self: Pin<&mut Self>,
1028        quinn_transmit: &quinn_udp::Transmit,
1029        cx: &mut Context,
1030    ) -> Poll<io::Result<()>> {
1031        // On errors this methods prefers returning Ok(()) to Quinn.  Returning an error
1032        // should only happen if the error is permanent and fatal and it will never be
1033        // possible to send anything again.  Doing so kills the Quinn EndpointDriver.  Most
1034        // send errors are intermittent errors, returning Ok(()) in those cases will mean
1035        // Quinn eventually considers the packets that had send errors as lost and will try
1036        // and re-send them.
1037        let mapped_addr = self.mapped_addr(quinn_transmit)?;
1038
1039        let transport_addr = match mapped_addr {
1040            MultipathMappedAddr::Mixed(mapped_addr) => {
1041                let Some(endpoint_id) = self.sock.mapped_addrs.endpoint_addrs.lookup(&mapped_addr)
1042                else {
1043                    error!(dst = ?mapped_addr, "unknown NodeIdMappedAddr, dropped transmit");
1044                    return Poll::Ready(Ok(()));
1045                };
1046
1047                // Note we drop the src_ip set in the Quinn Transmit.  This is only the
1048                // Initial packet we are sending, so we do not yet have an src address we
1049                // need to respond from.
1050                if let Some(src_ip) = quinn_transmit.src_ip {
1051                    warn!(dst = ?mapped_addr, ?src_ip, dst_endpoint = %endpoint_id.fmt_short(),
1052                        "oops, flub didn't think this would happen");
1053                }
1054
1055                match self.sock.try_send_remote_state_msg(
1056                    endpoint_id,
1057                    super::RemoteStateMessage::SendDatagram(
1058                        Box::new(self.sender.clone()),
1059                        OwnedTransmit::from(quinn_transmit),
1060                    ),
1061                ) {
1062                    Ok(()) => {
1063                        trace!(dst = ?mapped_addr, dst_endpoint = %endpoint_id.fmt_short(), "sent transmit");
1064                        return Poll::Ready(Ok(()));
1065                    }
1066                    Err(msg) => {
1067                        // We do not want to block the next send which might be on a
1068                        // different transport.  Instead we let Quinn handle this as
1069                        // a lost datagram.
1070                        // TODO: Revisit this: we might want to do something better.
1071                        debug!(
1072                            dst = ?mapped_addr,
1073                            dst_endpoint = %endpoint_id.fmt_short(),
1074                            ?msg,
1075                            "RemoteStateActor inbox dropped message"
1076                        );
1077                        return Poll::Ready(Ok(()));
1078                    }
1079                };
1080            }
1081            MultipathMappedAddr::Relay(relay_mapped_addr) => {
1082                match self
1083                    .sock
1084                    .mapped_addrs
1085                    .relay_addrs
1086                    .lookup(&relay_mapped_addr)
1087                {
1088                    Some((relay_url, endpoint_id)) => Addr::Relay(relay_url, endpoint_id),
1089                    None => {
1090                        error!("unknown RelayMappedAddr, dropped transmit");
1091                        return Poll::Ready(Ok(()));
1092                    }
1093                }
1094            }
1095            MultipathMappedAddr::Custom(custom_mapped_addr) => {
1096                match self
1097                    .sock
1098                    .mapped_addrs
1099                    .custom_addrs
1100                    .lookup(&custom_mapped_addr)
1101                {
1102                    Some(addr) => Addr::Custom(addr),
1103                    None => {
1104                        error!("unknown CustomMappedAddr, dropped transmit");
1105                        return Poll::Ready(Ok(()));
1106                    }
1107                }
1108            }
1109            MultipathMappedAddr::Ip(socket_addr) => {
1110                // Ensure IPv6 mapped addresses are converted back
1111                let socket_addr =
1112                    SocketAddr::new(socket_addr.ip().to_canonical(), socket_addr.port());
1113                Addr::Ip(socket_addr)
1114            }
1115        };
1116
1117        let transmit = Transmit {
1118            ecn: quinn_transmit.ecn,
1119            contents: quinn_transmit.contents,
1120            segment_size: quinn_transmit.segment_size,
1121        };
1122        let this = self.project();
1123
1124        match this
1125            .sender
1126            .poll_send(cx, &transport_addr, quinn_transmit.src_ip, &transmit)
1127        {
1128            Poll::Ready(Ok(())) => Poll::Ready(Ok(())),
1129            Poll::Ready(Err(ref err)) => {
1130                warn!(?transport_addr, "dropped transmit: {err:#}");
1131                Poll::Ready(Ok(()))
1132            }
1133            Poll::Pending => {
1134                // We do not want to block the next send which might be on a
1135                // different transport.  Instead we let Quinn handle this as a lost
1136                // datagram.
1137                // TODO: Revisit this: we might want to do something better.
1138                trace!(?transport_addr, "transport pending, dropped transmit");
1139                Poll::Ready(Ok(()))
1140            }
1141        }
1142    }
1143
1144    fn max_transmit_segments(&self) -> NonZeroUsize {
1145        self.sender.max_transmit_segments
1146    }
1147}
