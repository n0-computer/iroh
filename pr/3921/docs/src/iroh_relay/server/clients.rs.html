24/// Manages the connections to all currently connected clients.
25#[derive(Debug, Default, Clone)]
26/// Registry of connected relay clients.
27///
28/// This type manages the collection of active client connections and
29/// handles routing messages between them.
30pub struct Clients(Arc<Inner>);
31
32#[derive(Debug, Default)]
33struct Inner {
34    /// The list of all currently connected clients.
35    clients: DashMap<EndpointId, ClientState>,
36    // /// List of clients that are still connected, but not active anymore
37    // ///
38    // /// Clients are moved here if another endpoint with the same id connects.
39    // inactive_clients: DashMap<EndpointId, Vec<Client>>,
40    /// Map of which client has sent where
41    sent_to: DashMap<EndpointId, HashSet<EndpointId>>,
42    /// Connection ID Counter
43    next_connection_id: AtomicU64,
44}
45
46#[derive(Debug)]
47struct ClientState {
48    active: Client,
49    inactive: Vec<Client>,
50}
51
52impl ClientState {
53    async fn shutdown_all(mut self) {
54        [self.active]
55            .into_iter()
56            .chain(self.inactive.drain(..))
57            .map(Client::shutdown)
58            .join_all()
59            .await;
60    }
61}

63impl Clients {
64    /// Shuts down all connected clients.
65    ///
66    /// This method gracefully disconnects all active client connections managed by
67    /// this registry. It will wait for all clients to complete their shutdown before
68    /// returning.
69    pub async fn shutdown(&self) {
70        let keys: Vec<_> = self.0.clients.iter().map(|x| *x.key()).collect();
71        trace!("shutting down {} clients", keys.len());
72        let clients = keys.into_iter().filter_map(|k| self.0.clients.remove(&k));
73        n0_future::join_all(clients.map(|(_, state)| state.shutdown_all())).await;
74    }

76    /// Builds the client handler and starts the read & write loops for the connection.
77    pub fn register<S>(&self, client_config: Config<S>, metrics: Arc<Metrics>)
78    where
79        S: BytesStreamSink + Send + 'static,
80    {
81        let endpoint_id = client_config.endpoint_id;
82        let connection_id = self.get_connection_id();
83        trace!(remote_endpoint = %endpoint_id.fmt_short(), "registering client");
84
85        let client = Client::new(client_config, connection_id, self, metrics);
86        match self.0.clients.entry(endpoint_id) {
87            dashmap::Entry::Occupied(mut entry) => {
88                let state = entry.get_mut();
89                let old_client = std::mem::replace(&mut state.active, client);
90                debug!(
91                    remote_endpoint = %endpoint_id.fmt_short(),
92                    "multiple connections found, deactivating old connection",
93                );
94                old_client
95                    .try_send_health("Another endpoint connected with the same endpoint id. No more messages will be received".to_string())
96                    .ok();
97                state.inactive.push(old_client);
98            }
99            dashmap::Entry::Vacant(entry) => {
100                entry.insert(ClientState {
101                    active: client,
102                    inactive: Vec::new(),
103                });
104            }
105        }
106    }
107
108    fn get_connection_id(&self) -> u64 {
109        self.0.next_connection_id.fetch_add(1, Ordering::Relaxed)
110    }
111
112    /// Removes the client from the map of clients, & sends a notification
113    /// to each client that peers has sent data to, to let them know that
114    /// peer is gone from the network.
115    ///
116    /// Must be passed a matching connection_id.
117    pub(super) fn unregister(&self, connection_id: u64, endpoint_id: EndpointId) {
118        trace!(
119            endpoint_id = %endpoint_id.fmt_short(),
120            connection_id, "unregistering client"
121        );
122
123        self.0.clients.remove_if_mut(&endpoint_id, |_id, state| {
124            if state.active.connection_id() == connection_id {
125                // The unregistering client is the currently active client
126                let client = &state.active;
127                if let Some(last_inactive_client) = state.inactive.pop() {
128                    // There is an inactive client, promote to active again.
129                    state.active = last_inactive_client;
130                    // Don't remove the entry from client map.
131                    false
132                } else {
133                    // No inactive clients: Inform other peers that this peer is now gone.
134                    if let Some((_, sent_to)) = self.0.sent_to.remove(&endpoint_id) {
135                        for key in sent_to {
136                            match client.try_send_peer_gone(key) {
137                                Ok(_) => {}
138                                Err(TrySendError::Full(_)) => {
139                                    debug!(
140                                        dst = %key.fmt_short(),
141                                        "client too busy to receive packet, dropping packet"
142                                    );
143                                }
144                                Err(TrySendError::Closed(_)) => {
145                                    debug!(
146                                        dst = %key.fmt_short(),
147                                        "can no longer write to client, dropping packet"
148                                    );
149                                }
150                            }
151                        }
152                    }
153                    // Remove entry from the client map.
154                    true
155                }
156            } else {
157                // The unregistering client is already inactive. Remove from the list of inactive clients.
158                state
159                    .inactive
160                    .retain(|client| client.connection_id() != connection_id);
161                // Active client is unmodified: keep entry in map.
162                false
163            }
164        });
165    }

167    /// Attempt to send a packet to client with [`EndpointId`] `dst`.
168    pub(super) fn send_packet(
169        &self,
170        dst: EndpointId,
171        data: Datagrams,
172        src: EndpointId,
173        metrics: &Metrics,
174    ) -> Result<(), ForwardPacketError> {
175        let Some(client) = self.0.clients.get(&dst) else {
176            debug!(dst = %dst.fmt_short(), "no connected client, dropped packet");
177            metrics.send_packets_dropped.inc();
178            return Ok(());
179        };
180        match client.active.try_send_packet(src, data) {
181            Ok(_) => {
182                // Record sent_to relationship
183                self.0.sent_to.entry(src).or_default().insert(dst);
184                Ok(())
185            }
186            Err(TrySendError::Full(_)) => {
187                debug!(
188                    dst = %dst.fmt_short(),
189                    "client too busy to receive packet, dropping packet"
190                );
191                Err(ForwardPacketError::new(SendError::Full))
192            }
193            Err(TrySendError::Closed(_)) => {
194                debug!(
195                    dst = %dst.fmt_short(),
196                    "can no longer write to client, dropping message and pruning connection"
197                );
198                client.active.start_shutdown();
199                Err(ForwardPacketError::new(SendError::Closed))
200            }
201        }
202    }

230    async fn recv_frame<
231        E: std::error::Error + Sync + Send + 'static,
232        S: Stream<Item = Result<RelayToClientMsg, E>> + Unpin,
233    >(
234        frame_type: FrameType,
235        mut stream: S,
236    ) -> Result<RelayToClientMsg> {
237        match stream.next().await {
238            Some(Ok(frame)) => {
239                if frame_type != frame.typ() {
240                    n0_error::bail_any!(
241                        "Unexpected frame, got {:?}, but expected {:?}",
242                        frame.typ(),
243                        frame_type
244                    );
245                }
246                Ok(frame)
247            }
248            Some(Err(err)) => Err(err).anyerr(),
249            None => n0_error::bail_any!("Unexpected EOF, expected frame {frame_type:?}"),
250        }
251    }

268    #[tokio::test]
269    #[traced_test]
270    async fn test_clients() -> Result {
271        let mut rng = rand_chacha::ChaCha8Rng::seed_from_u64(0u64);
272        let a_key = SecretKey::generate(&mut rng).public();
273        let b_key = SecretKey::generate(&mut rng).public();
274
275        let (builder_a, mut a_rw) = test_client_builder(a_key);
276
277        let clients = Clients::default();
278        let metrics = Arc::new(Metrics::default());
279        clients.register(builder_a, metrics.clone());
280
281        // send packet
282        let data = b"hello world!";
283        clients.send_packet(a_key, Datagrams::from(&data[..]), b_key, &metrics)?;
284        let frame = recv_frame(FrameType::RelayToClientDatagram, &mut a_rw).await?;
285        assert_eq!(
286            frame,
287            RelayToClientMsg::Datagrams {
288                remote_endpoint_id: b_key,
289                datagrams: data.to_vec().into(),
290            }
291        );
292
293        {
294            let client = clients.0.clients.get(&a_key).unwrap();
295            // shutdown client a, this should trigger the removal from the clients list
296            client.active.start_shutdown();
297        }
298
299        // need to wait a moment for the removal to be processed
300        let c = clients.clone();
301        tokio::time::timeout(Duration::from_secs(1), async move {
302            loop {
303                if !c.0.clients.contains_key(&a_key) {
304                    break;
305                }
306                tokio::time::sleep(Duration::from_millis(100)).await;
307            }
308        })
309        .await
310        .std_context("timeout")?;
311        clients.shutdown().await;
312
313        Ok(())
314    }

316    #[tokio::test]
317    #[traced_test]
318    async fn test_clients_same_endpoint_id() -> Result {
319        let mut rng = rand_chacha::ChaCha8Rng::seed_from_u64(0u64);
320        let a_key = SecretKey::generate(&mut rng).public();
321        let b_key = SecretKey::generate(&mut rng).public();
322
323        let (a1_builder, mut a1_rw) = test_client_builder(a_key);
324
325        let clients = Clients::default();
326        let metrics = Arc::new(Metrics::default());
327
328        // register client a
329        clients.register(a1_builder, metrics.clone());
330        let a1_conn_id = clients.active_connection_id(a_key).unwrap();
331
332        // send packet and verify it is send to a1
333        let data = b"hello world!";
334        clients.send_packet(a_key, Datagrams::from(&data[..]), b_key, &metrics)?;
335        let frame = recv_frame(FrameType::RelayToClientDatagram, &mut a1_rw).await?;
336        assert_eq!(
337            frame,
338            RelayToClientMsg::Datagrams {
339                remote_endpoint_id: b_key,
340                datagrams: data.to_vec().into(),
341            }
342        );
343
344        // register new client with same endpoint id
345        let (a2_builder, mut a2_rw) = test_client_builder(a_key);
346        clients.register(a2_builder, metrics.clone());
347        let a2_conn_id = clients.active_connection_id(a_key).unwrap();
348        assert!(a2_conn_id != a1_conn_id);
349
350        // a1 is marked inactive and should receive a health frame
351        let _frame = recv_frame(FrameType::Health, &mut a1_rw).await?;
352
353        // send packet and verify it is send to a2
354        clients.send_packet(a_key, Datagrams::from(&data[..]), b_key, &metrics)?;
355        let frame = recv_frame(FrameType::RelayToClientDatagram, &mut a2_rw).await?;
356        assert_eq!(
357            frame,
358            RelayToClientMsg::Datagrams {
359                remote_endpoint_id: b_key,
360                datagrams: data.to_vec().into(),
361            }
362        );
363
364        // disconnect a2
365        clients
366            .0
367            .clients
368            .get(&a_key)
369            .unwrap()
370            .active
371            .start_shutdown();
372
373        // need to wait a moment for the removal to be processed
374        tokio::time::timeout(Duration::from_secs(1), {
375            let clients = clients.clone();
376            async move {
377                // wait until the active connection is no longer a2 (which we unregistered)
378                while clients.active_connection_id(a_key) == Some(a2_conn_id) {
379                    tokio::time::sleep(Duration::from_millis(100)).await;
380                }
381            }
382        })
383        .await
384        .std_context("timeout")?;
385
386        // a1 should be marked active again now, and receive sent messages
387        assert_eq!(clients.active_connection_id(a_key), Some(a1_conn_id));
388        clients.send_packet(a_key, Datagrams::from(&data[..]), b_key, &metrics)?;
389        let frame = recv_frame(FrameType::RelayToClientDatagram, &mut a1_rw).await?;
390        assert_eq!(
391            frame,
392            RelayToClientMsg::Datagrams {
393                remote_endpoint_id: b_key,
394                datagrams: data.to_vec().into(),
395            }
396        );
397
398        // after shutting down the now-active client, there should no longer be an entry for that endpoint id
399        clients
400            .0
401            .clients
402            .get(&a_key)
403            .unwrap()
404            .active
405            .start_shutdown();
406
407        // need to wait a moment for the removal to be processed
408        tokio::time::timeout(Duration::from_secs(1), {
409            let clients = clients.clone();
410            async move {
411                // wait until the active connection is no longer a2 (which we unregistered)
412                while clients.0.clients.contains_key(&a_key) {
413                    tokio::time::sleep(Duration::from_millis(100)).await;
414                }
415            }
416        })
417        .await
418        .std_context("timeout")?;
419
420        clients.shutdown().await;
421
422        Ok(())
423    }
