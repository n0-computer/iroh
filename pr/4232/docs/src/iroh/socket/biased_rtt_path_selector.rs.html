1//! Default [`PathSelector`] implementation.
2//!
3//! [`BiasedRttPathSelector`] preserves iroh's historical "lowest biased RTT wins, with
4//! stickiness against flapping" behaviour and is what's installed when no custom
5//! selector is provided.
6
7use std::{sync::Arc, time::Duration};
8
9use rustc_hash::FxHashMap;
10
11use super::{
12    remote_map::{PathSelection, PathSelectionContext, PathSelectionData, PathSelector},
13    transports::{Addr, AddrKind},
14};

16/// How much do we prefer IPv6 over IPv4 by default.
17const IPV6_RTT_ADVANTAGE: Duration = Duration::from_millis(3);
18
19/// Stickiness threshold for biased RTT comparisons.  Switching to a same-tier path only
20/// happens when its biased RTT is at least this much better than the current path's.
21const RTT_SWITCHING_MIN: Duration = Duration::from_millis(5);
22
23/// Whether a transport is a primary path or a backup.
24///
25/// Primary paths are used preferentially.  Backup paths are only used when no primary
26/// path is available.  This is independent of the QUIC `PathStatus`; today the only
27/// transport classified as backup is the relay transport.
28#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
29enum TransportType {
30    /// A primary path: used whenever available.
31    Primary,
32    /// A backup path: only used when no primary path is available.
33    Backup,
34}
35
36/// Bias configuration for a single transport kind.
37///
38/// Used by [`BiasedRttPathSelector`] to bias path selection per address kind.
39#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
40pub(crate) struct TransportBias {
41    transport_type: TransportType,
42    /// RTT bias in nanoseconds.  Negative values make this transport more preferred.
43    rtt_bias: i128,
44}
45
46impl TransportBias {
47    /// Creates a primary transport bias with no RTT advantage.
48    pub(crate) fn primary() -> Self {
49        Self {
50            transport_type: TransportType::Primary,
51            rtt_bias: 0,
52        }
53    }
54
55    /// Creates a backup transport bias with no RTT advantage.
56    fn backup() -> Self {
57        Self {
58            transport_type: TransportType::Backup,
59            rtt_bias: 0,
60        }
61    }
62
63    /// Adds an RTT advantage to this transport, making it more preferred.
64    pub(crate) fn with_rtt_advantage(mut self, advantage: Duration) -> Self {
65        self.rtt_bias -= advantage.as_nanos() as i128;
66        self
67    }
68
69    /// Adds an RTT disadvantage to this transport, making it less preferred.
70    #[cfg(all(test, feature = "unstable-custom-transports"))]
71    pub(crate) fn with_rtt_disadvantage(mut self, disadvantage: Duration) -> Self {
72        self.rtt_bias += disadvantage.as_nanos() as i128;
73        self
74    }
75}
76
77/// The default [`PathSelector`] used by iroh.
78///
79/// Sorts paths by `(transport_type, biased_rtt)` (primary tier wins, then lowest biased
80/// RTT).  Within the same tier, switching only happens once a candidate's biased RTT is
81/// at least 5ms better than the currently-selected path — this avoids flapping under
82/// jitter.  Across tiers, switching is immediate.
83///
84/// The biases are configured per [`AddrKind`].  Defaults: IPv4 and IPv6 are primary
85/// (IPv6 has a 3ms RTT advantage), Relay is backup, custom transports are primary with
86/// no advantage.
87#[derive(Debug, Clone)]
88pub(crate) struct BiasedRttPathSelector {
89    biases: Arc<FxHashMap<AddrKind, TransportBias>>,
90}
91
92impl Default for BiasedRttPathSelector {
93    fn default() -> Self {
94        let mut map = FxHashMap::default();
95        map.insert(AddrKind::IpV4, TransportBias::primary());
96        map.insert(
97            AddrKind::IpV6,
98            TransportBias::primary().with_rtt_advantage(IPV6_RTT_ADVANTAGE),
99        );
100        map.insert(AddrKind::Relay, TransportBias::backup());
101        Self {
102            biases: Arc::new(map),
103        }
104    }
105}
106
107impl BiasedRttPathSelector {
108    /// Returns a new selector with the given bias added or updated for `kind`.
109    #[cfg(all(test, feature = "unstable-custom-transports"))]
110    pub(crate) fn with_bias(self, kind: AddrKind, bias: TransportBias) -> Self {
111        let mut map = (*self.biases).clone();
112        map.insert(kind, bias);
113        Self {
114            biases: Arc::new(map),
115        }
116    }
117
118    /// Looks up the bias for an address.  Defaults to primary with no RTT bias.
119    fn bias_for(&self, addr: &Addr) -> TransportBias {
120        self.biases
121            .get(&addr.addr_kind())
122            .copied()
123            .unwrap_or_else(TransportBias::primary)
124    }
125
126    /// Computes the sort key for a path: lower is better.
127    fn sort_key(&self, addr: &Addr, rtt: Duration) -> (TransportType, i128) {
128        let bias = self.bias_for(addr);
129        let biased_rtt = (rtt.as_nanos() as i128).saturating_add(bias.rtt_bias);
130        (bias.transport_type, biased_rtt)
131    }
132}
133
134impl PathSelector for BiasedRttPathSelector {
135    fn select(&self, ctx: &PathSelectionContext<'_>) -> PathSelection {
136        // Single pass: track the best candidate by sort key, and the best (lowest)
137        // sort key seen for the currently-selected address.  When the same address
138        // appears multiple times (one path per connection), `min` over `sort_key`
139        // naturally picks the lowest-RTT instance — no separate aggregation needed.
140        let current = ctx.current();
141        let mut best: Option<(PathSelectionData<'_>, (TransportType, i128))> = None;
142        let mut current_key: Option<(TransportType, i128)> = None;
143
144        tracing::debug!("dumping path RTTs");
145        for psd in ctx.paths() {
146            let addr = psd.remote_addr();
147            // Skip paths whose stats can't be read (e.g. closed concurrently with select).
148            let Some(stats) = psd.stats() else {
149                continue;
150            };
151            let rtt = stats.rtt;
152            tracing::debug!(?addr, ?rtt);
153            let key = self.sort_key(addr, rtt);
154
155            if Some(addr) == current && current_key.is_none_or(|c| key < c) {
156                current_key = Some(key);
157            }
158            if best.as_ref().is_none_or(|(_, b)| key < *b) {
159                best = Some((psd, key));
160            }
161        }
162
163        let mut selection = PathSelection::none();
164        let Some((best_psd, (best_tier, best_biased))) = best else {
165            return selection;
166        };
167
168        // If we have no current path or no data for it, switch to the best.
169        let Some((current_tier, current_biased)) = current_key else {
170            selection.set(&best_psd);
171            return selection;
172        };
173
174        if current_tier != best_tier {
175            // Always switch across tiers (e.g. relay -> primary).
176            selection.set(&best_psd);
177        } else if best_biased + RTT_SWITCHING_MIN.as_nanos() as i128 <= current_biased {
178            // For the same tier, only switch when biased RTT is meaningfully better.
179            selection.set(&best_psd);
180        }
181        selection
182    }
183}

186mod tests {
187    use std::net::{Ipv4Addr, Ipv6Addr, SocketAddr, SocketAddrV4, SocketAddrV6};

189    use iroh_base::{EndpointId, RelayUrl};

195    fn v4(port: u16) -> TransportFourTuple {
196        TransportFourTuple::from_remote(Addr::Ip(SocketAddr::V4(SocketAddrV4::new(
197            Ipv4Addr::LOCALHOST,
198            port,
199        ))))
200    }
201
202    fn v6(port: u16) -> TransportFourTuple {
203        TransportFourTuple::from_remote(Addr::Ip(SocketAddr::V6(SocketAddrV6::new(
204            Ipv6Addr::LOCALHOST,
205            port,
206            0,
207            0,
208        ))))
209    }
210
211    fn relay(port: u16) -> TransportFourTuple {
212        let url = format!("https://relay{port}.iroh.computer")
213            .parse::<RelayUrl>()
214            .unwrap();
215        TransportFourTuple::from_remote(Addr::Relay(
216            url,
217            EndpointId::from_bytes(&[0u8; 32]).unwrap(),
218        ))
219    }
220
221    fn psd(addr: &TransportFourTuple, rtt_ms: u64) -> PathSelectionData<'_> {
222        // PathStats is #[non_exhaustive], so build via Default + field assignment.
223        let mut stats = PathStats::default();
224        stats.rtt = Duration::from_millis(rtt_ms);
225        PathSelectionData::for_test(addr, Some(stats))
226    }
227
228    /// Runs [`BiasedRttPathSelector::default`] against the given paths and current
229    /// selection, returning the selector's primary pick (cloned, for easier asserts).
230    fn select_with_default(
231        current: Option<&TransportFourTuple>,
232        paths: Vec<PathSelectionData<'_>>,
233    ) -> Option<TransportFourTuple> {
234        let ctx = PathSelectionContext::for_test(current, paths);
235        BiasedRttPathSelector::default()
236            .select(&ctx)
237            .selected()
238            .cloned()
239    }

241    #[test]
242    fn ipv6_wins_over_ipv4_within_bias() {
243        let v4 = v4(1);
244        let v6 = v6(1);
245
246        // Equal RTTs: IPv6 wins because of the bias advantage.
247        let chosen = select_with_default(None, vec![psd(&v4, 10), psd(&v6, 10)]);
248        assert_eq!(chosen.as_ref(), Some(&v6));
249
250        // IPv6 still wins when 2ms slower (within the 3ms bias).
251        let chosen = select_with_default(None, vec![psd(&v4, 10), psd(&v6, 12)]);
252        assert_eq!(chosen.as_ref(), Some(&v6));
253
254        // IPv4 wins when IPv6 is 10ms slower (exceeds 3ms bias).
255        let chosen = select_with_default(None, vec![psd(&v4, 10), psd(&v6, 20)]);
256        assert_eq!(chosen.as_ref(), Some(&v4));
257    }

259    #[test]
260    fn primary_wins_over_backup_regardless_of_rtt() {
261        let v4 = v4(1);
262        let relay = relay(1);
263
264        // Primary tier beats backup tier even when the backup has a much lower RTT.
265        let chosen = select_with_default(None, vec![psd(&v4, 100), psd(&relay, 10)]);
266        assert!(matches!(
267            chosen.as_ref().map(|t| t.remote()),
268            Some(Addr::Ip(_))
269        ));
270
271        // Even more extreme: 1000ms primary still wins over 1ms backup.
272        let chosen = select_with_default(None, vec![psd(&v4, 1000), psd(&relay, 1)]);
273        assert!(matches!(
274            chosen.as_ref().map(|t| t.remote()),
275            Some(Addr::Ip(_))
276        ));
277    }

279    #[test]
280    fn same_tier_only_switches_with_significant_rtt_diff() {
281        let v4_1 = v4(1);
282        let v4_2 = v4(2);
283
284        // 2ms diff < 5ms threshold → keep current (no switch, primary() == None).
285        let chosen = select_with_default(Some(&v4_1), vec![psd(&v4_1, 20), psd(&v4_2, 18)]);
286        assert_eq!(chosen, None);
287
288        // 4ms diff < 5ms → keep current.
289        let chosen = select_with_default(Some(&v4_1), vec![psd(&v4_1, 20), psd(&v4_2, 16)]);
290        assert_eq!(chosen, None);
291
292        // 5ms diff hits the threshold (the condition is `<=`) → switch.
293        let chosen = select_with_default(Some(&v4_1), vec![psd(&v4_1, 20), psd(&v4_2, 15)]);
294        assert_eq!(chosen.as_ref(), Some(&v4_2));
295
296        // 6ms diff > 5ms → switch.
297        let chosen = select_with_default(Some(&v4_1), vec![psd(&v4_1, 20), psd(&v4_2, 14)]);
298        assert_eq!(chosen.as_ref(), Some(&v4_2));
299    }
