1//! Default [`PathSelector`] implementation.
2//!
3//! [`BiasedRttPathSelector`] preserves iroh's historical "lowest biased RTT wins, with
4//! stickiness against flapping" behaviour and is what's installed when no custom
5//! selector is provided.
6
7#![allow(dead_code)]
8
9use std::{sync::Arc, time::Duration};
10
11use rustc_hash::FxHashMap;
12
13use super::{
14    remote_map::{PathSelection, PathSelectionContext, PathSelectionData, PathSelector},
15    transports::{Addr, AddrKind},
16};
17
18/// How much do we prefer IPv6 over IPv4 by default.
19const IPV6_RTT_ADVANTAGE: Duration = Duration::from_millis(3);
20
21/// Stickiness threshold for biased RTT comparisons.  Switching to a same-tier path only
22/// happens when its biased RTT is at least this much better than the current path's.
23const RTT_SWITCHING_MIN: Duration = Duration::from_millis(5);
24
25/// Whether a transport is a primary path or a backup.
26///
27/// Primary paths are used preferentially.  Backup paths are only used when no primary
28/// path is available.  This is independent of the QUIC `PathStatus`; today the only
29/// transport classified as backup is the relay transport.
30#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
31enum TransportType {
32    /// A primary path: used whenever available.
33    Primary,
34    /// A backup path: only used when no primary path is available.
35    Backup,
36}
37
38/// Bias configuration for a single transport kind.
39///
40/// Used by [`BiasedRttPathSelector`] to bias path selection per address kind.
41#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
42pub(crate) struct TransportBias {
43    transport_type: TransportType,
44    /// RTT bias in nanoseconds.  Negative values make this transport more preferred.
45    rtt_bias: i128,
46}
47
48impl TransportBias {
49    /// Creates a primary transport bias with no RTT advantage.
50    pub(crate) fn primary() -> Self {
51        Self {
52            transport_type: TransportType::Primary,
53            rtt_bias: 0,
54        }
55    }
56
57    /// Creates a backup transport bias with no RTT advantage.
58    fn backup() -> Self {
59        Self {
60            transport_type: TransportType::Backup,
61            rtt_bias: 0,
62        }
63    }
64
65    /// Adds an RTT advantage to this transport, making it more preferred.
66    pub(crate) fn with_rtt_advantage(mut self, advantage: Duration) -> Self {
67        self.rtt_bias -= advantage.as_nanos() as i128;
68        self
69    }
70
71    /// Adds an RTT disadvantage to this transport, making it less preferred.
72    pub(crate) fn with_rtt_disadvantage(mut self, disadvantage: Duration) -> Self {
73        self.rtt_bias += disadvantage.as_nanos() as i128;
74        self
75    }
76}
77
78/// The default [`PathSelector`] used by iroh.
79///
80/// Sorts paths by `(transport_type, biased_rtt)` (primary tier wins, then lowest biased
81/// RTT).  Within the same tier, switching only happens once a candidate's biased RTT is
82/// at least 5ms better than the currently-selected path — this avoids flapping under
83/// jitter.  Across tiers, switching is immediate.
84///
85/// The biases are configured per [`AddrKind`].  Defaults: IPv4 and IPv6 are primary
86/// (IPv6 has a 3ms RTT advantage), Relay is backup, custom transports are primary with
87/// no advantage.
88#[derive(Debug, Clone)]
89pub(crate) struct BiasedRttPathSelector {
90    biases: Arc<FxHashMap<AddrKind, TransportBias>>,
91}
92
93impl Default for BiasedRttPathSelector {
94    fn default() -> Self {
95        let mut map = FxHashMap::default();
96        map.insert(AddrKind::IpV4, TransportBias::primary());
97        map.insert(
98            AddrKind::IpV6,
99            TransportBias::primary().with_rtt_advantage(IPV6_RTT_ADVANTAGE),
100        );
101        map.insert(AddrKind::Relay, TransportBias::backup());
102        Self {
103            biases: Arc::new(map),
104        }
105    }
106}
107
108impl BiasedRttPathSelector {
109    /// Returns a new selector with the given bias added or updated for `kind`.
110    pub(crate) fn with_bias(self, kind: AddrKind, bias: TransportBias) -> Self {
111        let mut map = (*self.biases).clone();
112        map.insert(kind, bias);
113        Self {
114            biases: Arc::new(map),
115        }
116    }
117
118    /// Looks up the bias for an address.  Defaults to primary with no RTT bias.
119    fn bias_for(&self, addr: &Addr) -> TransportBias {
120        self.biases
121            .get(&addr.addr_kind())
122            .copied()
123            .unwrap_or_else(TransportBias::primary)
124    }
125
126    /// Computes the sort key for a path: lower is better.
127    fn sort_key(&self, addr: &Addr, rtt: Duration) -> (TransportType, i128) {
128        let bias = self.bias_for(addr);
129        let biased_rtt = (rtt.as_nanos() as i128).saturating_add(bias.rtt_bias);
130        (bias.transport_type, biased_rtt)
131    }
132}
133
134impl PathSelector for BiasedRttPathSelector {
135    fn select(&self, ctx: &PathSelectionContext<'_>) -> PathSelection {
136        // Single pass: track the best candidate by sort key, and the best (lowest)
137        // sort key seen for the currently-selected address.  When the same address
138        // appears multiple times (one path per connection), `min` over `sort_key`
139        // naturally picks the lowest-RTT instance — no separate aggregation needed.
140        let current = ctx.current();
141        let mut best: Option<(PathSelectionData<'_>, (TransportType, i128))> = None;
142        let mut current_key: Option<(TransportType, i128)> = None;
143
144        tracing::debug!("dumping path RTTs");
145        for psd in ctx.paths() {
146            let addr = psd.remote_addr();
147            // Skip paths whose stats can't be read (e.g. closed concurrently with select).
148            let Some(stats) = psd.stats() else {
149                continue;
150            };
151            let rtt = stats.rtt;
152            tracing::debug!(?addr, ?rtt);
153            let key = self.sort_key(addr, rtt);
154
155            if Some(addr) == current && current_key.is_none_or(|c| key < c) {
156                current_key = Some(key);
157            }
158            if best.as_ref().is_none_or(|(_, b)| key < *b) {
159                best = Some((psd, key));
160            }
161        }
162
163        let mut selection = PathSelection::none();
164        let Some((best_psd, (best_tier, best_biased))) = best else {
165            return selection;
166        };
167
168        // If we have no current path or no data for it, switch to the best.
169        let Some((current_tier, current_biased)) = current_key else {
170            selection.set(&best_psd);
171            return selection;
172        };
173
174        if current_tier != best_tier {
175            // Always switch across tiers (e.g. relay -> primary).
176            selection.set(&best_psd);
177        } else if best_biased + RTT_SWITCHING_MIN.as_nanos() as i128 <= current_biased {
178            // For the same tier, only switch when biased RTT is meaningfully better.
179            selection.set(&best_psd);
180        }
181        selection
182    }
183}
