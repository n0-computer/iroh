8#[derive(Debug)]
9pub(super) struct AlwaysResolvesCert {
10    key: Arc<rustls::sign::CertifiedKey>,
11}

13/// Error for generating TLS configs.
14#[add_meta]
15#[derive(Error)]
16#[error(from_sources, std_sources)]
17#[non_exhaustive]
18pub(super) enum CreateConfigError {
19    /// Rustls configuration error
20    #[display("rustls error")]
21    Rustls { source: rustls::Error },
22}
23
24impl AlwaysResolvesCert {
25    pub(super) fn new(secret_key: &SecretKey) -> Result<Self, CreateConfigError> {
26        // Directly use the key
27        let client_private_key = secret_key
28            .as_signing_key()
29            .to_pkcs8_pem(LineEnding::default())
30            .expect("key is valid");
31
32        let client_private_key = PrivatePkcs8KeyDer::from_pem_slice(client_private_key.as_bytes())
33            .expect("cannot open private key file");
34        let client_private_key = rustls::crypto::ring::sign::any_eddsa_type(&client_private_key)?;
35
36        let client_public_key = client_private_key
37            .public_key()
38            .ok_or(rustls::Error::InconsistentKeys(
39                rustls::InconsistentKeys::Unknown,
40            ))
41            .expect("cannot load public key");
42        let client_public_key_as_cert = CertificateDer::from(client_public_key.to_vec());
43
44        let certified_key =
45            rustls::sign::CertifiedKey::new(vec![client_public_key_as_cert], client_private_key);
46
47        let key = Arc::new(certified_key);
48
49        Ok(Self { key })
50    }
51}
