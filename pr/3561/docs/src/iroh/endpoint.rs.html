166    /// Binds the magic endpoint.
167    pub async fn bind(self) -> Result<Endpoint, BindError> {
168        let mut rng = rand::rng();
169        let relay_map = self.relay_mode.relay_map();
170        let secret_key = self
171            .secret_key
172            .unwrap_or_else(move || SecretKey::generate(&mut rng));
173        let static_config = StaticConfig {
174            transport_config: Arc::new(self.transport_config),
175            tls_config: tls::TlsConfig::new(secret_key.clone(), self.max_tls_tickets),
176            keylog: self.keylog,
177        };
178        let server_config = static_config.create_server_config(self.alpn_protocols);
179
180        #[cfg(not(wasm_browser))]
181        let dns_resolver = self.dns_resolver.unwrap_or_default();
182
183        let metrics = EndpointMetrics::default();
184
185        let msock_opts = magicsock::Options {
186            addr_v4: self.addr_v4,
187            addr_v6: self.addr_v6,
188            secret_key,
189            relay_map,
190            discovery_user_data: self.discovery_user_data,
191            proxy_url: self.proxy_url,
192            #[cfg(not(wasm_browser))]
193            dns_resolver,
194            server_config,
195            #[cfg(any(test, feature = "test-utils"))]
196            insecure_skip_relay_cert_verify: self.insecure_skip_relay_cert_verify,
197            #[cfg(any(test, feature = "test-utils"))]
198            path_selection: self.path_selection,
199            metrics,
200        };
201
202        let msock = magicsock::MagicSock::spawn(msock_opts).await?;
203        trace!("created magicsock");
204        debug!(version = env!("CARGO_PKG_VERSION"), "iroh Endpoint created");
205
206        let metrics = msock.metrics.magicsock.clone();
207        let ep = Endpoint {
208            msock,
209            rtt_actor: Arc::new(rtt_actor::RttHandle::new(metrics)),
210            static_config: Arc::new(static_config),
211        };
212
213        // Add discovery mechanisms
214        for create_service in self.discovery {
215            let service = create_service.into_discovery(&ep)?;
216            ep.discovery().add_boxed(service);
217        }
218
219        Ok(ep)
220    }

449/// Controls an iroh endpoint, establishing connections with other endpoints.
450///
451/// This is the main API interface to create connections to, and accept connections from
452/// other iroh endpoints.  The connections are peer-to-peer and encrypted, a Relay server is
453/// used to make the connections reliable.  See the [crate docs] for a more detailed
454/// overview of iroh.
455///
456/// It is recommended to only create a single instance per application.  This ensures all
457/// the connections made share the same peer-to-peer connections to other iroh endpoints,
458/// while still remaining independent connections.  This will result in more optimal network
459/// behaviour.
460///
461/// The endpoint is created using the [`Builder`], which can be created using
462/// [`Endpoint::builder`].
463///
464/// Once an endpoint exists, new connections are typically created using the
465/// [`Endpoint::connect`] and [`Endpoint::accept`] methods.  Once established, the
466/// [`Connection`] gives access to most [QUIC] features.  Individual streams to send data to
467/// the peer are created using the [`Connection::open_bi`], [`Connection::accept_bi`],
468/// [`Connection::open_uni`] and [`Connection::open_bi`] functions.
469///
470/// Note that due to the light-weight properties of streams a stream will only be accepted
471/// once the initiating peer has sent some data on it.
472///
473/// [QUIC]: https://quicwg.org
474#[derive(Clone, Debug)]
475pub struct Endpoint {
476    /// Handle to the magicsocket/actor
477    msock: Handle,
478    /// Handle to the actor that resets the quinn RTT estimator
479    rtt_actor: Arc<rtt_actor::RttHandle>,
480    /// Configuration structs for quinn, holds the transport config, certificate setup, secret key etc.
481    static_config: Arc<StaticConfig>,
482}

602    /// Connects to a remote [`Endpoint`].
603    ///
604    /// A value that can be converted into a [`EndpointAddr`] is required. This can be either a
605    /// [`EndpointAddr`], a [`EndpointId`] or a [`iroh_base::ticket::EndpointTicket`].
606    ///
607    /// The [`EndpointAddr`] must contain the [`EndpointId`] to dial and may also contain a [`RelayUrl`]
608    /// and direct addresses. If direct addresses are provided, they will be used to try and
609    /// establish a direct connection without involving a relay server.
610    ///
611    /// If neither a [`RelayUrl`] or direct addresses are configured in the [`EndpointAddr`] it
612    /// may still be possible a connection can be established.  This depends on which, if any,
613    /// [`crate::discovery::Discovery`] services were configured using [`Builder::discovery`].  The discovery
614    /// service will also be used if the remote endpoint is not reachable on the provided direct
615    /// addresses and there is no [`RelayUrl`].
616    ///
617    /// If addresses or relay servers are neither provided nor can be discovered, the
618    /// connection attempt will fail with an error.
619    ///
620    /// The `alpn`, or application-level protocol identifier, is also required. The remote
621    /// endpoint must support this `alpn`, otherwise the connection attempt will fail with
622    /// an error.
623    ///
624    /// [`RelayUrl`]: crate::RelayUrl
625    pub async fn connect(
626        &self,
627        endpoint_addr: impl Into<EndpointAddr>,
628        alpn: &[u8],
629    ) -> Result<Connection, ConnectError> {
630        let endpoint_addr = endpoint_addr.into();
631        let remote = endpoint_addr.endpoint_id;
632        let connecting = self
633            .connect_with_opts(endpoint_addr, alpn, Default::default())
634            .await?;
635        let conn = connecting.await?;
636
637        debug!(
638            me = %self.id().fmt_short(),
639            remote = %remote.fmt_short(),
640            alpn = %String::from_utf8_lossy(alpn),
641            "Connection established."
642        );
643        Ok(conn)
644    }

648    /// Like [`Endpoint::connect`] (see also its docs for general details), but allows for a more
649    /// advanced connection setup with more customization in two aspects:
650    /// 1. The returned future resolves to a [`Connecting`], which can be further processed into
651    ///    a [`Connection`] by awaiting, or alternatively allows connecting with 0RTT via
652    ///    [`Connecting::into_0rtt`].
653    ///    **Note:** Please read the documentation for `into_0rtt` carefully to assess
654    ///    security concerns.
655    /// 2. The [`TransportConfig`] for the connection can be modified via the provided
656    ///    [`ConnectOptions`].
657    ///    **Note:** Please be aware that changing transport config settings may have adverse effects on
658    ///    establishing and maintaining direct connections.  Carefully test settings you use and
659    ///    consider this currently as still rather experimental.
660    #[instrument(name = "connect", skip_all, fields(
661        me = %self.id().fmt_short(),
662        remote = tracing::field::Empty,
663        alpn = String::from_utf8_lossy(alpn).to_string(),
664    ))]
665    pub async fn connect_with_opts(
666        &self,
667        endpoint_addr: impl Into<EndpointAddr>,
668        alpn: &[u8],
669        options: ConnectOptions,
670    ) -> Result<Connecting, ConnectWithOptsError> {
671        let endpoint_addr: EndpointAddr = endpoint_addr.into();
672        tracing::Span::current().record(
673            "remote",
674            tracing::field::display(endpoint_addr.endpoint_id.fmt_short()),
675        );
676
677        // Connecting to ourselves is not supported.
678        n0_error::ensure!(
679            endpoint_addr.endpoint_id != self.id(),
680            e!(ConnectWithOptsError::SelfConnect)
681        );
682
683        if !endpoint_addr.is_empty() {
684            self.add_endpoint_addr(endpoint_addr.clone(), Source::App)
685                .map_err(|err| e!(ConnectWithOptsError::AddEndpointAddr { source: err }))?;
686        }
687        let endpoint_id = endpoint_addr.endpoint_id;
688        let direct_addresses = endpoint_addr.direct_addresses.clone();
689        let relay_url = endpoint_addr.relay_url.clone();
690
691        // Get the mapped IPv6 address from the magic socket. Quinn will connect to this
692        // address.  Start discovery for this endpoint if it's enabled and we have no valid or
693        // verified address information for this endpoint.  Dropping the discovery cancels any
694        // still running task.
695        let (mapped_addr, _discovery_drop_guard) = self
696            .get_mapping_addr_and_maybe_start_discovery(endpoint_addr)
697            .await
698            .map_err(|err| e!(ConnectWithOptsError::NoAddress { source: err }))?;
699
700        let transport_config = options
701            .transport_config
702            .unwrap_or(self.static_config.transport_config.clone());
703
704        // Start connecting via quinn. This will time out after 10 seconds if no reachable
705        // address is available.
706
707        debug!(
708            ?mapped_addr,
709            ?direct_addresses,
710            ?relay_url,
711            "Attempting connection..."
712        );
713        let client_config = {
714            let mut alpn_protocols = vec![alpn.to_vec()];
715            alpn_protocols.extend(options.additional_alpns);
716            let quic_client_config = self
717                .static_config
718                .tls_config
719                .make_client_config(alpn_protocols, self.static_config.keylog);
720            let mut client_config = quinn::ClientConfig::new(Arc::new(quic_client_config));
721            client_config.transport_config(transport_config);
722            client_config
723        };
724
725        let server_name = &tls::name::encode(endpoint_id);
726        let connect = self
727            .msock
728            .endpoint()
729            .connect_with(
730                client_config,
731                mapped_addr.private_socket_addr(),
732                server_name,
733            )
734            .map_err(|err| e!(ConnectWithOptsError::Quinn { source: err }))?;
735
736        Ok(Connecting {
737            inner: connect,
738            ep: self.clone(),
739            remote_endpoint_id: Some(endpoint_id),
740            _discovery_drop_guard,
741        })
742    }

825    /// Returns a [`Watcher`] for the current [`EndpointAddr`] for this endpoint.
826    ///
827    /// The observed [`EndpointAddr`] will have the current [`RelayUrl`] and direct addresses.
828    ///
829    /// ```no_run
830    /// # async fn wrapper() -> n0_error::Result<()> {
831    /// use iroh::{Endpoint, Watcher};
832    ///
833    /// let endpoint = Endpoint::builder()
834    ///     .alpns(vec![b"my-alpn".to_vec()])
835    ///     .bind()
836    ///     .await?;
837    /// let endpoint_addr = endpoint.watch_addr().get();
838    /// # let _ = endpoint_addr;
839    /// # Ok(())
840    /// # }
841    /// ```
842    ///
843    /// The [`Endpoint::online`] method can be used as a convenience method to
844    /// understand if the endpoint has ever been considered "online". But after
845    /// that initial call to [`Endpoint::online`], to understand if your
846    /// endpoint is no longer able to be connected to by endpoints outside
847    /// of the private or local network, watch for changes in it's [`EndpointAddr`].
848    /// If the `relay_url` is `None` or if there are no `direct_addresses` in
849    /// the [`EndpointAddr`], you may not be dialable by other endpoints on the internet.
850    ///
851    ///
852    /// The `EndpointAddr` will change as:
853    /// - network conditions change
854    /// - the endpoint connects to a relay server
855    /// - the endpoint changes its preferred relay server
856    /// - more addresses are discovered for this endpoint
857    ///
858    /// [`RelayUrl`]: crate::RelayUrl
859    #[cfg(not(wasm_browser))]
860    pub fn watch_addr(&self) -> impl n0_watcher::Watcher<Value = EndpointAddr> + use<> {
861        let watch_addrs = self.msock.direct_addresses();
862        let watch_relay = self.msock.home_relay();
863        let endpoint_id = self.id();
864
865        watch_addrs
866            .or(watch_relay)
867            .map(move |(addrs, mut relays)| {
868                debug_assert!(!addrs.is_empty(), "direct addresses must never be empty");
869
870                EndpointAddr::from_parts(
871                    endpoint_id,
872                    relays.pop(),
873                    addrs.into_iter().map(|x| x.addr),
874                )
875            })
876            .expect("watchable is alive - cannot be disconnected yet")
877    }

881    /// When compiled to Wasm, this function returns a watcher that initializes
882    /// with a [`EndpointAddr`] that only contains a relay URL, but no direct addresses,

1016    /// Returns metrics collected for this endpoint.
1017    ///
1018    /// The endpoint internally collects various metrics about its operation.
1019    /// The returned [`EndpointMetrics`] struct contains all of these metrics.
1020    ///
1021    /// You can access individual metrics directly by using the public fields:
1022    /// ```rust
1023    /// # use std::collections::BTreeMap;
1024    /// # use iroh::endpoint::Endpoint;
1025    /// # async fn wrapper() -> n0_error::Result<()> {
1026    /// let endpoint = Endpoint::bind().await?;
1027    /// assert_eq!(endpoint.metrics().magicsock.recv_datagrams.get(), 0);
1028    /// # Ok(())
1029    /// # }
1030    /// ```
1031    ///
1032    /// [`EndpointMetrics`] implements [`MetricsGroupSet`], and each field
1033    /// implements [`MetricsGroup`]. These traits provide methods to iterate over
1034    /// the groups in the set, and over the individual metrics in each group, without having
1035    /// to access each field manually. With these methods, it is straightforward to collect
1036    /// all metrics into a map or push their values to a metrics collector.
1037    ///
1038    /// For example, the following snippet collects all metrics into a map:
1039    /// ```rust
1040    /// # use std::collections::BTreeMap;
1041    /// # use iroh_metrics::{Metric, MetricsGroup, MetricValue, MetricsGroupSet};
1042    /// # use iroh::endpoint::Endpoint;
1043    /// # async fn wrapper() -> n0_error::Result<()> {
1044    /// let endpoint = Endpoint::bind().await?;
1045    /// let metrics: BTreeMap<String, MetricValue> = endpoint
1046    ///     .metrics()
1047    ///     .iter()
1048    ///     .map(|(group, metric)| {
1049    ///         let name = [group, metric.name()].join(":");
1050    ///         (name, metric.value())
1051    ///     })
1052    ///     .collect();
1053    ///
1054    /// assert_eq!(metrics["magicsock:recv_datagrams"], MetricValue::Counter(0));
1055    /// # Ok(())
1056    /// # }
1057    /// ```
1058    ///
1059    /// The metrics can also be encoded into the OpenMetrics text format, as used by Prometheus.
1060    /// To do so, use the [`iroh_metrics::Registry`], add the endpoint metrics to the
1061    /// registry with [`Registry::register_all`], and encode the metrics to a string with
1062    /// [`encode_openmetrics_to_string`]:
1063    /// ```rust
1064    /// # use iroh_metrics::{Registry, MetricsSource};
1065    /// # use iroh::endpoint::Endpoint;
1066    /// # async fn wrapper() -> n0_error::Result<()> {
1067    /// let endpoint = Endpoint::bind().await?;
1068    /// let mut registry = Registry::default();
1069    /// registry.register_all(endpoint.metrics());
1070    /// let s = registry.encode_openmetrics_to_string()?;
1071    /// assert!(s.contains(r#"TYPE magicsock_recv_datagrams counter"#));
1072    /// assert!(s.contains(r#"magicsock_recv_datagrams_total 0"#));
1073    /// # Ok(())
1074    /// # }
1075    /// ```
1076    ///
1077    /// Through a registry, you can also add labels or prefixes to metrics with
1078    /// [`Registry::sub_registry_with_label`] or [`Registry::sub_registry_with_prefix`].
1079    /// Furthermore, [`iroh_metrics::service`] provides functions to easily start services
1080    /// to serve the metrics with a HTTP server, dump them to a file, or push them
1081    /// to a Prometheus gateway.
1082    ///
1083    /// For example, the following snippet launches an HTTP server that serves the metrics in the
1084    /// OpenMetrics text format:
1085    /// ```no_run
1086    /// # use std::{sync::{Arc, RwLock}, time::Duration};
1087    /// # use iroh_metrics::{Registry, MetricsSource};
1088    /// # use iroh::endpoint::Endpoint;
1089    /// # use n0_error::StackResultExt;
1090    /// # async fn wrapper() -> n0_error::Result<()> {
1091    /// // Create a registry, wrapped in a read-write lock so that we can register and serve
1092    /// // the metrics independently.
1093    /// let registry = Arc::new(RwLock::new(Registry::default()));
1094    /// // Spawn a task to serve the metrics on an OpenMetrics HTTP endpoint.
1095    /// let metrics_task = tokio::task::spawn({
1096    ///     let registry = registry.clone();
1097    ///     async move {
1098    ///         let addr = "0.0.0.0:9100".parse().unwrap();
1099    ///         iroh_metrics::service::start_metrics_server(addr, registry).await
1100    ///     }
1101    /// });

1103    /// // Spawn an endpoint and add the metrics to the registry.
1104    /// let endpoint = Endpoint::bind().await?;
1105    /// registry.write().unwrap().register_all(endpoint.metrics());

1107    /// // Wait for the metrics server to bind, then fetch the metrics via HTTP.
1108    /// tokio::time::sleep(Duration::from_millis(500));
1109    /// let res = reqwest::get("http://localhost:9100/metrics")
1110    ///     .await
1111    ///     .context("get")?
1112    ///     .text()
1113    ///     .await
1114    ///     .context("text")?;
1115    ///
1116    /// assert!(res.contains(r#"TYPE magicsock_recv_datagrams counter"#));
1117    /// assert!(res.contains(r#"magicsock_recv_datagrams_total 0"#));
1118    /// # metrics_task.abort();
1119    /// # Ok(())
1120    /// # }
1121    /// ```
1122    ///
1123    /// [`Registry`]: iroh_metrics::Registry
1124    /// [`Registry::register_all`]: iroh_metrics::Registry::register_all
1125    /// [`Registry::sub_registry_with_label`]: iroh_metrics::Registry::sub_registry_with_label

1166    /// Closes the QUIC endpoint and the magic socket.
1167    ///
1168    /// This will close any remaining open [`Connection`]s with an error code
1169    /// of `0` and an empty reason.  Though it is best practice to close those
1170    /// explicitly before with a custom error code and reason.
1171    ///
1172    /// It will then make a best effort to wait for all close notifications to be
1173    /// acknowledged by the peers, re-transmitting them if needed. This ensures the
1174    /// peers are aware of the closed connections instead of having to wait for a timeout
1175    /// on the connection. Once all connections are closed or timed out, the future
1176    /// finishes.
1177    ///
1178    /// The maximum time-out that this future will wait for depends on QUIC transport
1179    /// configurations of non-drained connections at the time of calling, and their current
1180    /// estimates of round trip time. With default parameters and a conservative estimate
1181    /// of round trip time, this call's future should take 3 seconds to resolve in cases of
1182    /// bad connectivity or failed connections. In the usual case, this call's future should
1183    /// return much more quickly.
1184    ///
1185    /// It is highly recommended you *do* wait for this close call to finish, if possible.
1186    /// Not doing so will make connections that were still open while closing the endpoint
1187    /// time out on the remote end. Thus remote ends will assume connections to have failed
1188    /// even if all application data was transmitted successfully.
1189    ///
1190    /// Note: Someone used to closing TCP sockets might wonder why it is necessary to wait
1191    /// for timeouts when closing QUIC endpoints, while they don't have to do this for TCP
1192    /// sockets. This is due to QUIC and its acknowledgments being implemented in user-land,
1193    /// while TCP sockets usually get closed and drained by the operating system in the
1194    /// kernel during the "Time-Wait" period of the TCP socket.
1195    ///
1196    /// Be aware however that the underlying UDP sockets are only closed once all clones of
1197    /// the the respective [`Endpoint`] are dropped.
1198    pub async fn close(&self) {
1199        if self.is_closed() {
1200            return;
1201        }
1202
1203        tracing::debug!("Connections closed");
1204        self.msock.close().await;
1205    }

1214    /// Checks if the given `EndpointId` needs discovery.
1215    pub(crate) fn needs_discovery(&self, endpoint_id: EndpointId, max_age: Duration) -> bool {
1216        match self.msock.remote_info(endpoint_id) {
1217            // No info means no path to endpoint -> start discovery.
1218            None => true,
1219            Some(info) => {
1220                match (
1221                    info.last_received(),
1222                    info.relay_url.as_ref().and_then(|r| r.last_alive),
1223                ) {
1224                    // No path to endpoint -> start discovery.
1225                    (None, None) => true,
1226                    // If we haven't received on direct addresses or the relay for MAX_AGE,
1227                    // start discovery.
1228                    (Some(elapsed), Some(elapsed_relay)) => {
1229                        elapsed > max_age && elapsed_relay > max_age
1230                    }
1231                    (Some(elapsed), _) | (_, Some(elapsed)) => elapsed > max_age,
1232                }
1233            }
1234        }
1235    }

1237    /// Return the quic mapped address for this `endpoint_id` and possibly start discovery
1238    /// services if discovery is enabled on this magic endpoint.
1239    ///
1240    /// This will launch discovery in all cases except if:
1241    /// 1) we do not have discovery enabled
1242    /// 2) we have discovery enabled, but already have at least one verified, unexpired
1243    ///    addresses for this `endpoint_id`
1244    ///
1245    /// # Errors
1246    ///
1247    /// This method may fail if we have no way of dialing the endpoint. This can occur if
1248    /// we were given no dialing information in the [`EndpointAddr`] and no discovery
1249    /// services were configured or if discovery failed to fetch any dialing information.
1250    async fn get_mapping_addr_and_maybe_start_discovery(
1251        &self,
1252        endpoint_addr: EndpointAddr,
1253    ) -> Result<(EndpointIdMappedAddr, Option<DiscoveryTask>), GetMappingAddressError> {
1254        let endpoint_id = endpoint_addr.endpoint_id;
1255
1256        // Only return a mapped addr if we have some way of dialing this endpoint, in other
1257        // words, we have either a relay URL or at least one direct address.
1258        let addr = if self.msock.has_send_address(endpoint_id) {
1259            self.msock.get_mapping_addr(endpoint_id)
1260        } else {
1261            None
1262        };
1263        match addr {
1264            Some(addr) => {
1265                // We have some way of dialing this endpoint, but that doesn't actually mean
1266                // we can actually connect to any of these addresses.
1267                // Therefore, we will invoke the discovery service if we haven't received from the
1268                // endpoint on any of the existing paths recently.
1269                // If the user provided addresses in this connect call, we will add a delay
1270                // followed by a recheck before starting the discovery, to give the magicsocket a
1271                // chance to test the newly provided addresses.
1272                let delay = (!endpoint_addr.is_empty()).then_some(DISCOVERY_WAIT_PERIOD);
1273                let discovery = DiscoveryTask::maybe_start_after_delay(self, endpoint_id, delay)
1274                    .ok()
1275                    .flatten();
1276                Ok((addr, discovery))
1277            }
1278
1279            None => {
1280                // We have no known addresses or relay URLs for this endpoint.
1281                // So, we start a discovery task and wait for the first result to arrive, and
1282                // only then continue, because otherwise we wouldn't have any
1283                // path to the remote endpoint.
1284                let res = DiscoveryTask::start(self.clone(), endpoint_id);
1285                let mut discovery =
1286                    res.map_err(|source| e!(GetMappingAddressError::DiscoveryStart { source }))?;
1287                discovery
1288                    .first_arrived()
1289                    .await
1290                    .map_err(|source| e!(GetMappingAddressError::Discover { source }))?;
1291                if let Some(addr) = self.msock.get_mapping_addr(endpoint_id) {
1292                    Ok((addr, Some(discovery)))
1293                } else {
1294                    Err!(GetMappingAddressError::NoAddress)
1295                }
1296            }
1297        }
1298    }

1538impl Connecting {
1539    /// Converts this [`Connecting`] into a 0-RTT or 0.5-RTT connection at the cost of weakened
1540    /// security.
1541    ///
1542    /// Returns `Ok` immediately if the local endpoint is able to attempt sending 0/0.5-RTT data.
1543    /// If so, the returned [`Connection`] can be used to send application data without waiting for
1544    /// the rest of the handshake to complete, at the cost of weakened cryptographic security
1545    /// guarantees. The returned [`ZeroRttAccepted`] future resolves when the handshake does
1546    /// complete, at which point subsequently opened streams and written data will have full
1547    /// cryptographic protection.
1548    ///
1549    /// Once the [`ZeroRttAccepted`] future completed, a full handshake has been carried through
1550    /// and any data sent and any streams opened on the [`Connection`] will operate with the same
1551    /// security as on normal 1-RTT connections.
1552    ///
1553    /// ## Outgoing
1554    ///
1555    /// For outgoing connections, the initial attempt to convert to a [`Connection`] which sends
1556    /// 0-RTT data will attempt to resume a previous TLS session. However, **the remote endpoint
1557    /// may not actually _accept_ the 0-RTT data**--yet still accept the connection attempt in
1558    /// general. This possibility is conveyed through the [`ZeroRttAccepted`] future--when the
1559    /// handshake completes, it resolves to true if the 0-RTT data was accepted and false if it was
1560    /// rejected. If it was rejected, the existence of streams opened and other application data
1561    /// sent prior to the handshake completing will not be conveyed to the remote application, and
1562    /// local operations on them will return `ZeroRttRejected` errors.
1563    ///
1564    /// A server may reject 0-RTT data at its discretion, but accepting 0-RTT data requires the
1565    /// relevant resumption state to be stored in the server, which servers may limit or lose for
1566    /// various reasons including not persisting resumption state across server restarts.
1567    ///
1568    /// ## Incoming
1569    ///
1570    /// For incoming connections, conversion to 0.5-RTT will always fully succeed. `into_0rtt` will
1571    /// always return `Ok` and the [`ZeroRttAccepted`] will always resolve to true.
1572    ///
1573    /// ## Security
1574    ///
1575    /// On outgoing connections, this enables transmission of 0-RTT data, which is vulnerable to
1576    /// replay attacks, and should therefore never invoke non-idempotent operations.
1577    ///
1578    /// On incoming connections, this enables transmission of 0.5-RTT data, which may be sent
1579    /// before TLS client authentication has occurred, and should therefore not be used to send
1580    /// data for which client authentication is being used.
1581    ///
1582    /// You can use [`RecvStream::is_0rtt`] to check whether a stream has been opened in 0-RTT
1583    /// and thus whether parts of the stream are operating under this reduced security level.
1584    #[allow(clippy::result_large_err)]
1585    pub fn into_0rtt(self) -> Result<(Connection, ZeroRttAccepted), Self> {
1586        match self.inner.into_0rtt() {
1587            Ok((inner, zrtt_accepted)) => {
1588                let conn = Connection { inner };
1589                let zrtt_accepted = ZeroRttAccepted {
1590                    inner: zrtt_accepted,
1591                    _discovery_drop_guard: self._discovery_drop_guard,
1592                };
1593                // This call is why `self.remote_endpoint_id` was introduced.
1594                // When we `Connecting::into_0rtt`, then we don't yet have `handshake_data`
1595                // in our `Connection`, thus `try_send_rtt_msg` won't be able to pick up
1596                // `Connection::remote_endpoint_id`.
1597                // Instead, we provide `self.remote_endpoint_id` here - we know it in advance,
1598                // after all.
1599                try_send_rtt_msg(&conn, &self.ep, self.remote_endpoint_id);
1600                Ok((conn, zrtt_accepted))
1601            }
1602            Err(inner) => Err(Self {
1603                inner,
1604                ep: self.ep,
1605                remote_endpoint_id: self.remote_endpoint_id,
1606                _discovery_drop_guard: self._discovery_drop_guard,
1607            }),
1608        }
1609    }

1616    /// Extracts the ALPN protocol from the peer's handshake data.
1617    pub async fn alpn(&mut self) -> Result<Vec<u8>, AlpnError> {
1618        let data = self.handshake_data().await?;
1619        match data.downcast::<quinn::crypto::rustls::HandshakeData>() {
1620            Ok(data) => match data.protocol {
1621                Some(protocol) => Ok(protocol),
1622                None => Err!(AlpnError::Unavailable),
1623            },
1624            Err(_) => Err!(AlpnError::UnknownHandshake),
1625        }
1626    }

1762    /// Closes the connection immediately.

1764    /// Pending operations will fail immediately with [`ConnectionError::LocallyClosed`]. No
1765    /// more data is sent to the peer and the peer may drop buffered data upon receiving the
1766    /// CONNECTION_CLOSE frame.
1767    ///
1768    /// `error_code` and `reason` are not interpreted, and are provided directly to the
1769    /// peer.
1770    ///
1771    /// `reason` will be truncated to fit in a single packet with overhead; to improve odds
1772    /// that it is preserved in full, it should be kept under 1KiB.
1773    ///
1774    /// # Gracefully closing a connection
1775    ///
1776    /// Only the peer last receiving application data can be certain that all data is
1777    /// delivered. The only reliable action it can then take is to close the connection,
1778    /// potentially with a custom error code. The delivery of the final CONNECTION_CLOSE
1779    /// frame is very likely if both endpoints stay online long enough, calling
1780    /// [`Endpoint::close`] will wait to provide sufficient time. Otherwise, the remote peer
1781    /// will time out the connection, provided that the idle timeout is not disabled.
1782    ///
1783    /// The sending side can not guarantee all stream data is delivered to the remote
1784    /// application. It only knows the data is delivered to the QUIC stack of the remote
1785    /// endpoint. Once the local side sends a CONNECTION_CLOSE frame in response to calling
1786    /// [`close`] the remote endpoint may drop any data it received but is as yet
1787    /// undelivered to the application, including data that was acknowledged as received to
1788    /// the local endpoint.
1789    ///
1790    /// [`close`]: Connection::close

1994/// Try send a message to the rtt-actor.
1995///
1996/// If we can't notify the actor that will impact performance a little, but we can still
1997/// function.
1998fn try_send_rtt_msg(
1999    conn: &Connection,
2000    magic_ep: &Endpoint,
2001    remote_endpoint_id: Option<EndpointId>,
2002) {
2003    // If we can't notify the rtt-actor that's not great but not critical.
2004    let Some(endpoint_id) = remote_endpoint_id.or_else(|| conn.remote_id().ok()) else {
2005        warn!(?conn, "failed to get remote endpoint id");
2006        return;
2007    };
2008    let Some(conn_type_changes) = magic_ep.conn_type(endpoint_id) else {
2009        warn!(?conn, "failed to create conn_type stream");
2010        return;
2011    };
2012    let rtt_msg = RttMessage::NewConnection {
2013        connection: conn.inner.weak_handle(),
2014        conn_type_changes: conn_type_changes.stream(),
2015        endpoint_id,
2016    };
2017    if let Err(err) = magic_ep.rtt_actor.msg_tx.try_send(rtt_msg) {
2018        warn!(?conn, "rtt-actor not reachable: {err:#}");
2019    }
2020}

2752                                }

2754                        }

2782                    }
2783
2784                    // Stay alive until the other side closes the connection.
2785                    conn.closed().await;
2786                    tracing::trace!("Connection closed, ready for next");
2787                }
2788                tracing::trace!("Server accept loop exiting");
