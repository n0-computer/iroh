1//! Contains the discovery message types.
2//!
3//! A discovery message is:
4//!
5//! Header:
6//!
7//! ```ignore
8//! magic:            [u8; 6]  // ‚ÄúTSüí¨‚Äù (0x54 53 f0 9f 92 ac)
9//! sender_disco_pub: [u8; 32] // nacl public key
10//! nonce:            [u8; 24]
11//! ````
12//! The recipient then decrypts the bytes following (the nacl secretbox)
13//! and then the inner payload structure is:
14//!
15//! ```ignore
16//! message_type:    u8   // (the MessageType constants below)
17//! message_version: u8   // (0 for now; but always ignore bytes at the end)
18//! message_payload: &[u8]
19//! ```
20

36// TODO: custom magicn
37/// The 6 byte header of all discovery messages.
38pub const MAGIC: &str = "TSüí¨"; // 6 bytes: 0x54 53 f0 9f 92 ac
39pub const MAGIC_LEN: usize = MAGIC.len();
40
41/// Current Version.
42const V0: u8 = 0;
43
44pub(crate) const KEY_LEN: usize = 32;
45const TX_LEN: usize = 12;
46
47// Sizes for the inner message structure.
48
49/// Header: Type | Version
50const HEADER_LEN: usize = 2;
51
52const PING_LEN: usize = TX_LEN + iroh_base::PublicKey::LENGTH;
53const EP_LENGTH: usize = 16 + 2; // 16 byte IP address + 2 byte port

55#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
56#[repr(u8)]
57pub enum MessageType {
58    Ping = 0x01,
59    Pong = 0x02,
60    CallMeMaybe = 0x03,
61}
62
63impl TryFrom<u8> for MessageType {
64    type Error = u8;
65
66    fn try_from(value: u8) -> std::result::Result<Self, Self::Error> {
67        match value {
68            0x01 => Ok(MessageType::Ping),
69            0x02 => Ok(MessageType::Pong),
70            0x03 => Ok(MessageType::CallMeMaybe),
71            _ => Err(value),
72        }
73    }
74}

76const MESSAGE_HEADER_LEN: usize = MAGIC_LEN + KEY_LEN;
77
78pub fn encode_message(sender: &PublicKey, seal: Vec<u8>) -> Vec<u8> {
79    let mut out = Vec::with_capacity(MESSAGE_HEADER_LEN);
80    out.extend_from_slice(MAGIC.as_bytes());
81    out.extend_from_slice(sender.as_bytes());
82    out.extend(seal);
83
84    out
85}

87/// Reports whether p looks like it's a packet containing an encrypted disco message.
88pub fn looks_like_disco_wrapper(p: &[u8]) -> bool {
89    if p.len() < MESSAGE_HEADER_LEN {
90        return false;
91    }
92
93    &p[..MAGIC_LEN] == MAGIC.as_bytes()
94}

96/// If `p` looks like a disco message it returns the slice of `p` that represents the disco public key source,
97/// and the part that is the box.
98pub fn source_and_box(p: &[u8]) -> Option<(PublicKey, &[u8])> {
99    if !looks_like_disco_wrapper(p) {
100        return None;
101    }
102
103    let source = &p[MAGIC_LEN..MAGIC_LEN + KEY_LEN];
104    let sender = PublicKey::try_from(source).ok()?;
105    let sealed_box = &p[MAGIC_LEN + KEY_LEN..];
106    Some((sender, sealed_box))
107}

203/// Message sent only over the relay to request that the recipient try
204/// to open up a magicsock path back to the sender.
205///
206/// The sender should've already sent UDP packets to the peer to open
207/// up the stateful firewall mappings inbound.
208///
209/// The recipient may choose to not open a path back, if it's already happy with its path.
210/// But usually it will.
211#[derive(Debug, Clone, PartialEq, Eq)]
212pub struct CallMeMaybe {
213    /// What the peer believes its endpoints are.
214    pub my_numbers: Vec<SocketAddr>,
215}

217impl Ping {
218    fn from_bytes(p: &[u8]) -> Result<Self, ParseError> {
219        // Deliberately lax on longer-than-expected messages, for future compatibility.
220        n0_error::ensure!(p.len() >= PING_LEN, e!(ParseError::TooShort));
221        let tx_id: [u8; TX_LEN] = p[..TX_LEN].try_into().expect("length checked");
222        let raw_key = &p[TX_LEN..TX_LEN + iroh_base::PublicKey::LENGTH];
223        let endpoint_key =
224            PublicKey::try_from(raw_key).map_err(|_| e!(ParseError::InvalidEncoding))?;
225        let tx_id = TransactionId::from(tx_id);
226
227        Ok(Ping {
228            tx_id,
229            endpoint_key,
230        })
231    }
232
233    fn as_bytes(&self) -> Vec<u8> {
234        let header = msg_header(MessageType::Ping, V0);
235        let mut out = vec![0u8; PING_LEN + HEADER_LEN];
236
237        out[..HEADER_LEN].copy_from_slice(&header);
238        out[HEADER_LEN..HEADER_LEN + TX_LEN].copy_from_slice(&self.tx_id);
239        out[HEADER_LEN + TX_LEN..].copy_from_slice(self.endpoint_key.as_ref());
240
241        out
242    }
243}

245#[allow(missing_docs)]
246#[add_meta]
247#[derive(Error)]
248#[non_exhaustive]
249pub enum ParseError {
250    #[display("message is too short")]
251    TooShort,
252    #[display("invalid encoding")]
253    InvalidEncoding,
254    #[display("unknown format")]
255    UnknownFormat,
256}

258fn send_addr_from_bytes(p: &[u8]) -> Result<SendAddr, ParseError> {
259    n0_error::ensure!(p.len() > 2, e!(ParseError::TooShort));
260    match p[0] {
261        0u8 => {
262            let bytes: [u8; EP_LENGTH] = p[1..].try_into().map_err(|_| e!(ParseError::TooShort))?;
263            let addr = socket_addr_from_bytes(bytes);
264            Ok(SendAddr::Udp(addr))
265        }
266        1u8 => {
267            let s = std::str::from_utf8(&p[1..]).map_err(|_| e!(ParseError::InvalidEncoding))?;
268            let u: Url = s.parse().map_err(|_| e!(ParseError::InvalidEncoding))?;
269            Ok(SendAddr::Relay(u.into()))
270        }
271        _ => Err!(ParseError::UnknownFormat),
272    }
273}

275fn send_addr_to_vec(addr: &SendAddr) -> Vec<u8> {
276    match addr {
277        SendAddr::Relay(url) => {
278            let mut out = vec![1u8];
279            out.extend_from_slice(url.to_string().as_bytes());
280            out
281        }
282        SendAddr::Udp(ip) => {
283            let mut out = vec![0u8];
284            out.extend_from_slice(&socket_addr_as_bytes(ip));
285            out
286        }
287    }
288}

290// Assumes p.len() == EP_LENGTH
291fn socket_addr_from_bytes(p: [u8; EP_LENGTH]) -> SocketAddr {
292    debug_assert_eq!(EP_LENGTH, 16 + 2);
293
294    let raw_src_ip: [u8; 16] = p[..16].try_into().expect("array long enough");
295    let raw_port: [u8; 2] = p[16..].try_into().expect("array long enough");
296
297    let src_ip = IpAddr::from(raw_src_ip).to_canonical();
298    let src_port = u16::from_le_bytes(raw_port);
299
300    SocketAddr::new(src_ip, src_port)
301}
302
303fn socket_addr_as_bytes(addr: &SocketAddr) -> [u8; EP_LENGTH] {
304    let mut out = [0u8; EP_LENGTH];
305    let ipv6 = match addr.ip() {
306        IpAddr::V4(v4) => v4.to_ipv6_mapped(),
307        IpAddr::V6(v6) => v6,
308    };
309    out[..16].copy_from_slice(&ipv6.octets());
310    out[16..].copy_from_slice(&addr.port().to_le_bytes());
311
312    out
313}

315impl Pong {
316    fn from_bytes(p: &[u8]) -> Result<Self, ParseError> {
317        let tx_id: [u8; TX_LEN] = p[..TX_LEN]
318            .try_into()
319            .map_err(|_| e!(ParseError::TooShort))?;
320
321        let tx_id = TransactionId::from(tx_id);
322        let src = send_addr_from_bytes(&p[TX_LEN..])?;
323
324        Ok(Pong {
325            tx_id,
326            ping_observed_addr: src,
327        })
328    }
329
330    fn as_bytes(&self) -> Vec<u8> {
331        let header = msg_header(MessageType::Pong, V0);
332        let mut out = header.to_vec();
333        out.extend_from_slice(&self.tx_id);
334
335        let src_bytes = send_addr_to_vec(&self.ping_observed_addr);
336        out.extend(src_bytes);
337        out
338    }
339}

341impl CallMeMaybe {
342    fn from_bytes(p: &[u8]) -> Result<Self, ParseError> {
343        n0_error::ensure!(p.len() % EP_LENGTH == 0, e!(ParseError::InvalidEncoding));
344
345        let num_entries = p.len() / EP_LENGTH;
346        let mut m = CallMeMaybe {
347            my_numbers: Vec::with_capacity(num_entries),
348        };
349
350        for chunk in p.chunks_exact(EP_LENGTH) {
351            let bytes: [u8; EP_LENGTH] = chunk
352                .try_into()
353                .map_err(|_| e!(ParseError::InvalidEncoding))?;
354            let src = socket_addr_from_bytes(bytes);
355            m.my_numbers.push(src);
356        }
357
358        Ok(m)
359    }
360
361    fn as_bytes(&self) -> Vec<u8> {
362        let header = msg_header(MessageType::CallMeMaybe, V0);
363        let mut out = vec![0u8; HEADER_LEN + self.my_numbers.len() * EP_LENGTH];
364        out[..HEADER_LEN].copy_from_slice(&header);
365
366        for (m, chunk) in self
367            .my_numbers
368            .iter()
369            .zip(out[HEADER_LEN..].chunks_exact_mut(EP_LENGTH))
370        {
371            let raw = socket_addr_as_bytes(m);
372            chunk.copy_from_slice(&raw);
373        }
374
375        out
376    }
377}

379impl Message {
380    /// Parses the encrypted part of the message from inside the nacl secretbox.
381    pub fn from_bytes(p: &[u8]) -> Result<Self, ParseError> {
382        n0_error::ensure!(p.len() >= 2, e!(ParseError::TooShort));
383
384        let t = MessageType::try_from(p[0]).map_err(|_| e!(ParseError::UnknownFormat))?;
385        let version = p[1];
386        n0_error::ensure!(version == V0, e!(ParseError::UnknownFormat));
387
388        let p = &p[2..];
389        match t {
390            MessageType::Ping => {
391                let ping = Ping::from_bytes(p)?;
392                Ok(Message::Ping(ping))
393            }
394            MessageType::Pong => {
395                let pong = Pong::from_bytes(p)?;
396                Ok(Message::Pong(pong))
397            }
398            MessageType::CallMeMaybe => {
399                let cm = CallMeMaybe::from_bytes(p)?;
400                Ok(Message::CallMeMaybe(cm))
401            }
402        }
403    }
404
405    /// Serialize this message to bytes.
406    pub fn as_bytes(&self) -> Vec<u8> {
407        match self {
408            Message::Ping(ping) => ping.as_bytes(),
409            Message::Pong(pong) => pong.as_bytes(),
410            Message::CallMeMaybe(cm) => cm.as_bytes(),
411        }
412    }
413}

435const TRANSACTION_ID_SIZE: usize = 12;
436
437/// The transaction ID is a 96-bit identifier
438///
439/// It is used to uniquely identify STUN transactions.
440/// It primarily serves to correlate requests with responses,
441/// though it also plays a small role in helping to prevent
442/// certain types of attacks. The server also uses the transaction ID as
443/// a key to identify each transaction uniquely across all clients.
444#[derive(Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]
445pub(crate) struct TransactionId([u8; TRANSACTION_ID_SIZE]);
446
447impl fmt::Debug for TransactionId {
448    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
449        write!(f, "TransactionId(0x")?;
450        fmt_transcation_id(self.as_ref(), f)
451    }
452}
453
454impl fmt::Display for TransactionId {
455    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
456        write!(f, "transaction id (0x")?;
457        fmt_transcation_id(self.as_ref(), f)
458    }
459}
460
461fn fmt_transcation_id(bytes: &[u8], f: &mut fmt::Formatter) -> fmt::Result {
462    for byte in bytes {
463        write!(f, "{:02X}", byte)?;
464    }
465    write!(f, ")")
466}

592    #[test]
593    fn test_extraction() {
594        let mut rng = rand_chacha::ChaCha8Rng::seed_from_u64(0u64);
595        let sender_key = SecretKey::generate(&mut rng);
596        let recv_key = SecretKey::generate(&mut rng);
597
598        let msg = Message::Ping(Ping {
599            tx_id: TransactionId::default(),
600            endpoint_key: sender_key.public(),
601        });
602
603        let sender_secret = secret_ed_box(&sender_key);
604        let shared = SharedSecret::new(&sender_secret, &public_ed_box(&recv_key.public()));
605        let mut seal = msg.as_bytes();
606        shared.seal(&mut seal);
607
608        let bytes = encode_message(&sender_key.public(), seal.clone());
609
610        assert!(looks_like_disco_wrapper(&bytes));
611        assert_eq!(source_and_box(&bytes).unwrap().0, sender_key.public());
612
613        let (raw_key, seal_back) = source_and_box(&bytes).unwrap();
614        assert_eq!(raw_key, sender_key.public());
615        assert_eq!(seal_back, seal);
616
617        let recv_secret = secret_ed_box(&recv_key);
618        let shared_recv = SharedSecret::new(&recv_secret, &public_ed_box(&sender_key.public()));
619        let mut open_seal = seal_back.to_vec();
620        shared_recv
621            .open(&mut open_seal)
622            .expect("failed to open seal_back");
623        let msg_back = Message::from_bytes(&open_seal).unwrap();
624        assert_eq!(msg_back, msg);
625    }
