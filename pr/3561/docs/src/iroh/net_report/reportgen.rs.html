1//! The reportgen actor is responsible for generating a single net_report report.
2//!
3//! It is implemented as an actor with [`Client`] as handle.
4//!
5//! The actor starts generating the report as soon as it is created, it does not receive any
6//! messages from the client.  It follows roughly these steps:
7//!
8//! - Determines host IPv6 support.
9//! - Creates portmapper future.
10//! - Creates captive portal detection future.
11//! - Creates Probe Set futures.
12//!   - These send messages to the reportgen actor.
13//! - Loops driving the futures and handling actor messages:
14//!   - Disables futures as they are completed or aborted.
15//!   - Stop if there are no outstanding tasks/futures, or on timeout.
16//! - Sends the completed report to the net_report actor.
17

118    /// Creates a new actor generating a single report.
119    ///
120    /// The actor starts running immediately and only generates a single report, after which
121    /// it shuts down.  Dropping this handle will abort the actor.
122    pub(super) fn new(
123        last_report: Option<Report>,
124        relay_map: RelayMap,
125        protocols: BTreeSet<Probe>,
126        if_state: IfStateDetails,
127        #[cfg(not(wasm_browser))] socket_state: SocketState,
128        #[cfg(any(test, feature = "test-utils"))] insecure_skip_relay_cert_verify: bool,
129    ) -> (Self, mpsc::Receiver<ProbeFinished>) {
130        let (msg_tx, msg_rx) = mpsc::channel(32);
131        let actor = Actor {
132            msg_tx,
133            last_report,
134            relay_map,
135            protocols,
136            #[cfg(not(wasm_browser))]
137            socket_state,
138            #[cfg(any(test, feature = "test-utils"))]
139            insecure_skip_relay_cert_verify,
140            if_state,
141        };
142        let task = task::spawn(actor.run().instrument(warn_span!("reportgen-actor")));
143        (
144            Self {
145                _drop_guard: AbortOnDropHandle::new(task),
146            },
147            msg_rx,
148        )
149    }

193#[derive(Debug)]

200impl Actor {
201    async fn run(self) {
202        match time::timeout(OVERALL_REPORT_TIMEOUT, self.run_inner()).await {
203            Ok(()) => debug!("reportgen actor finished"),
204            Err(time::Elapsed { .. }) => {
205                warn!("reportgen timed out");
206            }
207        }
208    }
209
210    /// Runs the main reportgen actor logic.
211    ///
212    /// This actor runs by:
213    ///
214    /// - Creates a captive portal future.
215    /// - Creates ProbeSet futures in a group of futures.
216    /// - Runs a main loop:
217    ///   - Drives all the above futures.
218    ///   - Receives actor messages (sent by those futures).
219    ///   - Updates the report, cancels unneeded futures.
220    /// - Sends the report to the net_report actor.
221    async fn run_inner(self) {
222        debug!("reportstate actor starting");
223
224        let mut probes = JoinSet::default();
225
226        let _probes_token = self.spawn_probes_task(self.if_state.clone(), &mut probes);
227        let mut num_probes = probes.len();
228
229        let captive_token = self.prepare_captive_portal_task(&mut probes);
230
231        // any reports of working UDP/QUIC?
232        let mut have_udp = false;
233
234        // Check for probes finishing.
235        while let Some(probe_result) = probes.join_next().await {
236            trace!(?probe_result, num_probes, "processing finished probe");
237            match probe_result {
238                Ok(report) => {
239                    #[cfg_attr(wasm_browser, allow(irrefutable_let_patterns))]
240                    if let ProbeFinished::Regular(report) = &report {
241                        have_udp |= report.as_ref().map(|r| r.is_udp()).unwrap_or_default();
242                        num_probes -= 1;
243
244                        // If all probes are done & we have_udp cancel captive
245                        if num_probes == 0 {
246                            debug!("all regular probes done");
247                            debug_assert!(probes.len() <= 1, "{} probes", probes.len());
248
249                            if have_udp {
250                                captive_token.cancel();
251                            }
252                        }
253                    }
254                    self.msg_tx.send(report).await.ok();
255                }
256                Err(e) => {
257                    if e.is_panic() {
258                        error!("Task panicked {:?}", e);
259                        break;
260                    }
261                    warn!("probes task join error: {:?}", e);
262                }
263            }
264        }
265    }
266
267    /// Creates the future which will perform the captive portal check.
268    fn prepare_captive_portal_task(&self, tasks: &mut JoinSet<ProbeFinished>) -> CancellationToken {
269        let token = CancellationToken::new();
270
271        // If we're doing a full probe, also check for a captive portal. We
272        // delay by a bit to wait for UDP QAD to finish, to avoid the probe if
273        // it's unnecessary.
274        #[cfg(not(wasm_browser))]
275        if self.last_report.is_none() {
276            // Even if we're doing a non-incremental update, we may want to try our
277            // preferred relay for captive portal detection.
278            let preferred_relay = self
279                .last_report
280                .as_ref()
281                .and_then(|l| l.preferred_relay.clone());
282
283            let dns_resolver = self.socket_state.dns_resolver.clone();
284            let dm = self.relay_map.clone();
285            let token = token.clone();
286            tasks.spawn(
287                async move {
288                    let res = token
289                        .run_until_cancelled_owned(async move {
290                            time::sleep(CAPTIVE_PORTAL_DELAY).await;
291                            trace!("check started after {CAPTIVE_PORTAL_DELAY:?}");
292                            time::timeout(
293                                CAPTIVE_PORTAL_TIMEOUT,
294                                check_captive_portal(&dns_resolver, &dm, preferred_relay),
295                            )
296                            .await
297                        })
298                        .await;
299                    let res = match res {
300                        Some(Ok(Ok(found))) => Some(found),
301                        Some(Ok(Err(err))) => {
302                            match err {
303                                CaptivePortalError::CreateReqwestClient { source, .. }
304                                | CaptivePortalError::HttpRequest { source, .. }
305                                    if source.is_connect() =>
306                                {
307                                    debug!("check_captive_portal failed: {source:#}");
308                                }
309                                err => warn!("check_captive_portal error: {err:#}"),
310                            }
311                            None
312                        }
313                        Some(Err(time::Elapsed { .. })) => {
314                            warn!("probe timed out");
315                            None
316                        }
317                        None => {
318                            trace!("probe cancelled");
319                            None
320                        }
321                    };
322                    ProbeFinished::CaptivePortal(res)
323                }
324                .instrument(warn_span!("captive-portal")),
325            );
326        }
327        token
328    }
329
330    /// Prepares the future which will run all the probes as per generated ProbePlan.
331    ///
332    /// Probes operate like the following:
333    ///
334    /// - A future is created for each probe in all probe sets.
335    /// - All probes in a set are grouped in [`JoinSet`].
336    /// - All those probe sets are grouped in one overall [`JoinSet`].
337    ///   - This future is polled by the main actor loop to make progress.
338    /// - Once a probe future is polled:
339    ///   - Many probes start with a delay, they sleep during this time.
340    ///   - When a probe starts it first asks the reportgen [`Actor`] if it is still useful
341    ///     to run.  If not it aborts the entire probe set.
342    ///   - When a probe finishes, its [`ProbeReport`] is yielded to the reportgen actor.
343    /// - Probes get aborted in several ways:
344    ///   - A running it can fail and abort the entire probe set if it deems the
345    ///     failure permanent.  Probes in a probe set are essentially retries.
346    ///   - Once there are [`ProbeReport`]s from enough relays, all remaining probes are
347    ///     aborted.  That is, the main actor loop stops polling them.
348    fn spawn_probes_task(
349        &self,
350        if_state: IfStateDetails,
351        probes: &mut JoinSet<ProbeFinished>,
352    ) -> CancellationToken {
353        debug!(?if_state, "local interface details");
354        let plan = match self.last_report {
355            Some(ref report) => {
356                ProbePlan::with_last_report(&self.relay_map, report, &self.protocols)
357            }
358            None => ProbePlan::initial(&self.relay_map, &self.protocols),
359        };
360        trace!(%plan, "probe plan");
361
362        let token = CancellationToken::new();
363
364        for probe_set in plan.iter() {
365            let set_token = token.child_token();
366            let proto = probe_set.proto();
367            for (delay, relay) in probe_set.params() {
368                let probe_token = set_token.child_token();
369
370                let fut = probe_token.run_until_cancelled_owned(time::timeout(
371                    PROBES_TIMEOUT,
372                    proto.run(
373                        *delay,
374                        relay.clone(),
375                        #[cfg(not(wasm_browser))]
376                        self.socket_state.clone(),
377                        #[cfg(any(test, feature = "test-utils"))]
378                        self.insecure_skip_relay_cert_verify,
379                    ),
380                ));
381                probes.spawn(
382                    async move {
383                        let res = fut.await;
384                        let res = match res {
385                            Some(Ok(Ok(report))) => Ok(report),
386                            Some(Ok(Err(err))) => {
387                                warn!("probe failed: {:#}", err);
388                                Err(e!(ProbesError::ProbeFailure, err))
389                            }
390                            Some(Err(time::Elapsed { .. })) => Err(e!(ProbesError::Timeout)),
391                            None => Err(e!(ProbesError::Cancelled)),
392                        };
393                        ProbeFinished::Regular(res)
394                    }
395                    .instrument(warn_span!(
396                        "run-probe",
397                        ?proto,
398                        ?delay,
399                        ?relay,
400                    )),
401                );
402            }
403        }
404
405        token
406    }
407}

488impl Probe {
489    /// Executes this particular [`Probe`], including using a delayed start if needed.
490    async fn run(
491        self,
492        delay: Duration,
493        relay: Arc<RelayConfig>,
494        #[cfg(not(wasm_browser))] socket_state: SocketState,
495        #[cfg(any(test, feature = "test-utils"))] insecure_skip_relay_cert_verify: bool,
496    ) -> Result<ProbeReport, ProbeError> {
497        if !delay.is_zero() {
498            trace!("delaying probe");
499            time::sleep(delay).await;
500        }
501        debug!("starting probe");
502
503        match self {
504            Probe::Https => {
505                match run_https_probe(
506                    #[cfg(not(wasm_browser))]
507                    &socket_state.dns_resolver,
508                    relay.url.clone(),
509                    #[cfg(any(test, feature = "test-utils"))]
510                    insecure_skip_relay_cert_verify,
511                )
512                .await
513                {
514                    Ok(report) => Ok(ProbeReport::Https(report)),
515                    Err(err) => Err(e!(ProbeError::Https, err)),
516                }
517            }
518            #[cfg(not(wasm_browser))]
519            Probe::QadIpv4 | Probe::QadIpv6 => unreachable!("must not be used"),
520        }
521    }
522}

554/// Reports whether or not we think the system is behind a
555/// captive portal, detected by making a request to a URL that we know should
556/// return a "204 No Content" response and checking if that's what we get.
557///
558/// The boolean return is whether we think we have a captive portal.
559#[cfg(not(wasm_browser))]
560async fn check_captive_portal(
561    dns_resolver: &DnsResolver,
562    dm: &RelayMap,
563    preferred_relay: Option<RelayUrl>,
564) -> Result<bool, CaptivePortalError> {
565    // If we have a preferred relay and we can use it for non-QAD requests, try that;
566    // otherwise, pick a random one suitable for non-STUN requests.
567
568    use crate::util::reqwest_client_builder;
569
570    let preferred_relay = preferred_relay.and_then(|url| dm.get(&url).map(|_| url));
571
572    let url = match preferred_relay {
573        Some(url) => url,
574        None => {
575            let urls: Vec<_> = dm.urls();
576            if urls.is_empty() {
577                debug!("No suitable relay for captive portal check");
578                return Ok(false);
579            }
580
581            let i = (0..urls.len()).choose(&mut rand::rng()).unwrap_or_default();
582            urls[i].clone()
583        }
584    };
585
586    let mut builder = reqwest_client_builder().redirect(reqwest::redirect::Policy::none());
587
588    if let Some(Host::Domain(domain)) = url.host() {
589        // Use our own resolver rather than getaddrinfo
590        //
591        // Be careful, a non-zero port will override the port in the URI.
592        //
593        // Ideally we would try to resolve **both** IPv4 and IPv6 rather than purely race
594        // them.  But our resolver doesn't support that yet.
595        let addrs: Vec<_> = dns_resolver
596            .lookup_ipv4_ipv6_staggered(domain, DNS_TIMEOUT, DNS_STAGGERING_MS)
597            .await
598            .map_err(|source| e!(CaptivePortalError::DnsLookup { source }))?
599            .map(|ipaddr| SocketAddr::new(ipaddr, 0))
600            .collect();
601        builder = builder.resolve_to_addrs(domain, &addrs);
602    }
603    let client = builder
604        .build()
605        .map_err(|source| e!(CaptivePortalError::CreateReqwestClient { source }))?;
606
607    // Note: the set of valid characters in a challenge and the total
608    // length is limited; see is_challenge_char in bin/iroh-relay for more
609    // details.
610
611    let host_name = url.host_str().unwrap_or_default();
612    let challenge = format!("ts_{host_name}");
613    let portal_url = format!("http://{host_name}/generate_204");
614    let res = client
615        .request(reqwest::Method::GET, portal_url)
616        .header("X-Iroh-Challenge", &challenge)
617        .send()
618        .await
619        .map_err(|source| e!(CaptivePortalError::HttpRequest { source }))?;
620
621    let expected_response = format!("response {challenge}");
622    let is_valid_response = res
623        .headers()
624        .get("X-Iroh-Response")
625        .map(|s| s.to_str().unwrap_or_default())
626        == Some(&expected_response);
627
628    debug!(
629        "check_captive_portal url={} status_code={} valid_response={}",
630        res.url(),
631        res.status(),
632        is_valid_response,
633    );
634    let has_captive = res.status() != 204 || !is_valid_response;
635
636    Ok(has_captive)
637}

678    let port = get_quic_port(relay).ok_or_else(|| e!(GetRelayAddrError::MissingPort))?;
679    relay_lookup_ipv4_staggered(dns_resolver, relay, port).await
680}

691/// Do a staggared ipv4 DNS lookup based on [`RelayConfig`]
692///
693/// `port` is combined with the resolved [`std::net::Ipv4Addr`] to return a [`SocketAddr`]
694#[cfg(not(wasm_browser))]
695async fn relay_lookup_ipv4_staggered(
696    dns_resolver: &DnsResolver,
697    relay: &RelayConfig,
698    port: u16,
699) -> Result<SocketAddrV4, GetRelayAddrError> {
700    match relay.url.host() {
701        Some(url::Host::Domain(hostname)) => {
702            trace!(%hostname, "Performing DNS A lookup for relay addr");
703            match dns_resolver
704                .lookup_ipv4_staggered(hostname, DNS_TIMEOUT, DNS_STAGGERING_MS)
705                .await
706            {
707                Ok(mut addrs) => addrs
708                    .next()
709                    .map(|ip| ip.to_canonical())
710                    .map(|addr| match addr {
711                        IpAddr::V4(ip) => SocketAddrV4::new(ip, port),
712                        IpAddr::V6(_) => unreachable!("bad DNS lookup: {:?}", addr),
713                    })
714                    .ok_or_else(|| e!(GetRelayAddrError::NoAddrFound)),
715                Err(err) => Err(e!(GetRelayAddrError::DnsLookup, err)),
716            }
717        }
718        Some(url::Host::Ipv4(addr)) => Ok(SocketAddrV4::new(addr, port)),
719        Some(url::Host::Ipv6(_addr)) => Err(e!(GetRelayAddrError::NoAddrFound)),
720        None => Err(e!(GetRelayAddrError::InvalidHostname)),
721    }
722}

724/// Do a staggared ipv6 DNS lookup based on [`RelayConfig`]
725///
726/// `port` is combined with the resolved [`std::net::Ipv6Addr`] to return a [`SocketAddr`]
727#[cfg(not(wasm_browser))]
728async fn relay_lookup_ipv6_staggered(
729    dns_resolver: &DnsResolver,
730    relay: &RelayConfig,
731    port: u16,
732) -> Result<SocketAddrV6, GetRelayAddrError> {
733    match relay.url.host() {
734        Some(url::Host::Domain(hostname)) => {
735            trace!(%hostname, "Performing DNS AAAA lookup for relay addr");
736            match dns_resolver
737                .lookup_ipv6_staggered(hostname, DNS_TIMEOUT, DNS_STAGGERING_MS)
738                .await
739            {
740                Ok(mut addrs) => addrs
741                    .next()
742                    .map(|addr| match addr {
743                        IpAddr::V4(_) => unreachable!("bad DNS lookup: {:?}", addr),
744                        IpAddr::V6(ip) => SocketAddrV6::new(ip, port, 0, 0),
745                    })
746                    .ok_or_else(|| e!(GetRelayAddrError::NoAddrFound)),
747                Err(err) => Err(e!(GetRelayAddrError::DnsLookup, err)),
748            }
749        }
750        Some(url::Host::Ipv4(_addr)) => Err(e!(GetRelayAddrError::NoAddrFound)),
751        Some(url::Host::Ipv6(addr)) => Ok(SocketAddrV6::new(addr, port, 0, 0)),
752        None => Err(e!(GetRelayAddrError::InvalidHostname)),
753    }
754}

756#[add_meta]
757#[derive(Error)]
758#[non_exhaustive]
759pub enum MeasureHttpsLatencyError {
760    #[error(transparent)]
761    InvalidUrl {
762        #[error(std_err)]
763        source: url::ParseError,
764    },
765    #[cfg(not(wasm_browser))]
766    #[error(transparent)]
767    DnsLookup { source: StaggeredError },
768    #[display("Creating HTTP client failed")]
769    CreateReqwestClient {
770        #[error(std_err)]
771        source: reqwest::Error,
772    },
773    #[display("HTTP request failed")]
774    HttpRequest {
775        #[error(std_err)]
776        source: reqwest::Error,
777    },
778    #[display("Error response from server {status}: {:?}", status.canonical_reason())]
779    InvalidResponse { status: StatusCode },
780}

782/// Executes an HTTPS probe.
783///
784/// If `certs` is provided they will be added to the trusted root certificates, allowing the
785/// use of self-signed certificates for servers.  Currently this is used for testing.
786#[allow(clippy::unused_async)]
787async fn run_https_probe(
788    #[cfg(not(wasm_browser))] dns_resolver: &DnsResolver,
789    relay: RelayUrl,
790    #[cfg(any(test, feature = "test-utils"))] insecure_skip_relay_cert_verify: bool,
791) -> Result<HttpsProbeReport, MeasureHttpsLatencyError> {
792    trace!("HTTPS probe start");
793    let url = relay
794        .join(RELAY_PROBE_PATH)
795        .map_err(|source| e!(MeasureHttpsLatencyError::InvalidUrl { source }))?;
796
797    // This should also use same connection establishment as relay client itself, which
798    // needs to be more configurable so users can do more crazy things:
799    // https://github.com/n0-computer/iroh/issues/2901
800    let mut builder = reqwest_client_builder();
801
802    #[cfg(not(wasm_browser))]
803    {
804        builder = builder.redirect(reqwest::redirect::Policy::none());
805    }
806
807    #[cfg(not(wasm_browser))]
808    if let Some(Host::Domain(domain)) = url.host() {
809        // Use our own resolver rather than getaddrinfo
810        //
811        // Be careful, a non-zero port will override the port in the URI.
812        //
813        // The relay Client uses `.lookup_ipv4_ipv6` to connect, so use the same function
814        // but staggered for reliability.  Ideally this tries to resolve **both** IPv4 and
815        // IPv6 though.  But our resolver does not have a function for that yet.
816        let addrs: Vec<_> = dns_resolver
817            .lookup_ipv4_ipv6_staggered(domain, DNS_TIMEOUT, DNS_STAGGERING_MS)
818            .await
819            .map_err(|source| e!(MeasureHttpsLatencyError::DnsLookup { source }))?
820            .map(|ipaddr| SocketAddr::new(ipaddr, 0))
821            .collect();
822        trace!(?addrs, "resolved addrs");
823        builder = builder.resolve_to_addrs(domain, &addrs);
824    }
825
826    #[cfg(all(not(wasm_browser), any(test, feature = "test-utils")))]
827    let builder = builder.danger_accept_invalid_certs(insecure_skip_relay_cert_verify);
828
829    let client = builder
830        .build()
831        .map_err(|source| e!(MeasureHttpsLatencyError::CreateReqwestClient { source }))?;
832
833    let start = Instant::now();
834    let response = client
835        .request(reqwest::Method::GET, url)
836        .send()
837        .await
838        .map_err(|source| e!(MeasureHttpsLatencyError::HttpRequest { source }))?;
839    let latency = start.elapsed();
840    if response.status().is_success() {
841        // Drain the response body to be nice to the server, up to a limit.
842        const MAX_BODY_SIZE: usize = 8 << 10; // 8 KiB
843        let mut body_size = 0;
844        let mut stream = response.bytes_stream();
845        // ignore failing frames
846        while let Some(Ok(chunk)) = stream.next().await {
847            body_size += chunk.len();
848            if body_size >= MAX_BODY_SIZE {
849                break;
850            }
851        }
852
853        Ok(HttpsProbeReport { relay, latency })
854    } else {
855        Err(e!(MeasureHttpsLatencyError::InvalidResponse {
856            status: response.status()
857        }))
858    }
859}
