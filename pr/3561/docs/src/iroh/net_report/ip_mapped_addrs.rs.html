12/// Can occur when converting a [`SocketAddr`] to an [`IpMappedAddr`]
13#[add_meta]
14#[derive(Error)]
15#[display("Failed to convert")]
16pub struct IpMappedAddrError;
17
18/// A map fake Ipv6 address with an actual IP address.
19///
20/// It is essentially a lookup key for an IP that iroh's magicsocket knows about.
21#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, Ord, PartialOrd)]
22pub(crate) struct IpMappedAddr(Ipv6Addr);
23
24/// Counter to always generate unique addresses for [`IpMappedAddr`].
25static IP_ADDR_COUNTER: AtomicU64 = AtomicU64::new(1);
26
27impl IpMappedAddr {
28    /// The Prefix/L of our Unique Local Addresses.
29    const ADDR_PREFIXL: u8 = 0xfd;
30    /// The Global ID used in our Unique Local Addresses.
31    const ADDR_GLOBAL_ID: [u8; 5] = [21, 7, 10, 81, 11];
32    /// The Subnet ID used in our Unique Local Addresses.
33    const ADDR_SUBNET: [u8; 2] = [0, 1];
34
35    /// The dummy port used for all mapped addresses.
36    const MAPPED_ADDR_PORT: u16 = 12345;
37
38    /// Generates a globally unique fake UDP address.
39    ///
40    /// This generates a new IPv6 address in the Unique Local Address range (RFC 4193)
41    /// which is recognised by iroh as an IP mapped address.
42    pub(super) fn generate() -> Self {
43        let mut addr = [0u8; 16];
44        addr[0] = Self::ADDR_PREFIXL;
45        addr[1..6].copy_from_slice(&Self::ADDR_GLOBAL_ID);
46        addr[6..8].copy_from_slice(&Self::ADDR_SUBNET);
47
48        let counter = IP_ADDR_COUNTER.fetch_add(1, Ordering::Relaxed);
49        addr[8..16].copy_from_slice(&counter.to_be_bytes());
50
51        Self(Ipv6Addr::from(addr))
52    }
53
54    /// Returns a consistent [`SocketAddr`] for the [`IpMappedAddr`].
55    ///
56    /// This does not have a routable IP address.
57    ///
58    /// This uses a made-up, but fixed port number.  The [IpMappedAddresses`] map this is
59    /// made for creates a unique [`IpMappedAddr`] for each IP+port and thus does not use
60    /// the port to map back to the original [`SocketAddr`].
61    pub(crate) fn private_socket_addr(&self) -> SocketAddr {
62        SocketAddr::new(IpAddr::from(self.0), Self::MAPPED_ADDR_PORT)
63    }
64}
65
66impl TryFrom<Ipv6Addr> for IpMappedAddr {
67    type Error = IpMappedAddrError;
68
69    fn try_from(value: Ipv6Addr) -> std::result::Result<Self, Self::Error> {
70        let octets = value.octets();
71        if octets[0] == Self::ADDR_PREFIXL
72            && octets[1..6] == Self::ADDR_GLOBAL_ID
73            && octets[6..8] == Self::ADDR_SUBNET
74        {
75            return Ok(Self(value));
76        }
77        Err!(IpMappedAddrError)
78    }
79}

81impl std::fmt::Display for IpMappedAddr {
82    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
83        write!(f, "IpMappedAddr({})", self.0)
84    }
85}

87/// A Map of [`IpMappedAddresses`] to [`SocketAddr`].
88// TODO(ramfox): before this is ready to be used beyond QAD, we should add
89// mechanisms for keeping track of "aliveness" and pruning address, as we do
90// with the `EndpointMap`
91#[derive(Debug, Clone, Default)]
92pub(crate) struct IpMappedAddresses(Arc<std::sync::Mutex<Inner>>);
93
94#[derive(Debug, Default)]
95pub(super) struct Inner {
96    by_mapped_addr: BTreeMap<IpMappedAddr, SocketAddr>,
97    /// Because [`std::net::SocketAddrV6`] contains extra fields besides the IP
98    /// address and port (ie, flow_info and scope_id), the a [`std::net::SocketAddrV6`]
99    /// with the same IP addr and port might Hash to something different.
100    /// So to get a hashable key for the map, we are using `(IpAddr, u6)`.
101    by_ip_port: BTreeMap<(IpAddr, u16), IpMappedAddr>,
102}
103
104impl IpMappedAddresses {
105    /// Adds a [`SocketAddr`] to the map and returns the generated [`IpMappedAddr`].
106    ///
107    /// If this [`SocketAddr`] already exists in the map, it returns its
108    /// associated [`IpMappedAddr`].
109    ///
110    /// Otherwise a new [`IpMappedAddr`] is generated for it and returned.
111    pub(super) fn get_or_register(&self, socket_addr: SocketAddr) -> IpMappedAddr {
112        let ip_port = (socket_addr.ip(), socket_addr.port());
113        let mut inner = self.0.lock().expect("poisoned");
114        if let Some(mapped_addr) = inner.by_ip_port.get(&ip_port) {
115            return *mapped_addr;
116        }
117        let ip_mapped_addr = IpMappedAddr::generate();
118        inner.by_mapped_addr.insert(ip_mapped_addr, socket_addr);
119        inner.by_ip_port.insert(ip_port, ip_mapped_addr);
120        ip_mapped_addr
121    }
122
123    /// Returns the [`IpMappedAddr`] for the given [`SocketAddr`].
124    pub(crate) fn get_mapped_addr(&self, socket_addr: &SocketAddr) -> Option<IpMappedAddr> {
125        let ip_port = (socket_addr.ip(), socket_addr.port());
126        let inner = self.0.lock().expect("poisoned");
127        inner.by_ip_port.get(&ip_port).copied()
128    }
129
130    /// Returns the [`SocketAddr`] for the given [`IpMappedAddr`].
131    pub(crate) fn get_ip_addr(&self, mapped_addr: &IpMappedAddr) -> Option<SocketAddr> {
132        let inner = self.0.lock().expect("poisoned");
133        inner.by_mapped_addr.get(mapped_addr).copied()
134    }
135}
