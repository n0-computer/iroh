1//! Presets allow configuring an endpoint quickly with a chosen set of defaults.
2//!
3//! # Example
4//!
5//! ```no_run
6//! # async fn wrapper() -> n0_error::Result<()> {
7//! use iroh::{Endpoint, RelayMode, Watcher, endpoint::presets};
8//!
9//! let endpoint = Endpoint::empty_builder(RelayMode::Disabled)
10//!     .preset(presets::N0)
11//!     .bind()
12//!     .await?;
13//! # let _ = endpoint;
14//! # Ok(())
15//! # }
16//! ```
17
18use crate::{
19    discovery::pkarr::PkarrPublisher,
20    endpoint::{Builder, default_relay_mode},
21};
22
23/// Defines a preset
24pub trait Preset {
25    /// Applies the configuration to the passed in [`Builder`].
26    fn apply(self, builder: Builder) -> Builder;
27}
28
29/// Configures the endpoint to use the n0 defaults
30///
31/// Currently this consists of
32/// - the DNS discovery service.
33/// - the default relay servers provided by Number 0.
34///
35/// The default discovery service publishes to and resolves from the
36/// n0.computer dns server `iroh.link`.
37///
38/// This is equivalent to adding both a [`crate::discovery::pkarr::PkarrPublisher`]
39/// and a [`crate::discovery::dns::DnsDiscovery`], both configured to use the
40/// n0.computer dns server.
41///
42/// This will by default use [`N0_DNS_PKARR_RELAY_PROD`].
43/// When in tests, or when the `test-utils` feature is enabled, this will use the
44/// [`N0_DNS_PKARR_RELAY_STAGING`].
45///
46/// [`N0_DNS_PKARR_RELAY_PROD`]: crate::discovery::pkarr::N0_DNS_PKARR_RELAY_PROD
47/// [`N0_DNS_PKARR_RELAY_STAGING`]: crate::discovery::pkarr::N0_DNS_PKARR_RELAY_STAGING
48#[derive(Debug, Copy, Clone, Default)]
49pub struct N0;
50
51impl Preset for N0 {
52    fn apply(self, mut builder: Builder) -> Builder {
53        builder = builder.discovery(PkarrPublisher::n0_dns());
54        // Resolve using HTTPS requests to our DNS server's /pkarr path in browsers
55        #[cfg(wasm_browser)]
56        {
57            use crate::discovery::pkarr::PkarrResolver;
58
59            builder = builder.discovery(PkarrResolver::n0_dns());
60        }
61        // Resolve using DNS queries outside browsers.
62        #[cfg(not(wasm_browser))]
63        {
64            use crate::discovery::dns::DnsDiscovery;
65
66            builder = builder.discovery(DnsDiscovery::n0_dns());
67        }
68
69        builder = builder.relay_mode(default_relay_mode());
70
71        builder
72    }
73}
