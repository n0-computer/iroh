22/// Server configuration
23///
24/// The config is usually loaded from a file with [`Self::load`].
25///
26/// The struct also implements [`Default`] which creates a config suitable for local development
27/// and testing.
28#[derive(Debug, Serialize, Deserialize)]
29pub struct Config {
30    /// Config for the HTTP server
31    ///
32    /// If set to `None` no HTTP server will be started.
33    pub http: Option<HttpConfig>,
34    /// Config for the HTTPS server
35    ///
36    /// If set to `None` no HTTPS server will be started.
37    pub https: Option<HttpsConfig>,
38    /// Config for the DNS server.
39    pub dns: DnsConfig,
40    /// Config for the metrics server.
41    ///
42    /// The metrics server is started by default. To disable the metrics server, set to
43    /// `Some(MetricsConfig::disabled())`.
44    pub metrics: Option<MetricsConfig>,
45
46    /// Config for the mainline lookup.
47    pub mainline: Option<MainlineConfig>,
48
49    /// Config for the zone store.
50    pub zone_store: Option<StoreConfig>,
51
52    /// Config for pkarr rate limit
53    #[serde(default)]
54    pub pkarr_put_rate_limit: RateLimitConfig,
55}

57/// The config for the store.
58#[derive(Debug, Serialize, Deserialize, Clone)]
59pub struct StoreConfig {
60    /// Maximum number of packets to process in a single write transaction.
61    max_batch_size: usize,
62
63    /// Maximum time to keep a write transaction open.
64    #[serde(with = "humantime_serde")]
65    max_batch_time: Duration,
66
67    /// Time to keep packets in the store before eviction.
68    #[serde(with = "humantime_serde")]
69    eviction: Duration,
70
71    /// Pause between eviction checks.
72    #[serde(with = "humantime_serde")]
73    eviction_interval: Duration,
74}
75
76impl Default for StoreConfig {
77    fn default() -> Self {
78        ZoneStoreOptions::default().into()
79    }
80}
81
82impl From<ZoneStoreOptions> for StoreConfig {

86            max_batch_time: value.max_batch_time,
87            eviction: value.eviction,
88            eviction_interval: value.eviction_interval,
89        }
90    }
91}

93impl From<StoreConfig> for ZoneStoreOptions {

123/// The config for the metrics server.
124#[derive(Debug, Serialize, Deserialize)]
125pub struct MainlineConfig {
126    /// Set to true to enable the mainline lookup.
127    pub enabled: bool,
128    /// Set custom bootstrap nodes.
129    ///
130    /// Addresses can either be `domain:port` or `ipv4:port`.
131    ///
132    /// If empty this will use the default bittorrent mainline bootstrap nodes as defined by pkarr.
133    pub bootstrap: Option<Vec<String>>,
134}
135
136/// Configure the bootstrap servers for mainline DHT resolution.
137#[derive(Debug, Serialize, Deserialize, Default)]
138pub enum BootstrapOption {
139    /// Use the default bootstrap servers.
140    #[default]
141    Default,
142    /// Use custom bootstrap servers.
143    Custom(Vec<String>),
144}
145
146#[allow(clippy::derivable_impls)]
147impl Default for MainlineConfig {
148    fn default() -> Self {
149        Self {
150            enabled: false,
151            bootstrap: None,
152        }
153    }
154}

156impl Config {
157    /// Load the config from a file.
158    pub async fn load(path: impl AsRef<Path>) -> Result<Config> {
159        info!(
160            "loading config file from {}",
161            path.as_ref().to_string_lossy()
162        );
163        let s = tokio::fs::read_to_string(path.as_ref())
164            .await
165            .with_std_context(|_| format!("failed to read {}", path.as_ref().to_string_lossy()))?;
166        let config: Config = toml::from_str(&s).e()?;
167        Ok(config)
168    }
169
170    /// Get the data directory.
171    pub fn data_dir() -> Result<PathBuf> {
172        let dir = if let Some(val) = env::var_os("IROH_DNS_DATA_DIR") {
173            PathBuf::from(val)
174        } else {
175            let path = dirs_next::data_dir()
176                .std_context("operating environment provides no directory for application data")?;
177
178            path.join("iroh-dns")
179        };
180        Ok(dir)
181    }
182
183    /// Get the path to the store database file.
184    pub fn signed_packet_store_path() -> Result<PathBuf> {
185        Ok(Self::data_dir()?.join("signed-packets-1.db"))
186    }
187
188    /// Get the address where the metrics server should be bound, if set.
189    pub(crate) fn metrics_addr(&self) -> Option<SocketAddr> {
190        match &self.metrics {
191            None => Some(DEFAULT_METRICS_ADDR),
192            Some(conf) => match conf.disabled {
193                true => None,
194                false => Some(conf.bind_addr.unwrap_or(DEFAULT_METRICS_ADDR)),
195            },
196        }
197    }
198
199    pub(crate) fn mainline_enabled(&self) -> Option<BootstrapOption> {
200        match self.mainline.as_ref() {
201            None => None,
202            Some(MainlineConfig { enabled: false, .. }) => None,
203            Some(MainlineConfig {
204                bootstrap: Some(bootstrap),
205                ..
206            }) => Some(BootstrapOption::Custom(bootstrap.clone())),
207            Some(MainlineConfig {
208                bootstrap: None, ..
209            }) => Some(BootstrapOption::Default),
210        }
211    }
212}
