1//! Implements the handshake protocol that authenticates and authorizes clients connecting to the relays.
2//!
3//! The purpose of the handshake is to
4//! 1. Inform the relay of the client's EndpointId
5//! 2. Check that the connecting client owns the secret key for its EndpointId ("is authentic"/"authentication")
6//! 3. Possibly check that the client has access to this relay, if the relay requires authorization.
7//!
8//! Additional complexity comes from the fact that there's two ways that clients can authenticate with
9//! relays.
10//!
11//! One way is via an explicitly sent challenge:
12//!
13//! 1. Once a websocket connection is opened, a client receives a challenge (the `ServerChallenge` frame)
14//! 2. The client sends back what is essentially a signature of that challenge with their secret key
15//!    that matches the EndpointId they have, as well as the EndpointId (the `ClientAuth` frame)
16//!
17//! The second way is very similar to the [Concealed HTTP Auth RFC], and involves send a header that
18//! contains a signature of some shared keying material extracted from TLS ([RFC 5705]).
19//!
20//! The second way can save a full round trip, because the challenge doesn't have to be sent to the client
21//! first, however, it won't always work, as it relies on the keying material extraction feature of TLS,
22//! which is not available in browsers (but might be in the future?) and might break when there's an
23//! HTTPS proxy that doesn't properly deal with this TLS feature.
24//!
25//! [Concealed HTTP Auth RFC]: https://datatracker.ietf.org/doc/rfc9729/
26//! [RFC 5705]: https://datatracker.ietf.org/doc/html/rfc5705

174pub(crate) enum VerificationError {
175    #[display("Couldn't export TLS keying material on our end")]
176    NoKeyingMaterial,

195impl ServerChallenge {
196    /// Generates a new challenge.
197    #[cfg(feature = "server")]
198    pub(crate) fn new<R: CryptoRng + ?Sized>(rng: &mut R) -> Self {
199        let mut challenge = [0u8; 16];
200        rng.fill_bytes(&mut challenge);
201        Self { challenge }
202    }
203
204    /// The actual message bytes to sign (and verify against) for this challenge.
205    fn message_to_sign(&self) -> [u8; 32] {
206        // We're signing a key instead of the direct challenge.
207        // This gives us domain separation protecting from multiple possible attacks,
208        // but especially this one:
209        // Assume a malicious relay. If the protocol required the client to sign the
210        // challenge directly, this would allow the relay to obtain an arbitrary 16-byte
211        // signature, if it maliciously choses the challenge instead of generating it
212        // randomly.
213        // Deriving a key to sign instead mitigates this attack.
214        blake3::derive_key(DOMAIN_SEP_CHALLENGE, &self.challenge)
215    }
216}

218impl ClientAuth {
219    /// Generates a signature for the given challenge from the server.
220    pub(crate) fn new(secret_key: &SecretKey, challenge: &ServerChallenge) -> Self {
221        Self {
222            public_key: secret_key.public(),
223            signature: secret_key.sign(&challenge.message_to_sign()).to_bytes(),
224        }
225    }
226
227    /// Verifies this client's authentication given the challenge this was sent in response to.
228    #[cfg(feature = "server")]
229    pub(crate) fn verify(&self, challenge: &ServerChallenge) -> Result<(), Box<VerificationError>> {
230        let message = challenge.message_to_sign();
231        self.public_key
232            .verify(&message, &Signature::from_bytes(&self.signature))
233            .map_err(|err| {
234                e!(VerificationError::SignatureInvalid {
235                    source: err,
236                    message: message.to_vec(),
237                    signature: self.signature,
238                    public_key: self.public_key
239                })
240            })
241            .map_err(Box::new)
242    }
243}

245#[cfg(not(wasm_browser))]
246impl KeyMaterialClientAuth {
247    /// Generates a client's authentication, similar to [`ClientAuth`], but by using TLS keying material
248    /// instead of a received challenge.
249    pub(crate) fn new(secret_key: &SecretKey, io: &impl ExportKeyingMaterial) -> Option<Self> {
250        let public_key = secret_key.public();
251        let key_material = io.export_keying_material(
252            [0u8; 32],
253            DOMAIN_SEP_TLS_EXPORT_LABEL,
254            Some(secret_key.public().as_bytes()),
255        )?;
256        // We split the export and only sign the first 16 bytes, and
257        // pass through the last 16 bytes. See also the note in [Self::verify].
258        let (message, suffix) = key_material.split_at(16);
259        Some(Self {
260            public_key,
261            signature: secret_key.sign(message).to_bytes(),
262            key_material_suffix: suffix.try_into().expect("hardcoded length"),
263        })
264    }
265
266    /// Generate the base64url-nopad-encoded header value.
267    pub(crate) fn into_header_value(self) -> HeaderValue {
268        HeaderValue::from_str(
269            &data_encoding::BASE64URL_NOPAD
270                .encode(&postcard::to_allocvec(&self).expect("encoding never fails")),
271        )
272        .expect("BASE64URL_NOPAD encoding contained invisible ascii characters")
273    }
274
275    /// Verifies this client auth on the server side using the same key material.
276    ///
277    /// This might return false for a couple of reasons:
278    /// 1. The exported keying material might not be the same between both ends of the TLS session
279    ///    (e.g. there's an HTTPS proxy in between that doesn't think/care about the TLS keying material exporter).
280    ///    This situation is detected when the key material suffix mismatches.
281    /// 2. The signature itself doesn't verify.
282    #[cfg(feature = "server")]
283    pub(crate) fn verify(
284        &self,
285        io: &impl ExportKeyingMaterial,
286    ) -> Result<(), Box<VerificationError>> {
287        let key_material = io
288            .export_keying_material(
289                [0u8; 32],
290                DOMAIN_SEP_TLS_EXPORT_LABEL,
291                Some(self.public_key.as_bytes()),
292            )
293            .ok_or_else(|| e!(VerificationError::NoKeyingMaterial))?;
294        // We split the export and only sign the first 16 bytes, and
295        // pass through the last 16 bytes.
296        // Passing on the suffix helps the verifying end figure out what
297        // went wrong: If there's a suffix mismatch, then the exported keying
298        // material on both ends wasn't the same - so perhaps there was a
299        // TLS proxy in between or similar.
300        // If the suffix does match, but the signature doesn't verify, then
301        // there must be something wrong with the client's secret key or signature.
302        let (message, suffix) = key_material.split_at(16);
303        let suffix: [u8; 16] = suffix.try_into().expect("hardcoded length");
304        n0_error::ensure!(
305            suffix == self.key_material_suffix,
306            e!(VerificationError::MismatchedSuffix {
307                expected: self.key_material_suffix,
308                actual: suffix
309            })
310        );
311        // NOTE: We don't blake3-hash here as we do it in [`ServerChallenge::message_to_sign`],
312        // because we already have a domain separation string and keyed hashing step in
313        // the TLS export keying material above.
314        self.public_key
315            .verify(message, &Signature::from_bytes(&self.signature))
316            .map_err(|err| {
317                e!(VerificationError::SignatureInvalid {
318                    source: err,
319                    message: message.to_vec(),
320                    public_key: self.public_key,
321                    signature: self.signature
322                })
323            })
324            .map_err(Box::new)
325    }
326}

328/// Runs the client side of the handshake protocol.
329///
330/// See the module docs for details on the protocol.
331/// This is already after having potentially transferred a [`KeyMaterialClientAuth`],
332/// but before having received a response for whether that worked or not.
333///
334/// This requires access to the client's secret key to sign a challenge.
335pub(crate) async fn clientside(
336    io: &mut (impl BytesStreamSink + ExportKeyingMaterial),
337    secret_key: &SecretKey,
338) -> Result<ServerConfirmsAuth, Error> {
339    let (tag, frame) = read_frame(io, &[ServerChallenge::TAG, ServerConfirmsAuth::TAG]).await?;
340
341    let (tag, frame) = if tag == ServerChallenge::TAG {
342        let challenge: ServerChallenge = deserialize_frame(frame)?;
343
344        let client_info = ClientAuth::new(secret_key, &challenge);
345        write_frame(io, client_info).await?;
346
347        read_frame(io, &[ServerConfirmsAuth::TAG, ServerDeniesAuth::TAG]).await?
348    } else {
349        (tag, frame)
350    };
351
352    match tag {
353        FrameType::ServerConfirmsAuth => {
354            let confirmation: ServerConfirmsAuth = deserialize_frame(frame)?;
355            Ok(confirmation)
356        }
357        FrameType::ServerDeniesAuth => {
358            let denial: ServerDeniesAuth = deserialize_frame(frame)?;
359            Err!(Error::ServerDeniedAuth {
360                reason: denial.reason
361            })
362        }
363        _ => unreachable!(),
364    }
365}

367/// This represents successful authentication for the client with the `client_key` public key
368/// via the authentication [`Mechanism`] `mechanism`.
369///
370/// You must call [`SuccessfulAuthentication::authorize_if`] to finish the protocol.
371#[cfg(feature = "server")]
372#[derive(Debug)]
373#[must_use = "the protocol is not finished unless `authorize_if` is called"]
374pub(crate) struct SuccessfulAuthentication {
375    pub(crate) client_key: PublicKey,
376    pub(crate) mechanism: Mechanism,
377}

379/// The mechanism that was used for authentication.
380#[cfg(feature = "server")]
381#[derive(Debug, Clone, Copy, PartialEq, Eq)]
382pub(crate) enum Mechanism {
383    /// Authentication was performed by verifying a signature of a challenge we sent
384    SignedChallenge,
385    /// Authentication was performed by verifying a signature of shared extracted TLS keying material
386    SignedKeyMaterial,
387}

389/// Runs the server side of the handshaking protocol.
390///
391/// See the module documentation for an overview of the handshaking protocol.
392///
393/// This takes `rng` to generate cryptographic randomness for the authentication challenge.
394///
395/// This also takes the `client_auth_header`, if present, to perform authentication without
396/// requiring sending a challenge, saving a round-trip, if possible.
397///
398/// If this fails, the protocol falls back to doing a normal extra round trip with a challenge.
399///
400/// The return value [`SuccessfulAuthentication`] still needs to be resolved by calling
401/// [`SuccessfulAuthentication::authorize_if`] to finish the whole authorization protocol
402/// (otherwise the client won't be notified about auth success or failure).
403#[cfg(feature = "server")]
404pub(crate) async fn serverside(
405    io: &mut (impl BytesStreamSink + ExportKeyingMaterial),
406    client_auth_header: Option<HeaderValue>,
407) -> Result<SuccessfulAuthentication, Error> {
408    if let Some(client_auth_header) = client_auth_header {
409        let client_auth_bytes = data_encoding::BASE64URL_NOPAD
410            .decode(client_auth_header.as_ref())
411            .map_err(|_| {
412                e!(Error::ClientAuthHeaderInvalid {
413                    value: client_auth_header.clone()
414                })
415            })?;
416
417        let client_auth: KeyMaterialClientAuth =
418            postcard::from_bytes(&client_auth_bytes).map_err(|_| {
419                e!(Error::ClientAuthHeaderInvalid {
420                    value: client_auth_header.clone()
421                })
422            })?;
423
424        if client_auth.verify(io).is_ok() {
425            trace!(?client_auth.public_key, "authentication succeeded via keying material");
426            return Ok(SuccessfulAuthentication {
427                client_key: client_auth.public_key,
428                mechanism: Mechanism::SignedKeyMaterial,
429            });
430        }
431        // Verification not succeeding is part of normal operation: The TLS exporter isn't required to match.
432        // We'll fall back to verification that takes another round trip more time.
433    }
434
435    let challenge = ServerChallenge::new(&mut rand::rng());
436    write_frame(io, &challenge).await?;
437
438    let (_, frame) = read_frame(io, &[ClientAuth::TAG]).await?;
439    let client_auth: ClientAuth = deserialize_frame(frame)?;
440
441    if let Err(err) = client_auth.verify(&challenge) {
442        trace!(?client_auth.public_key, ?err, "authentication failed");
443        let denial = ServerDeniesAuth {
444            reason: "signature invalid".into(),
445        };
446        write_frame(io, denial.clone()).await?;
447        Err!(Error::ServerDeniedAuth {
448            reason: denial.reason
449        })
450    } else {
451        trace!(?client_auth.public_key, "authentication succeeded via challenge");
452        Ok(SuccessfulAuthentication {
453            client_key: client_auth.public_key,
454            mechanism: Mechanism::SignedChallenge,
455        })
456    }
457}

459#[cfg(feature = "server")]
460impl SuccessfulAuthentication {
461    pub async fn authorize_if(
462        self,
463        is_authorized: bool,
464        io: &mut (impl BytesStreamSink + ExportKeyingMaterial),
465    ) -> Result<PublicKey, Error> {
466        if is_authorized {
467            trace!("authorizing client");
468            write_frame(io, ServerConfirmsAuth).await?;
469            Ok(self.client_key)
470        } else {
471            trace!("denying client auth");
472            let denial = ServerDeniesAuth {
473                reason: "not authorized".into(),
474            };
475            write_frame(io, denial.clone()).await?;
476            Err!(Error::ServerDeniedAuth {
477                reason: denial.reason
478            })
479        }
480    }
481}

483async fn write_frame<F: serde::Serialize + Frame>(
484    io: &mut impl BytesStreamSink,
485    frame: F,
486) -> Result<(), Error> {
487    let mut bytes = BytesMut::new();
488    trace!(frame_type = ?F::TAG, "Writing frame");
489    F::TAG.write_to(&mut bytes);
490    let bytes = postcard::to_io(&frame, bytes.writer())
491        .expect("serialization failed") // buffer can't become "full" without being a critical failure, datastructures shouldn't ever fail serialization
492        .into_inner()
493        .freeze();
494    io.send(bytes)
495        .await
496        .map_err(|err| e!(Error::Websocket, err))?;
497    io.flush().await.map_err(|err| e!(Error::Websocket, err))?;
498    Ok(())
499}
500
501async fn read_frame(
502    io: &mut impl BytesStreamSink,
503    expected_types: &[FrameType],
504) -> Result<(FrameType, Bytes), Error> {
505    let mut payload = io
506        .try_next()
507        .await
508        .map_err(|err| e!(Error::Websocket, err))?
509        .ok_or_else(|| e!(Error::UnexpectedEnd))?;
510
511    let frame_type =
512        FrameType::from_bytes(&mut payload).map_err(|err| e!(Error::FrameTypeError, err))?;
513    trace!(?frame_type, "Reading frame");
514    n0_error::ensure!(
515        expected_types.contains(&frame_type),
516        e!(Error::UnexpectedFrameType {
517            frame_type,
518            expected_types: expected_types.to_vec()
519        })
520    );
521
522    Ok((frame_type, payload))
523}
524
525fn deserialize_frame<F: Frame + serde::de::DeserializeOwned>(frame: Bytes) -> Result<F, Error> {
526    postcard::from_bytes(&frame).map_err(|err| {
527        e!(Error::DeserializationError {
528            frame_type: F::TAG,
529            source: err
530        })
531    })
532}

631        restricted_to: Option<PublicKey>,
632    ) -> (Result<ServerConfirmsAuth>, Result<(PublicKey, Mechanism)>) {
633        let (client, server) = tokio::io::duplex(1024);
634
635        let mut client_io = Framed::new(client, LengthDelimitedCodec::new())
636            .map_ok(BytesMut::freeze)
637            .map_err(tokio_websockets::Error::Io)
638            .sink_map_err(tokio_websockets::Error::Io)
639            .with_shared_secret(client_shared_secret);
640        let mut server_io = Framed::new(server, LengthDelimitedCodec::new())
641            .map_ok(BytesMut::freeze)
642            .map_err(tokio_websockets::Error::Io)
643            .sink_map_err(tokio_websockets::Error::Io)
644            .with_shared_secret(server_shared_secret);
645
646        let client_auth_header = KeyMaterialClientAuth::new(secret_key, &client_io)
647            .map(KeyMaterialClientAuth::into_header_value);
648
649        n0_future::future::zip(
650            async {
651                super::clientside(&mut client_io, secret_key)
652                    .await
653                    .std_context("clientside")
654            }
655            .instrument(info_span!("clientside")),
656            async {
657                let auth_n = super::serverside(&mut server_io, client_auth_header)
658                    .await
659                    .std_context("serverside")?;
660                let mechanism = auth_n.mechanism;
661                let is_authorized = restricted_to.is_none_or(|key| key == auth_n.client_key);
662                let key = auth_n.authorize_if(is_authorized, &mut server_io).await?;
663                Ok((key, mechanism))
664            }
665            .instrument(info_span!("serverside")),
666        )
667        .await
668    }
