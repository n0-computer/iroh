7/// A URL identifying a relay server.
8///
9/// It is cheaply clonable, as the underlying type is wrapped into an `Arc`.
10/// The main type under the hood though is [`Url`], with a few custom tweaks:
11///
12/// - A relay URL is never a relative URL, so an implicit `.` is added at the end of the
13///   domain name if missing.
14///
15/// - [`fmt::Debug`] is implemented so it prints the URL rather than the URL struct fields.
16///   Useful when logging e.g. `Option<RelayUrl>`.
17///
18/// To create a [`RelayUrl`] use the `From<Url>` implementation.
19#[derive(
20    Clone, derive_more::Display, PartialEq, Eq, PartialOrd, Ord, Hash, Serialize, Deserialize,
21)]
22pub struct RelayUrl(Arc<Url>);
23
24impl From<Url> for RelayUrl {
25    fn from(mut url: Url) -> Self {
26        if let Some(domain) = url.domain() {
27            if !domain.ends_with('.') {
28                let domain = String::from(domain) + ".";
29
30                // This can fail, though it is unlikely the resulting URL is usable as a
31                // relay URL, probably it has the wrong scheme or is not a base URL or the
32                // like.  We don't do full URL validation however, so just silently leave
33                // this bad URL in place.  Something will fail later.
34                url.set_host(Some(&domain)).ok();
35            }
36        }
37        Self(Arc::new(url))
38    }
39}

41/// Can occur when parsing a string into a [`RelayUrl`].
42#[add_meta]
43#[derive(Error)]
44#[error(from_sources, std_sources)]
45#[allow(missing_docs)]
46pub enum RelayUrlParseError {
47    /// Parsing the URL failed.
48    #[display("Failed to parse")]
49    Parse { source: url::ParseError },
50}
51
52/// Support for parsing strings directly.
53///
54/// If you need more control over the error first create a [`Url`] and use [`RelayUrl::from`]
55/// instead.
56impl FromStr for RelayUrl {
57    type Err = RelayUrlParseError;
58
59    fn from_str(s: &str) -> Result<Self, Self::Err> {
60        let inner = Url::from_str(s)?;
61        Ok(RelayUrl::from(inner))
62    }
63}
64
65impl From<RelayUrl> for Url {
66    fn from(value: RelayUrl) -> Self {
67        Arc::unwrap_or_clone(value.0)
68    }
69}

71/// Dereferences to the wrapped [`Url`].
72///
73/// Note that [`DerefMut`] is not implemented on purpose, so this type has more flexibility
74/// to change the inner later.
75///
76/// [`DerefMut`]: std::ops::DerefMut
77impl Deref for RelayUrl {
78    type Target = Url;
79
80    fn deref(&self) -> &Self::Target {
81        &self.0
82    }
83}

85impl fmt::Debug for RelayUrl {
86    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
87        f.debug_tuple("RelayUrl")
88            .field(&DbgStr(self.0.as_str()))
89            .finish()
90    }
91}

93/// Helper struct to format a &str without allocating a String.
94///
95/// Maybe this is entirely unneeded and the compiler would be smart enough to never allocate
96/// the String anyway.  Who knows.  Writing this was faster than checking the assembler
97/// output.
98struct DbgStr<'a>(&'a str);
99
100impl fmt::Debug for DbgStr<'_> {
101    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
102        write!(f, r#""{}""#, self.0)
103    }
104}

106#[cfg(test)]

110    #[test]
111    fn test_relay_url_debug_display() {
112        let url = RelayUrl::from(Url::parse("https://example.com").unwrap());
113
114        assert_eq!(format!("{url:?}"), r#"RelayUrl("https://example.com./")"#);
115
116        assert_eq!(format!("{url}"), "https://example.com./");
117    }
118
119    #[test]
120    fn test_relay_url_absolute() {
121        let url = RelayUrl::from(Url::parse("https://example.com").unwrap());
122
123        assert_eq!(url.domain(), Some("example.com."));
124
125        let url1 = RelayUrl::from(Url::parse("https://example.com.").unwrap());
126        assert_eq!(url, url1);
127
128        let url2 = RelayUrl::from(Url::parse("https://example.com./").unwrap());
129        assert_eq!(url, url2);
130
131        let url3 = RelayUrl::from(Url::parse("https://example.com/").unwrap());
132        assert_eq!(url, url3);
133    }
