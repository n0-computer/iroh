1//! TLS configuration for iroh.
2//!
3//! Currently there is one mechanisms available
4//! - Raw Public Keys, using the TLS extension described in [RFC 7250]
5//!
6//! [RFC 7250]: https://datatracker.ietf.org/doc/html/rfc7250
7
8use std::sync::Arc;
9
10use iroh_base::SecretKey;
11use quinn::crypto::rustls::{QuicClientConfig, QuicServerConfig};
12use tracing::warn;
13
14use self::resolver::AlwaysResolvesCert;
15
16pub(crate) mod name;
17mod resolver;
18mod verifier;
19
20/// Maximum amount of TLS tickets we will cache (by default) for 0-RTT connection
21/// establishment.
22///
23/// 8 tickets per remote endpoint, 32 different endpoints would max out the required storage:
24/// ~200 bytes per session + certificates (which are ~387 bytes)
25/// So 8 * 32 * (200 + 387) = 150.272 bytes, assuming pointers to certificates
26/// are never aliased pointers (they're Arc'ed).
27/// I think 150KB is an acceptable default upper limit for such a cache.
28const MAX_TLS_TICKETS: usize = 8 * 32;
29
30/// TLS Authentication mechanism
31#[derive(Default, Debug, Copy, Clone, PartialEq, Eq)]
32pub(crate) enum Authentication {
33    /// RFC 7250 TLS extension: Raw Public Keys.
34    #[default]
35    RawPublicKey,
36}
37
38/// Configuration for TLS.
39///
40/// The main point of this struct is to keep state that should be kept the same
41/// over multiple TLS sessions the same.
42/// E.g. the `server_verifier` and `client_verifier` Arc pointers are checked to be
43/// the same between different TLS session calls with 0-RTT data in rustls.
44/// This makes sure that's the case.
45#[derive(Debug)]
46pub(crate) struct TlsConfig {
47    pub(crate) auth: Authentication,
48    pub(crate) secret_key: SecretKey,
49    cert_resolver: Arc<AlwaysResolvesCert>,
50    server_verifier: Arc<verifier::ServerCertificateVerifier>,
51    client_verifier: Arc<verifier::ClientCertificateVerifier>,
52    session_store: Arc<dyn rustls::client::ClientSessionStore>,
53}
54
55impl TlsConfig {
56    pub(crate) fn new(auth: Authentication, secret_key: SecretKey) -> Self {
57        let cert_resolver = Arc::new(
58            AlwaysResolvesCert::new(auth, &secret_key).expect("Client cert key DER is valid; qed"),
59        );
60        Self {
61            auth,
62            secret_key,
63            cert_resolver,
64            server_verifier: Arc::new(verifier::ServerCertificateVerifier::new(auth)),
65            client_verifier: Arc::new(verifier::ClientCertificateVerifier::new(auth)),
66            session_store: Arc::new(rustls::client::ClientSessionMemoryCache::new(
67                MAX_TLS_TICKETS,
68            )),
69        }
70    }
71
72    /// Create a TLS client configuration.
73    ///
74    /// If *keylog* is `true` this will enable logging of the pre-master key to the file in the
75    /// `SSLKEYLOGFILE` environment variable.  This can be used to inspect the traffic for
76    /// debugging purposes.
77    pub(crate) fn make_client_config(
78        &self,
79        alpn_protocols: Vec<Vec<u8>>,
80        keylog: bool,
81    ) -> QuicClientConfig {
82        let mut crypto = rustls::ClientConfig::builder_with_provider(Arc::new(
83            rustls::crypto::ring::default_provider(),
84        ))
85        .with_protocol_versions(verifier::PROTOCOL_VERSIONS)
86        .expect("version supported by ring")
87        .dangerous()
88        .with_custom_certificate_verifier(self.server_verifier.clone())
89        .with_client_cert_resolver(self.cert_resolver.clone());
90        crypto.alpn_protocols = alpn_protocols;
91
92        // TODO: enable/disable 0-RTT/storing tickets
93        crypto.resumption = rustls::client::Resumption::store(self.session_store.clone());
94        crypto.enable_early_data = true;
95
96        if keylog {
97            warn!("enabling SSLKEYLOGFILE for TLS pre-master keys");
98            crypto.key_log = Arc::new(rustls::KeyLogFile::new());
99        }
100
101        crypto
102            .try_into()
103            .expect("expected to have a TLS1.3-compatible crypto provider set (hardcoded)")
104    }
105
106    /// Create a TLS server configuration.
107    ///
108    /// If *keylog* is `true` this will enable logging of the pre-master key to the file in the
109    /// `SSLKEYLOGFILE` environment variable.  This can be used to inspect the traffic for
110    /// debugging purposes.
111    pub(crate) fn make_server_config(
112        &self,
113        alpn_protocols: Vec<Vec<u8>>,
114        keylog: bool,
115    ) -> QuicServerConfig {
116        let mut crypto = rustls::ServerConfig::builder_with_provider(Arc::new(
117            rustls::crypto::ring::default_provider(),
118        ))
119        .with_protocol_versions(verifier::PROTOCOL_VERSIONS)
120        .expect("fixed config")
121        .with_client_cert_verifier(self.client_verifier.clone())
122        .with_cert_resolver(self.cert_resolver.clone());
123        crypto.alpn_protocols = alpn_protocols;
124        if keylog {
125            warn!("enabling SSLKEYLOGFILE for TLS pre-master keys");
126            crypto.key_log = Arc::new(rustls::KeyLogFile::new());
127        }
128
129        // must be u32::MAX or 0 (the default). Any other value panics with QUIC
130        // This is specified in RFC 9001: https://www.rfc-editor.org/rfc/rfc9001#section-4.6.1
131        crypto.max_early_data_size = u32::MAX;
132        crypto
133            .try_into()
134            .expect("expected to have a TLS1.3-compatible crypto provider set (hardcoded)")
135    }
136}
