37/// Map containing all the state for endpoints.
38///
39/// - Has actors which each manage all the connection state for a remote endpoint.
40///
41/// - Has the mapped addresses we use to refer to non-IP transports destinations into IPv6
42///   addressing space that is used by Quinn.
43#[derive(Debug)]
44pub(crate) struct EndpointMap {
45    //
46    // State we keep about remote endpoints.
47    //
48    /// The actors tracking each remote endpoint.
49    actor_handles: Mutex<FxHashMap<EndpointId, EndpointStateHandle>>,
50    /// The mapping between [`EndpointId`]s and [`EndpointIdMappedAddr`]s.
51    pub(super) endpoint_mapped_addrs: AddrMap<EndpointId, EndpointIdMappedAddr>,
52    /// The mapping between endpoints via a relay and their [`RelayMappedAddr`]s.
53    pub(super) relay_mapped_addrs: AddrMap<(RelayUrl, EndpointId), RelayMappedAddr>,
54
55    //
56    // State needed to start a new EndpointStateHandle.
57    //
58    /// The endpoint ID of the local endpoint.
59    local_endpoint_id: EndpointId,
60    metrics: Arc<MagicsockMetrics>,
61    local_addrs: n0_watcher::Direct<BTreeSet<DirectAddr>>,
62    disco: DiscoState,
63    sender: TransportsSender,
64    discovery: ConcurrentDiscovery,
65}

95    /// Removes the handles for terminated [`EndpointStateActor`]s from the endpoint map.
96    ///
97    /// This should be called periodically to remove handles to endpoint state actors
98    /// that have shutdown after their idle timeout expired.
99    pub(super) fn remove_closed_endpoint_state_actors(&self) {
100        let mut handles = self.actor_handles.lock().expect("poisoned");
101        handles.retain(|_eid, handle| !handle.sender.is_closed())
102    }
103
104    /// Returns the sender for the [`EndpointStateActor`].
105    ///
106    /// If needed a new actor is started on demand.
107    ///
108    /// [`EndpointStateActor`]: endpoint_state::EndpointStateActor
109    pub(super) fn endpoint_state_actor(
110        &self,
111        eid: EndpointId,
112    ) -> mpsc::Sender<EndpointStateMessage> {
113        let mut handles = self.actor_handles.lock().expect("poisoned");
114        match handles.entry(eid) {
115            hash_map::Entry::Occupied(mut entry) => {
116                if let Some(sender) = entry.get().sender.get() {
117                    sender
118                } else {
119                    // The actor is dead: Start a new actor.
120                    let (handle, sender) = self.start_endpoint_state_actor(eid);
121                    entry.insert(handle);
122                    sender
123                }
124            }
125            hash_map::Entry::Vacant(entry) => {
126                let (handle, sender) = self.start_endpoint_state_actor(eid);
127                entry.insert(handle);
128                sender
129            }
130        }
131    }
132
133    /// Starts a new endpoint state actor and returns a handle and a sender.
134    ///
135    /// The handle is not inserted into the endpoint map, this must be done by the caller of this function.
136    fn start_endpoint_state_actor(
137        &self,
138        eid: EndpointId,
139    ) -> (EndpointStateHandle, mpsc::Sender<EndpointStateMessage>) {
140        // Ensure there is a EndpointMappedAddr for this EndpointId.
141        self.endpoint_mapped_addrs.get(&eid);
142        let handle = EndpointStateActor::new(
143            eid,
144            self.local_endpoint_id,
145            self.local_addrs.clone(),
146            self.disco.clone(),
147            self.relay_mapped_addrs.clone(),
148            self.metrics.clone(),
149            self.sender.clone(),
150            self.discovery.clone(),
151        )
152        .start();
153        let sender = handle.sender.get().expect("just created");
154        (handle, sender)
155    }

157    pub(super) fn handle_ping(&self, msg: disco::Ping, sender: EndpointId, src: transports::Addr) {
158        if msg.endpoint_key != sender {
159            warn!("DISCO Ping EndpointId mismatch, ignoring ping");
160            return;
161        }
162        let endpoint_state = self.endpoint_state_actor(sender);
163        if let Err(err) = endpoint_state.try_send(EndpointStateMessage::PingReceived(msg, src)) {
164            // TODO: This is really, really bad and will drop pings under load.  But
165            //    DISCO pings are going away with QUIC-NAT-TRAVERSAL so I don't care.
166            warn!("DISCO Ping dropped: {err:#}");
167        }
168    }

171        let actor = self.endpoint_state_actor(sender);
172        if let Err(err) = actor.try_send(EndpointStateMessage::PongReceived(msg, src)) {

179    pub(super) fn handle_call_me_maybe(
180        &self,
181        msg: disco::CallMeMaybe,
182        sender: EndpointId,
183        src: transports::Addr,
184    ) {
185        if !src.is_relay() {
186            warn!("DISCO CallMeMaybe packets should only come via relay");
187            return;
188        }
189        let actor = self.endpoint_state_actor(sender);
190        if let Err(err) = actor.try_send(EndpointStateMessage::CallMeMaybeReceived(msg)) {
191            // TODO: This is bad and will drop call-me-maybe's under load.  But
192            //    DISCO CallMeMaybe going away with QUIC-NAT-TRAVERSAL so I don't care.
193            warn!("DISCO CallMeMaybe dropped: {err:#}");
194        }
195    }

198/// The origin or *source* through which an address associated with a remote endpoint
199/// was discovered.
200///
201/// An aggregate of the [`Source`]s of all the addresses of an endpoint describe the
202/// [`Source`]s of the endpoint itself.
203///
204/// A [`Source`] helps track how and where an address was learned. Multiple
205/// sources can be associated with a single address, if we have discovered this
206/// address through multiple means.
207#[derive(Serialize, Deserialize, strum::Display, Debug, Clone, Eq, PartialEq, Hash)]
208#[strum(serialize_all = "kebab-case")]
209#[allow(private_interfaces)]
210pub enum Source {
211    /// An endpoint communicated with us first via UDP.
212    Udp,
213    /// An endpoint communicated with us first via relay.
214    Relay,
215    /// Application layer added the address directly.
216    App,
217    /// The address was discovered by a discovery service.
218    #[strum(serialize = "{name}")]
219    Discovery {
220        /// The name of the discovery service that discovered the address.
221        name: String,
222    },
223    /// Application layer with a specific name added the endpoint directly.
224    #[strum(serialize = "{name}")]
225    NamedApp {
226        /// The name of the application that added the endpoint
227        name: String,
228    },
229    /// The address was advertised by a call-me-maybe DISCO message.
230    #[strum(serialize = "CallMeMaybe")]
231    CallMeMaybe {
232        /// private marker
233        _0: Private,
234    },
235    /// We received a ping on the path.
236    #[strum(serialize = "Ping")]
237    Ping {
238        /// private marker
239        _0: Private,
240    },
241    /// We established a connection on this address.
242    ///
243    /// Currently this means the path was in uses as [`PathId::ZERO`] when the a connection
244    /// was added to the `EndpointStateActor`.
245    ///
246    /// [`PathId::ZERO`]: quinn_proto::PathId::ZERO
247    #[strum(serialize = "Connection")]
248    Connection {
249        /// private marker
250        _0: Private,
251    },
252}

258/// An (Ip, Port) pair.
259///
260/// NOTE: storing an [`IpPort`] is safer than storing a [`SocketAddr`] because for IPv6 socket
261/// addresses include fields that can't be assumed consistent even within a single connection.
262#[derive(Debug, derive_more::Display, Clone, Copy, Hash, PartialEq, Eq, PartialOrd, Ord)]
263#[display("{}", SocketAddr::from(*self))]
264pub struct IpPort {
265    ip: IpAddr,
266    port: u16,
267}
268
269impl From<SocketAddr> for IpPort {
270    fn from(socket_addr: SocketAddr) -> Self {
271        Self {
272            ip: socket_addr.ip(),
273            port: socket_addr.port(),
274        }
275    }
276}

278impl From<IpPort> for SocketAddr {
279    fn from(ip_port: IpPort) -> Self {
280        let IpPort { ip, port } = ip_port;
281        (ip, port).into()
282    }
283}
