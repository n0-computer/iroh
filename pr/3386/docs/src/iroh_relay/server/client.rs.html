50/// The [`Server`] side representation of a [`Client`]'s connection.
51///
52/// [`Server`]: crate::server::Server
53/// [`Client`]: crate::client::Client
54#[derive(Debug)]
55pub(super) struct Client {
56    /// Identity of the connected peer.
57    node_id: NodeId,
58    /// Connection identifier.
59    connection_id: u64,
60    /// Used to close the connection loop.
61    done: CancellationToken,
62    /// Actor handle.
63    handle: AbortOnDropHandle<()>,
64    /// Queue of packets intended for the client.
65    send_queue: mpsc::Sender<Packet>,
66    /// Queue of disco packets intended for the client.
67    disco_send_queue: mpsc::Sender<Packet>,
68    /// Channel to notify the client that a previous sender has disconnected.
69    peer_gone: mpsc::Sender<NodeId>,
70}
71
72impl Client {
73    /// Creates a client from a connection & starts a read and write loop to handle io to and from
74    /// the client
75    /// Call [`Client::shutdown`] to close the read and write loops before dropping the [`Client`]
76    pub(super) fn new(
77        config: Config,
78        connection_id: u64,
79        clients: &Clients,
80        metrics: Arc<Metrics>,
81    ) -> Client {
82        let Config {
83            node_id,
84            stream,
85            write_timeout,
86            channel_capacity,
87        } = config;
88
89        let done = CancellationToken::new();
90        let (send_queue_s, send_queue_r) = mpsc::channel(channel_capacity);
91
92        let (disco_send_queue_s, disco_send_queue_r) = mpsc::channel(channel_capacity);
93        let (peer_gone_s, peer_gone_r) = mpsc::channel(channel_capacity);
94
95        let actor = Actor {
96            stream,
97            timeout: write_timeout,
98            send_queue: send_queue_r,
99            disco_send_queue: disco_send_queue_r,
100            node_gone: peer_gone_r,
101            node_id,
102            connection_id,
103            clients: clients.clone(),
104            client_counter: ClientCounter::default(),
105            ping_tracker: PingTracker::default(),
106            metrics,
107        };
108
109        // start io loop
110        let io_done = done.clone();
111        let handle = tokio::task::spawn(actor.run(io_done).instrument(tracing::info_span!(
112            "client-connection-actor",
113            remote_node = %node_id.fmt_short(),
114            connection_id = connection_id
115        )));
116
117        Client {
118            node_id,
119            connection_id,
120            handle: AbortOnDropHandle::new(handle),
121            done,
122            send_queue: send_queue_s,
123            disco_send_queue: disco_send_queue_s,
124            peer_gone: peer_gone_s,
125        }
126    }
127
128    pub(super) fn connection_id(&self) -> u64 {
129        self.connection_id
130    }
131
132    /// Shutdown the reader and writer loops and closes the connection.
133    ///
134    /// Any shutdown errors will be logged as warnings.
135    pub(super) async fn shutdown(self) {
136        self.start_shutdown();
137        if let Err(e) = self.handle.await {
138            warn!(
139                remote_node = %self.node_id.fmt_short(),
140                "error closing actor loop: {e:#?}",
141            );
142        };
143    }
144
145    /// Starts the process of shutdown.
146    pub(super) fn start_shutdown(&self) {
147        self.done.cancel();
148    }

151        &self,
152        src: NodeId,

266/// Manages all the reads and writes to this client. It periodically sends a `KEEP_ALIVE`
267/// message to the client to keep the connection alive.
268///
269/// Call `run` to manage the input and output to and from the connection and the server.
270/// Once it hits its first write error or error receiving off a channel,
271/// it errors on return.
272/// If writes do not complete in the given `timeout`, it will also error.
273///
274/// On the "write" side, the [`Actor`] can send the client:
275///  - a KEEP_ALIVE frame
276///  - a PEER_GONE frame to inform the client that a peer they have previously sent messages to
277///    is gone from the network
278///  - packets from other peers
279///
280/// On the "read" side, it can:
281///     - receive a ping and write a pong back
282///     to speak to the node ID associated with that client.
283#[derive(Debug)]
284struct Actor {
285    /// IO Stream to talk to the client
286    stream: RelayedStream,
287    /// Maximum time we wait to complete a write to the client
288    timeout: Duration,
289    /// Packets queued to send to the client
290    send_queue: mpsc::Receiver<Packet>,
291    /// Important packets queued to send to the client
292    disco_send_queue: mpsc::Receiver<Packet>,
293    /// Notify the client that a previous sender has disconnected
294    node_gone: mpsc::Receiver<NodeId>,
295    /// [`NodeId`] of this client
296    node_id: NodeId,
297    /// Connection identifier.
298    connection_id: u64,
299    /// Reference to the other connected clients.
300    clients: Clients,
301    /// Statistics about the connected clients
302    client_counter: ClientCounter,
303    ping_tracker: PingTracker,
304    metrics: Arc<Metrics>,
305}
306
307impl Actor {
308    async fn run(mut self, done: CancellationToken) {
309        // Note the accept and disconnects metrics must be in a pair.  Technically the
310        // connection is accepted long before this in the HTTP server, but it is clearer to
311        // handle the metric here.
312        self.metrics.accepts.inc();
313        if self.client_counter.update(self.node_id) {
314            self.metrics.unique_client_keys.inc();
315        }
316        match self.run_inner(done).await {
317            Err(e) => {
318                warn!("actor errored {e:#}, exiting");
319            }
320            Ok(()) => {
321                debug!("actor finished, exiting");
322            }
323        }
324
325        self.clients.unregister(self.connection_id, self.node_id);
326        self.metrics.disconnects.inc();
327    }
328
329    async fn run_inner(&mut self, done: CancellationToken) -> Result<(), RunError> {
330        use snafu::ResultExt;
331
332        // Add some jitter to ping pong interactions, to avoid all pings being sent at the same time
333        let next_interval = || {
334            let random_secs = rand::rngs::OsRng.gen_range(1..=5);
335            Duration::from_secs(random_secs) + PING_INTERVAL
336        };
337
338        let mut ping_interval = tokio::time::interval(next_interval());
339        // ticks immediately
340        ping_interval.set_missed_tick_behavior(MissedTickBehavior::Delay);
341        ping_interval.tick().await;
342
343        loop {
344            tokio::select! {
345                biased;
346
347                _ = done.cancelled() => {
348                    trace!("actor loop cancelled, exiting");
349                    // final flush
350                    self.stream.flush().await.map_err(|_| FlushSnafu.build())?;
351                    break;
352                }
353                maybe_frame = self.stream.next() => {
354                    self.handle_frame(maybe_frame).await?;
355                    // reset the ping interval, we just received a message
356                    ping_interval.reset();
357                }
358                // First priority, disco packets
359                packet = self.disco_send_queue.recv() => {
360                    let packet = packet.ok_or(DiscoSendQueuePacketDropSnafu.build())?;
361                    self.send_disco_packet(packet).await.context(DiscoPacketSendSnafu)?;
362                }
363                // Second priority, sending regular packets
364                packet = self.send_queue.recv() => {
365                    let packet = packet.ok_or(SendQueuePacketDropSnafu.build())?;
366                    self.send_packet(packet).await.context(PacketSendSnafu)?;
367                }
368                // Last priority, sending left nodes
369                node_id = self.node_gone.recv() => {
370                    let node_id = node_id.ok_or(NodeGoneDropSnafu.build())?;
371                    trace!("node_id gone: {:?}", node_id);
372                    self.write_frame(RelayToClientMsg::NodeGone(node_id)).await.context(NodeGoneWriteFrameSnafu)?;
373                }
374                _ = self.ping_tracker.timeout() => {
375                    trace!("pong timed out");
376                    break;
377                }
378                _ = ping_interval.tick() => {
379                    trace!("keep alive ping");
380                    // new interval
381                    ping_interval.reset_after(next_interval());
382                    let data = self.ping_tracker.new_ping();
383                    self.write_frame(RelayToClientMsg::Ping(data)).await.context(KeepAliveWriteFrameSnafu)?;
384                }
385            }
386
387            self.stream
388                .flush()
389                .await
390                .map_err(|_| TickFlushSnafu.build())?;
391        }
392        Ok(())
393    }
394
395    /// Writes the given frame to the connection.
396    ///
397    /// Errors if the send does not happen within the `timeout` duration
398    async fn write_frame(&mut self, frame: RelayToClientMsg) -> Result<(), WriteFrameError> {
399        tokio::time::timeout(self.timeout, self.stream.send(frame)).await??;
400        Ok(())
401    }
402
403    /// Writes contents to the client in a `RECV_PACKET` frame.
404    ///
405    /// Errors if the send does not happen within the `timeout` duration
406    /// Does not flush.
407    async fn send_raw(&mut self, packet: Packet) -> Result<(), WriteFrameError> {
408        let src_key = packet.src;
409        let content = packet.data;
410
411        if let Ok(len) = content.len().try_into() {
412            self.metrics.bytes_sent.inc_by(len);
413        }
414        self.write_frame(RelayToClientMsg::ReceivedPacket { src_key, content })
415            .await
416    }
417
418    async fn send_packet(&mut self, packet: Packet) -> Result<(), WriteFrameError> {
419        trace!("send packet");
420        match self.send_raw(packet).await {
421            Ok(()) => {
422                self.metrics.send_packets_sent.inc();
423                Ok(())
424            }
425            Err(err) => {
426                self.metrics.send_packets_dropped.inc();
427                Err(err)
428            }
429        }
430    }
431
432    async fn send_disco_packet(&mut self, packet: Packet) -> Result<(), WriteFrameError> {
433        trace!("send disco packet");
434        match self.send_raw(packet).await {
435            Ok(()) => {
436                self.metrics.disco_packets_sent.inc();
437                Ok(())
438            }
439            Err(err) => {
440                self.metrics.disco_packets_dropped.inc();
441                Err(err)
442            }
443        }
444    }
445
446    /// Handles frame read results.
447    async fn handle_frame(
448        &mut self,
449        maybe_frame: Option<Result<ClientToRelayMsg, RelayRecvError>>,
450    ) -> Result<(), HandleFrameError> {
451        trace!(?maybe_frame, "handle incoming frame");
452        let frame = match maybe_frame {
453            Some(frame) => frame?,
454            None => return Err(StreamTerminatedSnafu.build()),
455        };
456
457        match frame {
458            ClientToRelayMsg::SendPacket { dst_key, packet } => {
459                let packet_len = packet.len();
460                if let Err(err @ ForwardPacketError { .. }) =
461                    self.handle_frame_send_packet(dst_key, packet)
462                {
463                    warn!("failed to handle send packet frame: {err:#}");
464                }
465                self.metrics.bytes_recv.inc_by(packet_len as u64);
466            }
467            ClientToRelayMsg::Ping(data) => {
468                self.metrics.got_ping.inc();
469                // TODO: add rate limiter
470                self.write_frame(RelayToClientMsg::Pong(data)).await?;
471                self.metrics.sent_pong.inc();
472            }
473            ClientToRelayMsg::Pong(data) => {
474                self.ping_tracker.pong_received(data);
475            }
476        }
477        Ok(())
478    }
479
480    fn handle_frame_send_packet(&self, dst: NodeId, data: Bytes) -> Result<(), ForwardPacketError> {
481        if disco::looks_like_disco_wrapper(&data) {
482            self.metrics.disco_packets_recv.inc();
483            self.clients
484                .send_disco_packet(dst, data, self.node_id, &self.metrics)?;
485        } else {
486            self.metrics.send_packets_recv.inc();
487            self.clients
488                .send_packet(dst, data, self.node_id, &self.metrics)?;
489        }
490        Ok(())
491    }
492}

524/// Tracks how many unique nodes have been seen during the last day.
525#[derive(Debug)]
526struct ClientCounter {
527    clients: HashSet<NodeId>,
528    last_clear_date: Date,
529}
530
531impl Default for ClientCounter {
532    fn default() -> Self {
533        Self {
534            clients: HashSet::new(),
535            last_clear_date: OffsetDateTime::now_utc().date(),
536        }
537    }
538}
539
540impl ClientCounter {
541    fn check_and_clear(&mut self) {
542        let today = OffsetDateTime::now_utc().date();
543        if today != self.last_clear_date {
544            self.clients.clear();
545            self.last_clear_date = today;
546        }
547    }
548
549    /// Marks this node as seen, returns whether it is new today or not.
550    fn update(&mut self, client: NodeId) -> bool {
551        self.check_and_clear();
552        self.clients.insert(client)
553    }
554}

567    async fn recv_frame<
568        E: snafu::Error + Sync + Send + 'static,
569        S: Stream<Item = Result<RelayToClientMsg, E>> + Unpin,
570    >(
571        frame_type: FrameType,
572        mut stream: S,
573    ) -> Result<RelayToClientMsg> {
574        match stream.next().await {
575            Some(Ok(frame)) => {
576                if frame_type != frame.typ() {
577                    snafu::whatever!(
578                        "Unexpected frame, got {:?}, but expected {:?}",
579                        frame.typ(),
580                        frame_type
581                    );
582                }
583                Ok(frame)
584            }
585            Some(Err(err)) => Err(err).e(),
586            None => snafu::whatever!("Unexpected EOF, expected frame {frame_type:?}"),
587        }
588    }

709    async fn test_rate_limit() -> Result {
710        const LIMIT: u32 = 50;
711        const MAX_FRAMES: u32 = 100;
712
713        // Build the rate limited stream.
714        let (io_read, io_write) = tokio::io::duplex((LIMIT * MAX_FRAMES) as _);
715        let mut frame_writer = Conn::test(io_write);
716        // Rate limiter allowing LIMIT bytes/s
717        let mut stream = RelayedStream::test_limited(io_read, LIMIT / 10, LIMIT)?;
718
719        // Prepare a frame to send, assert its size.
720        let data = Bytes::from_static(b"hello world!1eins");
721        let target = SecretKey::generate(rand::thread_rng()).public();
722        let frame = ClientToRelayMsg::SendPacket {
723            dst_key: target,
724            packet: data.clone(),
725        };
726        let frame_len = frame.to_bytes().len();
727        assert_eq!(frame_len, LIMIT as usize);
728
729        // Send a frame, it should arrive.
730        info!("-- send packet");
731        frame_writer.send(frame.clone()).await.context("send")?;
732        frame_writer.flush().await.context("flush")?;
733        let recv_frame = tokio::time::timeout(Duration::from_millis(500), stream.next())
734            .await
735            .expect("timeout")
736            .expect("option")
737            .expect("ok");
738        assert_eq!(recv_frame, frame);
739
740        // Next frame does not arrive.
741        info!("-- send packet");
742        frame_writer.send(frame.clone()).await.context("send")?;
743        frame_writer.flush().await.context("flush")?;
744        let res = tokio::time::timeout(Duration::from_millis(100), stream.next()).await;
745        assert!(res.is_err(), "expecting a timeout");
746        info!("-- timeout happened");
747
748        // Wait long enough.
749        info!("-- sleep");
750        tokio::time::sleep(Duration::from_secs(1)).await;
751
752        // Frame arrives.
753        let recv_frame = tokio::time::timeout(Duration::from_millis(500), stream.next())
754            .await
755            .expect("timeout")
756            .expect("option")
757            .expect("ok");
