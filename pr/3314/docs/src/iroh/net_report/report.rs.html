13/// A net_report report.
14#[derive(Default, Debug, PartialEq, Eq, Clone)]
15pub struct Report {
16    /// A UDP STUN round trip completed.
17    pub udp: bool,
18    /// An IPv6 round trip completed.
19    pub ipv6: bool,
20    /// An IPv4 round trip completed.
21    pub ipv4: bool,
22    /// An IPv6 packet was able to be sent
23    pub ipv6_can_send: bool,
24    /// an IPv4 packet was able to be sent
25    pub ipv4_can_send: bool,
26    /// could bind a socket to ::1
27    pub os_has_ipv6: bool,
28    /// Whether the reported public address differs when probing different servers (on IPv4).
29    pub mapping_varies_by_dest_ip: Option<bool>,
30    /// Whether the reported public address differs when probing different servers (on IPv6).
31    ///
32    /// Note that we don't really expect this to happen and are merely logging this if
33    /// detecting rather than using it.  For now.
34    pub mapping_varies_by_dest_ipv6: Option<bool>,
35    /// Probe indicating the presence of port mapping protocols on the LAN.
36    #[cfg(not(wasm_browser))]
37    pub portmap_probe: Option<portmapper::ProbeOutput>,
38    /// `None` for unknown
39    pub preferred_relay: Option<RelayUrl>,
40    /// keyed by relay Url
41    pub relay_latency: RelayLatencies,
42    /// ip:port of global IPv4
43    pub global_v4: Option<SocketAddrV4>,
44    /// `[ip]:port` of global IPv6
45    pub global_v6: Option<SocketAddrV6>,
46    /// CaptivePortal is set when we think there's a captive portal that is
47    /// intercepting HTTP traffic.
48    pub captive_portal: Option<bool>,
49}

57impl Report {
58    /// Updates a net_report [`Report`] with a new [`ProbeReport`].
59    pub(super) fn update(&mut self, probe_report: &ProbeReport) {
60        let relay_node = &probe_report.node;
61        if let Some(latency) = probe_report.latency {
62            self.relay_latency
63                .update_relay(relay_node.url.clone(), latency, probe_report.probe);
64
65            #[cfg(not(wasm_browser))]
66            if matches!(probe_report.probe, Probe::QadIpv4 | Probe::QadIpv6) {
67                self.udp = true;
68
69                match probe_report.addr {
70                    Some(SocketAddr::V4(ipp)) => {
71                        self.ipv4 = true;
72                        if self.global_v4.is_none() {
73                            self.global_v4 = Some(ipp);
74                        } else if self.global_v4 != Some(ipp) {
75                            self.mapping_varies_by_dest_ip = Some(true);
76                        } else if self.mapping_varies_by_dest_ip.is_none() {
77                            self.mapping_varies_by_dest_ip = Some(false);
78                        }
79                    }
80                    Some(SocketAddr::V6(ipp)) => {
81                        self.ipv6 = true;
82                        if self.global_v6.is_none() {
83                            self.global_v6 = Some(ipp);
84                        } else if self.global_v6 != Some(ipp) {
85                            self.mapping_varies_by_dest_ipv6 = Some(true);
86                            warn!("IPv6 Address detected by STUN varies by destination");
87                        } else if self.mapping_varies_by_dest_ipv6.is_none() {
88                            self.mapping_varies_by_dest_ipv6 = Some(false);
89                        }
90                    }
91                    None => {
92                        // If we are here we had a relay server latency reported from a STUN probe.
93                        // Thus we must have a reported address.
94                        debug_assert!(probe_report.addr.is_some());
95                    }
96                }
97            }
98        }
99        self.ipv4_can_send |= probe_report.ipv4_can_send;
100        self.ipv6_can_send |= probe_report.ipv6_can_send;
101    }
102}

104/// Latencies per relay node.
105#[derive(Debug, Default, PartialEq, Eq, Clone)]
106pub struct RelayLatencies {
107    #[cfg(not(wasm_browser))]
108    ipv4: BTreeMap<RelayUrl, Duration>,
109    #[cfg(not(wasm_browser))]
110    ipv6: BTreeMap<RelayUrl, Duration>,
111    https: BTreeMap<RelayUrl, Duration>,
112}
113
114impl RelayLatencies {
115    /// Updates a relay's latency, if it is faster than before.
116    pub(super) fn update_relay(&mut self, url: RelayUrl, latency: Duration, probe: Probe) {
117        let list = match probe {
118            Probe::Https => &mut self.https,
119            #[cfg(not(wasm_browser))]
120            Probe::QadIpv4 => &mut self.ipv4,
121            #[cfg(not(wasm_browser))]
122            Probe::QadIpv6 => &mut self.ipv6,
123        };
124        let old_latency = list.entry(url).or_insert(latency);
125        if latency < *old_latency {
126            *old_latency = latency;
127        }
128    }
129
130    /// Merges another [`RelayLatencies`] into this one.
131    ///
132    /// For each relay the latency is updated using [`RelayLatencies::update_relay`].
133    pub(super) fn merge(&mut self, other: &RelayLatencies) {
134        for (url, latency) in other.https.iter() {
135            self.update_relay(url.clone(), *latency, Probe::Https);
136        }
137        #[cfg(not(wasm_browser))]
138        for (url, latency) in other.ipv4.iter() {
139            self.update_relay(url.clone(), *latency, Probe::QadIpv4);
140        }
141        #[cfg(not(wasm_browser))]
142        for (url, latency) in other.ipv6.iter() {
143            self.update_relay(url.clone(), *latency, Probe::QadIpv6);
144        }
145    }
146
147    /// Returns an iterator over all the relays and their latencies.
148    #[cfg(not(wasm_browser))]
149    pub fn iter(&self) -> impl Iterator<Item = (&'_ RelayUrl, Duration)> + '_ {
150        self.https
151            .iter()
152            .chain(self.ipv4.iter())
153            .chain(self.ipv6.iter())
154            .map(|(k, v)| (k, *v))
155    }
156
157    /// Returns an iterator over all the relays and their latencies.
158    #[cfg(wasm_browser)]
159    pub fn iter(&self) -> impl Iterator<Item = (&'_ RelayUrl, Duration)> + '_ {
160        self.https.iter().map(|(k, v)| (k, *v))
161    }
162
163    #[cfg(not(wasm_browser))]
164    pub(super) fn is_empty(&self) -> bool {
165        self.https.is_empty() && self.ipv4.is_empty() && self.ipv6.is_empty()
166    }
167
168    #[cfg(wasm_browser)]
169    pub(super) fn is_empty(&self) -> bool {
170        self.https.is_empty()
171    }
172
173    #[cfg(test)]
174    pub(super) fn len(&self) -> usize {
175        self.https.len() + self.ipv4.len() + self.ipv6.len()
176    }
177
178    /// Returns the lowest latency across records.
179    pub(super) fn get(&self, url: &RelayUrl) -> Option<Duration> {
180        let mut list = Vec::with_capacity(3);
181        if let Some(val) = self.https.get(url) {
182            list.push(*val);
183        }
184        #[cfg(not(wasm_browser))]
185        if let Some(val) = self.ipv4.get(url) {
186            list.push(*val);
187        }
188        #[cfg(not(wasm_browser))]
189        if let Some(val) = self.ipv6.get(url) {
190            list.push(*val);
191        }
192        list.into_iter().min()
193    }
194
195    #[cfg(not(wasm_browser))]
196    pub(super) fn ipv4(&self) -> &BTreeMap<RelayUrl, Duration> {
197        &self.ipv4
198    }
199
200    #[cfg(not(wasm_browser))]
201    pub(super) fn ipv6(&self) -> &BTreeMap<RelayUrl, Duration> {
202        &self.ipv6
203    }
