2//!
3//! All the probes try and establish the latency to the relay servers.  Preferably the STUN
4//! probes work and we also learn about our public IP addresses and ports.  But fallback
5//! probes for HTTPS and ICMP exist as well.

16/// The retransmit interval used when net_report first runs.
17///
18/// We have no past context to work with, and we want answers relatively quickly, so it's
19/// biased slightly more aggressive than [`DEFAULT_ACTIVE_RETRANSMIT_DELAY`]. A few extra
20/// packets at startup is fine.
21const DEFAULT_INITIAL_RETRANSMIT: Duration = Duration::from_millis(100);
22
23/// The retransmit interval used when a previous report exists but is missing latency.
24///
25/// When in an active steady-state, i.e. a previous report exists, we use the latency of the
26/// previous report to determine the retransmit interval.  However when this previous relay
27/// latency is missing this default is used.
28///
29/// This is a somewhat conservative guess because if we have no data, likely the relay node
30/// is very far away and we have no data because we timed out the last time we probed it.
31const DEFAULT_ACTIVE_RETRANSMIT_DELAY: Duration = Duration::from_millis(200);
32
33/// The extra time to add to retransmits if a previous report exists.
34///
35/// When in an active steady-state, i.e. a previous report exists, we add this delay
36/// multiplied with the attempt to probe retries to give later attempts increasingly more
37/// time.
38const ACTIVE_RETRANSMIT_EXTRA_DELAY: Duration = Duration::from_millis(50);
39
40/// The number of fastest relays to periodically re-query during incremental net_report
41/// reports. (During a full report, all relay servers are scanned.)
42const NUM_INCREMENTAL_RELAYS: usize = 3;
43
44/// The protocol used to time a node's latency.
45#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, derive_more::Display)]
46#[repr(u8)]
47pub enum Probe {
48    /// HTTPS
49    Https,
50    /// QUIC Address Discovery Ipv4
51    #[cfg(not(wasm_browser))]
52    QadIpv4,
53    /// QUIC Address Discovery Ipv6
54    #[cfg(not(wasm_browser))]
55    QadIpv6,
56}
57
58impl Probe {
59    #[cfg(not(wasm_browser))]
60    pub(super) fn is_udp(&self) -> bool {
61        matches!(self, Self::QadIpv4 | Self::QadIpv6)
62    }
63
64    #[cfg(wasm_browser)]
65    pub(super) fn is_udp(&self) -> bool {
66        false
67    }
68}

70/// A probe set is a sequence of similar [`Probe`]s with delays between them.
71///
72/// The probes are to the same Relayer and of the same [`Probe`] but will have different
73/// delays.  The delays are effectively retries, though they do not wait for the previous
74/// probe to be finished.  The first successful probe will cancel all other probes in the
75/// set.
76///
77/// This is a lot of type-safety by convention.  It would be so much nicer to have this
78/// compile-time checked but that introduces a giant mess of generics and traits and
79/// associated exploding types.
80///
81/// A [`ProbeSet`] implements [`IntoIterator`] similar to how [`Vec`] does.
82#[derive(Debug, PartialEq, Eq, PartialOrd, Ord)]
83pub(super) struct ProbeSet {
84    /// The [`Probe`] all the probes in this set have.
85    proto: Probe,
86    /// The data in the set.
87    probes: Vec<(Duration, Arc<RelayNode>)>,
88}
89
90#[derive(Debug, Snafu)]
91#[snafu(display("Mismatching probe"))]
92struct PushError;
93
94impl ProbeSet {
95    fn new(proto: Probe) -> Self {
96        Self {
97            probes: Vec::new(),
98            proto,
99        }
100    }
101
102    pub(super) fn proto(&self) -> Probe {
103        self.proto
104    }
105
106    fn push(&mut self, delay: Duration, node: Arc<RelayNode>) {
107        self.probes.push((delay, node));
108    }
109
110    fn is_empty(&self) -> bool {
111        self.probes.is_empty()
112    }
113
114    fn delays(&self) -> impl Iterator<Item = &Duration> {
115        self.probes.iter().map(|(d, _)| d)
116    }
117
118    pub(super) fn params(&self) -> impl Iterator<Item = &(Duration, Arc<RelayNode>)> {
119        self.probes.iter()
120    }
121}

123/// A probe plan.
124///
125/// A probe plan contains a number of [`ProbeSet`]s containing probes to be executed.
126/// Generally the first probe of of a set which completes aborts the remaining probes of a
127/// set.  Sometimes a failing probe can also abort the remaining probes of a set.
128///
129/// The [`reportgen`] actor will also abort all the remaining [`ProbeSet`]s once it has
130/// sufficient information for a report.
131///
132/// [`reportgen`]: crate::net_report::reportgen
133#[derive(Debug, Default, PartialEq, Eq)]
134pub(super) struct ProbePlan {
135    set: BTreeSet<ProbeSet>,
136}
137
138impl ProbePlan {
139    /// Creates an initial probe plan.
140    #[cfg(not(wasm_browser))]
141    pub(super) fn initial(
142        relay_map: &RelayMap,
143        protocols: &BTreeSet<Probe>,
144        if_state: &super::IfStateDetails,
145    ) -> Self {
146        let mut plan = Self {
147            set: Default::default(),
148        };
149
150        // The first time we need add probes after the STUN we record this delay, so that
151        // further relay server can reuse this delay.
152        let mut max_high_prio_delay: Option<Duration> = None;
153
154        for relay_node in relay_map.nodes() {
155            let mut quic_ipv4_probes = ProbeSet::new(Probe::QadIpv4);
156            let mut quic_ipv6_probes = ProbeSet::new(Probe::QadIpv6);
157
158            for attempt in 0..3 {
159                let delay = DEFAULT_INITIAL_RETRANSMIT * attempt as u32;
160
161                if if_state.have_v4 {
162                    quic_ipv4_probes.push(delay, relay_node.clone());
163                }
164                if if_state.have_v6 {
165                    quic_ipv6_probes.push(delay, relay_node.clone());
166                }
167            }
168            // plan.add_if_enabled(protocols, quic_ipv4_probes);
169            // plan.add_if_enabled(protocols, quic_ipv6_probes);
170
171            // The HTTP probes only start after the QAD probes have had a chance.
172            let mut https_probes = ProbeSet::new(Probe::Https);
173
174            for attempt in 0..3 {
175                let mut start = *max_high_prio_delay.get_or_insert_with(|| plan.max_delay());
176                // if there are high priority probes, ensure there is a buffer between
177                // the highest probe delay and the next probes we create
178                // if there are no high priority probes, we don't need a buffer
179                if plan.has_priority_probes() {
180                    start += DEFAULT_INITIAL_RETRANSMIT;
181                }
182                let delay = start + DEFAULT_INITIAL_RETRANSMIT * attempt as u32;
183                https_probes.push(delay, relay_node.clone());
184            }
185
186            plan.add_if_enabled(protocols, https_probes);
187        }
188        plan
189    }
190
191    /// Creates an initial probe plan for browsers.
192    ///
193    /// Here, we essentially only run HTTPS probes without any delays waiting for STUN.
194    #[cfg(wasm_browser)]
195    pub(super) fn initial(relay_map: &RelayMap, protocols: &BTreeSet<Probe>) -> Self {
196        let mut plan = Self {
197            set: Default::default(),
198        };
199
200        for relay_node in relay_map.nodes() {
201            let mut https_probes = ProbeSet::new(Probe::Https);
202
203            for attempt in 0u32..3 {
204                let delay = DEFAULT_INITIAL_RETRANSMIT * attempt;
205                https_probes.push(delay, relay_node.clone());
206            }
207
208            plan.add_if_enabled(protocols, https_probes);
209        }
210        plan
211    }
212
213    /// Creates a follow up probe plan using a previous net_report report in browsers.
214    ///
215    /// This will only schedule HTTPS probes.
216    pub(super) fn with_last_report(
217        relay_map: &RelayMap,
218        last_report: &Report,
219        protocols: &BTreeSet<Probe>,
220        if_state: &super::IfStateDetails,
221    ) -> Self {
222        if last_report.relay_latency.is_empty() {
223            return Self::initial(relay_map, protocols, if_state);
224        }
225        Self::default()
226    }
227
228    /// Returns an iterator over the [`ProbeSet`]s in this plan.
229    pub(super) fn iter(&self) -> impl Iterator<Item = &ProbeSet> {
230        self.set.iter()
231    }
232
233    /// Adds a [`ProbeSet`] if it contains probes and the protocol indicated in
234    /// the [`ProbeSet] matches a protocol in our set of [`Probe`]s.
235    fn add_if_enabled(&mut self, protocols: &BTreeSet<Probe>, set: ProbeSet) {
236        if !set.is_empty() && protocols.contains(&set.proto) {
237            self.set.insert(set);
238        }
239    }
240
241    /// Returns the delay of the last probe in the probe plan.
242    fn max_delay(&self) -> Duration {
243        self.set
244            .iter()
245            .flat_map(|probe_set| probe_set.delays())
246            .max()
247            .copied()
248            .unwrap_or_default()
249    }
250
251    /// Any UDP based probes are "priority" probes
252    fn has_priority_probes(&self) -> bool {
253        self.set.iter().any(|p| p.proto.is_udp())
254    }
255}

257impl fmt::Display for ProbePlan {
258    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
259        writeln!(f, "ProbePlan {{")?;
260        for probe_set in self.set.iter() {
261            writeln!(f, r#"    ProbeSet("{}") {{"#, probe_set.proto)?;
262            for (delay, node) in probe_set.probes.iter() {
263                writeln!(f, "        {delay:?} to {node},")?;
264            }
265            writeln!(f, "    }}")?;
266        }
267        writeln!(f, "}}")
268    }
269}

279/// Sorts the nodes in the [`RelayMap`] from fastest to slowest.
280///
281/// This uses the latencies from the last report to determine the order. Relay Nodes with no
282/// data are at the end.
283fn sort_relays<'a>(
284    relay_map: &'a RelayMap,
285    last_report: &Report,
286) -> Vec<(&'a RelayUrl, &'a Arc<RelayNode>)> {
287    let mut prev: Vec<_> = relay_map.nodes().collect();
288    prev.sort_by(|a, b| {
289        let latencies_a = last_report.relay_latency.get(&a.url);
290        let latencies_b = last_report.relay_latency.get(&b.url);
291        match (latencies_a, latencies_b) {
292            (Some(_), None) => {
293                // Non-zero sorts before zero.
294                std::cmp::Ordering::Less
295            }
296            (None, Some(_)) => {
297                // Zero can't sort before anything else.
298                std::cmp::Ordering::Greater
299            }
300            (None, None) => {
301                // For both empty latencies sort by relay_id.
302                a.url.cmp(&b.url)
303            }
304            (Some(_), Some(_)) => match latencies_a.cmp(&latencies_b) {
305                std::cmp::Ordering::Equal => a.url.cmp(&b.url),
306                x => x,
307            },
308        }
309    });
310
311    prev.into_iter().map(|n| (&n.url, n)).collect()
312}

322    /// Shorthand which declares a new ProbeSet.
323    ///
324    /// `$kind`: The `Probe`.
325    /// `$node`: Expression which will be an `Arc<RelayNode>`.
326    /// `$delays`: A `Vec` of the delays for this probe.
327    macro_rules! probeset {
328        (proto: Probe::$kind:ident, relay: $node:expr, delays: $delays:expr,) => {
329            ProbeSet {
330                proto: Probe::$kind,
331                probes: $delays.iter().map(|delay| (*delay, $node)).collect(),
332            }
333        };
334    }
335
336    fn default_protocols() -> BTreeSet<Probe> {
337        BTreeSet::from([Probe::QadIpv4, Probe::QadIpv6, Probe::Https])
338    }

539    fn create_last_report(
540        url_1: &RelayUrl,
541        latency_1: Option<Duration>,
542        url_2: &RelayUrl,
543        latency_2: Option<Duration>,
544    ) -> Report {
545        let mut latencies = RelayLatencies::default();
546        if let Some(latency_1) = latency_1 {
547            latencies.update_relay(url_1.clone(), latency_1, Probe::QadIpv4);
548        }
549        if let Some(latency_2) = latency_2 {
550            latencies.update_relay(url_2.clone(), latency_2, Probe::QadIpv4);
551        }
552        Report {
553            udp_v6: true,
554            udp_v4: true,
555            mapping_varies_by_dest_ipv4: Some(false),
556            mapping_varies_by_dest_ipv6: Some(false),
557            preferred_relay: Some(url_1.clone()),
558            relay_latency: latencies,
559            global_v4: None,
560            global_v6: None,
561            captive_portal: None,
562        }
563    }
