1//! Default values used in net_report.
2
3/// Contains all timeouts that we use in `iroh-net-report`.
4pub(crate) mod timeouts {
5    use n0_future::time::Duration;
6
7    // Timeouts for net_report
8
9    /// The maximum amount of time net_report will spend gathering a single report.
10    pub(crate) const OVERALL_REPORT_TIMEOUT: Duration = Duration::from_secs(5);
11
12    /// The total time we wait for all the probes.
13    ///
14    /// This includes the STUN, ICMP and HTTPS probes, which will all
15    /// start at different times based on the ProbePlan.
16    pub(crate) const PROBES_TIMEOUT: Duration = Duration::from_secs(3);
17
18    /// How long to await for a captive-portal result.
19    ///
20    /// This delay is chosen so it starts after good-working STUN probes
21    /// would have finished, but not too long so the delay is bearable if
22    /// STUN is blocked.
23    pub(crate) const CAPTIVE_PORTAL_DELAY: Duration = Duration::from_millis(200);
24
25    /// Timeout for captive portal checks
26    ///
27    /// Must be lower than [`OVERALL_REPORT_TIMEOUT`] minus
28    /// [`CAPTIVE_PORTAL_DELAY`].
29    pub(crate) const CAPTIVE_PORTAL_TIMEOUT: Duration = Duration::from_secs(2);
30
31    pub(crate) const DNS_TIMEOUT: Duration = Duration::from_secs(3);
32}
