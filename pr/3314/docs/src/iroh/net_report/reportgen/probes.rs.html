3//! All the probes try and establish the latency to the relay servers.  Preferably the STUN

16/// The retransmit interval used when net_report first runs.
17///
18/// We have no past context to work with, and we want answers relatively quickly, so it's
19/// biased slightly more aggressive than [`DEFAULT_ACTIVE_RETRANSMIT_DELAY`]. A few extra
20/// packets at startup is fine.
21const DEFAULT_INITIAL_RETRANSMIT: Duration = Duration::from_millis(100);
22
23/// The retransmit interval used when a previous report exists but is missing latency.
24///
25/// When in an active steady-state, i.e. a previous report exists, we use the latency of the
26/// previous report to determine the retransmit interval.  However when this previous relay
27/// latency is missing this default is used.
28///
29/// This is a somewhat conservative guess because if we have no data, likely the relay node
30/// is very far away and we have no data because we timed out the last time we probed it.
31const DEFAULT_ACTIVE_RETRANSMIT_DELAY: Duration = Duration::from_millis(200);
32
33/// The extra time to add to retransmits if a previous report exists.
34///
35/// When in an active steady-state, i.e. a previous report exists, we add this delay
36/// multiplied with the attempt to probe retries to give later attempts increasingly more
37/// time.
38const ACTIVE_RETRANSMIT_EXTRA_DELAY: Duration = Duration::from_millis(50);
39
40/// The number of fastest relays to periodically re-query during incremental net_report
41/// reports. (During a full report, all relay servers are scanned.)
42const NUM_INCREMENTAL_RELAYS: usize = 3;
43
44/// The protocol used to time a node's latency.
45#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, derive_more::Display)]
46#[repr(u8)]
47pub enum ProbeProto {
48    /// HTTPS
49    Https,
50    /// QUIC Address Discovery Ipv4
51    #[cfg(not(wasm_browser))]
52    QadIpv4,
53    /// QUIC Address Discovery Ipv6
54    #[cfg(not(wasm_browser))]
55    QadIpv6,
56}
57
58#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, derive_more::Display)]
59pub(crate) enum Probe {
60    #[display("HTTPS after {delay:?} to {node}")]
61    Https {
62        delay: Duration,
63        node: Arc<RelayNode>,
64    },
65    #[display("QAD Ipv4 after {delay:?} to {node}")]
66    #[cfg(not(wasm_browser))]
67    QadIpv4 {
68        delay: Duration,
69        node: Arc<RelayNode>,
70    },
71    #[display("QAD Ipv6 after {delay:?} to {node}")]
72    #[cfg(not(wasm_browser))]
73    QadIpv6 {
74        delay: Duration,
75        node: Arc<RelayNode>,
76    },
77}

79impl Probe {
80    pub(super) fn delay(&self) -> Duration {
81        match self {
82            #[cfg(not(wasm_browser))]
83            Probe::Https { delay, .. }
84            | Probe::QadIpv4 { delay, .. }
85            | Probe::QadIpv6 { delay, .. } => *delay,
86            #[cfg(wasm_browser)]
87            Probe::Https { delay, .. } => *delay,
88        }
89    }
90
91    pub(crate) fn proto(&self) -> ProbeProto {
92        match self {
93            Probe::Https { .. } => ProbeProto::Https,
94            #[cfg(not(wasm_browser))]
95            Probe::QadIpv4 { .. } => ProbeProto::QadIpv4,
96            #[cfg(not(wasm_browser))]
97            Probe::QadIpv6 { .. } => ProbeProto::QadIpv6,
98        }
99    }
100
101    pub(crate) fn node(&self) -> &Arc<RelayNode> {
102        match self {
103            #[cfg(not(wasm_browser))]
104            Probe::Https { node, .. }
105            | Probe::QadIpv4 { node, .. }
106            | Probe::QadIpv6 { node, .. } => node,
107            #[cfg(wasm_browser)]
108            Probe::Https { node, .. } => node,
109        }
110    }
111
112    #[cfg(not(wasm_browser))]
113    pub(super) fn is_udp(&self) -> bool {
114        matches!(self, Self::QadIpv4 { .. } | Self::QadIpv6 { .. })
115    }
116
117    #[cfg(wasm_browser)]
118    pub(super) fn is_udp(&self) -> bool {
119        false
120    }
121}

123/// A probe set is a sequence of similar [`Probe`]s with delays between them.
124///
125/// The probes are to the same Relayer and of the same [`ProbeProto`] but will have different
126/// delays.  The delays are effectively retries, though they do not wait for the previous
127/// probe to be finished.  The first successful probe will cancel all other probes in the
128/// set.
129///
130/// This is a lot of type-safety by convention.  It would be so much nicer to have this
131/// compile-time checked but that introduces a giant mess of generics and traits and
132/// associated exploding types.
133///
134/// A [`ProbeSet`] implements [`IntoIterator`] similar to how [`Vec`] does.
135#[derive(Debug, PartialEq, Eq, PartialOrd, Ord)]
136pub(super) struct ProbeSet {
137    /// The [`ProbeProto`] all the probes in this set have.
138    proto: ProbeProto,
139    /// The probes in the set.
140    probes: Vec<Probe>,
141}
142
143#[derive(Debug, Snafu)]
144#[snafu(display("Mismatching probe"))]
145struct PushError;
146
147impl ProbeSet {
148    fn new(proto: ProbeProto) -> Self {
149        Self {
150            probes: Vec::new(),
151            proto,
152        }
153    }
154
155    fn push(&mut self, probe: Probe) -> Result<(), PushError> {
156        if probe.proto() != self.proto {
157            return Err(PushError);
158        }
159        self.probes.push(probe);
160        Ok(())
161    }
162
163    fn is_empty(&self) -> bool {
164        self.probes.is_empty()
165    }
166}
167
168impl<'a> IntoIterator for &'a ProbeSet {
169    type Item = &'a Probe;
170
171    type IntoIter = std::slice::Iter<'a, Probe>;
172
173    fn into_iter(self) -> Self::IntoIter {
174        self.probes.iter()
175    }
176}

178impl fmt::Display for ProbeSet {
179    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
180        writeln!(f, r#"ProbeSet("{}") {{"#, self.proto)?;
181        for probe in self.probes.iter() {
182            writeln!(f, "        {probe},")?;
183        }
184        writeln!(f, "}}")
185    }
186}

188/// A probe plan.
189///
190/// A probe plan contains a number of [`ProbeSet`]s containing probes to be executed.
191/// Generally the first probe of of a set which completes aborts the remaining probes of a
192/// set.  Sometimes a failing probe can also abort the remaining probes of a set.
193///
194/// The [`reportgen`] actor will also abort all the remaining [`ProbeSet`]s once it has
195/// sufficient information for a report.
196///
197/// [`reportgen`]: crate::net_report::reportgen
198#[derive(Debug, PartialEq, Eq)]
199pub(super) struct ProbePlan {
200    set: BTreeSet<ProbeSet>,
201    protocols: BTreeSet<ProbeProto>,
202}
203
204impl ProbePlan {
205    /// Creates an initial probe plan.
206    #[cfg(not(wasm_browser))]
207    pub(super) fn initial(
208        relay_map: &RelayMap,
209        protocols: &BTreeSet<ProbeProto>,
210        if_state: &super::IfStateDetails,
211    ) -> Self {
212        let mut plan = Self {
213            set: BTreeSet::new(),
214            protocols: protocols.clone(),
215        };
216
217        // The first time we need add probes after the STUN we record this delay, so that
218        // further relay server can reuse this delay.
219        let mut max_high_prio_delay: Option<Duration> = None;
220
221        for relay_node in relay_map.nodes() {
222            let mut quic_ipv4_probes = ProbeSet::new(ProbeProto::QadIpv4);
223            let mut quic_ipv6_probes = ProbeSet::new(ProbeProto::QadIpv6);
224
225            for attempt in 0..3 {
226                let delay = DEFAULT_INITIAL_RETRANSMIT * attempt as u32;
227
228                if if_state.have_v4 {
229                    quic_ipv4_probes
230                        .push(Probe::QadIpv4 {
231                            delay,
232                            node: relay_node.clone(),
233                        })
234                        .expect("adding QuicIpv4 probe to a QuicIpv4 probe set");
235                }
236                if if_state.have_v6 {
237                    quic_ipv6_probes
238                        .push(Probe::QadIpv6 {
239                            delay,
240                            node: relay_node.clone(),
241                        })
242                        .expect("adding QuicIpv6 probe to a QuicAddrIpv6 probe set");
243                }
244            }
245            plan.add_if_enabled(quic_ipv4_probes);
246            plan.add_if_enabled(quic_ipv6_probes);
247
248            // The HTTP probes only start after the QUAD probes have had a chance.
249            let mut https_probes = ProbeSet::new(ProbeProto::Https);
250
251            for attempt in 0..3 {
252                let mut start = *max_high_prio_delay.get_or_insert_with(|| plan.max_delay());
253                // if there are high priority probes, ensure there is a buffer between
254                // the highest probe delay and the next probes we create
255                // if there are no high priority probes, we don't need a buffer
256                if plan.has_priority_probes() {
257                    start += DEFAULT_INITIAL_RETRANSMIT;
258                }
259                let delay = start + DEFAULT_INITIAL_RETRANSMIT * attempt as u32;
260                https_probes
261                    .push(Probe::Https {
262                        delay,
263                        node: relay_node.clone(),
264                    })
265                    .expect("adding Https probe to a Https probe set");
266            }
267
268            plan.add_if_enabled(https_probes);
269        }
270        plan
271    }

273    /// Creates an initial probe plan for browsers.
274    ///
275    /// Here, we essentially only run HTTPS probes without any delays waiting for STUN.
276    #[cfg(wasm_browser)]
277    pub(super) fn initial(relay_map: &RelayMap, protocols: &BTreeSet<ProbeProto>) -> Self {
278        let mut plan = Self {
279            set: BTreeSet::new(),
280            protocols: protocols.clone(),
281        };
282
283        for relay_node in relay_map.nodes() {
284            let mut https_probes = ProbeSet::new(ProbeProto::Https);
285
286            for attempt in 0u32..3 {
287                let delay = DEFAULT_INITIAL_RETRANSMIT * attempt;
288                https_probes
289                    .push(Probe::Https {
290                        delay,
291                        node: relay_node.clone(),
292                    })
293                    .expect("adding Https probe to a Https probe set");
294            }
295
296            plan.add_if_enabled(https_probes);
297        }
298        plan
299    }

301    /// Creates a follow up probe plan using a previous net_report report in browsers.
302    ///
303    /// This will only schedule HTTPS probes.
304    #[cfg(not(wasm_browser))]
305    pub(super) fn with_last_report(
306        relay_map: &RelayMap,
307        last_report: &Report,
308        protocols: &BTreeSet<ProbeProto>,
309        if_state: &super::IfStateDetails,
310    ) -> Self {
311        if last_report.relay_latency.is_empty() {
312            return Self::initial(relay_map, protocols, if_state);
313        }
314        let mut plan = Self {
315            set: Default::default(),
316            protocols: protocols.clone(),
317        };
318
319        // The first time we need to add probes after the STUN we record this delay, so that
320        // further relay servers can reuse this delay.
321        let mut max_stun_delay: Option<Duration> = None;
322
323        let had_stun_ipv4 = !last_report.relay_latency.ipv4().is_empty();
324        let had_stun_ipv6 = !last_report.relay_latency.ipv6().is_empty();
325        let had_both = if_state.have_v6 && had_stun_ipv4 && had_stun_ipv6;
326        let sorted_relays = sort_relays(relay_map, last_report);
327        for (ri, (url, relay_node)) in sorted_relays.into_iter().enumerate() {
328            if ri == NUM_INCREMENTAL_RELAYS {
329                break;
330            }
331            let mut do4 = if_state.have_v4;
332            let mut do6 = if_state.have_v6;
333
334            // By default, each node only gets one STUN packet sent,
335            // except the fastest two from the previous round.
336            let mut attempts = 1;
337            let is_fastest_two = ri < 2;
338
339            if is_fastest_two {
340                attempts = 2;
341            } else if had_both {
342                // For dual stack machines, make the 3rd & slower nodes alternate between
343                // IPv4 and IPv6 for STUN and ICMP probes.
344                if ri % 2 == 0 {
345                    (do4, do6) = (true, false);
346                } else {
347                    (do4, do6) = (false, true);
348                }
349            }
350            if !is_fastest_two && !had_stun_ipv6 {
351                do6 = false;
352            }
353            if Some(url) == last_report.preferred_relay.as_ref() {
354                // But if we already had a relay home, try extra hard to
355                // make sure it's there so we don't flip flop around.
356                attempts = 4;
357            }
358            let retransmit_delay = last_report
359                .relay_latency
360                .get(url)
361                .map(|l| l * 120 / 100) // increases latency by 20%, why?
362                .unwrap_or(DEFAULT_ACTIVE_RETRANSMIT_DELAY);
363
364            let mut quic_ipv4_probes = ProbeSet::new(ProbeProto::QadIpv4);
365            let mut quic_ipv6_probes = ProbeSet::new(ProbeProto::QadIpv6);
366
367            for attempt in 0..attempts {
368                let delay = (retransmit_delay * attempt as u32)
369                    + (ACTIVE_RETRANSMIT_EXTRA_DELAY * attempt as u32);
370                if do4 {
371                    quic_ipv4_probes
372                        .push(Probe::QadIpv4 {
373                            delay,
374                            node: relay_node.clone(),
375                        })
376                        .expect("adding QuicIpv4 probe to a QuicAddrIpv4 probe set");
377                }
378                if do6 {
379                    quic_ipv6_probes
380                        .push(Probe::QadIpv6 {
381                            delay,
382                            node: relay_node.clone(),
383                        })
384                        .expect("adding QuicIpv6 probe to a QuicAddrIpv6 probe set");
385                }
386            }
387            plan.add_if_enabled(quic_ipv4_probes);
388            plan.add_if_enabled(quic_ipv6_probes);
389
390            // The HTTP and ICMP probes only start after the STUN probes have had a chance.
391            let mut https_probes = ProbeSet::new(ProbeProto::Https);
392            let start = *max_stun_delay.get_or_insert_with(|| plan.max_delay());
393            for attempt in 0..attempts {
394                let delay = start
395                    + (retransmit_delay * attempt as u32)
396                    + (ACTIVE_RETRANSMIT_EXTRA_DELAY * (attempt as u32 + 1));
397                https_probes
398                    .push(Probe::Https {
399                        delay,
400                        node: relay_node.clone(),
401                    })
402                    .expect("Pushing Https Probe to an Https ProbeSet");
403            }
404
405            plan.add_if_enabled(https_probes);
406        }
407        plan
408    }

410    #[cfg(wasm_browser)]
411    pub(super) fn with_last_report(
412        relay_map: &RelayMap,
413        last_report: &Report,
414        protocols: &BTreeSet<ProbeProto>,
415    ) -> Self {
416        if last_report.relay_latency.is_empty() {
417            return Self::initial(relay_map, protocols);
418        }
419        let mut plan = Self {
420            set: Default::default(),
421            protocols: protocols.clone(),
422        };
423
424        let sorted_relays = sort_relays(relay_map, last_report);
425        for (ri, (url, relay_node)) in sorted_relays.into_iter().enumerate() {
426            if ri == NUM_INCREMENTAL_RELAYS {
427                break;
428            }
429
430            // By default, each node only gets one probe sent,
431            let mut attempts: u32 = 1;
432            // except the fastest two from the previous round.
433            if ri < 2 {
434                attempts = 2;
435            }
436            if Some(url) == last_report.preferred_relay.as_ref() {
437                // But if we already had a relay home, try extra hard to
438                // make sure it's there so we don't flip flop around.
439                attempts = 4;
440            }
441            let retransmit_delay = last_report
442                .relay_latency
443                .get(url)
444                .map(|l| l * 120 / 100) // increases latency by 20%, why?
445                .unwrap_or(DEFAULT_ACTIVE_RETRANSMIT_DELAY);
446
447            let mut https_probes = ProbeSet::new(ProbeProto::Https);
448            for attempt in 0..attempts {
449                let delay =
450                    (retransmit_delay * attempt) + (ACTIVE_RETRANSMIT_EXTRA_DELAY * (attempt + 1));
451                https_probes
452                    .push(Probe::Https {
453                        delay,
454                        node: relay_node.clone(),
455                    })
456                    .expect("Pushing Https Probe to an Https ProbeSet");
457            }
458
459            plan.add_if_enabled(https_probes);
460        }
461        plan
462    }

464    /// Returns an iterator over the [`ProbeSet`]s in this plan.
465    pub(super) fn iter(&self) -> impl Iterator<Item = &ProbeSet> {
466        self.set.iter()
467    }

469    /// Adds a [`ProbeSet`] if it contains probes and the protocol indicated in
470    /// the [`ProbeSet] matches a protocol in our set of [`ProbeProto`]s.
471    fn add_if_enabled(&mut self, set: ProbeSet) {
472        if !set.is_empty() && self.protocols.contains(&set.proto) {
473            self.set.insert(set);
474        }
475    }

477    /// Returns the delay of the last probe in the probe plan.
478    fn max_delay(&self) -> Duration {
479        self.set
480            .iter()
481            .flatten()
482            .map(|probe| probe.delay())
483            .max()
484            .unwrap_or_default()
485    }
486
487    /// Stun & Quic probes are "priority" probes
488    fn has_priority_probes(&self) -> bool {
489        #[cfg(not(wasm_browser))]
490        for probe in &self.set {
491            if matches!(probe.proto, ProbeProto::QadIpv4 | ProbeProto::QadIpv6) {
492                return true;
493            }
494        }
495        false
496    }

522/// Sorts the nodes in the [`RelayMap`] from fastest to slowest.
523///
524/// This uses the latencies from the last report to determine the order. Relay Nodes with no
525/// data are at the end.
526fn sort_relays<'a>(
527    relay_map: &'a RelayMap,
528    last_report: &Report,
529) -> Vec<(&'a RelayUrl, &'a Arc<RelayNode>)> {
530    let mut prev: Vec<_> = relay_map.nodes().collect();
531    prev.sort_by(|a, b| {
532        let latencies_a = last_report.relay_latency.get(&a.url);
533        let latencies_b = last_report.relay_latency.get(&b.url);
534        match (latencies_a, latencies_b) {
535            (Some(_), None) => {
536                // Non-zero sorts before zero.
537                std::cmp::Ordering::Less
538            }
539            (None, Some(_)) => {
540                // Zero can't sort before anything else.
541                std::cmp::Ordering::Greater
542            }
543            (None, None) => {
544                // For both empty latencies sort by relay_id.
545                a.url.cmp(&b.url)
546            }
547            (Some(_), Some(_)) => match latencies_a.cmp(&latencies_b) {
548                std::cmp::Ordering::Equal => a.url.cmp(&b.url),
549                x => x,
550            },
551        }
552    });
553
554    prev.into_iter().map(|n| (&n.url, n)).collect()
555}

796    fn create_last_report(
797        url_1: &RelayUrl,
798        latency_1: Option<Duration>,
799        url_2: &RelayUrl,
800        latency_2: Option<Duration>,
801    ) -> Report {
802        let mut latencies = RelayLatencies::default();
803        if let Some(latency_1) = latency_1 {
804            latencies.update_relay(url_1.clone(), latency_1, ProbeProto::QadIpv4);
805        }
806        if let Some(latency_2) = latency_2 {
807            latencies.update_relay(url_2.clone(), latency_2, ProbeProto::QadIpv4);
808        }
809        Report {
810            udp: true,
811            ipv6: true,
812            ipv4: true,
813            ipv6_can_send: true,
814            ipv4_can_send: true,
815            os_has_ipv6: true,
816            mapping_varies_by_dest_ip: Some(false),
817            mapping_varies_by_dest_ipv6: Some(false),
818            portmap_probe: None,
819            preferred_relay: Some(url_1.clone()),
820            relay_latency: latencies,
821            global_v4: None,
822            global_v6: None,
823            captive_portal: None,
824        }
825    }
