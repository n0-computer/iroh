23/// Production configuration.
24pub mod prod {
25    use iroh_relay::{RelayMap, RelayNode, RelayQuicConfig};
26
27    use super::*;
28
29    /// Hostname of the default NA relay.
30    pub const NA_RELAY_HOSTNAME: &str = "use1-1.relay.iroh.network.";
31    /// Hostname of the default EU relay.
32    pub const EU_RELAY_HOSTNAME: &str = "euw1-1.relay.iroh.network.";
33    /// Hostname of the default Asia-Pacific relay.
34    pub const AP_RELAY_HOSTNAME: &str = "aps1-1.relay.iroh.network.";
35
36    /// Get the default [`RelayMap`].
37    pub fn default_relay_map() -> RelayMap {
38        RelayMap::from_iter([
39            default_na_relay_node(),
40            default_eu_relay_node(),
41            default_ap_relay_node(),
42        ])
43    }
44
45    /// Get the default [`RelayNode`] for NA.
46    pub fn default_na_relay_node() -> RelayNode {
47        // The default NA relay server run by number0.
48        let url: Url = format!("https://{NA_RELAY_HOSTNAME}")
49            .parse()
50            .expect("default url");
51        RelayNode {
52            url: url.into(),
53            quic: Some(RelayQuicConfig::default()),
54        }
55    }
56
57    /// Get the default [`RelayNode`] for EU.
58    pub fn default_eu_relay_node() -> RelayNode {
59        // The default EU relay server run by number0.
60        let url: Url = format!("https://{EU_RELAY_HOSTNAME}")
61            .parse()
62            .expect("default_url");
63        RelayNode {
64            url: url.into(),
65            quic: Some(RelayQuicConfig::default()),
66        }
67    }
68
