1//! Implements a socket that can change its communication path while in use, actively searching for the best way to communicate.
2//!
3//! Based on tailscale/wgengine/magicsock
4//!
5//! ### `RelayOnly` path selection:
6//! When set this will force all packets to be sent over
7//! the relay connection, regardless of whether or
8//! not we have a direct UDP address for the given node.
9//!
10//! The intended use is for testing the relay protocol inside the MagicSock
11//! to ensure that we can rely on the relay to send packets when two nodes
12//! are unable to find direct UDP connections to each other.
13//!
14//! This also prevent this node from attempting to hole punch and prevents it
15//! from responding to any hole punching attempts. This node will still,
16//! however, read any packets that come off the UDP sockets.

388    /// Add addresses for a node to the magic socket's addresbook.
389    #[instrument(skip_all)]
390    pub fn add_node_addr(
391        &self,
392        mut addr: NodeAddr,
393        source: node_map::Source,
394    ) -> Result<(), AddNodeAddrError> {
395        let mut pruned: usize = 0;
396        for my_addr in self.direct_addrs.sockaddrs() {
397            if addr.direct_addresses.remove(&my_addr) {
398                warn!( node_id=addr.node_id.fmt_short(), %my_addr, %source, "not adding our addr for node");
399                pruned += 1;
400            }
401        }
402        if !addr.is_empty() {
403            self.node_map
404                .add_node_addr(addr, source, &self.metrics.magicsock);
405            Ok(())
406        } else if pruned != 0 {
407            Err(EmptyPrunedSnafu { pruned }.build())
408        } else {
409            Err(EmptySnafu.build())
410        }
411    }

468    #[cfg_attr(windows, allow(dead_code))]
469    fn normalized_local_addr(&self) -> io::Result<SocketAddr> {
470        let addrs = self.local_addrs_watch.get().expect("disconnected");
471
472        let mut ipv4_addr = None;
473        for addr in addrs {
474            let Some(addr) = addr.into_socket_addr() else {
475                continue;
476            };
477            if addr.is_ipv6() {
478                return Ok(addr);
479            }
480            if addr.is_ipv4() && ipv4_addr.is_none() {
481                ipv4_addr.replace(addr);
482            }
483        }
484        match ipv4_addr {
485            Some(addr) => Ok(addr),
486            None => Err(io::Error::other("no valid socket available")),
487        }
488    }

490    /// Searches the `node_map` to determine the current transports to be used.
491    #[instrument(skip_all)]
492    fn prepare_send(
493        &self,
494        transmit: &quinn_udp::Transmit,
495    ) -> io::Result<SmallVec<[transports::Addr; 3]>> {
496        self.metrics
497            .magicsock
498            .send_data
499            .inc_by(transmit.contents.len() as _);
500
501        if self.is_closed() {
502            self.metrics
503                .magicsock
504                .send_data_network_down
505                .inc_by(transmit.contents.len() as _);
506            return Err(io::Error::new(
507                io::ErrorKind::NotConnected,
508                "connection closed",
509            ));
510        }
511
512        let mut active_paths = SmallVec::<[_; 3]>::new();
513
514        match MappedAddr::from(transmit.destination) {
515            MappedAddr::None(dest) => {
516                error!(%dest, "Cannot convert to a mapped address.");
517            }
518            MappedAddr::NodeId(dest) => {
519                trace!(
520                    dst = %dest,
521                    src = ?transmit.src_ip,
522                    len = %transmit.contents.len(),
523                    "sending",
524                );
525
526                // Get the node's relay address and best direct address, as well
527                // as any pings that need to be sent for hole-punching purposes.
528                match self.node_map.get_send_addrs(
529                    dest,
530                    self.ipv6_reported.load(Ordering::Relaxed),
531                    &self.metrics.magicsock,
532                ) {
533                    Some((node_id, udp_addr, relay_url, ping_actions)) => {
534                        if !ping_actions.is_empty() {
535                            self.actor_sender
536                                .try_send(ActorMessage::PingActions(ping_actions))
537                                .ok();
538                        }
539                        if let Some(addr) = udp_addr {
540                            active_paths.push(transports::Addr::from(addr));
541                        }
542                        if let Some(url) = relay_url {
543                            active_paths.push(transports::Addr::Relay(url, node_id));
544                        }
545                    }
546                    None => {
547                        error!(%dest, "no NodeState for mapped address");
548                    }
549                }
550            }
551            #[cfg(not(wasm_browser))]
552            MappedAddr::Ip(dest) => {
553                trace!(
554                    dst = %dest,
555                    src = ?transmit.src_ip,
556                    len = %transmit.contents.len(),
557                    "sending",
558                );
559
560                // Check if this is a known IpMappedAddr, and if so, send over UDP
561                // Get the socket addr
562                match self.ip_mapped_addrs.get_ip_addr(&dest) {
563                    Some(addr) => {
564                        active_paths.push(transports::Addr::from(addr));
565                    }
566                    None => {
567                        error!(%dest, "unknown mapped address");
568                    }
569                }
570            }
571        }
572
573        Ok(active_paths)
574    }

576    /// Process datagrams received from UDP sockets.
577    ///
578    /// All the `bufs` and `metas` should have initialized packets in them.
579    ///
580    /// This fixes up the datagrams to use the correct [`NodeIdMappedAddr`] and extracts DISCO
581    /// packets, processing them inside the magic socket.
582    fn process_datagrams(
583        &self,
584        bufs: &mut [io::IoSliceMut<'_>],
585        metas: &mut [quinn_udp::RecvMeta],
586        source_addrs: &[transports::Addr],
587    ) {
588        debug_assert_eq!(bufs.len(), metas.len(), "non matching bufs & metas");
589        debug_assert_eq!(
590            bufs.len(),
591            source_addrs.len(),
592            "non matching bufs & source_addrs"
593        );
594
595        // Adding the IP address we received something on results in Quinn using this
596        // address on the send path to send from.  However we let Quinn use a
597        // NodeIdMappedAddress, not a real address.  So we used to substitute our bind address
598        // here so that Quinn would send on the right address.  But that would sometimes
599        // result in the wrong address family and Windows trips up on that.
600        //
601        // What should be done is that this dst_ip from the RecvMeta is stored in the
602        // NodeState/PathState.  Then on the send path it should be retrieved from the
603        // NodeState/PathSate together with the send address and substituted at send time.
604        // This is relevant for IPv6 link-local addresses where the OS otherwise does not
605        // know which interface to send from.
606        #[cfg(not(windows))]
607        let dst_ip = self.normalized_local_addr().ok().map(|addr| addr.ip());
608        // Reasoning for this here:
609        // https://github.com/n0-computer/iroh/pull/2595#issuecomment-2290947319
610        #[cfg(windows)]
611        let dst_ip = None;
612
613        let mut quic_packets_total = 0;
614
615        for ((quinn_meta, buf), source_addr) in metas
616            .iter_mut()
617            .zip(bufs.iter_mut())
618            .zip(source_addrs.iter())
619        {
620            let mut buf_contains_quic_datagrams = false;
621            let mut quic_datagram_count = 0;
622            if quinn_meta.len > quinn_meta.stride {
623                trace!(%quinn_meta.len, %quinn_meta.stride, "GRO datagram received");
624                self.metrics.magicsock.recv_gro_datagrams.inc();
625            }
626
627            // Chunk through the datagrams in this GRO payload to find disco
628            // packets and forward them to the actor
629            for datagram in buf[..quinn_meta.len].chunks_mut(quinn_meta.stride) {
630                if datagram.len() < quinn_meta.stride {
631                    trace!(
632                        len = %datagram.len(),
633                        %quinn_meta.stride,
634                        "Last GRO datagram smaller than stride",
635                    );
636                }
637
638                // Detect DISCO datagrams and process them.  Overwrite the first
639                // byte of those packets with zero to make Quinn ignore the packet.  This
640                // relies on quinn::EndpointConfig::grease_quic_bit being set to `false`,
641                // which we do in Endpoint::bind.
642                if let Some((sender, sealed_box)) = disco::source_and_box(datagram) {
643                    trace!(src = ?source_addr, len = %quinn_meta.stride, "UDP recv: disco packet");
644                    self.handle_disco_message(sender, sealed_box, source_addr);
645                    datagram[0] = 0u8;
646                } else {
647                    trace!(src = ?source_addr, len = %quinn_meta.stride, "UDP recv: quic packet");
648                    match source_addr {
649                        transports::Addr::Ip(SocketAddr::V4(..)) => {
650                            self.metrics
651                                .magicsock
652                                .recv_data_ipv4
653                                .inc_by(datagram.len() as _);
654                        }
655                        transports::Addr::Ip(SocketAddr::V6(..)) => {
656                            self.metrics
657                                .magicsock
658                                .recv_data_ipv6
659                                .inc_by(datagram.len() as _);
660                        }
661                        transports::Addr::Relay(..) => {
662                            self.metrics
663                                .magicsock
664                                .recv_data_relay
665                                .inc_by(datagram.len() as _);
666                        }
667                    }
668
669                    quic_datagram_count += 1;
670                    buf_contains_quic_datagrams = true;
671                }
672            }
673
674            if buf_contains_quic_datagrams {
675                match source_addr {
676                    #[cfg(wasm_browser)]
677                    transports::Addr::Ip(_addr) => {
678                        panic!("cannot use IP based addressing in the browser");
679                    }
680                    #[cfg(not(wasm_browser))]
681                    transports::Addr::Ip(addr) => {
682                        // UDP
683
684                        // Update the NodeMap and remap RecvMeta to the NodeIdMappedAddr.
685                        match self.node_map.receive_udp(*addr) {
686                            None => {
687                                // Check if this address is mapped to an IpMappedAddr
688                                if let Some(ip_mapped_addr) =
689                                    self.ip_mapped_addrs.get_mapped_addr(addr)
690                                {
691                                    trace!(
692                                        src = %addr,
693                                        count = %quic_datagram_count,
694                                        len = quinn_meta.len,
695                                        "UDP recv QUIC address discovery packets",
696                                    );
697                                    quic_packets_total += quic_datagram_count;
698                                    quinn_meta.addr = ip_mapped_addr.private_socket_addr();
699                                } else {
700                                    warn!(
701                                        src = %addr,
702                                        count = %quic_datagram_count,
703                                        len = quinn_meta.len,
704                                        "UDP recv quic packets: no node state found, skipping",
705                                    );
706                                    // If we have no node state for the from addr, set len to 0 to make
707                                    // quinn skip the buf completely.
708                                    quinn_meta.len = 0;
709                                }
710                            }
711                            Some((node_id, quic_mapped_addr)) => {
712                                trace!(
713                                    src = %addr,
714                                    node = %node_id.fmt_short(),
715                                    count = %quic_datagram_count,
716                                    len = quinn_meta.len,
717                                    "UDP recv quic packets",
718                                );
719                                quic_packets_total += quic_datagram_count;
720                                quinn_meta.addr = quic_mapped_addr.private_socket_addr();
721                            }
722                        }
723                    }
724                    transports::Addr::Relay(src_url, src_node) => {
725                        // Relay
726                        let quic_mapped_addr = self.node_map.receive_relay(src_url, *src_node);
727                        quinn_meta.addr = quic_mapped_addr.private_socket_addr();
728                    }
729                }
730            } else {
731                // If all datagrams in this buf are DISCO, set len to zero to make
732                // Quinn skip the buf completely.
733                quinn_meta.len = 0;
734            }
735            // Normalize local_ip
736            quinn_meta.dst_ip = dst_ip;
737        }
738
739        if quic_packets_total > 0 {
740            self.metrics
741                .magicsock
742                .recv_datagrams
743                .inc_by(quic_packets_total as _);
744            trace!("UDP recv: {} packets", quic_packets_total);
745        }
746    }

748    /// Handles a discovery message.
749    #[instrument("disco_in", skip_all, fields(node = %sender.fmt_short(), ?src))]
750    fn handle_disco_message(&self, sender: PublicKey, sealed_box: &[u8], src: &transports::Addr) {
751        trace!("handle_disco_message start");
752        if self.is_closed() {
753            return;
754        }
755
756        if let transports::Addr::Relay(_, node_id) = src {
757            if node_id != &sender {
758                // TODO: return here?
759                warn!(
760                    "Received relay disco message from connection for {}, but with message from {}",
761                    node_id.fmt_short(),
762                    sender.fmt_short()
763                );
764            }
765        }
766
767        // We're now reasonably sure we're expecting communication from
768        // this node, do the heavy crypto lifting to see what they want.
769        let dm = match self.disco.unseal_and_decode(sender, sealed_box) {
770            Ok(dm) => dm,
771            Err(DiscoBoxError::Open { source, .. }) => {
772                warn!(?source, "failed to open disco box");
773                self.metrics.magicsock.recv_disco_bad_key.inc();
774                return;
775            }
776            Err(DiscoBoxError::Parse { source, .. }) => {
777                // Couldn't parse it, but it was inside a correctly
778                // signed box, so just ignore it, assuming it's from a
779                // newer version of Tailscale that we don't
780                // understand. Not even worth logging about, lest it
781                // be too spammy for old clients.
782
783                self.metrics.magicsock.recv_disco_bad_parse.inc();
784                debug!(?source, "failed to parse disco message");
785                return;
786            }
787        };
788
789        if src.is_relay() {
790            self.metrics.magicsock.recv_disco_relay.inc();
791        } else {
792            self.metrics.magicsock.recv_disco_udp.inc();
793        }
794
795        let span = trace_span!("handle_disco", ?dm);
796        let _guard = span.enter();
797        trace!("receive disco message");
798        match dm {
799            disco::Message::Ping(ping) => {
800                self.metrics.magicsock.recv_disco_ping.inc();
801                self.handle_ping(ping, sender, src);
802            }
803            disco::Message::Pong(pong) => {
804                self.metrics.magicsock.recv_disco_pong.inc();
805                self.node_map.handle_pong(sender, src, pong);
806            }
807            disco::Message::CallMeMaybe(cm) => {
808                self.metrics.magicsock.recv_disco_call_me_maybe.inc();
809                match src {
810                    transports::Addr::Relay(url, _) => {
811                        event!(
812                            target: "iroh::_events::call-me-maybe::recv",
813                            Level::DEBUG,
814                            remote_node = sender.fmt_short(),
815                            via = ?url,
816                            their_addrs = ?cm.my_numbers,
817                        );
818                    }
819                    _ => {
820                        warn!("call-me-maybe packets should only come via relay");
821                        return;
822                    }
823                }
824                let ping_actions =
825                    self.node_map
826                        .handle_call_me_maybe(sender, cm, &self.metrics.magicsock);
827                for action in ping_actions {
828                    match action {
829                        PingAction::SendCallMeMaybe { .. } => {
830                            warn!("Unexpected CallMeMaybe as response of handling a CallMeMaybe");
831                        }
832                        PingAction::SendPing(ping) => {
833                            self.send_ping_queued(ping);
834                        }
835                    }
836                }
837            }
838        }
839        trace!("disco message handled");
840    }

842    /// Handle a ping message.
843    fn handle_ping(&self, dm: disco::Ping, sender: NodeId, src: &transports::Addr) {
844        // Insert the ping into the node map, and return whether a ping with this tx_id was already
845        // received.
846        let addr: SendAddr = src.clone().into();
847        let handled = self.node_map.handle_ping(sender, addr.clone(), dm.tx_id);
848        match handled.role {
849            PingRole::Duplicate => {
850                debug!(?src, tx = %HEXLOWER.encode(&dm.tx_id), "received ping: path already confirmed, skip");
851                return;
852            }
853            PingRole::LikelyHeartbeat => {}
854            PingRole::NewPath => {
855                debug!(?src, tx = %HEXLOWER.encode(&dm.tx_id), "received ping: new path");
856            }
857            PingRole::Activate => {
858                debug!(?src, tx = %HEXLOWER.encode(&dm.tx_id), "received ping: path active");
859            }
860        }
861
862        // Send a pong.
863        debug!(tx = %HEXLOWER.encode(&dm.tx_id), %addr, dstkey = %sender.fmt_short(),
864               "sending pong");
865        let pong = disco::Message::Pong(disco::Pong {
866            tx_id: dm.tx_id,
867            ping_observed_addr: addr.clone(),
868        });
869        event!(
870            target: "iroh::_events::pong::sent",
871            Level::DEBUG,
872            remote_node = %sender.fmt_short(),
873            dst = ?addr,
874            txn = ?dm.tx_id,
875        );
876
877        if !self.disco.try_send(addr.clone(), sender, pong) {
878            warn!(%addr, "failed to queue pong");
879        }
880
881        if let Some(ping) = handled.needs_ping_back {
882            debug!(
883                %addr,
884                dstkey = %sender.fmt_short(),
885                "sending direct ping back",
886            );
887            self.send_ping_queued(ping);
888        }
889    }

891    fn send_ping_queued(&self, ping: SendPing) {
892        let SendPing {
893            id,
894            dst,
895            dst_node,
896            tx_id,
897            purpose,
898        } = ping;
899        let msg = disco::Message::Ping(disco::Ping {
900            tx_id,
901            node_key: self.public_key,
902        });
903        let sent = self.disco.try_send(dst.clone(), dst_node, msg);
904        if sent {
905            let msg_sender = self.actor_sender.clone();
906            trace!(%dst, tx = %HEXLOWER.encode(&tx_id), ?purpose, "ping sent (queued)");
907            self.node_map
908                .notify_ping_sent(id, dst, tx_id, purpose, msg_sender);
909        } else {
910            warn!(dst = ?dst, tx = %HEXLOWER.encode(&tx_id), ?purpose, "failed to send ping: queues full");
911        }
912    }

914    /// Send the given ping actions out.
915    async fn send_ping_actions(&self, sender: &UdpSender, msgs: Vec<PingAction>) -> io::Result<()> {
916        for msg in msgs {
917            // Abort sending as soon as we know we are shutting down.
918            if self.is_closing() || self.is_closed() {
919                return Ok(());
920            }
921            match msg {
922                PingAction::SendCallMeMaybe {
923                    relay_url,
924                    dst_node,
925                } => {
926                    // Sends the call-me-maybe DISCO message, queuing if addresses are too stale.
927                    //
928                    // To send the call-me-maybe message, we need to know our current direct addresses.  If
929                    // this information is too stale, the call-me-maybe is queued while a net_report run is
930                    // scheduled.  Once this run finishes, the call-me-maybe will be sent.
931                    match self.direct_addrs.fresh_enough() {
932                        Ok(()) => {
933                            let msg = disco::Message::CallMeMaybe(
934                                self.direct_addrs.to_call_me_maybe_message(),
935                            );
936                            if !self.disco.try_send(
937                                SendAddr::Relay(relay_url.clone()),
938                                dst_node,
939                                msg.clone(),
940                            ) {
941                                warn!(dstkey = %dst_node.fmt_short(), %relay_url, "relay channel full, dropping call-me-maybe");
942                            } else {
943                                debug!(dstkey = %dst_node.fmt_short(), %relay_url, "call-me-maybe sent");
944                            }
945                        }
946                        Err(last_refresh_ago) => {
947                            debug!(
948                                ?last_refresh_ago,
949                                "want call-me-maybe but direct addrs stale; queuing after restun",
950                            );
951                            self.actor_sender
952                                .try_send(ActorMessage::ScheduleDirectAddrUpdate(
953                                    UpdateReason::RefreshForPeering,
954                                    Some((dst_node, relay_url)),
955                                ))
956                                .ok();
957                        }
958                    }
959                }
960                PingAction::SendPing(SendPing {
961                    id,
962                    dst,
963                    dst_node,
964                    tx_id,
965                    purpose,
966                }) => {
967                    let msg = disco::Message::Ping(disco::Ping {
968                        tx_id,
969                        node_key: self.public_key,
970                    });
971
972                    self.send_disco_message(sender, dst.clone(), dst_node, msg)
973                        .await?;
974                    debug!(%dst, tx = %HEXLOWER.encode(&tx_id), ?purpose, "ping sent");
975                    let msg_sender = self.actor_sender.clone();
976                    self.node_map
977                        .notify_ping_sent(id, dst, tx_id, purpose, msg_sender);
978                }
979            }
980        }
981        Ok(())
982    }

984    /// Sends out a disco message.
985    async fn send_disco_message(
986        &self,
987        sender: &UdpSender,
988        dst: SendAddr,
989        dst_key: PublicKey,
990        msg: disco::Message,
991    ) -> io::Result<()> {
992        let dst = match dst {
993            SendAddr::Udp(addr) => transports::Addr::Ip(addr),
994            SendAddr::Relay(url) => transports::Addr::Relay(url, dst_key),
995        };
996
997        trace!(?dst, %msg, "send disco message (UDP)");
998        if self.is_closed() {
999            return Err(io::Error::new(
1000                io::ErrorKind::NotConnected,
1001                "connection closed",
1002            ));
1003        }
1004
1005        let pkt = self.disco.encode_and_seal(self.public_key, dst_key, &msg);
1006
1007        let transmit = transports::Transmit {
1008            contents: &pkt,
1009            ecn: None,
1010            segment_size: None,
1011        };
1012
1013        let dst2 = dst.clone();
1014        match sender.send(&dst2, None, &transmit).await {
1015            Ok(()) => {
1016                trace!(?dst, %msg, "sent disco message");
1017                self.metrics.magicsock.sent_disco_udp.inc();
1018                disco_message_sent(&msg, &self.metrics.magicsock);
1019                Ok(())
1020            }
1021            Err(err) => {
1022                warn!(?dst, ?msg, ?err, "failed to send disco message");
1023                Err(err)
1024            }
1025        }
1026    }

1078/// Manages currently running direct addr discovery, aka net_report runs.
1079///
1080/// Invariants:
1081/// - only one direct addr update must be running at a time
1082/// - if an update is scheduled while another one is running, remember that
1083///   and start a new one when the current one has finished
1084#[derive(Debug)]
1085struct DirectAddrUpdateState {
1086    /// If set, start a new update as soon as the current one is finished.
1087    want_update: Option<UpdateReason>,
1088    msock: Arc<MagicSock>,
1089    #[cfg(not(wasm_browser))]
1090    port_mapper: portmapper::Client,
1091    /// The prober that discovers local network conditions, including the closest relay relay and NAT mappings.
1092    net_reporter: Arc<Mutex<net_report::Client>>,
1093    relay_map: RelayMap,
1094    run_done: mpsc::Sender<()>,
1095}
1096
1097#[derive(Default, Debug, PartialEq, Eq, Clone, Copy)]
1098enum UpdateReason {
1099    /// Initial state
1100    #[default]
1101    None,
1102    RefreshForPeering,
1103    Periodic,
1104    PortmapUpdated,
1105    LinkChangeMajor,
1106    LinkChangeMinor,
1107}
1108
1109impl UpdateReason {
1110    fn is_major(self) -> bool {
1111        matches!(self, Self::LinkChangeMajor)
1112    }
1113}
1114
1115impl DirectAddrUpdateState {
1116    fn new(
1117        msock: Arc<MagicSock>,
1118        #[cfg(not(wasm_browser))] port_mapper: portmapper::Client,
1119        net_reporter: Arc<Mutex<net_report::Client>>,
1120        relay_map: RelayMap,
1121        run_done: mpsc::Sender<()>,
1122    ) -> Self {
1123        DirectAddrUpdateState {
1124            want_update: Default::default(),
1125            #[cfg(not(wasm_browser))]
1126            port_mapper,
1127            net_reporter,
1128            msock,
1129            relay_map,
1130            run_done,
1131        }
1132    }
1133
1134    /// Schedules a new run, either starting it immediately if none is running or
1135    /// scheduling it for later.
1136    fn schedule_run(&mut self, why: UpdateReason, if_state: IfStateDetails) {
1137        match self.net_reporter.clone().try_lock_owned() {
1138            Ok(net_reporter) => {
1139                self.run(why, if_state, net_reporter);
1140            }
1141            Err(_) => {
1142                let _ = self.want_update.insert(why);
1143            }
1144        }
1145    }
1146
1147    /// If another run is needed, triggers this run, otherwise does nothing.
1148    fn try_run(&mut self, if_state: IfStateDetails) {
1149        match self.net_reporter.clone().try_lock_owned() {
1150            Ok(net_reporter) => {
1151                if let Some(why) = self.want_update.take() {
1152                    self.run(why, if_state, net_reporter);
1153                }
1154            }
1155            Err(_) => {
1156                // do nothing
1157            }
1158        }
1159    }
1160
1161    /// Trigger a new run.
1162    fn run(
1163        &mut self,
1164        why: UpdateReason,
1165        if_state: IfStateDetails,
1166        mut net_reporter: tokio::sync::OwnedMutexGuard<net_report::Client>,
1167    ) {
1168        debug!("starting direct addr update ({:?})", why);
1169        #[cfg(not(wasm_browser))]
1170        self.port_mapper.procure_mapping();
1171        // Don't start a net report probe if we know
1172        // we are shutting down
1173        if self.msock.is_closing() || self.msock.is_closed() {
1174            debug!("skipping net_report, socket is shutting down");
1175            return;
1176        }
1177        if self.relay_map.is_empty() {
1178            debug!("skipping net_report, empty RelayMap");
1179            self.msock.net_report.set((None, why)).ok();
1180            return;
1181        }
1182
1183        debug!("requesting net_report report");
1184        let msock = self.msock.clone();
1185
1186        let run_done = self.run_done.clone();
1187        task::spawn(async move {
1188            let fut = time::timeout(
1189                NET_REPORT_TIMEOUT,
1190                net_reporter.get_report(if_state, why.is_major()),
1191            );
1192            match fut.await {
1193                Ok(report) => {
1194                    msock.net_report.set((Some(report), why)).ok();
1195                }
1196                Err(time::Elapsed { .. }) => {
1197                    warn!("net_report report timed out");
1198                }
1199            }
1200
1201            // mark run as finished
1202            debug!("direct addr update done ({:?})", why);
1203            run_done.send(()).await.ok();
1204        });
1205    }
1206}

1425    /// The underlying [`quinn::Endpoint`]
1426    pub fn endpoint(&self) -> &quinn::Endpoint {

1428    }

1430    /// Closes the connection.
1431    ///
1432    /// Only the first close does anything. Any later closes return nil.
1433    /// Polling the socket ([`AsyncUdpSocket::poll_recv`]) will return [`Poll::Pending`]
1434    /// indefinitely after this call.
1435    #[instrument(skip_all)]
1436    pub(crate) async fn close(&self) {
1437        trace!(me = ?self.public_key, "magicsock closing...");
1438        // Initiate closing all connections, and refuse future connections.
1439        self.endpoint.close(0u16.into(), b"");
1440
1441        // In the history of this code, this call had been
1442        // - removed: https://github.com/n0-computer/iroh/pull/1753
1443        // - then added back in: https://github.com/n0-computer/iroh/pull/2227/files#diff-ba27e40e2986a3919b20f6b412ad4fe63154af648610ea5d9ed0b5d5b0e2d780R573
1444        // - then removed again: https://github.com/n0-computer/iroh/pull/3165
1445        // and finally added back in together with this comment.
1446        // So before removing this call, please consider carefully.
1447        // Among other things, this call tries its best to make sure that any queued close frames
1448        // (e.g. via the call to `endpoint.close(...)` above), are flushed out to the sockets
1449        // *and acknowledged* (or time out with the "probe timeout" of usually 3 seconds).
1450        // This allows the other endpoints for these connections to be notified to release
1451        // their resources, or - depending on the protocol - that all data was received.
1452        // With the current quinn API, this is the only way to ensure protocol code can use
1453        // connection close codes, and close the endpoint properly.
1454        // If this call is skipped, then connections that protocols close just shortly before the
1455        // call to `Endpoint::close` will in most cases cause connection time-outs on remote ends.
1456        self.endpoint.wait_idle().await;
1457
1458        if self.msock.is_closed() {
1459            return;
1460        }
1461        self.msock.closing.store(true, Ordering::Relaxed);
1462        self.actor_token.cancel();
1463
1464        if let Some(task) = self.actor_task.lock().await.take() {
1465            // give the tasks a moment to shutdown cleanly
1466            let shutdown_done = time::timeout(Duration::from_millis(100), async move {
1467                if let Err(err) = task.await {
1468                    warn!("unexpected error in task shutdown: {:?}", err);
1469                }
1470            })
1471            .await;
1472            match shutdown_done {
1473                Ok(_) => trace!("tasks finished in time, shutdown complete"),
1474                Err(time::Elapsed { .. }) => {
1475                    // Dropping the task will abort itt
1476                    warn!("tasks didn't finish in time, aborting");
1477                }
1478            }
1479        }
1480
1481        self.msock.closed.store(true, Ordering::SeqCst);
1482
1483        trace!("magicsock closed");
1484    }

1510impl DiscoState {
1511    fn new(
1512        secret_encryption_key: crypto_box::SecretKey,
1513    ) -> (Self, mpsc::Receiver<(SendAddr, PublicKey, disco::Message)>) {
1514        let (disco_sender, disco_receiver) = mpsc::channel(256);
1515
1516        (
1517            Self {
1518                secret_encryption_key,
1519                secrets: Default::default(),
1520                sender: disco_sender,
1521            },
1522            disco_receiver,

1526    fn try_send(&self, dst: SendAddr, node_id: PublicKey, msg: disco::Message) -> bool {
1527        self.sender.try_send((dst, node_id, msg)).is_ok()
1528    }
1529
1530    fn encode_and_seal(
1531        &self,
1532        this_node_id: NodeId,
1533        other_node_id: NodeId,
1534        msg: &disco::Message,
1535    ) -> Bytes {
1536        let mut seal = msg.as_bytes();
1537        self.get_secret(other_node_id, |secret| secret.seal(&mut seal));
1538        disco::encode_message(&this_node_id, seal).into()
1539    }
1540
1541    fn unseal_and_decode(
1542        &self,
1543        node_id: PublicKey,
1544        sealed_box: &[u8],
1545    ) -> Result<disco::Message, DiscoBoxError> {
1546        let mut sealed_box = sealed_box.to_vec();
1547        self.get_secret(node_id, |secret| secret.open(&mut sealed_box))
1548            .context(OpenSnafu)?;
1549        disco::Message::from_bytes(&sealed_box).context(ParseSnafu)
1550    }
1551
1552    fn get_secret<F, T>(&self, node_id: PublicKey, cb: F) -> T
1553    where
1554        F: FnOnce(&mut SharedSecret) -> T,
1555    {
1556        let mut inner = self.secrets.lock().expect("poisoned");
1557        let x = inner.entry(node_id).or_insert_with(|| {
1558            let public_key = public_ed_box(&node_id.public());
1559            SharedSecret::new(&self.secret_encryption_key, &public_key)
1560        });
1561        cb(x)
1562    }
1563}

1675#[cfg(not(wasm_browser))]
1676fn bind_ip(
1677    addr_v4: SocketAddrV4,
1678    addr_v6: Option<SocketAddrV6>,
1679    metrics: &EndpointMetrics,
1680) -> io::Result<(Vec<IpTransport>, portmapper::Client)> {
1681    let port_mapper =
1682        portmapper::Client::with_metrics(Default::default(), metrics.portmapper.clone());
1683
1684    let v4 = Arc::new(bind_with_fallback(SocketAddr::V4(addr_v4))?);
1685    let ip4_port = v4.local_addr()?.port();
1686    let ip6_port = ip4_port.checked_add(1).unwrap_or(ip4_port - 1);
1687
1688    let addr_v6 =
1689        addr_v6.unwrap_or_else(|| SocketAddrV6::new(Ipv6Addr::UNSPECIFIED, ip6_port, 0, 0));
1690
1691    let v6 = match bind_with_fallback(SocketAddr::V6(addr_v6)) {
1692        Ok(sock) => Some(Arc::new(sock)),
1693        Err(err) => {
1694            info!("bind ignoring IPv6 bind failure: {:?}", err);
1695            None
1696        }
1697    };
1698
1699    let port = v4.local_addr().map_or(0, |p| p.port());
1700
1701    let mut ip = vec![IpTransport::new(
1702        addr_v4.into(),
1703        v4,
1704        metrics.magicsock.clone(),
1705    )];
1706    if let Some(v6) = v6 {
1707        ip.push(IpTransport::new(
1708            addr_v6.into(),
1709            v6,
1710            metrics.magicsock.clone(),
1711        ))
1712    }
1713
1714    // NOTE: we can end up with a zero port if `netwatch::UdpSocket::socket_addr` fails
1715    match port.try_into() {
1716        Ok(non_zero_port) => {
1717            port_mapper.update_local_port(non_zero_port);
1718        }
1719        Err(_zero_port) => debug!("Skipping port mapping with zero local port"),
1720    }
1721
1722    Ok((ip, port_mapper))
1723}

1725impl Actor {
1726    async fn run(
1727        mut self,
1728        shutdown_token: CancellationToken,
1729        mut watcher: impl Watcher<Value = Vec<transports::Addr>> + Send + Sync,
1730        sender: UdpSender,
1731    ) {
1732        // Initialize addresses
1733        self.update_direct_addresses(None);
1734
1735        // Setup network monitoring
1736        let mut current_netmon_state = self.netmon_watcher.get().expect("missing network state");
1737
1738        #[cfg(not(wasm_browser))]
1739        let mut direct_addr_heartbeat_timer = time::interval(HEARTBEAT_INTERVAL);
1740
1741        #[cfg(not(wasm_browser))]
1742        let mut portmap_watcher = self
1743            .direct_addr_update_state
1744            .port_mapper
1745            .watch_external_address();
1746
1747        let mut discovery_events: BoxStream<DiscoveryItem> = Box::pin(n0_future::stream::empty());
1748        if let Some(d) = self.msock.discovery() {
1749            if let Some(events) = d.subscribe() {
1750                discovery_events = events;
1751            }
1752        }
1753
1754        let mut receiver_closed = false;
1755        #[cfg_attr(wasm_browser, allow(unused_mut))]
1756        let mut portmap_watcher_closed = false;
1757
1758        let mut net_report_watcher = self.msock.net_report.watch();
1759
1760        loop {
1761            self.msock.metrics.magicsock.actor_tick_main.inc();
1762            #[cfg(not(wasm_browser))]
1763            let portmap_watcher_changed = portmap_watcher.changed();
1764            #[cfg(wasm_browser)]
1765            let portmap_watcher_changed = n0_future::future::pending();
1766
1767            #[cfg(not(wasm_browser))]
1768            let direct_addr_heartbeat_timer_tick = direct_addr_heartbeat_timer.tick();
1769            #[cfg(wasm_browser)]
1770            let direct_addr_heartbeat_timer_tick = n0_future::future::pending();
1771
1772            tokio::select! {
1773                _ = shutdown_token.cancelled() => {
1774                    debug!("shutting down");
1775                    return;
1776                }
1777                msg = self.msg_receiver.recv(), if !receiver_closed => {
1778                    let Some(msg) = msg else {
1779                        trace!("tick: magicsock receiver closed");
1780                        self.msock.metrics.magicsock.actor_tick_other.inc();
1781
1782                        receiver_closed = true;
1783                        continue;
1784                    };
1785
1786                    trace!(?msg, "tick: msg");
1787                    self.msock.metrics.magicsock.actor_tick_msg.inc();
1788                    self.handle_actor_message(msg, &sender).await;
1789                }
1790                tick = self.periodic_re_stun_timer.tick() => {
1791                    trace!("tick: re_stun {:?}", tick);
1792                    self.msock.metrics.magicsock.actor_tick_re_stun.inc();
1793                    self.re_stun(UpdateReason::Periodic);
1794                }
1795                new_addr = watcher.updated() => {
1796                    match new_addr {
1797                        Ok(addrs) => {
1798                            if !addrs.is_empty() {
1799                                trace!(?addrs, "local addrs");
1800                                self.msock.publish_my_addr();
1801                            }
1802                        }
1803                        Err(_) => {
1804                            warn!("local addr watcher stopped");
1805                        }
1806                    }
1807                }
1808                report = net_report_watcher.updated() => {
1809                    match report {
1810                        Ok((report, _)) => {
1811                            self.handle_net_report_report(report);
1812                            #[cfg(not(wasm_browser))]
1813                            {
1814                                self.periodic_re_stun_timer = new_re_stun_timer(true);
1815                            }
1816                        }
1817                        Err(_) => {
1818                            warn!("net report watcher stopped");
1819                        }
1820                    }
1821                }
1822                reason = self.direct_addr_done_rx.recv() => {
1823                    match reason {
1824                        Some(()) => {
1825                            // check if a new run needs to be scheduled
1826                            let state = self.netmon_watcher.get().expect("disconnected");
1827                            self.direct_addr_update_state.try_run(state.into());
1828                        }
1829                        None => {
1830                            warn!("direct addr watcher died");
1831                        }
1832                    }
1833                }
1834                change = portmap_watcher_changed, if !portmap_watcher_closed => {
1835                    #[cfg(not(wasm_browser))]
1836                    {
1837                        if change.is_err() {
1838                            trace!("tick: portmap watcher closed");
1839                            self.msock.metrics.magicsock.actor_tick_other.inc();
1840
1841                            portmap_watcher_closed = true;
1842                            continue;
1843                        }
1844
1845                        trace!("tick: portmap changed");
1846                        self.msock.metrics.magicsock.actor_tick_portmap_changed.inc();
1847                        let new_external_address = *portmap_watcher.borrow();
1848                        debug!("external address updated: {new_external_address:?}");
1849                        self.re_stun(UpdateReason::PortmapUpdated);
1850                    }
1851                    #[cfg(wasm_browser)]
1852                    let _unused_in_browsers = change;
1853                },
1854                _ = direct_addr_heartbeat_timer_tick => {
1855                    #[cfg(not(wasm_browser))]
1856                    {
1857                        trace!(

1898                }

1901                        warn!(%dst, node = %dst_key.fmt_short(), ?err, "failed to send disco message (UDP)");

1903                }
1904            }
1905        }

1967    /// Updates the direct addresses of this magic socket.
1968    ///
1969    /// Updates the [`DiscoveredDirectAddrs`] of this [`MagicSock`] with the current set of
1970    /// direct addresses from:
1971    ///
1972    /// - The portmapper.
1973    /// - A net_report report.
1974    /// - The local interfaces IP addresses.
1975    #[cfg(not(wasm_browser))]
1976    fn update_direct_addresses(&mut self, net_report_report: Option<&net_report::Report>) {
1977        let portmap_watcher = self
1978            .direct_addr_update_state
1979            .port_mapper
1980            .watch_external_address();
1981
1982        // We only want to have one DirectAddr for each SocketAddr we have.  So we store
1983        // this as a map of SocketAddr -> DirectAddrType.  At the end we will construct a
1984        // DirectAddr from each entry.
1985        let mut addrs: BTreeMap<SocketAddr, DirectAddrType> = BTreeMap::new();
1986
1987        // First add PortMapper provided addresses.
1988        let maybe_port_mapped = *portmap_watcher.borrow();
1989        if let Some(portmap_ext) = maybe_port_mapped.map(SocketAddr::V4) {
1990            addrs
1991                .entry(portmap_ext)
1992                .or_insert(DirectAddrType::Portmapped);
1993        }
1994
1995        // Next add STUN addresses from the net_report report.
1996        if let Some(net_report_report) = net_report_report {
1997            if let Some(global_v4) = net_report_report.global_v4 {
1998                addrs.entry(global_v4.into()).or_insert(DirectAddrType::Qad);
1999
2000                // If they're behind a hard NAT and are using a fixed
2001                // port locally, assume they might've added a static
2002                // port mapping on their router to the same explicit
2003                // port that we are running with. Worst case it's an invalid candidate mapping.
2004                let port = self.msock.ip_bind_addrs().iter().find_map(|addr| {
2005                    if addr.port() != 0 {
2006                        Some(addr.port())
2007                    } else {
2008                        None
2009                    }
2010                });
2011
2012                if let Some(port) = port {
2013                    if net_report_report
2014                        .mapping_varies_by_dest()
2015                        .unwrap_or_default()
2016                    {
2017                        let mut addr = global_v4;
2018                        addr.set_port(port);
2019                        addrs
2020                            .entry(addr.into())
2021                            .or_insert(DirectAddrType::Qad4LocalPort);
2022                    }
2023                }
2024            }
2025            if let Some(global_v6) = net_report_report.global_v6 {
2026                addrs.entry(global_v6.into()).or_insert(DirectAddrType::Qad);
2027            }
2028        }
2029
2030        let local_addrs: Vec<(SocketAddr, SocketAddr)> = self
2031            .msock
2032            .ip_bind_addrs()
2033            .iter()
2034            .copied()
2035            .zip(self.msock.ip_local_addrs())
2036            .collect();
2037
2038        let has_ipv4_unspecified = local_addrs.iter().find_map(|(_, a)| {
2039            if a.is_ipv4() && a.ip().is_unspecified() {
2040                Some(a.port())
2041            } else {
2042                None
2043            }
2044        });
2045        let has_ipv6_unspecified = local_addrs.iter().find_map(|(_, a)| {
2046            if a.is_ipv6() && a.ip().is_unspecified() {
2047                Some(a.port())
2048            } else {
2049                None
2050            }
2051        });
2052
2053        // If a socket is bound to the unspecified address, create SocketAddrs for
2054        // each local IP address by pairing it with the port the socket is bound on.
2055        if local_addrs
2056            .iter()
2057            .any(|(_, local)| local.ip().is_unspecified())
2058        {
2059            let LocalAddresses {
2060                regular: mut ips,
2061                loopback,
2062            } = self
2063                .netmon_watcher
2064                .get()
2065                .expect("netmon disconnected")
2066                .local_addresses;
2067            if ips.is_empty() && addrs.is_empty() {
2068                // Include loopback addresses only if there are no other interfaces
2069                // or public addresses, this allows testing offline.
2070                ips = loopback;
2071            }
2072
2073            for ip in ips {
2074                let port_if_unspecified = match ip {
2075                    IpAddr::V4(_) => has_ipv4_unspecified,
2076                    IpAddr::V6(_) => has_ipv6_unspecified,
2077                };
2078                if let Some(port) = port_if_unspecified {
2079                    let addr = SocketAddr::new(ip, port);
2080                    addrs.entry(addr).or_insert(DirectAddrType::Local);
2081                }
2082            }
2083        }
2084
2085        // If a socket is bound to a specific address, add it.
2086        for (bound, local) in local_addrs {
2087            if !bound.ip().is_unspecified() {
2088                addrs.entry(local).or_insert(DirectAddrType::Local);
2089            }
2090        }
2091
2092        // Finally create and store store all these direct addresses and send any
2093        // queued call-me-maybe messages.
2094        self.msock.store_direct_addresses(
2095            addrs
2096                .iter()
2097                .map(|(addr, typ)| DirectAddr {
2098                    addr: *addr,
2099                    typ: *typ,
2100                })
2101                .collect(),
2102        );
2103        self.send_queued_call_me_maybes();
2104    }

2263/// The fake address used by the QUIC layer to address a node.
2264///
2265/// You can consider this as nothing more than a lookup key for a node the [`MagicSock`] knows
2266/// about.
2267///
2268/// [`MagicSock`] can reach a node by several real socket addresses, or maybe even via the relay
2269/// node.  The QUIC layer however needs to address a node by a stable [`SocketAddr`] so
2270/// that normal socket APIs can function.  Thus when a new node is introduced to a [`MagicSock`]
2271/// it is given a new fake address.  This is the type of that address.
2272///
2273/// It is but a newtype.  And in our QUIC-facing socket APIs like [`AsyncUdpSocket`] it
2274/// comes in as the inner [`Ipv6Addr`], in those interfaces we have to be careful to do
2275/// the conversion to this type.
2276#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
2277pub(crate) struct NodeIdMappedAddr(Ipv6Addr);
2278
2279/// Can occur when converting a [`SocketAddr`] to an [`NodeIdMappedAddr`]
2280#[derive(Debug, Snafu)]
2281#[snafu(display("Failed to convert"))]
2282pub struct NodeIdMappedAddrError;
2283
2284/// Counter to always generate unique addresses for [`NodeIdMappedAddr`].
2285static NODE_ID_ADDR_COUNTER: AtomicU64 = AtomicU64::new(1);
2286
2287impl NodeIdMappedAddr {
2288    /// The Prefix/L of our Unique Local Addresses.
2289    const ADDR_PREFIXL: u8 = 0xfd;
2290    /// The Global ID used in our Unique Local Addresses.
2291    const ADDR_GLOBAL_ID: [u8; 5] = [21, 7, 10, 81, 11];
2292    /// The Subnet ID used in our Unique Local Addresses.
2293    const ADDR_SUBNET: [u8; 2] = [0; 2];
2294
2295    /// The dummy port used for all [`NodeIdMappedAddr`]s.
2296    const NODE_ID_MAPPED_PORT: u16 = 12345;
2297
2298    /// Generates a globally unique fake UDP address.
2299    ///
2300    /// This generates and IPv6 Unique Local Address according to RFC 4193.
2301    pub(crate) fn generate() -> Self {
2302        let mut addr = [0u8; 16];
2303        addr[0] = Self::ADDR_PREFIXL;
2304        addr[1..6].copy_from_slice(&Self::ADDR_GLOBAL_ID);
2305        addr[6..8].copy_from_slice(&Self::ADDR_SUBNET);
2306
2307        let counter = NODE_ID_ADDR_COUNTER.fetch_add(1, Ordering::Relaxed);
2308        addr[8..16].copy_from_slice(&counter.to_be_bytes());
2309
2310        Self(Ipv6Addr::from(addr))
2311    }

2314    ///
2315    /// This socket address does not have a routable IP address.
2316    ///
2317    /// This uses a made-up port number, since the port does not play a role in looking up
2318    /// the node in the [`NodeMap`].  This socket address is only to be used to pass into
2319    /// Quinn.
2320    pub(crate) fn private_socket_addr(&self) -> SocketAddr {
2321        SocketAddr::new(IpAddr::from(self.0), Self::NODE_ID_MAPPED_PORT)
2322    }

2361///
2362/// Direct addresses are UDP socket addresses on which an iroh node could potentially be
2363/// contacted.  These can come from various sources depending on the network topology of the
2364/// iroh node, see [`DirectAddrType`] for the several kinds of sources.
2365#[derive(Debug, Clone, PartialEq, Eq, Hash, PartialOrd, Ord)]
2366pub struct DirectAddr {
2367    /// The address.
2368    pub addr: SocketAddr,
2369    /// The origin of this direct address.
2370    pub typ: DirectAddrType,
2371}

2373/// The type of direct address.

2375/// These are the various sources or origins from which an iroh node might have found a
2376/// possible [`DirectAddr`].

2380    Unknown,

2382    Local,

2384    ///

2386    /// from which it sends data on the public internet.  This can be different from locally
2387    /// bound addresses when the node is on a local network which performs NAT or similar.
2388    Qad,
2389    /// An address assigned by the router using port mapping.
2390    ///
2391    /// When possible an iroh node will request a port mapping from the local router to
2392    /// get a publicly routable direct address.
2393    Portmapped,
2394    /// Hard NAT: QAD'ed IPv4 address + local fixed port.
2395    ///
2396    /// It is possible to configure iroh to bound to a specific port and independently
2397    /// configure the router to forward this port to the iroh node.  This indicates a
2398    /// situation like this, which still uses QAD to discover the public address.

2400}

2530    /// Monitors endpoint changes and plumbs things together.
2531    ///
2532    /// This is a way of connecting endpoints without a relay server.  Whenever the local
2533    /// endpoints of a magic endpoint change this address is added to the other magic
2534    /// sockets.  This function will await until the endpoints are connected the first time
2535    /// before returning.
2536    ///
2537    /// When the returned drop guard is dropped, the tasks doing this updating are stopped.
2538    #[instrument(skip_all)]
2539    async fn mesh_stacks(stacks: Vec<MagicStack>) -> Result<JoinSet<()>> {
2540        /// Registers endpoint addresses of a node to all other nodes.
2541        fn update_direct_addrs(
2542            stacks: &[MagicStack],
2543            my_idx: usize,
2544            new_addrs: BTreeSet<DirectAddr>,
2545        ) {
2546            let me = &stacks[my_idx];
2547            for (i, m) in stacks.iter().enumerate() {
2548                if i == my_idx {
2549                    continue;
2550                }
2551
2552                let addr = NodeAddr {
2553                    node_id: me.public(),
2554                    relay_url: None,
2555                    direct_addresses: new_addrs.iter().map(|ep| ep.addr).collect(),
2556                };
2557                m.endpoint.magic_sock().add_test_addr(addr);
2558            }
2559        }
2560
2561        // For each node, start a task which monitors its local endpoints and registers them
2562        // with the other nodes as local endpoints become known.
2563        let mut tasks = JoinSet::new();
2564        for (my_idx, m) in stacks.iter().enumerate() {
2565            let m = m.clone();
2566            let stacks = stacks.clone();
2567            tasks.spawn(async move {
2568                let me = m.endpoint.node_id().fmt_short();
2569                let mut stream = m.endpoint.direct_addresses().stream().filter_map(|i| i);
2570                while let Some(new_eps) = stream.next().await {
2571                    info!(%me, "conn{} endpoints update: {:?}", my_idx + 1, new_eps);
2572                    update_direct_addrs(&stacks, my_idx, new_eps);
2573                }
2574            });
2575        }
2576
2577        // Wait for all nodes to be registered with each other.
2578        time::timeout(Duration::from_secs(10), async move {
2579            let all_node_ids: Vec<_> = stacks.iter().map(|ms| ms.endpoint.node_id()).collect();
2580            loop {
2581                let mut ready = Vec::with_capacity(stacks.len());
2582                for ms in stacks.iter() {
2583                    let endpoints = ms.tracked_endpoints();
2584                    let my_node_id = ms.endpoint.node_id();
2585                    let all_nodes_meshed = all_node_ids
2586                        .iter()
2587                        .filter(|node_id| **node_id != my_node_id)
2588                        .all(|node_id| endpoints.contains(node_id));
2589                    ready.push(all_nodes_meshed);
2590                }
2591                if ready.iter().all(|meshed| *meshed) {
2592                    break;
2593                }
2594                time::sleep(Duration::from_millis(200)).await;
2595            }
2596        })
2597        .await
2598        .context("timeout")?;
2599        info!("all nodes meshed");
2600        Ok(tasks)
2601    }

2702    /// Runs a roundtrip between the [`echo_sender`] and [`echo_receiver`].
2703    async fn run_roundtrip(
2704        sender: MagicStack,
2705        receiver: MagicStack,
2706        payload: &[u8],
2707        loss: ExpectedLoss,
2708    ) {
2709        let send_node_id = sender.endpoint.node_id();
2710        let recv_node_id = receiver.endpoint.node_id();
2711        info!("\nroundtrip: {send_node_id:#} -> {recv_node_id:#}");
2712
2713        let receiver_task = tokio::spawn(echo_receiver(receiver, loss));
2714        let sender_res = echo_sender(sender, recv_node_id, payload, loss).await;
2715        let sender_is_err = match sender_res {
2716            Ok(()) => false,
2717            Err(err) => {
2718                eprintln!("[sender] Error:\n{err:#?}");
2719                true
2720            }
2721        };
2722        let receiver_is_err = match receiver_task.await {
2723            Ok(Ok(())) => false,
2724            Ok(Err(err)) => {
2725                eprintln!("[receiver] Error:\n{err:#?}");
2726                true
2727            }
2728            Err(joinerr) => {
2729                if joinerr.is_panic() {
2730                    std::panic::resume_unwind(joinerr.into_panic());
2731                } else {
2732                    eprintln!("[receiver] Error:\n{joinerr:#?}");
2733                }
2734                true
2735            }
2736        };
2737        if sender_is_err || receiver_is_err {
2738            panic!("Sender or receiver errored");
2739        }
2740    }
