1//! Tools for spawning an accept loop that routes incoming requests to the right protocol.
2//!
3//! ## Example
4//!
5//! ```no_run
6//! # use iroh::{
7//! #     endpoint::{Connection, BindError},
8//! #     protocol::{AcceptError, ProtocolHandler, Router},
9//! #     Endpoint,
10//! #     EndpointAddr
11//! # };
12//! #
13//! # async fn test_compile() -> Result<(), BindError> {
14//! let endpoint = Endpoint::bind().await?;
15//!
16//! let router = Router::builder(endpoint).accept(b"/my/alpn", Echo).spawn();
17//! # Ok(())
18//! # }
19//!
20//! // The protocol definition:
21//! #[derive(Debug, Clone)]
22//! struct Echo;
23//!
24//! impl ProtocolHandler for Echo {
25//!     async fn accept(&self, connection: Connection) -> Result<(), AcceptError> {
26//!         let (mut send, mut recv) = connection.accept_bi().await?;
27//!
28//!         // Echo any bytes received back directly.
29//!         let bytes_sent = tokio::io::copy(&mut recv, &mut send).await?;
30//!
31//!         send.finish()?;
32//!         connection.closed().await;
33//!
34//!         Ok(())
35//!     }
36//! }
37//! ```

60/// The built router.
61///
62/// Construct this using [`Router::builder`].
63///
64/// When dropped, this will abort listening the tasks, so make sure to store it.
65///
66/// Even with this abort-on-drop behaviour, it's recommended to call and await
67/// [`Router::shutdown`] before ending the process.
68///
69/// As an example for graceful shutdown, e.g. for tests or CLI tools,
70/// wait for [`tokio::signal::ctrl_c()`]:
71///
72/// ```no_run
73/// # use std::sync::Arc;
74/// # use n0_error::StdResultExt;
75/// # use iroh::{endpoint::Connecting, protocol::{ProtocolHandler, Router}, Endpoint, EndpointAddr};
76/// #
77/// # async fn test_compile() -> n0_error::Result<()> {
78/// let endpoint = Endpoint::bind().await?;
79///
80/// let router = Router::builder(endpoint)
81///     // .accept(&ALPN, <something>)
82///     .spawn();
83///
84/// // wait until the user wants to
85/// tokio::signal::ctrl_c().await.std_context("ctrl+c")?;
86/// router.shutdown().await.std_context("shutdown")?;
87/// # Ok(())
88/// # }
89/// ```
90#[derive(Clone, Debug)]
91pub struct Router {
92    endpoint: Endpoint,
93    // `Router` needs to be `Clone + Send`, and we need to `task.await` in its `shutdown()` impl.
94    task: Arc<Mutex<Option<AbortOnDropHandle<()>>>>,
95    cancel_token: CancellationToken,
96}

157/// Outcome when filtering incoming connections by their remote address.
158///
159/// We can either accept the connection, use retry to make sure the socket
160/// addr is not spoofed, reject it, or ignore the connection entirely.
161#[derive(Debug, Clone, Copy, PartialEq, Eq)]
162pub enum AcceptAddrOutcome {
163    /// Accept the connection. Later filters will still run.
164    Accept,
165    /// Tell the remote to retry. If the socket address was spoofed, this will
166    /// send the retry token to the spoofed address, so we will never hear from
167    /// the attacker again. If the socket address was correct, we will get
168    /// a connection attempt for which the socket address is validated.
169    Retry,
170    /// Actively refuse the connection. The remote will receive a
171    /// CONNECTION_REFUSED error immediately.
172    Reject,
173    /// Ignore the connection entirely. The remote gets no response and will
174    /// eventually time out. No further filters will run.
175    Ignore,
176}
177
178/// Outcome of filtering a connection by endpoint ID.
179///
180/// Unlike [`AcceptAddrOutcome`], this does not have a `Retry` variant since
181/// address validation is not applicable at this stage.
182#[derive(Debug, Clone, Copy, PartialEq, Eq)]
183pub enum AcceptOutcome {
184    /// Accept the connection. Later filters will still run.
185    Accept,
186    /// Actively refuse the connection. The remote will receive an error.
187    Reject,
188    /// Ignore the connection entirely. The remote gets no response.
189    Ignore,
190}

192/// Outcome of filtering a connection by ALPN.
193///
194/// At this stage, the connection is fully established, so the filter can
195/// provide a close error code and reason that will be sent to the remote peer.
196#[derive(Debug, Clone, PartialEq, Eq)]
197pub enum AcceptAlpnOutcome {
198    /// Accept the connection.
199    Accept,
200    /// Close the connection with the given error code and reason.
201    Close {
202        /// The error code to send to the remote peer.
203        error_code: VarInt,
204        /// The reason to send to the remote peer.
205        reason: Vec<u8>,
206    },
207}
208
209impl AcceptAlpnOutcome {
210    /// Create a `Close` outcome with the given error code and reason.
211    pub fn close(error_code: impl Into<VarInt>, reason: impl Into<Vec<u8>>) -> Self {
212        Self::Close {
213            error_code: error_code.into(),
214            reason: reason.into(),
215        }
216    }
217}

219/// A filter that allows rejecting incoming connections early, based on socket
220/// address, endpoint ID, or ALPN.
221pub trait ConnectionFilter: std::fmt::Debug + Send + Sync + 'static {
222    /// Accept an incoming connection by its socket address.
223    ///
224    /// If `validated` is false, this is just the socket address in the UDP packet,
225    /// which might be spoofed. If `validated` is true, the remote has responded
226    /// to a retry and we know that the socket address is correct.
227    fn accept_addr(&self, _addr: SocketAddr, _validated: bool) -> AcceptAddrOutcome {
228        AcceptAddrOutcome::Accept
229    }
230
231    /// Accept an incoming connection by its endpoint ID.
232    fn accept_endpoint_id(&self, _endpoint_id: EndpointId) -> AcceptOutcome {
233        AcceptOutcome::Accept
234    }
235
236    /// Accept an incoming connection by its ALPN. This can be used to implement
237    /// protocol level access control without having to wrap the protocol handler.
238    ///
239    /// This is called after the connection is fully established, so the filter
240    /// can provide a close error code and reason.
241    fn accept_alpn(&self, _endpoint_id: EndpointId, _alpn: &[u8]) -> AcceptAlpnOutcome {
242        AcceptAlpnOutcome::Accept
243    }
244}

246/// Handler for incoming connections.
247///
248/// A router accepts connections for arbitrary ALPN protocols.
249///
250/// With this trait, you can handle incoming connections for any protocol.
251///
252/// Implement this trait on a struct that should handle incoming connections.
253/// The protocol handler must then be registered on the endpoint for an ALPN protocol with
254/// [`crate::protocol::RouterBuilder::accept`].
255///
256/// See the [module documentation](crate::protocol) for an example.
257pub trait ProtocolHandler: Send + Sync + std::fmt::Debug + 'static {
258    /// Optional interception point to handle the `Accepting` state.
259    ///
260    /// Can be implemented as `async fn on_accepting(&self, accepting: Accepting) -> Result<Connection>`.
261    ///
262    /// Typically, this method is used as an early interception point to accept
263    /// or reject a connection.
264    ///
265    /// However, this method can also be used to implement the accept side of a
266    /// 0-RTT connection.
267    ///
268    /// ## 0-RTT
269    ///
270    /// `ProtocolHandler::on_accepting` allows you to take over the connection
271    /// state machine early in the handshake processes, by calling [`Accepting::into_0rtt`].
272    ///
273    /// When working with 0-RTT, you may want to implement all of your protocol
274    /// logic in `on_accepting`. This is fine because `on_accepting` can handle
275    /// long-running processes. In this case, the [`ProtocolHandler::accept`] method
276    /// can simply return `Ok(())`.
277    fn on_accepting(
278        &self,
279        accepting: Accepting,
280    ) -> impl Future<Output = Result<Connection, AcceptError>> + Send {
281        async move {
282            let conn = accepting.await?;
283            Ok(conn)
284        }
285    }
286
287    /// Handle an incoming connection.
288    ///
289    /// Can be implemented as `async fn accept(&self, connection: Connection) -> Result<()>`.
290    ///
291    /// The returned future runs on a freshly spawned tokio task so it can be long-running. Once
292    /// `accept()` returns, the connection is dropped. This means that it will be closed
293    /// if there are no other clones of the connection.  If there is a protocol error, you
294    /// can use [`Connection::close`] to send an error code to the remote peer. Returning
295    /// an `Err<AcceptError>` will also drop the connection and log a warning, but no
296    /// dedicated error code will be sent to the peer, so it's recommended to explicitly
297    /// close the connection within your accept handler.
298    ///
299    /// When [`Router::shutdown`] is called, no further connections will be accepted, and
300    /// the futures returned by [`Self::accept`] will be aborted after the future returned
301    /// from [`ProtocolHandler::shutdown`] completes.
302    fn accept(
303        &self,
304        connection: Connection,
305    ) -> impl Future<Output = Result<(), AcceptError>> + Send;
306
307    /// Called when the router shuts down.
308    ///
309    /// Can be implemented as `async fn shutdown(&self)`.
310    ///
311    /// This is called from [`Router::shutdown`]. The returned future is awaited before
312    /// the router closes the endpoint.
313    fn shutdown(&self) -> impl Future<Output = ()> + Send {
314        async move {}
315    }
316}

352/// A dyn-compatible version of [`ProtocolHandler`] that returns boxed futures.
353///
354/// Any type that implements [`ProtocolHandler`] automatically also implements [`DynProtocolHandler`].
355/// There is a also [`From`] impl to turn any type that implements [`ProtocolHandler`] into a
356/// `Box<dyn DynProtocolHandler>`.
357//
358// We are not using [`n0_future::boxed::BoxFuture] because we don't need a `'static` bound
359// on these futures.
360pub trait DynProtocolHandler: Send + Sync + std::fmt::Debug + 'static {
361    /// See [`ProtocolHandler::on_accepting`].
362    fn on_accepting(
363        &self,
364        accepting: Accepting,
365    ) -> Pin<Box<dyn Future<Output = Result<Connection, AcceptError>> + Send + '_>> {
366        Box::pin(async move {
367            let conn = accepting.await?;
368            Ok(conn)
369        })
370    }
371
372    /// See [`ProtocolHandler::accept`].
373    fn accept(
374        &self,
375        connection: Connection,
376    ) -> Pin<Box<dyn Future<Output = Result<(), AcceptError>> + Send + '_>>;
377
378    /// See [`ProtocolHandler::shutdown`].
379    fn shutdown(&self) -> Pin<Box<dyn Future<Output = ()> + Send + '_>> {
380        Box::pin(async move {})
381    }
382}

384impl<P: ProtocolHandler> DynProtocolHandler for P {
385    fn accept(
386        &self,
387        connection: Connection,
388    ) -> Pin<Box<dyn Future<Output = Result<(), AcceptError>> + Send + '_>> {
389        Box::pin(<Self as ProtocolHandler>::accept(self, connection))
390    }
391
392    fn on_accepting(
393        &self,
394        accepting: Accepting,
395    ) -> Pin<Box<dyn Future<Output = Result<Connection, AcceptError>> + Send + '_>> {
396        Box::pin(<Self as ProtocolHandler>::on_accepting(self, accepting))
397    }
398
399    fn shutdown(&self) -> Pin<Box<dyn Future<Output = ()> + Send + '_>> {
400        Box::pin(<Self as ProtocolHandler>::shutdown(self))
401    }

404/// A typed map of protocol handlers, mapping them from ALPNs.
405#[derive(Debug, Default)]
406pub(crate) struct ProtocolMap(BTreeMap<Vec<u8>, Box<dyn DynProtocolHandler>>);
407
408impl ProtocolMap {
409    /// Returns the registered protocol handler for an ALPN as a [`Arc<dyn ProtocolHandler>`].
410    pub(crate) fn get(&self, alpn: &[u8]) -> Option<&dyn DynProtocolHandler> {
411        self.0.get(alpn).map(|p| &**p)
412    }
413
414    /// Inserts a protocol handler.
415    pub(crate) fn insert(&mut self, alpn: Vec<u8>, handler: Box<dyn DynProtocolHandler>) {
416        self.0.insert(alpn, handler);
417    }
418
419    /// Returns an iterator of all registered ALPN protocol identifiers.
420    pub(crate) fn alpns(&self) -> impl Iterator<Item = &Vec<u8>> {
421        self.0.keys()
422    }
423
424    /// Shuts down all protocol handlers.
425    ///
426    /// Calls and awaits [`ProtocolHandler::shutdown`] for all registered handlers concurrently.
427    pub(crate) async fn shutdown(&self) {
428        let handlers = self.0.values().map(|p| p.shutdown());
429        join_all(handlers).await;
430    }
431}

433impl Router {
434    /// Creates a new [`Router`] using given [`Endpoint`].
435    pub fn builder(endpoint: Endpoint) -> RouterBuilder {
436        RouterBuilder::new(endpoint)
437    }
438
439    /// Returns the [`Endpoint`] stored in this router.
440    pub fn endpoint(&self) -> &Endpoint {
441        &self.endpoint
442    }
443
444    /// Checks if the router is already shutdown.
445    pub fn is_shutdown(&self) -> bool {
446        self.cancel_token.is_cancelled()
447    }
448
449    /// Shuts down the accept loop cleanly.
450    ///
451    /// When this function returns, all [`ProtocolHandler`]s will be shutdown and
452    /// `Endpoint::close` will have been called.
453    ///
454    /// If already shutdown, it returns `Ok`.
455    ///
456    /// If some [`ProtocolHandler`] panicked in the accept loop, this will propagate
457    /// that panic into the result here.
458    pub async fn shutdown(&self) -> Result<(), n0_future::task::JoinError> {
459        if self.is_shutdown() {
460            return Ok(());
461        }
462
463        // Trigger shutdown of the main run task by activating the cancel token.
464        self.cancel_token.cancel();
465
466        // Wait for the main task to terminate.
467
468        // MutexGuard is not held across await point
469        let task = self.task.lock().expect("poisoned").take();
470        if let Some(task) = task {
471            task.await?;
472        }
473
474        Ok(())
475    }
476}

479    /// Creates a new router builder using given [`Endpoint`].

489    pub fn connection_filter(mut self, filter: Arc<dyn ConnectionFilter>) -> Self {
490        self.connection_filter = Some(filter);

492    }
493
494    /// Configures the router to accept the [`ProtocolHandler`] when receiving a connection
495    /// with this `alpn`.
496    ///
497    /// `handler` can either be a type that implements [`ProtocolHandler`] or a
498    /// [`Box<dyn DynProtocolHandler>`].
499    ///
500    /// [`Box<dyn DynProtocolHandler>`]: DynProtocolHandler
501    pub fn accept(
502        mut self,
503        alpn: impl AsRef<[u8]>,
504        handler: impl Into<Box<dyn DynProtocolHandler>>,
505    ) -> Self {
506        self.protocols
507            .insert(alpn.as_ref().to_vec(), handler.into());
508        self
509    }
510
511    /// Returns the [`Endpoint`] of the endpoint.
512    pub fn endpoint(&self) -> &Endpoint {
513        &self.endpoint
514    }
515
516    /// Spawns an accept loop and returns a handle to it encapsulated as the [`Router`].
517    #[must_use = "Router aborts when dropped, use Router::shutdown to shut the router down cleanly"]
518    pub fn spawn(self) -> Router {
519        // Update the endpoint with our alpns.
520        let alpns = self
521            .protocols
522            .alpns()
523            .map(|alpn| alpn.to_vec())
524            .collect::<Vec<_>>();
525
526        let protocols = Arc::new(self.protocols);
527        let connection_filter = self.connection_filter;
528        self.endpoint.set_alpns(alpns);
529
530        let mut join_set = JoinSet::new();
531        let endpoint = self.endpoint.clone();
532
533        // Our own shutdown works with a cancellation token.
534        let cancel = CancellationToken::new();
535        let cancel_token = cancel.clone();
536
537        let run_loop_fut = async move {
538            // Make sure to cancel the token, if this future ever exits.
539            let _cancel_guard = cancel_token.clone().drop_guard();
540            // We create a separate cancellation token to stop any `ProtocolHandler::accept` futures
541            // that are still running after `ProtocolHandler::shutdown` was called.
542            let handler_cancel_token = CancellationToken::new();
543
544            loop {
545                tokio::select! {
546                    biased;
547                    _ = cancel_token.cancelled() => {
548                        break;
549                    },
550                    // handle task terminations and quit on panics.
551                    Some(res) = join_set.join_next() => {
552                        match res {
553                            Err(outer) => {
554                                if outer.is_panic() {
555                                    error!("Task panicked: {outer:?}");
556                                    break;
557                                } else if outer.is_cancelled() {
558                                    trace!("Task cancelled: {outer:?}");
559                                } else {
560                                    error!("Task failed: {outer:?}");
561                                    break;
562                                }
563                            }
564                            Ok(Some(())) => {
565                                trace!("Task finished");
566                            }
567                            Ok(None) => {
568                                trace!("Task cancelled");
569                            }
570                        }
571                    },
572
573                    // handle incoming p2p connections.
574                    incoming = endpoint.accept() => {
575                        let Some(incoming) = incoming else {
576                            break; // Endpoint is closed.
577                        };
578
579                        let addr = incoming.remote_addr();
580                        if let Some(filter) = &connection_filter {
581                            match &addr {
582                                // Addr filter: direct connections only.
583                                IncomingAddr::Ip(socket_addr) => {
584                                    let validated = incoming.remote_addr_validated();
585                                    match filter.accept_addr(*socket_addr, validated) {
586                                        AcceptAddrOutcome::Accept => {}
587                                        AcceptAddrOutcome::Retry => {
588                                            if validated {
589                                                incoming.refuse();
590                                            } else if let Err(err) = incoming.retry() {
591                                                err.into_incoming().refuse();
592                                            }
593                                            continue;
594                                        }
595                                        AcceptAddrOutcome::Reject => {
596                                            incoming.refuse();
597                                            continue;
598                                        }
599                                        AcceptAddrOutcome::Ignore => {
600                                            incoming.ignore();
601                                            continue;
602                                        }
603                                    }
604                                }
605                                // Endpoint ID filter: relay connections only.
606                                // We have the endpoint ID from the relay frame,
607                                // before the TLS handshake.
608                                IncomingAddr::Relay { endpoint_id, .. } => {
609                                    match filter.accept_endpoint_id(*endpoint_id) {
610                                        AcceptOutcome::Accept => {}
611                                        AcceptOutcome::Reject => {
612                                            incoming.refuse();
613                                            continue;
614                                        }
615                                        AcceptOutcome::Ignore => {
616                                            incoming.ignore();
617                                            continue;
618                                        }
619                                    }
620                                }
621                            }
622                        }
623
624                        let protocols = protocols.clone();
625                        let filter = connection_filter.clone();
626                        let token = handler_cancel_token.child_token();
627                        let span = info_span!("router.accept", me=%endpoint.id().fmt_short(), remote=Empty, alpn=Empty);
628                        join_set.spawn(async move {
629                            token.run_until_cancelled(handle_connection(incoming, protocols, filter)).await
630                        }.instrument(span));
631                    },
632                }
633            }
634
635            // We first shutdown the protocol handlers to give them a chance to close connections gracefully.
636            protocols.shutdown().await;
637            // We now cancel the remaining `ProtocolHandler::accept` futures.
638            handler_cancel_token.cancel();
639            // Now we close the endpoint. This will force-close all connections that are not yet closed.
640            endpoint.close().await;
641            // Finally, we abort the remaining accept tasks. This should be a noop because we already cancelled
642            // the futures above.
643            tracing::debug!("Shutting down remaining tasks");
644            join_set.abort_all();
645            while let Some(res) = join_set.join_next().await {
646                match res {
647                    Err(err) if err.is_panic() => error!("Task panicked: {err:?}"),
648                    _ => {}
649                }
650            }
651        };
652        let task = task::spawn(run_loop_fut.instrument(tracing::Span::current()));
653        let task = AbortOnDropHandle::new(task);
654
655        Router {
656            endpoint: self.endpoint,
657            task: Arc::new(Mutex::new(Some(task))),
658            cancel_token: cancel,
659        }
660    }
661}

663async fn handle_connection(
664    incoming: crate::endpoint::Incoming,
665    protocols: Arc<ProtocolMap>,
666    filter: Option<Arc<dyn ConnectionFilter>>,
667) {
668    let mut accepting = match incoming.accept() {
669        Ok(conn) => conn,
670        Err(err) => {
671            warn!("Ignoring connection: accepting failed: {err:#}");
672            return;
673        }
674    };
675    let alpn = match accepting.alpn().await {
676        Ok(alpn) => alpn,
677        Err(err) => {
678            warn!("Ignoring connection: invalid handshake: {err:#}");
679            return;
680        }
681    };
682    tracing::Span::current().record("alpn", String::from_utf8_lossy(&alpn).to_string());
683
684    let Some(handler) = protocols.get(&alpn) else {
685        warn!("Ignoring connection: unsupported ALPN protocol");
686        return;
687    };
688    match handler.on_accepting(accepting).await {
689        Ok(connection) => {
690            tracing::Span::current().record(
691                "remote",
692                tracing::field::display(connection.remote_id().fmt_short()),
693            );
694
695            // ALPN filter: runs after the connection is established for both
696            // relay and direct connections.
697            if let Some(filter) = &filter {
698                match filter.accept_alpn(connection.remote_id(), &alpn) {
699                    AcceptAlpnOutcome::Accept => {}
700                    AcceptAlpnOutcome::Close {
701                        error_code,
702                        reason,
703                    } => {
704                        trace!(
705                            "Connection rejected by ALPN filter: endpoint {}",
706                            connection.remote_id().fmt_short()
707                        );
708                        connection.close(error_code, &reason);
709                        return;
710                    }
711                }
712            }
713
714            if let Err(err) = handler.accept(connection).await {
715                warn!("Handling incoming connection ended with error: {err}");
716            }
717        }
718        Err(err) => {
719            warn!("Accepting incoming connection ended with error: {err}");
720        }
721    }
722}

724/// Wraps an existing protocol, limiting its access,
725/// based on the provided function.
726///
727/// Any refused connection will be closed with an error code of `0` and reason `not allowed`.
728#[derive(derive_more::Debug, Clone)]
729pub struct AccessLimit<P: ProtocolHandler + Clone> {
730    proto: P,
731    #[debug("limiter")]
732    limiter: Arc<dyn Fn(EndpointId) -> bool + Send + Sync + 'static>,
733}
734
735impl<P: ProtocolHandler + Clone> AccessLimit<P> {
736    /// Create a new `AccessLimit`.
737    ///
738    /// The function should return `true` for endpoints that are allowed to
739    /// connect, and `false` otherwise.
740    pub fn new<F>(proto: P, limiter: F) -> Self
741    where
742        F: Fn(EndpointId) -> bool + Send + Sync + 'static,
743    {
744        Self {
745            proto,
746            limiter: Arc::new(limiter),
747        }
748    }
749}
750
751impl<P: ProtocolHandler + Clone> ProtocolHandler for AccessLimit<P> {
752    fn on_accepting(
753        &self,
754        accepting: Accepting,
755    ) -> impl Future<Output = Result<Connection, AcceptError>> + Send {
756        self.proto.on_accepting(accepting)
757    }
758
759    async fn accept(&self, conn: Connection) -> Result<(), AcceptError> {
760        let remote = conn.remote_id();
761        let is_allowed = (self.limiter)(remote);
762        if !is_allowed {
763            conn.close(0u32.into(), b"not allowed");
764            return Err(e!(AcceptError::NotAllowed));
765        }
766        self.proto.accept(conn).await?;
767        Ok(())
768    }
769
770    fn shutdown(&self) -> impl Future<Output = ()> + Send {
771        self.proto.shutdown()
772    }
773}
