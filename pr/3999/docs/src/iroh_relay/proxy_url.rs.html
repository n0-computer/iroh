1//! A wrapper type for proxy URL configuration that compiles to a ZST when
2//! the `proxy` feature is disabled, eliminating the need for `#[cfg(feature = "proxy")]`
3//! on every struct field and constructor that threads proxy configuration.
4
5use url::Url;
6
7/// Proxy URL configuration.
8///
9/// When the `proxy` feature is enabled, this wraps an `Option<Url>`.
10/// When disabled, this is a zero-size type whose methods are no-ops.
11#[derive(Debug, Clone, Default)]
12#[cfg(feature = "proxy")]
13pub struct ProxyUrl(Option<Url>);
14
15/// Proxy URL configuration.
16///
17/// The `proxy` feature is disabled, so this is a zero-size no-op type.
18#[derive(Debug, Clone, Default)]
19#[cfg(not(feature = "proxy"))]
20pub struct ProxyUrl;
21
22#[cfg(feature = "proxy")]
23impl ProxyUrl {
24    /// Create a new `ProxyUrl` wrapping the given URL.
25    pub fn new(url: Url) -> Self {
26        Self(Some(url))
27    }
28
29    /// View as `Option<&Url>`.
30    pub fn as_option(&self) -> Option<&Url> {
31        self.0.as_ref()
32    }
33
34    /// Read a proxy URL from the environment, in this order:
35    ///
36    /// - `HTTP_PROXY`
37    /// - `http_proxy`
38    /// - `HTTPS_PROXY`
39    /// - `https_proxy`
40    pub fn from_env() -> Self {
41        if let Some(url) = std::env::var("HTTP_PROXY")
42            .ok()
43            .and_then(|s| s.parse::<Url>().ok())
44        {
45            if is_cgi() {
46                tracing::warn!("HTTP_PROXY environment variable ignored in CGI");
47            } else {
48                return Self::new(url);
49            }
50        }
51        if let Some(url) = std::env::var("http_proxy")
52            .ok()
53            .and_then(|s| s.parse::<Url>().ok())
54        {
55            return Self::new(url);
56        }
57        if let Some(url) = std::env::var("HTTPS_PROXY")
58            .ok()
59            .and_then(|s| s.parse::<Url>().ok())
60        {
61            return Self::new(url);
62        }
63        if let Some(url) = std::env::var("https_proxy")
64            .ok()
65            .and_then(|s| s.parse::<Url>().ok())
66        {
67            return Self::new(url);
68        }
69        Self::default()
70    }
71}
72
73/// Check if we are being executed in a CGI context.
74///
75/// If so, a malicious client can send the `Proxy:` header, and it will
76/// be in the `HTTP_PROXY` env var. So we don't use it :)
77#[cfg(feature = "proxy")]
78fn is_cgi() -> bool {
79    std::env::var_os("REQUEST_METHOD").is_some()
80}
81
82#[cfg(not(feature = "proxy"))]
83impl ProxyUrl {
84    /// View as `Option<&Url>` (always `None` without the proxy feature).
85    pub fn as_option(&self) -> Option<&Url> {
86        None
87    }
88
89    /// No-op without the proxy feature.
90    pub fn from_env() -> Self {
91        Self
92    }
93}
