1//! HTTP-specific constants for the relay server and client.
2
3pub(crate) const HTTP_UPGRADE_PROTOCOL: &str = "iroh derp http";
4pub(crate) const WEBSOCKET_UPGRADE_PROTOCOL: &str = "websocket";
5#[cfg(feature = "server")] // only used in the server for now
6pub(crate) const SUPPORTED_WEBSOCKET_VERSION: &str = "13";
7
8/// The HTTP path under which the relay accepts relaying connections
9/// (over websockets and a custom upgrade protocol).
10pub const RELAY_PATH: &str = "/relay";
11/// The HTTP path under which the relay allows doing latency queries for testing.
12pub const RELAY_PROBE_PATH: &str = "/ping";
13/// The legacy HTTP path under which the relay used to accept relaying connections.
14/// We keep this for backwards compatibility.
15#[cfg(feature = "server")] // legacy paths only used on server-side for backwards compat
16pub(crate) const LEGACY_RELAY_PATH: &str = "/derp";
17
18/// The HTTP upgrade protocol used for relaying.
19#[derive(Debug, Copy, Clone, PartialEq, Eq)]
20pub enum Protocol {
21    /// Relays over the custom relaying protocol with a custom HTTP upgrade header.
22    Relay,
23    /// Relays over websockets.
24    ///
25    /// Originally introduced to support browser connections.
26    Websocket,
27}
28
29impl Default for Protocol {
30    fn default() -> Self {
31        #[cfg(not(wasm_browser))]
32        return Self::Relay;
33        #[cfg(wasm_browser)]
34        return Self::Websocket;
35    }
36}
37
38impl Protocol {
39    /// The HTTP upgrade header used or expected.
40    pub const fn upgrade_header(&self) -> &'static str {
41        match self {
42            Protocol::Relay => HTTP_UPGRADE_PROTOCOL,
43            Protocol::Websocket => WEBSOCKET_UPGRADE_PROTOCOL,
44        }
45    }
46
47    /// Tries to match the value of an HTTP upgrade header to figure out which protocol should be initiated.
48    pub fn parse_header(header: &http::HeaderValue) -> Option<Self> {
49        let header_bytes = header.as_bytes();
50        if header_bytes == Protocol::Relay.upgrade_header().as_bytes() {
51            Some(Protocol::Relay)
52        } else if header_bytes == Protocol::Websocket.upgrade_header().as_bytes() {
53            Some(Protocol::Websocket)
54        } else {
55            None
56        }
57    }
58}
