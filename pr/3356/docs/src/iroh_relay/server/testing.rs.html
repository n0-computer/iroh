1//! Exposes functions to quickly configure a server suitable for testing.

8/// Creates a [`StunConfig`] suitable for testing.
9///
10/// To ensure port availability for testing, the port is configured to be assigned by the OS.
11pub fn stun_config() -> StunConfig {
12    StunConfig {
13        bind_addr: (Ipv4Addr::LOCALHOST, 0).into(),
14    }
15}

17/// Creates a [`rustls::ServerConfig`] and certificates suitable for testing.
18///
19/// - Uses a self signed certificate valid for the `"localhost"` and `"127.0.0.1"` domains.
20pub fn self_signed_tls_certs_and_config() -> (
21    Vec<rustls::pki_types::CertificateDer<'static>>,
22    rustls::ServerConfig,
23) {
24    let cert = rcgen::generate_simple_self_signed(vec![
25        "localhost".to_string(),
26        "127.0.0.1".to_string(),
27        "::1".to_string(),
28    ])
29    .expect("valid");
30    let rustls_cert = cert.cert.der();
31    let private_key = rustls::pki_types::PrivatePkcs8KeyDer::from(cert.key_pair.serialize_der());
32    let private_key = rustls::pki_types::PrivateKeyDer::from(private_key);
33    let certs = vec![rustls_cert.clone()];
34    let server_config = rustls::ServerConfig::builder_with_provider(std::sync::Arc::new(
35        rustls::crypto::ring::default_provider(),
36    ))
37    .with_safe_default_protocol_versions()
38    .expect("protocols supported by ring")
39    .with_no_client_auth();
40
41    let server_config = server_config
42        .with_single_cert(certs.clone(), private_key)
43        .expect("valid");
44    (certs, server_config)
45}

47/// Creates a [`TlsConfig`] suitable for testing.
48///
49/// - Uses a self signed certificate valid for the `"localhost"` and `"127.0.0.1"` domains.
50/// - Configures https to be served on an OS assigned port on ipv4.
51pub fn tls_config() -> TlsConfig<()> {
52    let (certs, server_config) = self_signed_tls_certs_and_config();
53    TlsConfig {
54        server_config,
55        cert: CertConfig::<(), ()>::Manual { certs },
56        https_bind_addr: (Ipv4Addr::LOCALHOST, 0).into(),
57        quic_bind_addr: (Ipv4Addr::UNSPECIFIED, 0).into(),
58    }
59}
60
61/// Creates a [`RelayConfig`] suitable for testing.
62///
63/// - Binds http to an OS assigned port on ipv4.
64/// - Uses [`tls_config`] to enable TLS.
65/// - Uses default limits.
66pub fn relay_config() -> RelayConfig<()> {
67    RelayConfig {
68        http_bind_addr: (Ipv4Addr::LOCALHOST, 0).into(),
69        tls: Some(tls_config()),
70        limits: Default::default(),
71        key_cache_capacity: Some(1024),
72        access: AccessConfig::Everyone,
73    }
74}
75
76/// Creates a [`QuicConfig`] suitable for testing.
77///
78/// - Binds to an OS assigned port on ipv4
79/// - Uses [`self_signed_tls_certs_and_config`] to create tls certificates
80pub fn quic_config() -> QuicConfig {
81    let (_, server_config) = self_signed_tls_certs_and_config();
82    QuicConfig {
83        bind_addr: (Ipv4Addr::UNSPECIFIED, 0).into(),
84        server_config,
85    }
86}

88/// Creates a [`ServerConfig`] suitable for testing.
89///
90/// - Relaying is enabled using [`relay_config`]
91/// - Stun is enabled using [`stun_config`]
92/// - QUIC addr discovery is disabled.
93/// - Metrics are not enabled.
94pub fn server_config() -> ServerConfig<()> {
95    ServerConfig {
96        relay: Some(relay_config()),
97        stun: Some(stun_config()),
98        quic: Some(quic_config()),
99        #[cfg(feature = "metrics")]
100        metrics_addr: None,
101    }
