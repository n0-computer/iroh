14/// A net_report report.
15#[derive(Default, Debug, PartialEq, Eq, Clone, Serialize, Deserialize)]
16pub struct Report {
17    /// A QAD IPv4 round trip completed.
18    pub udp_v4: bool,
19    /// A QAD IPv6 round trip completed.
20    pub udp_v6: bool,
21    /// Whether the reported public address differs when probing different servers (on IPv4).
22    pub mapping_varies_by_dest_ipv4: Option<bool>,
23    /// Whether the reported public address differs when probing different servers (on IPv6).
24    pub mapping_varies_by_dest_ipv6: Option<bool>,
25    /// Probe indicating the presence of port mapping protocols on the LAN.
26    /// `None` for unknown
27    pub preferred_relay: Option<RelayUrl>,
28    /// keyed by relay Url
29    pub relay_latency: RelayLatencies,
30    /// ip:port of global IPv4
31    pub global_v4: Option<SocketAddrV4>,
32    /// `[ip]:port` of global IPv6
33    pub global_v6: Option<SocketAddrV6>,
34    /// CaptivePortal is set when we think there's a captive portal that is
35    /// intercepting HTTP traffic.
36    pub captive_portal: Option<bool>,
37}
38
39impl fmt::Display for Report {
40    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
41        fmt::Debug::fmt(&self, f)
42    }
43}

45impl Report {
46    /// Do we have any indication that UDP is working?
47    pub fn has_udp(&self) -> bool {
48        self.udp_v4 || self.udp_v6
49    }
50
51    /// Whether the reported public address differs when probing different servers.
52    pub fn mapping_varies_by_dest(&self) -> Option<bool> {
53        match (
54            self.mapping_varies_by_dest_ipv4,
55            self.mapping_varies_by_dest_ipv6,
56        ) {
57            (Some(v4), Some(v6)) => Some(v4 || v6),
58            (None, Some(v6)) => Some(v6),
59            (Some(v4), None) => Some(v4),
60            (None, None) => None,
61        }
62    }
63
64    /// Updates a net_report [`Report`] with a new [`ProbeReport`].
65    pub(super) fn update(&mut self, report: &ProbeReport) {
66        match report {
67            ProbeReport::Https(report) => {
68                self.relay_latency
69                    .update_relay(report.node.clone(), report.latency, Probe::Https);
70            }
71            #[cfg(not(wasm_browser))]
72            ProbeReport::QadIpv4(report) => {
73                self.relay_latency.update_relay(
74                    report.node.clone(),
75                    report.latency,
76                    Probe::QadIpv4,
77                );
78                let SocketAddr::V4(ipp) = report.addr else {
79                    warn!("received IPv6 address from IPv4 QAD: {}", report.addr);
80                    return;
81                };
82
83                self.udp_v4 = true;
84
85                tracing::debug!(?self.global_v4, ?self.mapping_varies_by_dest_ipv4, %ipp,"got");
86                if let Some(global) = self.global_v4 {
87                    if global == ipp {
88                        if self.mapping_varies_by_dest_ipv4.is_none() {
89                            self.mapping_varies_by_dest_ipv4 = Some(false);
90                        }
91                    } else {
92                        self.mapping_varies_by_dest_ipv4 = Some(true);
93                        warn!("IPv4 address detected by QAD varies by destination");
94                    }
95                } else {
96                    self.global_v4 = Some(ipp);
97                }
98            }
99            #[cfg(not(wasm_browser))]
100            ProbeReport::QadIpv6(report) => {
101                self.relay_latency.update_relay(
102                    report.node.clone(),
103                    report.latency,
104                    Probe::QadIpv6,
105                );
106                let SocketAddr::V6(ipp) = report.addr else {
107                    warn!("received IPv4 address from IPv6 QAD: {}", report.addr);
108                    return;
109                };
110
111                self.udp_v6 = true;
112                tracing::debug!(?self.global_v6, ?self.mapping_varies_by_dest_ipv6, %ipp,"got");
113                if let Some(global) = self.global_v6 {
114                    if global == ipp {
115                        if self.mapping_varies_by_dest_ipv6.is_none() {
116                            self.mapping_varies_by_dest_ipv6 = Some(false);
117                        }
118                    } else {
119                        self.mapping_varies_by_dest_ipv6 = Some(true);
120                        warn!("IPv6 address detected by QAD varies by destination");
121                    }
122                } else {
123                    self.global_v6 = Some(ipp);
124                }
125            }
126        }
127    }
128}

130/// Latencies per relay node.
131#[derive(Debug, Default, PartialEq, Eq, Clone, Serialize, Deserialize)]
132pub struct RelayLatencies {
133    #[cfg(not(wasm_browser))]
134    ipv4: BTreeMap<RelayUrl, Duration>,
135    #[cfg(not(wasm_browser))]
136    ipv6: BTreeMap<RelayUrl, Duration>,
137    https: BTreeMap<RelayUrl, Duration>,
138}
139
140impl RelayLatencies {
141    /// Updates a relay's latency, if it is faster than before.
142    pub(super) fn update_relay(&mut self, url: RelayUrl, latency: Duration, probe: Probe) {
143        let list = match probe {
144            Probe::Https => &mut self.https,
145            #[cfg(not(wasm_browser))]
146            Probe::QadIpv4 => &mut self.ipv4,
147            #[cfg(not(wasm_browser))]
148            Probe::QadIpv6 => &mut self.ipv6,
149        };
150        let old_latency = list.entry(url).or_insert(latency);
151        if latency < *old_latency {
152            *old_latency = latency;
153        }
154    }
155
156    /// Merges another [`RelayLatencies`] into this one.
157    ///
158    /// For each relay the latency is updated using [`RelayLatencies::update_relay`].
159    pub(super) fn merge(&mut self, other: &RelayLatencies) {
160        for (url, latency) in other.https.iter() {
161            self.update_relay(url.clone(), *latency, Probe::Https);
162        }
163        #[cfg(not(wasm_browser))]
164        for (url, latency) in other.ipv4.iter() {
165            self.update_relay(url.clone(), *latency, Probe::QadIpv4);
166        }
167        #[cfg(not(wasm_browser))]
168        for (url, latency) in other.ipv6.iter() {
169            self.update_relay(url.clone(), *latency, Probe::QadIpv6);
170        }
171    }
172
173    /// Returns an iterator over all the relays and their latencies.
174    #[cfg(not(wasm_browser))]
175    pub fn iter(&self) -> impl Iterator<Item = (&'_ RelayUrl, Duration)> + '_ {
176        self.https
177            .iter()
178            .chain(self.ipv4.iter())
179            .chain(self.ipv6.iter())
180            .map(|(k, v)| (k, *v))
181    }
182
183    /// Returns an iterator over all the relays and their latencies.
184    #[cfg(wasm_browser)]
185    pub fn iter(&self) -> impl Iterator<Item = (&'_ RelayUrl, Duration)> + '_ {
186        self.https.iter().map(|(k, v)| (k, *v))
187    }
188
189    #[cfg(not(wasm_browser))]
190    pub(super) fn is_empty(&self) -> bool {
191        self.https.is_empty() && self.ipv4.is_empty() && self.ipv6.is_empty()
192    }
193
194    #[cfg(wasm_browser)]
195    pub(super) fn is_empty(&self) -> bool {
196        self.https.is_empty()
197    }
198
199    /// Returns the lowest latency across records.
200    pub(super) fn get(&self, url: &RelayUrl) -> Option<Duration> {
201        let mut list = Vec::with_capacity(3);
202        if let Some(val) = self.https.get(url) {
203            list.push(*val);
204        }
205        #[cfg(not(wasm_browser))]
206        if let Some(val) = self.ipv4.get(url) {
207            list.push(*val);
208        }
209        #[cfg(not(wasm_browser))]
210        if let Some(val) = self.ipv6.get(url) {
211            list.push(*val);
212        }
213        list.into_iter().min()
214    }
