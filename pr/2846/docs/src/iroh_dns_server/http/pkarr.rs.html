12pub async fn put(
13    State(state): State<AppState>,
14    Path(key): Path<String>,
15    body: Bytes,
16) -> Result<impl IntoResponse, AppError> {
17    let key = pkarr::PublicKey::try_from(key.as_str())
18        .map_err(|e| AppError::new(StatusCode::BAD_REQUEST, Some(format!("invalid key: {e}"))))?;
19    let label = &key.to_z32()[..10];
20    let signed_packet = pkarr::SignedPacket::from_relay_payload(&key, &body).map_err(|e| {
21        AppError::new(
22            StatusCode::BAD_REQUEST,
23            Some(format!("invalid body payload: {e}")),
24        )
25    })?;
26
27    let updated = state
28        .store
29        .insert(signed_packet, PacketSource::PkarrPublish)
30        .await?;
31    info!(key = %label, ?updated, "pkarr upsert");
32    Ok(StatusCode::NO_CONTENT)
33}

35pub async fn get(
36    State(state): State<AppState>,
37    Path(pubkey): Path<String>,
38) -> Result<impl IntoResponse, AppError> {
39    let pubkey = PublicKeyBytes::from_z32(&pubkey)
40        .map_err(|e| AppError::new(StatusCode::BAD_REQUEST, Some(format!("invalid key: {e}"))))?;
41    let signed_packet = state
42        .store
43        .get_signed_packet(&pubkey)
44        .await?
45        .ok_or_else(|| AppError::with_status(StatusCode::NOT_FOUND))?;
46    let body = signed_packet.to_relay_payload();
47    let headers = [(header::CONTENT_TYPE, "application/x-pkarr-signed-packet")];
48    Ok((headers, body))
49}
