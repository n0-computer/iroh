27/// A static endpoint discovery to manually add endpoint addressing information.
28///
29/// Often an application might get endpoint addressing information out-of-band in an
30/// application-specific way.  [`EndpointTicket`]'s are one common way used to achieve this.
31/// This "static" addressing information is often only usable for a limited time so needs to
32/// be able to be removed again once know it is no longer useful.
33///
34/// This is where the [`StaticProvider`] is useful: it allows applications to add and
35/// retract endpoint addressing information that is otherwise out-of-band to iroh.
36///
37/// # Examples
38///
39/// ```rust
40/// use iroh::{Endpoint, EndpointAddr, discovery::static_provider::StaticProvider};
41/// use iroh_base::SecretKey;
42///
43/// # #[tokio::main]
44/// # async fn main() -> n0_snafu::Result<()> {
45/// // Create the discovery service and endpoint.
46/// let discovery = StaticProvider::new();
47///
48/// let _ep = Endpoint::builder()
49///     .discovery(discovery.clone())
50///     .bind()
51///     .await?;
52///
53/// // Sometime later add a RelayUrl for a fake EndpointId.
54/// let endpoint_id = SecretKey::from_bytes(&[0u8; 32]).public(); // Do not use fake secret keys!
55/// // You can pass either `EndpointInfo` or `EndpointAddr` to `add_endpoint_info`.
56/// discovery.add_endpoint_info(EndpointAddr {
57///     endpoint_id,
58///     relay_url: Some("https://example.com".parse()?),
59///     direct_addresses: Default::default(),
60/// });
61///
62/// # Ok(())
63/// # }
64/// ```
65///
66/// [`EndpointTicket`]: https://docs.rs/iroh-base/latest/iroh_base/ticket/struct.EndpointTicket
67#[derive(Debug, Clone)]
68pub struct StaticProvider {
69    endpoints: Arc<RwLock<BTreeMap<EndpointId, StoredEndpointInfo>>>,
70    provenance: &'static str,
71}
72
73impl Default for StaticProvider {
74    fn default() -> Self {
75        Self {
76            endpoints: Default::default(),
77            provenance: Self::PROVENANCE,
78        }
79    }
80}
81
82#[derive(Debug)]
83struct StoredEndpointInfo {
84    data: EndpointData,
85    last_updated: SystemTime,
86}
87
88impl StaticProvider {
89    /// The provenance string for this discovery implementation.
90    ///
91    /// This is mostly used for debugging information and allows understanding the origin of
92    /// addressing information used by an iroh [`Endpoint`].
93    ///
94    /// [`Endpoint`]: crate::Endpoint
95    pub const PROVENANCE: &'static str = "static_discovery";
96
97    /// Creates a new static discovery instance.
98    pub fn new() -> Self {
99        Self::default()
100    }
101
102    /// Creates a new static discovery instance with the provided `provenance`.
103    ///
104    /// The provenance is part of [`DiscoveryItem`]s returned from [`Self::resolve`].
105    /// It is mostly used for debugging information and allows understanding the origin of
106    /// addressing information used by an iroh [`Endpoint`].
107    ///
108    /// [`Endpoint`]: crate::Endpoint
109    pub fn with_provenance(provenance: &'static str) -> Self {
110        Self {
111            endpoints: Default::default(),
112            provenance,
113        }
114    }
115
116    /// Creates a static discovery instance from endpoint addresses.
117    ///
118    /// # Examples
119    ///
120    /// ```rust
121    /// use std::{net::SocketAddr, str::FromStr};
122    ///
123    /// use iroh::{Endpoint, EndpointAddr, discovery::static_provider::StaticProvider};
124    ///
125    /// # fn get_addrs() -> Vec<EndpointAddr> {
126    /// #     Vec::new()
127    /// # }
128    /// # #[tokio::main]
129    /// # async fn main() -> n0_snafu::Result<()> {
130    /// // get addrs from somewhere
131    /// let addrs = get_addrs();
132    ///
133    /// // create a StaticProvider from the list of addrs.
134    /// let discovery = StaticProvider::from_endpoint_info(addrs);
135    /// // create an endpoint with the discovery
136    /// let endpoint = Endpoint::builder().discovery(discovery).bind().await?;
137    /// # Ok(())
138    /// # }
139    /// ```
140    pub fn from_endpoint_info(infos: impl IntoIterator<Item = impl Into<EndpointInfo>>) -> Self {
141        let res = Self::default();
142        for info in infos {
143            res.add_endpoint_info(info);
144        }
145        res
146    }
147
148    /// Sets endpoint addressing information for the given endpoint ID.
149    ///
150    /// This will completely overwrite any existing info for the endpoint.
151    ///
152    /// Returns the [`EndpointData`] of the previous entry, or `None` if there was no previous
153    /// entry for this endpoint ID.
154    pub fn set_endpoint_info(
155        &self,
156        endpoint_info: impl Into<EndpointInfo>,
157    ) -> Option<EndpointData> {
158        let last_updated = SystemTime::now();
159        let EndpointInfo { endpoint_id, data } = endpoint_info.into();
160        let mut guard = self.endpoints.write().expect("poisoned");
161        let previous = guard.insert(endpoint_id, StoredEndpointInfo { data, last_updated });
162        previous.map(|x| x.data)
163    }
164
165    /// Augments endpoint addressing information for the given endpoint ID.
166    ///
167    /// The provided addressing information is combined with the existing info in the static
168    /// provider.  Any new direct addresses are added to those already present while the
169    /// relay URL is overwritten.
170    pub fn add_endpoint_info(&self, endpoint_info: impl Into<EndpointInfo>) {
171        let last_updated = SystemTime::now();
172        let EndpointInfo { endpoint_id, data } = endpoint_info.into();
173        let mut guard = self.endpoints.write().expect("poisoned");
174        match guard.entry(endpoint_id) {
175            Entry::Occupied(mut entry) => {
176                let existing = entry.get_mut();
177                existing
178                    .data
179                    .add_direct_addresses(data.direct_addresses().iter().copied());
180                existing.data.set_relay_url(data.relay_url().cloned());
181                existing.data.set_user_data(data.user_data().cloned());
182                existing.last_updated = last_updated;
183            }
184            Entry::Vacant(entry) => {
185                entry.insert(StoredEndpointInfo { data, last_updated });
186            }
187        }
188    }
189
190    /// Returns endpoint addressing information for the given endpoint ID.
191    pub fn get_endpoint_info(&self, endpoint_id: EndpointId) -> Option<EndpointInfo> {
192        let guard = self.endpoints.read().expect("poisoned");
193        let info = guard.get(&endpoint_id)?;
194        Some(EndpointInfo::from_parts(endpoint_id, info.data.clone()))
195    }
196
197    /// Removes all endpoint addressing information for the given endpoint ID.
198    ///
199    /// Any removed information is returned.
200    pub fn remove_endpoint_info(&self, endpoint_id: EndpointId) -> Option<EndpointInfo> {
201        let mut guard = self.endpoints.write().expect("poisoned");
202        let info = guard.remove(&endpoint_id)?;
203        Some(EndpointInfo::from_parts(endpoint_id, info.data))
204    }
205}
206
207impl Discovery for StaticProvider {
208    fn publish(&self, _data: &EndpointData) {}
209
210    fn resolve(
211        &self,
212        endpoint_id: EndpointId,
213    ) -> Option<BoxStream<Result<super::DiscoveryItem, DiscoveryError>>> {
214        let guard = self.endpoints.read().expect("poisoned");
215        let info = guard.get(&endpoint_id);
216        match info {
217            Some(endpoint_info) => {
218                let last_updated = endpoint_info
219                    .last_updated
220                    .duration_since(SystemTime::UNIX_EPOCH)
221                    .expect("time drift")
222                    .as_micros() as u64;
223                let item = DiscoveryItem::new(
224                    EndpointInfo::from_parts(endpoint_id, endpoint_info.data.clone()),
225                    self.provenance,
226                    Some(last_updated),
227                );
228                Some(stream::iter(Some(Ok(item))).boxed())
229            }
230            None => None,
231        }
232    }
233}

235#[cfg(test)]
236mod tests {
237    use iroh_base::{EndpointAddr, SecretKey};
238    use n0_snafu::{Result, ResultExt};
239
240    use super::*;
241    use crate::Endpoint;
242
243    #[tokio::test]
244    async fn test_basic() -> Result {
245        let discovery = StaticProvider::new();
246
247        let _ep = Endpoint::empty_builder()
248            .discovery(discovery.clone())
249            .bind()
250            .await?;
251
252        let key = SecretKey::from_bytes(&[0u8; 32]);
253        let addr = EndpointAddr {
254            endpoint_id: key.public(),
255            relay_url: Some("https://example.com".parse()?),
256            direct_addresses: Default::default(),
257        };
258        let user_data = Some("foobar".parse().unwrap());
259        let endpoint_info = EndpointInfo::from(addr.clone()).with_user_data(user_data.clone());
260        discovery.add_endpoint_info(endpoint_info.clone());
261
262        let back = discovery
263            .get_endpoint_info(key.public())
264            .context("no addr")?;
265
266        assert_eq!(back, endpoint_info);
267        assert_eq!(back.user_data(), user_data.as_ref());
268        assert_eq!(back.into_endpoint_addr(), addr);
269
270        let removed = discovery
271            .remove_endpoint_info(key.public())
272            .context("nothing removed")?;
273        assert_eq!(removed, endpoint_info);
274        let res = discovery.get_endpoint_info(key.public());
275        assert!(res.is_none());
276
277        Ok(())
278    }
279
280    #[tokio::test]
281    async fn test_provenance() -> Result {
282        let discovery = StaticProvider::with_provenance("foo");
283        let key = SecretKey::from_bytes(&[0u8; 32]);
284        let addr = EndpointAddr {
285            endpoint_id: key.public(),
286            relay_url: Some("https://example.com".parse()?),
287            direct_addresses: Default::default(),
288        };
289        discovery.add_endpoint_info(addr);
290        let mut stream = discovery.resolve(key.public()).unwrap();
291        let item = stream.next().await.unwrap()?;
292        assert_eq!(item.provenance(), "foo");
293        assert_eq!(item.relay_url(), Some(&("https://example.com".parse()?)));
294
295        Ok(())
296    }
