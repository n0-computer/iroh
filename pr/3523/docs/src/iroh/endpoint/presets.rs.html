1//! Presets allow configuring an endpoint quickly with a chosen set of defaults.
2//!
3//! # Example
4//!
5//! ```no_run
6//! # async fn main() -> n0_snafu::Result {
7//! use iroh::{Endpoint, Watcher, endpoint::presets};
8//!
9//! let endpoint = Endpoint::bind_preset(presets::N0).await?;
10//! # let _ = endpoint;
11//! # Ok(())
12//! # }
13//! ```
14use super::Builder;
15use crate::discovery::pkarr::PkarrPublisher;
16
17/// Defines a preset
18pub trait Preset {
19    /// Applies the configuration to the passed in [`Builder`].
20    fn apply(self, builder: Builder) -> Builder;
21}
22
23/// Configures the endpoint to use the n0 defaults
24///
25/// Currently this consists of the DNS discovery service.
26///
27/// The default discovery service publishes to and resolves from the
28/// n0.computer dns server `iroh.link`.
29///
30/// This is equivalent to adding both a [`crate::discovery::pkarr::PkarrPublisher`]
31/// and a [`crate::discovery::dns::DnsDiscovery`], both configured to use the
32/// n0.computer dns server.
33///
34/// This will by default use [`N0_DNS_PKARR_RELAY_PROD`].
35/// When in tests, or when the `test-utils` feature is enabled, this will use the
36/// [`N0_DNS_PKARR_RELAY_STAGING`].
37///
38/// [`N0_DNS_PKARR_RELAY_PROD`]: crate::discovery::pkarr::N0_DNS_PKARR_RELAY_PROD
39/// [`N0_DNS_PKARR_RELAY_STAGING`]: crate::discovery::pkarr::N0_DNS_PKARR_RELAY_STAGING
40#[derive(Debug, Copy, Clone, Default)]
41pub struct N0;
42
43impl Preset for N0 {
44    fn apply(self, mut builder: Builder) -> Builder {
45        builder = builder.add_discovery(PkarrPublisher::n0_dns());
46        // Resolve using HTTPS requests to our DNS server's /pkarr path in browsers
47        #[cfg(wasm_browser)]
48        {
49            use crate::discovery::pkarr::PkarrResolver;
50
51            builder = self.add_discovery(PkarrResolver::n0_dns());
52        }
53        // Resolve using DNS queries outside browsers.
54        #[cfg(not(wasm_browser))]
55        {
56            use crate::discovery::dns::DnsDiscovery;
57
58            builder = builder.add_discovery(DnsDiscovery::n0_dns());
59        }
60        builder
61    }
62}
