1//! This module exports [`looks_like_disco_wrapper`] as the only disco-related relay
2//! functionality.
3//!
4//! Despite the relay not being able to read disco messages by design, it does attempt to
5//! identify this traffic to ensure hole-punching messages are not lost do to congestion.
6
7/// The 6 byte header of all discovery messages.
8pub const MAGIC: &str = "TSðŸ’¬"; // 6 bytes: 0x54 53 f0 9f 92 ac
9pub(crate) const MAGIC_LEN: usize = MAGIC.len();
10pub(crate) const KEY_LEN: usize = 32;
11
12const MESSAGE_HEADER_LEN: usize = MAGIC_LEN + KEY_LEN;
13
14/// Reports whether p looks like it's a packet containing an encrypted disco message.
15pub fn looks_like_disco_wrapper(p: &[u8]) -> bool {
16    if p.len() < MESSAGE_HEADER_LEN {
17        return false;
18    }
19
20    &p[..MAGIC_LEN] == MAGIC.as_bytes()
