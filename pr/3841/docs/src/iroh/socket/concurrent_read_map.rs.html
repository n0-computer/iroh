1//! This module implements a map that can be modified from only one task but read from many others.
2//!
3//! The combination of only being able to write using `&mut self` in [`ConcurrentReadMap`] and
4//! only allowing reading these from other places via read-only [`ReadOnlyMap`]s with `&self` is
5//! what ensures that this map avoids race conditions from multiple racing writers.
6
7use std::{hash::Hash, sync::Arc};
8
9use rustc_hash::FxBuildHasher;
10
11/// A map that can be modified from only one task but read from many others.
12///
13/// Only the single `&mut self`-based "leader" which owns this map can modify it, but cheaply
14/// clonable "followers" can be created with [`ConcurrentReadMap::read_only`].
15/// These clones will update with recent writes from the leader.
16#[derive(Debug)]
17pub(crate) struct ConcurrentReadMap<K: Hash + Eq, V>(Arc<papaya::HashMap<K, V, FxBuildHasher>>);
18
19impl<K: Hash + Eq, V> Default for ConcurrentReadMap<K, V> {
20    fn default() -> Self {
21        Self(Default::default())
22    }
23}
24
25impl<K: Hash + Eq, V> ConcurrentReadMap<K, V> {
26    pub(crate) fn get_or_insert_with<F>(&mut self, key: K, f: F) -> V
27    where
28        F: FnOnce() -> V,
29        V: Clone,
30    {
31        self.0.get_or_insert_with(key, f, &self.0.guard()).clone()
32    }
33
34    pub(crate) fn insert(&mut self, key: K, value: V) {
35        self.0.insert(key, value, &self.0.guard());
36    }
37
38    pub(crate) fn remove(&mut self, key: &K) {
39        self.0.remove(key, &self.0.guard());
40    }
41
42    pub(crate) fn get(&self, key: &K) -> Option<V>
43    where
44        V: Clone,
45    {
46        self.0.get(key, &self.0.guard()).cloned()
47    }
48
49    pub(crate) fn read_only(&self) -> ReadOnlyMap<K, V> {
50        ReadOnlyMap(self.0.clone())
51    }
52}
53
54/// A read replica for a [`ConcurrentReadMap`].
55///
56/// These can be cloned cheaply and will read recent writes from the original map.
57#[derive(Clone, Debug)]
58pub(crate) struct ReadOnlyMap<K: Hash + Eq, V>(Arc<papaya::HashMap<K, V, FxBuildHasher>>);
59
60impl<K: Hash + Eq, V> ReadOnlyMap<K, V> {
61    pub(crate) fn get(&self, key: &K) -> Option<V>
62    where
63        V: Clone,
64    {
65        self.0.get(key, &self.0.guard()).cloned()
66    }
67
68    pub(crate) fn guard(&self) -> papaya::LocalGuard<'_> {
69        self.0.guard()
70    }
71
72    pub(crate) fn values<'g, G: papaya::Guard>(&self, guard: &'g G) -> papaya::Values<'g, K, V, G> {
73        self.0.values(guard)
74    }
75}
