79/// The maximum latency of all nodes, if none are found yet.
80///
81/// Normally the max latency of all nodes is computed, but if we don't yet know any nodes
82/// latencies we return this as default.  This is the value of the initial STUN probe
83/// delays.  It is only used as time to wait for further latencies to arrive, which *should*
84/// never happen unless there already is at least one latency.  Yet here we are, defining a
85/// default which will never be used.
86const DEFAULT_MAX_LATENCY: Duration = Duration::from_millis(100);

144/// Latencies per relay node.
145#[derive(Debug, Default, PartialEq, Eq, Clone)]
146pub struct RelayLatencies(BTreeMap<RelayUrl, Duration>);
147
148impl RelayLatencies {
149    fn new() -> Self {
150        Default::default()
151    }
152
153    /// Updates a relay's latency, if it is faster than before.
154    fn update_relay(&mut self, url: RelayUrl, latency: Duration) {
155        let val = self.0.entry(url).or_insert(latency);
156        if latency < *val {
157            *val = latency;
158        }
159    }
160
161    /// Merges another [`RelayLatencies`] into this one.
162    ///
163    /// For each relay the latency is updated using [`RelayLatencies::update_relay`].
164    fn merge(&mut self, other: &RelayLatencies) {
165        for (url, latency) in other.iter() {
166            self.update_relay(url.clone(), latency);
167        }
168    }
169
170    /// Returns the maximum latency for all relays.
171    ///
172    /// If there are not yet any latencies this will return [`DEFAULT_MAX_LATENCY`].
173    fn max_latency(&self) -> Duration {
174        self.0
175            .values()
176            .max()
177            .copied()
178            .unwrap_or(DEFAULT_MAX_LATENCY)
179    }
180
181    /// Returns an iterator over all the relays and their latencies.
182    pub fn iter(&self) -> impl Iterator<Item = (&'_ RelayUrl, Duration)> + '_ {
183        self.0.iter().map(|(k, v)| (k, *v))
184    }
185
186    fn len(&self) -> usize {
187        self.0.len()
188    }
189
190    fn is_empty(&self) -> bool {
191        self.0.is_empty()
192    }
193
194    fn get(&self, url: &RelayUrl) -> Option<Duration> {
195        self.0.get(url).copied()
196    }
197}

199/// Client to run net_reports.
200///
201/// Creating this creates a net_report actor which runs in the background.  Most of the time
202/// it is idle unless [`Client::get_report`] is called, which is the main interface.
203///
204/// The [`Client`] struct can be cloned and results multiple handles to the running actor.
205/// If all [`Client`]s are dropped the actor stops running.
206///
207/// While running the net_report actor expects to be passed all received stun packets using
208/// `Addr::receive_stun_packet`.
209#[derive(Debug)]
210pub struct Client {
211    /// Channel to send message to the [`Actor`].
212    ///
213    /// If all senders are dropped, in other words all clones of this struct are dropped,
214    /// the actor will terminate.
215    addr: Addr,
216    /// Ensures the actor is terminated when the client is dropped.
217    _drop_guard: Arc<AbortOnDropHandle<()>>,
218}

220#[derive(Debug)]
221struct Reports {
222    /// Do a full relay scan, even if last is `Some`.
223    next_full: bool,
224    /// Some previous reports.
225    prev: HashMap<Instant, Arc<Report>>,
226    /// Most recent report.
227    last: Option<Arc<Report>>,
228    /// Time of last full (non-incremental) report.
229    last_full: Instant,
230}
231
232impl Default for Reports {
233    fn default() -> Self {
234        Self {
235            next_full: Default::default(),
236            prev: Default::default(),
237            last: Default::default(),
238            last_full: Instant::now(),
239        }
240    }
241}

243impl Client {
244    /// Creates a new net_report client.
245    ///
246    /// This starts a connected actor in the background.  Once the client is dropped it will
247    /// stop running.
248    pub fn new(
249        #[cfg(not(wasm_browser))] port_mapper: Option<portmapper::Client>,
250        #[cfg(not(wasm_browser))] dns_resolver: DnsResolver,
251        #[cfg(not(wasm_browser))] ip_mapped_addrs: Option<IpMappedAddresses>,
252        metrics: Arc<Metrics>,
253    ) -> Self {
254        let mut actor = Actor::new(
255            #[cfg(not(wasm_browser))]
256            port_mapper,
257            #[cfg(not(wasm_browser))]
258            dns_resolver,
259            #[cfg(not(wasm_browser))]
260            ip_mapped_addrs,
261            metrics,
262        );
263        let addr = actor.addr();
264        let task = task::spawn(
265            async move { actor.run().await }.instrument(info_span!("net_report.actor")),
266        );
267        let drop_guard = AbortOnDropHandle::new(task);
268        Client {
269            addr,
270            _drop_guard: Arc::new(drop_guard),
271        }
272    }
273
274    /// Returns a new address to send messages to this actor.
275    ///
276    /// Unlike the client itself the returned [`Addr`] does not own the actor task, it only
277    /// allows sending messages to the actor.
278    pub fn addr(&self) -> Addr {
279        self.addr.clone()
280    }

282    /// Runs a net_report, returning the report.
283    ///
284    /// It may not be called concurrently with itself, `&mut self` takes care of that.
285    ///
286    /// The *stun_conn4* and *stun_conn6* endpoints are bound UDP sockets to use to send out
287    /// STUN packets.  This function **will not read from the sockets**, as they may be
288    /// receiving other traffic as well, normally they are the sockets carrying the real
289    /// traffic. Thus all stun packets received on those sockets should be passed to
290    /// `Addr::receive_stun_packet` in order for this function to receive the stun
291    /// responses and function correctly.
292    ///
293    /// If these are not passed in this will bind sockets for STUN itself, though results
294    /// may not be as reliable.
295    ///
296    /// The *quic_config* takes a [`QuicConfig`], a combination of a QUIC endpoint and
297    /// a client configuration that can be use for verifying the relay server connection.
298    /// When available, the report will attempt to get an observed public address
299    /// using QUIC address discovery.
300    ///
301    /// When `None`, it will disable the QUIC address discovery probes.
302    ///
303    /// This will attempt to use *all* probe protocols.
304    #[cfg(test)]
305    pub async fn get_report_all(
306        &mut self,
307        relay_map: RelayMap,
308        #[cfg(not(wasm_browser))] stun_sock_v4: Option<Arc<UdpSocket>>,
309        #[cfg(not(wasm_browser))] stun_sock_v6: Option<Arc<UdpSocket>>,
310        #[cfg(not(wasm_browser))] quic_config: Option<QuicConfig>,
311    ) -> Result<Arc<Report>, ReportError> {
312        #[cfg(not(wasm_browser))]
313        let opts = Options::default()
314            .stun_v4(stun_sock_v4)
315            .stun_v6(stun_sock_v6)
316            .quic_config(quic_config);
317        #[cfg(wasm_browser)]
318        let opts = Options::default();
319
320        let rx = self.get_report_channel(relay_map, opts).await?;
321        match rx.await {
322            Ok(res) => res,
323            Err(_) => Err(ActorGoneSnafu.build()),
324        }
325    }
326
327    /// Runs a net_report, returning the report.
328    ///
329    /// It may not be called concurrently with itself, `&mut self` takes care of that.
330    ///
331    /// Look at [`Options`] for the different configuration options.
332    pub async fn get_report(
333        &mut self,
334        relay_map: RelayMap,
335        opts: Options,
336    ) -> Result<Arc<Report>, ReportError> {
337        let rx = self.get_report_channel(relay_map, opts).await?;
338        match rx.await {
339            Ok(res) => res,
340            Err(_) => Err(ActorGoneSnafu.build()),
341        }
342    }
343
344    /// Get report with channel
345    ///
346    /// Look at [`Options`] for the different configuration options.
347    pub(crate) async fn get_report_channel(
348        &mut self,
349        relay_map: RelayMap,
350        opts: Options,
351    ) -> Result<oneshot::Receiver<Result<Arc<Report>, ReportError>>, ReportError> {
352        let (tx, rx) = oneshot::channel();
353        self.addr
354            .send(Message::RunCheck {
355                relay_map,
356                opts,
357                response_tx: tx,
358            })
359            .await
360            .map_err(|_| ActorGoneSnafu.build())?;
361        Ok(rx)
362    }
363}

368    txn: stun::TransactionId,
369    /// The time the STUN probe was sent.

375/// Messages to send to the [`Actor`].
376#[derive(Debug)]
377#[allow(clippy::large_enum_variant)]
378pub(crate) enum Message {
379    /// Run a net_report.
380    ///
381    /// Only one net_report can be run at a time, trying to run multiple concurrently will
382    /// fail.
383    RunCheck {
384        /// The map of relays we want to probe
385        relay_map: RelayMap,
386        /// Options for the report
387        opts: Options,
388        /// Channel to receive the response.
389        response_tx: oneshot::Sender<Result<Arc<Report>, ReportError>>,
390    },
391    /// A report produced by the [`reportgen`] actor.
392    ReportReady { report: Box<Report> },
393    /// The [`reportgen`] actor failed to produce a report.
394    ReportAborted { reason: ActorRunError },
395    /// An incoming STUN packet to parse.
396    StunPacket {
397        /// The raw UDP payload.
398        payload: Bytes,
399        /// The address this was claimed to be received from.
400        from_addr: SocketAddr,
401    },
402    /// A probe wants to register an in-flight STUN request.
403    ///
404    /// The sender is signalled once the STUN packet is registered with the actor and will
405    /// correctly accept the STUN response.
406    InFlightStun(Inflight, oneshot::Sender<()>),
407}

409/// Sender to the main service.
410///
411/// Unlike [`Client`] this is the raw channel to send messages over.  Keeping this alive
412/// will not keep the actor alive, which makes this handy to pass to internal tasks.
413#[derive(Debug, Clone)]
414pub struct Addr {
415    sender: mpsc::Sender<Message>,
416    metrics: Arc<Metrics>,
417}
418
419impl Addr {
420    /// Pass a received STUN packet to the net_reporter.
421    ///
422    /// Normally the UDP sockets to send STUN messages from are passed in so that STUN
423    /// packets are sent from the sockets that carry the real traffic.  However because
424    /// these sockets carry real traffic they will also receive non-STUN traffic, thus the
425    /// net_report actor does not read from the sockets directly.  If you receive a STUN
426    /// packet on the socket you should pass it to this method.
427    ///
428    /// It is safe to call this even when the net_report actor does not currently have any
429    /// in-flight STUN probes.  The actor will simply ignore any stray STUN packets.
430    ///
431    /// There is an implicit queue here which may drop packets if the actor does not keep up
432    /// consuming them.
433    pub fn receive_stun_packet(&self, payload: Bytes, src: SocketAddr) {
434        if let Err(mpsc::error::TrySendError::Full(_)) = self.sender.try_send(Message::StunPacket {
435            payload,
436            from_addr: src,
437        }) {
438            self.metrics.stun_packets_dropped.inc();
439            warn!("dropping stun packet from {}", src);
440        }
441    }
442
443    async fn send(&self, msg: Message) -> Result<(), mpsc::error::SendError<Message>> {
444        self.sender.send(msg).await.inspect_err(|_| {
445            error!("net_report actor lost");
446        })
447    }
448}

450/// The net_report actor.
451///
452/// This actor runs for the entire duration there's a [`Client`] connected.
453#[derive(Debug)]
454struct Actor {
455    // Actor plumbing.
456    /// Actor messages channel.
457    ///
458    /// If there are no more senders the actor stops.
459    receiver: mpsc::Receiver<Message>,
460    /// The sender side of the messages channel.
461    ///
462    /// This allows creating new [`Addr`]s from the actor.
463    sender: mpsc::Sender<Message>,
464    /// A collection of previously generated reports.
465    ///
466    /// Sometimes it is useful to look at past reports to decide what to do.
467    reports: Reports,
468
469    // Actor configuration.
470    /// The port mapper client, if those are requested.
471    ///
472    /// The port mapper is responsible for talking to routers via UPnP and the like to try
473    /// and open ports.
474    #[cfg(not(wasm_browser))]
475    port_mapper: Option<portmapper::Client>,
476
477    // Actor state.
478    /// Information about the currently in-flight STUN requests.
479    ///
480    /// This is used to complete the STUN probe when receiving STUN packets.
481    in_flight_stun_requests: HashMap<stun::TransactionId, Inflight>,
482    /// The [`reportgen`] actor currently generating a report.
483    current_report_run: Option<ReportRun>,
484
485    /// The DNS resolver to use for probes that need to perform DNS lookups
486    #[cfg(not(wasm_browser))]
487    dns_resolver: DnsResolver,
488
489    /// The [`IpMappedAddresses`] that allows you to do QAD in iroh
490    #[cfg(not(wasm_browser))]
491    ip_mapped_addrs: Option<IpMappedAddresses>,
492    metrics: Arc<Metrics>,
493}

495impl Actor {
496    /// Creates a new actor.
497    ///
498    /// This does not start the actor, see [`Actor::run`] for this.  You should not
499    /// normally create this directly but rather create a [`Client`].
500    fn new(
501        #[cfg(not(wasm_browser))] port_mapper: Option<portmapper::Client>,
502        #[cfg(not(wasm_browser))] dns_resolver: DnsResolver,
503        #[cfg(not(wasm_browser))] ip_mapped_addrs: Option<IpMappedAddresses>,
504        metrics: Arc<Metrics>,
505    ) -> Self {
506        // TODO: consider an instrumented flume channel so we have metrics.
507        let (sender, receiver) = mpsc::channel(32);
508        Self {
509            receiver,
510            sender,
511            reports: Default::default(),
512            #[cfg(not(wasm_browser))]
513            port_mapper,
514            in_flight_stun_requests: Default::default(),
515            current_report_run: None,
516            #[cfg(not(wasm_browser))]
517            dns_resolver,
518            #[cfg(not(wasm_browser))]
519            ip_mapped_addrs,
520            metrics,
521        }
522    }
523
524    /// Returns the channel to send messages to the actor.
525    fn addr(&self) -> Addr {
526        Addr {
527            sender: self.sender.clone(),
528            metrics: self.metrics.clone(),
529        }
530    }
531
532    /// Run the actor.
533    ///
534    /// It will now run and handle messages.  Once the connected [`Client`] (including all
535    /// its clones) is dropped this will terminate.
536    async fn run(&mut self) {
537        debug!("net_report actor starting");
538        while let Some(msg) = self.receiver.recv().await {
539            trace!(?msg, "handling message");
540            match msg {
541                Message::RunCheck {
542                    relay_map,
543                    opts,
544                    response_tx,
545                } => {
546                    self.handle_run_check(relay_map, opts, response_tx);
547                }
548                Message::ReportReady { report } => {
549                    self.handle_report_ready(*report);
550                }
551                Message::ReportAborted { reason: err } => {
552                    self.handle_report_aborted(err);
553                }
554                Message::StunPacket { payload, from_addr } => {
555                    self.handle_stun_packet(&payload, from_addr);
556                }
557                Message::InFlightStun(inflight, response_tx) => {
558                    self.handle_in_flight_stun(inflight, response_tx);
559                }
560            }
561        }
562    }
563
564    /// Starts a check run as requested by the [`Message::RunCheck`] message.
565    ///
566    /// If *stun_sock_v4* or *stun_sock_v6* are not provided this will bind the sockets
567    /// itself.  This is not ideal since really you want to send STUN probes from the
568    /// sockets you will be using.
569    fn handle_run_check(
570        &mut self,
571        relay_map: RelayMap,
572        opts: Options,
573        response_tx: oneshot::Sender<Result<Arc<Report>, ReportError>>,
574    ) {
575        let protocols = opts.to_protocols();
576        #[cfg(not(wasm_browser))]
577        let socket_state = SocketState {
578            port_mapper: self.port_mapper.clone(),
579            stun_sock4: opts.stun_sock_v4,
580            stun_sock6: opts.stun_sock_v6,
581            quic_config: opts.quic_config,
582            dns_resolver: self.dns_resolver.clone(),
583            ip_mapped_addrs: self.ip_mapped_addrs.clone(),
584        };
585        trace!("Attempting probes for protocols {protocols:#?}");
586        if self.current_report_run.is_some() {
587            response_tx.send(Err(AlreadyRunningSnafu.build())).ok();
588            return;
589        }
590
591        let now = Instant::now();
592
593        let mut do_full = self.reports.next_full
594            || now.duration_since(self.reports.last_full) > FULL_REPORT_INTERVAL;
595
596        // If the last report had a captive portal and reported no UDP access,
597        // it's possible that we didn't get a useful net_report due to the
598        // captive portal blocking us. If so, make this report a full (non-incremental) one.
599        if !do_full {
600            if let Some(ref last) = self.reports.last {
601                do_full = !last.udp && last.captive_portal.unwrap_or_default();
602            }
603        }
604        if do_full {
605            self.reports.last = None; // causes ProbePlan::new below to do a full (initial) plan
606            self.reports.next_full = false;
607            self.reports.last_full = now;
608            self.metrics.reports_full.inc();
609        }
610        self.metrics.reports.inc();
611
612        let actor = reportgen::Client::new(
613            self.addr(),
614            self.reports.last.clone(),
615            relay_map,
616            protocols,
617            self.metrics.clone(),
618            #[cfg(not(wasm_browser))]
619            socket_state,
620            #[cfg(any(test, feature = "test-utils"))]
621            opts.insecure_skip_relay_cert_verify,
622        );
623
624        self.current_report_run = Some(ReportRun {
625            _reportgen: actor,
626            report_tx: response_tx,
627        });
628    }
629
630    fn handle_report_ready(&mut self, report: Report) {
631        let report = self.finish_and_store_report(report);
632        self.in_flight_stun_requests.clear();
633        if let Some(ReportRun { report_tx, .. }) = self.current_report_run.take() {
634            report_tx.send(Ok(report)).ok();
635        }
636    }
637
638    fn handle_report_aborted(&mut self, reason: ActorRunError) {
639        self.in_flight_stun_requests.clear();
640        if let Some(ReportRun { report_tx, .. }) = self.current_report_run.take() {
641            report_tx.send(Err(AbortSnafu { reason }.build())).ok();
642        }
643    }
644
645    /// Handles [`Message::StunPacket`].
646    ///
647    /// If there are currently no in-flight stun requests registered this is dropped,
648    /// otherwise forwarded to the probe.
649    fn handle_stun_packet(&mut self, pkt: &[u8], src: SocketAddr) {
650        trace!(%src, "received STUN packet");
651        if self.in_flight_stun_requests.is_empty() {
652            return;
653        }
654
655        #[cfg(feature = "metrics")]
656        match &src {
657            SocketAddr::V4(_) => {
658                self.metrics.stun_packets_recv_ipv4.inc();
659            }
660            SocketAddr::V6(_) => {
661                self.metrics.stun_packets_recv_ipv6.inc();
662            }
663        }
664
665        match stun::parse_response(pkt) {
666            Ok((txn, addr_port)) => match self.in_flight_stun_requests.remove(&txn) {
667                Some(inf) => {
668                    debug!(%src, %txn, "received known STUN packet");
669                    let elapsed = inf.start.elapsed();
670                    inf.s.send((elapsed, addr_port)).ok();
671                }
672                None => {
673                    debug!(%src, %txn, "received unexpected STUN message response");
674                }
675            },
676            Err(err) => {
677                match stun::parse_binding_request(pkt) {
678                    Ok(txn) => {
679                        // Is this our hairpin request?
680                        match self.in_flight_stun_requests.remove(&txn) {
681                            Some(inf) => {
682                                debug!(%src, %txn, "received our hairpin STUN request");
683                                let elapsed = inf.start.elapsed();
684                                inf.s.send((elapsed, src)).ok();
685                            }
686                            None => {
687                                debug!(%src, %txn, "unknown STUN request");
688                            }
689                        }
690                    }
691                    Err(_) => {
692                        debug!(%src, "received invalid STUN response: {err:#}");
693                    }
694                }
695            }
696        }
697    }
698
699    /// Handles [`Message::InFlightStun`].
700    ///
701    /// The in-flight request is added to [`Actor::in_flight_stun_requests`] so that
702    /// [`Actor::handle_stun_packet`] can forward packets correctly.
703    ///
704    /// *response_tx* is to signal the actor message has been handled.
705    fn handle_in_flight_stun(&mut self, inflight: Inflight, response_tx: oneshot::Sender<()>) {
706        self.in_flight_stun_requests.insert(inflight.txn, inflight);
707        response_tx.send(()).ok();
708    }

710    fn finish_and_store_report(&mut self, report: Report) -> Arc<Report> {
711        let report = self.add_report_history_and_set_preferred_relay(report);
712        debug!("{report:?}");
713        report
714    }
715
716    /// Adds `r` to the set of recent Reports and mutates `r.preferred_relay` to contain the best recent one.
717    /// `r` is stored ref counted and a reference is returned.
718    fn add_report_history_and_set_preferred_relay(&mut self, mut r: Report) -> Arc<Report> {
719        let mut prev_relay = None;
720        if let Some(ref last) = self.reports.last {
721            prev_relay.clone_from(&last.preferred_relay);
722        }
723        let now = Instant::now();
724        const MAX_AGE: Duration = Duration::from_secs(5 * 60);
725
726        // relay ID => its best recent latency in last MAX_AGE
727        let mut best_recent = RelayLatencies::new();
728
729        // chain the current report as we are still mutating it
730        let prevs_iter = self
731            .reports
732            .prev
733            .iter()
734            .map(|(a, b)| -> (&Instant, &Report) { (a, b) })
735            .chain(std::iter::once((&now, &r)));
736
737        let mut to_remove = Vec::new();
738        for (t, pr) in prevs_iter {
739            if now.duration_since(*t) > MAX_AGE {
740                to_remove.push(*t);
741                continue;
742            }
743            best_recent.merge(&pr.relay_latency);
744        }
745
746        for t in to_remove {
747            self.reports.prev.remove(&t);
748        }
749
750        // Then, pick which currently-alive relay server from the
751        // current report has the best latency over the past MAX_AGE.
752        let mut best_any = Duration::default();
753        let mut old_relay_cur_latency = Duration::default();
754        {
755            for (url, duration) in r.relay_latency.iter() {
756                if Some(url) == prev_relay.as_ref() {
757                    old_relay_cur_latency = duration;
758                }
759                if let Some(best) = best_recent.get(url) {
760                    if r.preferred_relay.is_none() || best < best_any {
761                        best_any = best;
762                        r.preferred_relay.replace(url.clone());
763                    }
764                }
765            }
766
767            // If we're changing our preferred relay but the old one's still
768            // accessible and the new one's not much better, just stick with
769            // where we are.
770            if prev_relay.is_some()
771                && r.preferred_relay != prev_relay
772                && !old_relay_cur_latency.is_zero()
773                && best_any > old_relay_cur_latency / 3 * 2
774            {
775                r.preferred_relay = prev_relay;
776            }
777        }
778
779        let r = Arc::new(r);
780        self.reports.prev.insert(now, r.clone());
781        self.reports.last = Some(r.clone());
782
783        r
784    }
785}

832    /// Attempts to bind a local socket to send STUN packets from.
833    ///
834    /// If successful this returns the bound socket and will forward STUN responses to the
835    /// provided *actor_addr*.  The *cancel_token* serves to stop the packet forwarding when the
836    /// socket is no longer needed.
837    pub fn bind_local_stun_socket(
838        network: IpFamily,
839        actor_addr: Addr,
840        cancel_token: CancellationToken,
841    ) -> Option<Arc<UdpSocket>> {
842        let sock = match UdpSocket::bind(network, 0) {
843            Ok(sock) => Arc::new(sock),
844            Err(err) => {
845                debug!("failed to bind STUN socket: {}", err);
846                return None;
847            }
848        };
849        let span = info_span!(
850            "stun_udp_listener",
851            local_addr = sock
852                .local_addr()
853                .map(|a| a.to_string())
854                .unwrap_or(String::from("-")),
855        );
856        {
857            let sock = sock.clone();
858            task::spawn(
859                async move {
860                    debug!("udp stun socket listener started");
861                    // TODO: Can we do better for buffers here?  Probably doesn't matter much.
862                    let mut buf = vec![0u8; 64 << 10];
863                    loop {
864                        tokio::select! {
865                            biased;
866                            _ = cancel_token.cancelled() => break,
867                            res = recv_stun_once(&sock, &mut buf, &actor_addr) => {
868                                if let Err(err) = res {
869                                    warn!(%err, "stun recv failed");
870                                    break;
871                                }
872                            }
873                        }
874                    }
875                    debug!("udp stun socket listener stopped");
876                }
877                .instrument(span),
878            );
879        }
880        Some(sock)
881    }
882
883    #[derive(Debug, Snafu)]
884    enum RecvStunError {
885        #[snafu(transparent)]
886        Recv { source: std::io::Error },
887        #[snafu(display("Internal actor is gone"))]
888        ActorGone,
889    }
890
891    /// Receive STUN response from a UDP socket, pass it to the actor.
892    async fn recv_stun_once(
893        sock: &UdpSocket,
894        buf: &mut [u8],
895        actor_addr: &Addr,
896    ) -> Result<(), RecvStunError> {
897        let (count, mut from_addr) = sock.recv_from(buf).await?;
898
899        let payload = &buf[..count];
900        from_addr.set_ip(from_addr.ip().to_canonical());
901        let msg = Message::StunPacket {
902            payload: Bytes::from(payload.to_vec()),
903            from_addr,
904        };
905        actor_addr
906            .send(msg)
907            .await
908            .map_err(|_| ActorGoneSnafu.build())
909    }
910}

1029        pub(crate) async fn serve(
1030            ip: IpAddr,
1031        ) -> std::io::Result<(SocketAddr, StunStats, CleanupDropGuard)> {
1032            let stats = StunStats::default();
1033
1034            let pc = net::UdpSocket::bind((ip, 0)).await?;
1035            let mut addr = pc.local_addr()?;
1036            match addr.ip() {
1037                IpAddr::V4(ip) => {
1038                    if ip.octets() == [0, 0, 0, 0] {
1039                        addr.set_ip("127.0.0.1".parse().unwrap());
1040                    }
1041                }
1042                _ => unreachable!("using ipv4"),
1043            }

1046            let (_guard, r) = oneshot::channel();
1047            let stats_c = stats.clone();
1048            tokio::task::spawn(async move {
1049                run_stun(pc, stats_c, r).await;

1055        async fn run_stun(pc: net::UdpSocket, stats: StunStats, mut done: oneshot::Receiver<()>) {
1056            let mut buf = vec![0u8; 64 << 10];
1057            loop {
1058                trace!("read loop");
1059                tokio::select! {
1060                    _ = &mut done => {
1061                        debug!("shutting down");
1062                        break;
1063                    }
1064                    res = pc.recv_from(&mut buf) => match res {
1065                        Ok((n, addr)) => {
1066                            trace!("read packet {}bytes from {}", n, addr);
1067                            let pkt = &buf[..n];
1068                            if !stun::is(pkt) {
1069                                debug!("received non STUN pkt");
1070                                continue;
1071                            }
1072                            if let Ok(txid) = stun::parse_binding_request(pkt) {
1073                                debug!("received binding request");
1074                                let mut s = stats.0.lock().await;
1075                                if addr.is_ipv4() {
1076                                    s.0 += 1;
1077                                } else {
1078                                    s.1 += 1;
1079                                }
1080                                drop(s);
1081
1082                                let res = stun::response(txid, addr);
1083                                if let Err(err) = pc.send_to(&res, addr).await {
1084                                    eprintln!("STUN server write failed: {:?}", err);
1085                                }
1086                            }
1087                        }
1088                        Err(err) => {
1089                            eprintln!("failed to read: {:?}", err);
1090                        }
1091                    }
1092                }
1093            }
1094        }
1095    }

1321    #[tokio::test]
1322    async fn test_hairpin() -> Result {
1323        // Hairpinning is initiated after we discover our own IPv4 socket address (IP +
1324        // port) via STUN, so the test needs to have a STUN server and perform STUN over
1325        // IPv4 first.  Hairpinning detection works by sending a STUN *request* to **our own
1326        // public socket address** (IP + port).  If the router supports hairpinning the STUN
1327        // request is returned back to us and received on our public address.  This doesn't
1328        // need to be a STUN request, but STUN already has a unique transaction ID which we
1329        // can easily use to identify the packet.
1330
1331        // Setup STUN server and create relay_map.
1332        let (stun_addr, _stun_stats, _done) = stun_utils::serve_v4().await.e()?;
1333        let dm = stun_utils::relay_map_of([stun_addr].into_iter());
1334        dbg!(&dm);
1335
1336        let resolver = dns::tests::resolver().clone();
1337        let mut client = Client::new(None, resolver, None, Default::default());
1338
1339        // Set up an external socket to send STUN requests from, this will be discovered as
1340        // our public socket address by STUN.  We send back any packets received on this
1341        // socket to the net_report client using Client::receive_stun_packet.  Once we sent
1342        // the hairpin STUN request (from a different randomly bound socket) we are sending
1343        // it to this socket, which is forwarnding it back to our net_report client, because
1344        // this dumb implementation just forwards anything even if it would be garbage.
1345        // Thus hairpinning detection will declare hairpinning to work.
1346        let sock = UdpSocket::bind_local(netwatch::IpFamily::V4, 0).e()?;
1347        let sock = Arc::new(sock);
1348        info!(addr=?sock.local_addr().unwrap(), "Using local addr");
1349        let task = {
1350            let sock = sock.clone();
1351            let addr = client.addr.clone();
1352            tokio::spawn(
1353                async move {
1354                    let mut buf = BytesMut::zeroed(64 << 10);
1355                    loop {
1356                        let (count, src) = sock.recv_from(&mut buf).await.unwrap();
1357                        info!(
1358                            addr=?sock.local_addr().unwrap(),
1359                            %count,
1360                            "Forwarding payload to net_report client",
1361                        );
1362                        let payload = buf.split_to(count).freeze();
1363                        addr.receive_stun_packet(payload, src);
1364                    }
1365                }
1366                .instrument(info_span!("pkt-fwd")),
1367            )
1368        };
1369
1370        let r = client.get_report_all(dm, Some(sock), None, None).await?;
1371        dbg!(&r);
1372        assert_eq!(r.hair_pinning, Some(true));
1373
1374        task.abort();
1375        Ok(())
1376    }
