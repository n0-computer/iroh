1//! A discovery service which publishes and resolves endpoint information using a [pkarr] relay.
2//!
3//! Public-Key Addressable Resource Records, [pkarr], is a system which allows publishing
4//! [DNS Resource Records] owned by a particular [`SecretKey`] under a name derived from its
5//! corresponding [`PublicKey`], also known as the [`EndpointId`].  Additionally this pkarr
6//! Resource Record is signed using the same [`SecretKey`], ensuring authenticity of the
7//! record.
8//!
9//! Pkarr normally stores these records on the [Mainline DHT], but also provides two bridges
10//! that do not require clients to directly interact with the DHT:
11//!
12//! - Resolvers are servers which expose the pkarr Resource Record under a domain name,
13//!   e.g. `o3dks..6uyy.dns.iroh.link`.  This allows looking up the pkarr Resource Records
14//!   using normal DNS clients.  These resolvers would normally perform lookups on the
15//!   Mainline DHT augmented with a local cache to improve performance.
16//!
17//! - Relays are servers which allow both publishing and looking up of the pkarr Resource
18//!   Records using HTTP PUT and GET requests.  They will usually perform the publishing to
19//!   the Mainline DHT on behalf on the client as well as cache lookups performed on the DHT
20//!   to improve performance.
21//!
22//! For endpoint discovery in iroh the pkarr Resource Records contain the addressing information,
23//! providing endpoints which retrieve the pkarr Resource Record with enough detail
24//! to contact the iroh endpoint.
25//!
26//! There are several endpoint discovery services built on top of pkarr, which can be composed
27//! to the application's needs:
28//!
29//! - [`PkarrPublisher`], which publishes to a pkarr relay server using HTTP.
30//!
31//! - [`PkarrResolver`], which resolves from a pkarr relay server using HTTP.
32//!
33//! - [`DnsDiscovery`], which resolves from a DNS server.
34//!
35//! - [`DhtDiscovery`], which resolves and publishes from both pkarr relay servers and well
36//!   as the Mainline DHT.
37//!
38//! [pkarr]: https://pkarr.org
39//! [DNS Resource Records]: https://en.wikipedia.org/wiki/Domain_Name_System#Resource_records
40//! [Mainline DHT]: https://en.wikipedia.org/wiki/Mainline_DHT
41//! [`SecretKey`]: crate::SecretKey
42//! [`PublicKey`]: crate::PublicKey
43//! [`EndpointId`]: crate::EndpointId
44//! [`DnsDiscovery`]: crate::discovery::dns::DnsDiscovery
45//! [`DhtDiscovery`]: dht::DhtDiscovery
46

77#[allow(missing_docs)]
78#[derive(Debug, Snafu)]
79#[non_exhaustive]
80pub enum PkarrError {
81    #[snafu(display("Invalid public key"))]
82    PublicKey { source: PublicKeyError },
83    #[snafu(display("Packet failed to verify"))]
84    Verify { source: SignedPacketVerifyError },
85    #[snafu(display("Invalid relay URL"))]
86    InvalidRelayUrl { url: RelayUrl },
87    #[snafu(display("Error sending http request"))]
88    HttpSend { source: reqwest::Error },
89    #[snafu(display("Error resolving http request"))]
90    HttpRequest { status: reqwest::StatusCode },
91    #[snafu(display("Http payload error"))]
92    HttpPayload { source: reqwest::Error },
93    #[snafu(display("EncodingError"))]
94    Encoding { source: EncodingError },
95}

123/// `iroh-dns-server`, e.g. as running on [`N0_DNS_PKARR_RELAY_PROD`], as the home server

219/// Publisher of endpoint discovery information to a [pkarr] relay.
220///
221/// This publisher uses HTTP to publish endpoint discovery information to a pkarr relay
222/// server, see the [module docs] for details.
223///
224/// This implements the [`Discovery`] trait to be used as a endpoint discovery service.  Note
225/// that it only publishes endpoint discovery information, for the corresponding resolver use
226/// the [`PkarrResolver`] together with [`ConcurrentDiscovery`].
227///
228/// This publisher will **only** publish the [`RelayUrl`] if it is set, otherwise the *direct addresses* are published instead.
229///
230/// [pkarr]: https://pkarr.org
231/// [module docs]: crate::discovery::pkarr
232/// [`RelayUrl`]: crate::RelayUrl
233/// [`ConcurrentDiscovery`]: super::ConcurrentDiscovery
234#[derive(derive_more::Debug, Clone)]
235pub struct PkarrPublisher {
236    endpoint_id: EndpointId,
237    watchable: Watchable<Option<EndpointInfo>>,
238    _drop_guard: Arc<AbortOnDropHandle<()>>,
239}
240
241impl PkarrPublisher {
242    /// Returns a [`PkarrPublisherBuilder`] that publishes endpoint info to a [pkarr] relay at `pkarr_relay`.
243    ///
244    /// If no further options are set, the pkarr publisher  will use [`DEFAULT_PKARR_TTL`] as the
245    /// time-to-live value for the published packets, and it will republish discovery information
246    /// every [`DEFAULT_REPUBLISH_INTERVAL`], even if the information is unchanged.
247    ///
248    /// [`PkarrPublisherBuilder`] implements [`IntoDiscovery`], so it can be passed to [`add_discovery`].
249    /// It will then use the endpoint's secret key to sign published packets.
250    ///
251    /// [`add_discovery`]:  crate::endpoint::Builder::add_discovery
252    /// [pkarr]: https://pkarr.org
253    pub fn builder(pkarr_relay: Url) -> PkarrPublisherBuilder {
254        PkarrPublisherBuilder::new(pkarr_relay)
255    }
256
257    /// Creates a new [`PkarrPublisher`] with a custom TTL and republish intervals.
258    ///
259    /// This allows creating the publisher with custom time-to-live values of the
260    /// [`pkarr::SignedPacket`]s and well as a custom republish interval.
261    fn new(
262        secret_key: SecretKey,
263        pkarr_relay: Url,
264        ttl: u32,
265        republish_interval: Duration,
266        #[cfg(not(wasm_browser))] dns_resolver: Option<DnsResolver>,
267    ) -> Self {
268        debug!("creating pkarr publisher that publishes to {pkarr_relay}");
269        let endpoint_id = secret_key.public();
270
271        #[cfg(wasm_browser)]
272        let pkarr_client = PkarrRelayClient::new(pkarr_relay);
273
274        #[cfg(not(wasm_browser))]
275        let pkarr_client = if let Some(dns_resolver) = dns_resolver {
276            PkarrRelayClient::with_dns_resolver(pkarr_relay, dns_resolver)
277        } else {
278            PkarrRelayClient::new(pkarr_relay)
279        };
280
281        let watchable = Watchable::default();
282        let service = PublisherService {
283            ttl,
284            watcher: watchable.watch(),
285            secret_key,
286            pkarr_client,
287            republish_interval,
288        };
289        let join_handle = task::spawn(
290            service
291                .run()
292                .instrument(error_span!("pkarr_publish", me=%endpoint_id.fmt_short())),
293        );
294        Self {
295            watchable,
296            endpoint_id,
297            _drop_guard: Arc::new(AbortOnDropHandle::new(join_handle)),
298        }
299    }
300
301    /// Creates a pkarr publisher which uses the [number 0] pkarr relay server.
302    ///
303    /// This uses the pkarr relay server operated by [number 0], at
304    /// [`N0_DNS_PKARR_RELAY_PROD`].
305    ///
306    /// When running with the environment variable
307    /// `IROH_FORCE_STAGING_RELAYS` set to any non empty value [`N0_DNS_PKARR_RELAY_STAGING`]
308    /// server is used instead.
309    ///
310    /// [number 0]: https://n0.computer
311    pub fn n0_dns() -> PkarrPublisherBuilder {
312        PkarrPublisherBuilder::n0_dns()
313    }
314
315    /// Publishes the addressing information about this endpoint to a pkarr relay.
316    ///
317    /// This is a nonblocking function, the actual update is performed in the background.
318    pub fn update_endpoint_data(&self, data: &EndpointData) {
319        let mut data = data.clone();
320        if data.relay_url().is_some() {
321            // If relay url is set: only publish relay url, and no direct addrs.
322            data.clear_direct_addresses();
323        }
324        let info = EndpointInfo::from_parts(self.endpoint_id, data);
325        self.watchable.set(Some(info)).ok();
326    }
327}
328
329impl Discovery for PkarrPublisher {
330    fn publish(&self, data: &EndpointData) {
331        self.update_endpoint_data(data);
332    }

335/// Publish endpoint info to a pkarr relay.
336#[derive(derive_more::Debug, Clone)]
337struct PublisherService {
338    #[debug("SecretKey")]
339    secret_key: SecretKey,
340    #[debug("PkarrClient")]
341    pkarr_client: PkarrRelayClient,
342    watcher: n0_watcher::Direct<Option<EndpointInfo>>,
343    ttl: u32,
344    republish_interval: Duration,
345}
346
347impl PublisherService {
348    async fn run(mut self) {
349        let mut failed_attempts = 0;
350        let republish = time::sleep(Duration::MAX);
351        tokio::pin!(republish);
352        loop {
353            if !self.watcher.is_connected() {
354                break;
355            }
356            if let Some(info) = self.watcher.get() {
357                match self.publish_current(info).await {
358                    Err(err) => {
359                        failed_attempts += 1;
360                        // Retry after increasing timeout
361                        let retry_after = Duration::from_secs(failed_attempts);
362                        republish.as_mut().reset(Instant::now() + retry_after);
363                        warn!(
364                            err = %format!("{err:#}"),
365                            url = %self.pkarr_client.pkarr_relay_url ,
366                            ?retry_after,
367                            %failed_attempts,
368                            "Failed to publish to pkarr",
369                        );
370                    }
371                    _ => {
372                        failed_attempts = 0;
373                        // Republish after fixed interval
374                        republish
375                            .as_mut()
376                            .reset(Instant::now() + self.republish_interval);
377                    }
378                }
379            }
380            // Wait until either the retry/republish timeout is reached, or the endpoint info changed.
381            tokio::select! {
382                res = self.watcher.updated() => match res {
383                    Ok(_) => debug!("Publish endpoint info to pkarr (info changed)"),
384                    Err(Disconnected { .. }) => break,
385                },
386                _ = &mut republish => debug!("Publish endpoint info to pkarr (interval elapsed)"),
387            }
388        }
389    }
390
391    async fn publish_current(&self, info: EndpointInfo) -> Result<(), PkarrError> {
392        debug!(
393            data = ?info.data,
394            pkarr_relay = %self.pkarr_client.pkarr_relay_url,
395            "Publish endpoint info to pkarr"
396        );
397        let signed_packet = info
398            .to_pkarr_signed_packet(&self.secret_key, self.ttl)
399            .context(EncodingSnafu)?;
400        self.pkarr_client.publish(&signed_packet).await?;
401        Ok(())
402    }
403}

503impl Discovery for PkarrResolver {
504    fn resolve(
505        &self,
506        endpoint_id: EndpointId,
507    ) -> Option<BoxStream<Result<DiscoveryItem, DiscoveryError>>> {
508        let pkarr_client = self.pkarr_client.clone();
509        let fut = async move {
510            let signed_packet = pkarr_client.resolve(endpoint_id).await?;
511            let info = EndpointInfo::from_pkarr_signed_packet(&signed_packet)
512                .map_err(|err| DiscoveryError::from_err("pkarr", err))?;
513            let item = DiscoveryItem::new(info, "pkarr", None);
514            Ok(item)
515        };
516        let stream = n0_future::stream::once_future(fut);
517        Some(Box::pin(stream))
518    }

521/// A [pkarr] client to publish [`pkarr::SignedPacket`]s to a pkarr relay.
522///
523/// [pkarr]: https://pkarr.org
524#[derive(Debug, Clone)]
525pub struct PkarrRelayClient {
526    http_client: reqwest::Client,
527    pkarr_relay_url: Url,
528}
529
530impl PkarrRelayClient {
531    /// Creates a new client.
532    pub fn new(pkarr_relay_url: Url) -> Self {
533        Self {
534            http_client: reqwest_client_builder()
535                .build()
536                .expect("failed to create reqwest client"),
537            pkarr_relay_url,
538        }
539    }
540
541    /// Creates a new client while passing a DNS resolver to use.
542    #[cfg(not(wasm_browser))]
543    pub fn with_dns_resolver(pkarr_relay_url: Url, dns_resolver: crate::dns::DnsResolver) -> Self {
544        let http_client = reqwest_client_builder()
545            .dns_resolver(Arc::new(dns_resolver))
546            .build()
547            .expect("failed to create request client");
548        Self {
549            http_client,
550            pkarr_relay_url,
551        }
552    }
553
554    /// Resolves a [`SignedPacket`] for the given [`EndpointId`].
555    pub async fn resolve(&self, endpoint_id: EndpointId) -> Result<SignedPacket, DiscoveryError> {
556        // We map the error to string, as in browsers the error is !Send
557        let public_key =
558            pkarr::PublicKey::try_from(endpoint_id.as_bytes()).context(PublicKeySnafu)?;
559
560        let mut url = self.pkarr_relay_url.clone();
561        url.path_segments_mut()
562            .map_err(|_| {
563                InvalidRelayUrlSnafu {
564                    url: self.pkarr_relay_url.clone(),
565                }
566                .build()
567            })?
568            .push(&public_key.to_z32());
569
570        let response = self
571            .http_client
572            .get(url)
573            .send()
574            .await
575            .context(HttpSendSnafu)?;
576
577        if !response.status().is_success() {
578            return Err(HttpRequestSnafu {
579                status: response.status(),
580            }
581            .build()
582            .into());
583        }
584
585        let payload = response.bytes().await.context(HttpPayloadSnafu)?;
586        // We map the error to string, as in browsers the error is !Send
587        let packet =
588            SignedPacket::from_relay_payload(&public_key, &payload).context(VerifySnafu)?;
589        Ok(packet)
590    }
591
592    /// Publishes a [`SignedPacket`].
593    pub async fn publish(&self, signed_packet: &SignedPacket) -> Result<(), PkarrError> {
594        let mut url = self.pkarr_relay_url.clone();
595        url.path_segments_mut()
596            .map_err(|_| {
597                InvalidRelayUrlSnafu {
598                    url: self.pkarr_relay_url.clone(),
599                }
600                .build()
601            })?
602            .push(&signed_packet.public_key().to_z32());
603
604        let response = self
605            .http_client
606            .put(url)
607            .body(signed_packet.to_relay_payload())
608            .send()
609            .await
610            .context(HttpSendSnafu)?;
611
612        if !response.status().is_success() {
613            return Err(HttpRequestSnafu {
614                status: response.status(),
615            }
616            .build());
617        }
618
619        Ok(())
620    }
