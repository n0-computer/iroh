81    pub fn is_valid_send_addr(&self, dst: SocketAddr) -> bool {
82        match self {
83            Self::V4Default { .. } => matches!(dst, SocketAddr::V4(_)),
84            Self::V6Default { .. } => matches!(dst, SocketAddr::V6(_)),
85            Self::V4 { ip_addr, .. } => match dst {
86                SocketAddr::V4(dst_v4) => ip_addr.contains(dst_v4.ip()),
87                SocketAddr::V6(_) => false,
88            },
89            Self::V6 {
90                ip_addr, scope_id, ..
91            } => match dst {
92                SocketAddr::V6(dst_v6) => {
93                    if ip_addr.contains(dst_v6.ip()) {
94                        return true;
95                    }
96                    if dst_v6.ip().is_unicast_link_local() {
97                        // If we have a link local interface, use the scope id
98                        if *scope_id == dst_v6.scope_id() {
99                            return true;
100                        }
101                    }
102                    false
103                }
104                SocketAddr::V4(_) => false,
105            },
106        }
107    }

110impl From<Config> for SocketAddr {

112        match value {
113            Config::V4Default { ip_addr, port } => SocketAddr::V4(SocketAddrV4::new(ip_addr, port)),
114            Config::V6Default {
115                ip_addr,
116                scope_id,
117                port,
118            } => SocketAddr::V6(SocketAddrV6::new(ip_addr, port, 0, scope_id)),
119            Config::V4 { ip_addr, port } => SocketAddr::V4(SocketAddrV4::new(ip_addr.addr(), port)),
120            Config::V6 {
121                ip_addr,
122                scope_id,
123                port,
124            } => SocketAddr::V6(SocketAddrV6::new(ip_addr.addr(), port, 0, scope_id)),
125        }

129fn bind_with_fallback(mut addr: SocketAddr) -> io::Result<netwatch::UdpSocket> {
130    debug!(?addr, "binding");
131    // First try binding a preferred port, if specified
132    match netwatch::UdpSocket::bind_full(addr) {
133        Ok(socket) => {
134            let local_addr = socket.local_addr()?;
135            debug!(%addr, %local_addr, "successfully bound");
136            return Ok(socket);
137        }
138        Err(err) => {
139            debug!(%addr, "failed to bind: {err:#}");
140            // If that was already the fallback port, then error out
141            if addr.port() == 0 {
142                return Err(err);
143            }
144        }
145    }
146
147    // Otherwise, try binding with port 0
148    addr.set_port(0);
149    netwatch::UdpSocket::bind_full(addr)
150}

175    /// NOTE: Receiving on a closed socket will return [`Poll::Pending`] indefinitely.
176    pub(super) fn poll_recv(
177        &mut self,
178        cx: &mut Context,
179        bufs: &mut [io::IoSliceMut<'_>],
180        metas: &mut [quinn_udp::RecvMeta],
181        source_addrs: &mut [Addr],
182    ) -> Poll<io::Result<usize>> {
183        match self.socket.poll_recv_quinn(cx, bufs, metas) {
184            Poll::Pending => Poll::Pending,
185            Poll::Ready(Ok(n)) => {
186                for (source_addr, meta) in source_addrs.iter_mut().zip(metas.iter_mut()).take(n) {
187                    if meta.addr.is_ipv4() {
188                        // The AsyncUdpSocket is an AF_INET6 socket and needs to show this
189                        // as coming from an IPv4-mapped IPv6 addresses, since Quinn will
190                        // use those when sending on an INET6 socket.
191                        let v6_ip = match meta.addr.ip() {
192                            IpAddr::V4(ipv4_addr) => ipv4_addr.to_ipv6_mapped(),
193                            IpAddr::V6(ipv6_addr) => ipv6_addr,
194                        };
195                        meta.addr = SocketAddr::new(v6_ip.into(), meta.addr.port());
196                    }
197                    // The transport addresses are internal to iroh and we always want those
198                    // to remain the canonical address.
199                    *source_addr =
200                        SocketAddr::new(meta.addr.ip().to_canonical(), meta.addr.port()).into();
201                }
202                Poll::Ready(Ok(n))
203            }
204            Poll::Ready(Err(err)) => Poll::Ready(Err(err)),
205        }
206    }

252    pub(super) fn rebind(&self) -> io::Result<()> {
253        let old_addr = self.local_addr.get();
254        self.socket.rebind()?;
255        let addr = self.socket.local_addr()?;
256        self.local_addr.set(addr).ok();
257        trace!("rebound from {} to {}", old_addr, addr);
258
259        Ok(())
260    }

276impl IpSender {
277    pub(super) fn is_valid_send_addr(&self, dst: &SocketAddr) -> bool {
278        // Our net-tools crate binds sockets to their specific family.  This means an IPv6
279        // socket can not sent to IPv4, on any platform.  So we need to convert an
280        // IPv4-mapped IPv6 address back to it's canonical IPv4 address.
281        let mut dst = *dst;
282        dst.set_ip(dst.ip().to_canonical());
283
284        self.config.is_valid_send_addr(dst)
285    }
286
287    /// Creates a canonical socket address.
288    ///
289    /// We may be asked to send IPv4-mapped IPv6 addresses.  But our sockets are configured
290    /// to only send their actual family.  So we need to map those back to the canonical
291    /// addresses.
292    #[inline]
293    fn canonical_addr(addr: SocketAddr) -> SocketAddr {
294        SocketAddr::new(addr.ip().to_canonical(), addr.port())
295    }
296
297    pub(super) async fn send(
298        &self,
299        dst: SocketAddr,
300        src: Option<IpAddr>,
301        transmit: &Transmit<'_>,
302    ) -> io::Result<()> {
303        let total_bytes = transmit.contents.len() as u64;
304        let res = self
305            .sender
306            .send(&quinn_udp::Transmit {
307                destination: Self::canonical_addr(dst),
308                ecn: transmit.ecn,
309                contents: transmit.contents,
310                segment_size: transmit.segment_size,
311                src_ip: src,
312            })
313            .await;
314
315        match res {
316            Ok(res) => {
317                match dst {
318                    SocketAddr::V4(_) => {
319                        self.metrics.send_ipv4.inc_by(total_bytes);
320                    }
321                    SocketAddr::V6(_) => {
322                        self.metrics.send_ipv6.inc_by(total_bytes);
323                    }
324                }
325                Ok(res)
326            }
327            Err(err) => Err(err),
328        }
329    }

331    pub(super) fn poll_send(
332        mut self: Pin<&mut Self>,
333        cx: &mut std::task::Context,
334        dst: SocketAddr,
335        src: Option<IpAddr>,
336        transmit: &Transmit<'_>,
337    ) -> Poll<io::Result<()>> {
338        let total_bytes = transmit.contents.len() as u64;
339        let res = Pin::new(&mut self.sender).poll_send(
340            &quinn_udp::Transmit {
341                destination: Self::canonical_addr(dst),
342                ecn: transmit.ecn,
343                contents: transmit.contents,
344                segment_size: transmit.segment_size,
345                src_ip: src,
346            },
347            cx,
348        );
349
350        match res {
351            Poll::Ready(Ok(res)) => {
352                match dst {
353                    SocketAddr::V4(_) => {
354                        self.metrics.send_ipv4.inc_by(total_bytes);
355                    }
356                    SocketAddr::V6(_) => {
357                        self.metrics.send_ipv6.inc_by(total_bytes);
358                    }
359                }
360                Poll::Ready(Ok(res))
361            }
362            Poll::Ready(Err(err)) => Poll::Ready(Err(err)),
363            Poll::Pending => Poll::Pending,
364        }
365    }
