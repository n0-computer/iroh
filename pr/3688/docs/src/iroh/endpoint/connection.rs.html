1//! The [`Connection`] wraps a `quinn::Connection`.
2//!
3//! The [`Connection`] is how you send data to and receive data from the remote endpoint.
4//!
5//! There are many transitions states between attempting to start a connection and
6//! receiving a cryptographically secure connection.
7//!
8//! The main items in this module are:
9//!
10//! - [`Connection`] to create streams to talk to a remote endpoint.
11//! - [`Connecting`] for operating on connections that haven't finished their handshake yet.
12//! - [`Incoming`] to accept or reject an incoming connection.
13//! - [`OutgoingZeroRttConnection`] to attempt to send 0-RTT data before the cryptographic
14//!   handshake has completed.
15//! - [`IncomingZeroRttConnection`] to attempt to read 0-RTT or send 0.5-RTT data before the cryptographic
16//!   handshake has completed.
17//!
18//! [module docs]: crate

55    #[debug("quinn::Accept")]
56    pub(crate) inner: quinn::Accept<'a>,
57    pub(crate) ep: Endpoint,

60impl Future for Accept<'_> {
61    type Output = Option<Incoming>;
62
63    fn poll(self: Pin<&mut Self>, cx: &mut std::task::Context<'_>) -> Poll<Self::Output> {
64        let this = self.project();
65        match this.inner.poll(cx) {
66            Poll::Pending => Poll::Pending,
67            Poll::Ready(None) => Poll::Ready(None),
68            Poll::Ready(Some(inner)) => Poll::Ready(Some(Incoming {
69                inner,
70                ep: this.ep.clone(),
71            })),
72        }
73    }
74}

84impl Incoming {
85    /// Attempts to accept this incoming connection (an error may still occur).
86    ///
87    /// Errors occurring here are likely not caused by the application or remote.  The QUIC
88    /// connection listens on a normal UDP socket and any reachable network endpoint can
89    /// send datagrams to it, solicited or not.  Even if the first few bytes look like a
90    /// QUIC packet, it might not even be a QUIC packet that is being received.
91    ///
92    /// Thus it is common to simply log the errors here and accept them as something which
93    /// can happen.
94    pub fn accept(self) -> Result<Accepting, ConnectionError> {
95        self.inner
96            .accept()
97            .map(|conn| Accepting::new(conn, self.ep))
98    }

102    /// See [`accept()`] for more details.

107        server_config: Arc<ServerConfig>,
108    ) -> Result<Accepting, ConnectionError> {
109        self.inner
110            .accept_with(server_config)
111            .map(|conn| Accepting::new(conn, self.ep))
112    }

180/// Extracts the ALPN protocol from the peer's handshake data.
181fn alpn_from_quinn_conn(conn: &quinn::Connection) -> Option<Vec<u8>> {
182    let data = conn.handshake_data()?;
183    match data.downcast::<quinn::crypto::rustls::HandshakeData>() {
184        Ok(data) => data.protocol,
185        Err(_) => None,
186    }
187}
188
189async fn alpn_from_quinn_connecting(conn: &mut quinn::Connecting) -> Result<Vec<u8>, AlpnError> {
190    let data = conn.handshake_data().await?;
191    match data.downcast::<quinn::crypto::rustls::HandshakeData>() {
192        Ok(data) => match data.protocol {
193            Some(protocol) => Ok(protocol),
194            None => Err(e!(AlpnError::Unavailable)),
195        },
196        Err(_) => Err(e!(AlpnError::UnknownHandshake)),
197    }
198}

211/// Converts a `quinn::Connection` to a `Connection`.
212///
213/// Returns an error if there was a connection error, the handshake data has not completed
214/// or if the remote did not set an ALPN.
215///
216/// Otherwise returns a future that completes once the connection has been registered with the
217/// magicsock. This future can return an [`RemoteStateActorStoppedError`], which will only be
218/// emitted if the endpoint is closing.
219///
220/// The returned future is `'static`, so it can be stored without being lifetime-bound on `&ep`.
221fn conn_from_quinn_conn(
222    conn: quinn::Connection,
223    ep: &Endpoint,
224) -> Result<
225    impl Future<Output = Result<Connection, ConnectingError>> + Send + 'static,
226    ConnectingError,
227> {
228    let info = match static_info_from_conn(&conn) {
229        Ok(val) => val,
230        Err(auth_err) => {
231            // If the authentication error raced with a connection error, the connection
232            // error wins.
233            if let Some(conn_err) = conn.close_reason() {
234                return Err(e!(ConnectingError::ConnectionError { source: conn_err }));
235            } else {
236                return Err(e!(ConnectingError::HandshakeFailure { source: auth_err }));
237            }
238        }
239    };
240
241    // Register this connection with the magicsock.
242    let fut = ep
243        .msock
244        .register_connection(info.endpoint_id, conn.weak_handle());
245
246    // Check middleware.
247    // TODO: avoid endpoint clone?
248    let ep = ep.clone();
249    Ok(async move {
250        let paths = fut.await?;
251        let conn = Connection {
252            data: HandshakeCompletedData { info, paths },
253            inner: conn,
254        };
255
256        if let AfterHandshakeOutcome::Reject { error_code, reason } = ep
257            .msock
258            .middlewares
259            .handshake_completed(&conn.to_info())
260            .await
261        {
262            conn.close(error_code, &reason);
263            return Err(e!(ConnectingError::LocallyRejected));
264        }
265
266        Ok(conn)
267    })
268}

270fn static_info_from_conn(conn: &quinn::Connection) -> Result<StaticInfo, AuthenticationError> {
271    let endpoint_id = remote_id_from_quinn_conn(conn)?;
272    let alpn = alpn_from_quinn_conn(conn).ok_or_else(|| e!(AuthenticationError::NoAlpn))?;
273    Ok(StaticInfo { endpoint_id, alpn })
274}

276/// Returns the [`EndpointId`] from the peer's TLS certificate.
277///
278/// The [`PublicKey`] of an endpoint is also known as an [`EndpointId`].  This [`PublicKey`] is
279/// included in the TLS certificate presented during the handshake when connecting.
280/// This function allows you to get the [`EndpointId`] of the remote endpoint of this
281/// connection.
282///
283/// [`PublicKey`]: iroh_base::PublicKey
284fn remote_id_from_quinn_conn(
285    conn: &quinn::Connection,
286) -> Result<EndpointId, RemoteEndpointIdError> {
287    let data = conn.peer_identity();
288    match data {
289        None => {
290            warn!("no peer certificate found");
291            Err(RemoteEndpointIdError::new())
292        }
293        Some(data) => match data.downcast::<Vec<rustls::pki_types::CertificateDer>>() {
294            Ok(certs) => {
295                if certs.len() != 1 {
296                    warn!(
297                        "expected a single peer certificate, but {} found",
298                        certs.len()
299                    );
300                    return Err(RemoteEndpointIdError::new());
301                }
302
303                let peer_id = EndpointId::from_verifying_key(
304                    VerifyingKey::from_public_key_der(&certs[0])
305                        .map_err(|_| RemoteEndpointIdError::new())?,
306                );
307
308                Ok(peer_id)
309            }
310            Err(err) => {
311                warn!("invalid peer certificate: {:?}", err);
312                Err(RemoteEndpointIdError::new())
313            }
314        },
315    }
316}

401    /// Converts this [`Connecting`] into a 0-RTT connection at the cost of weakened security.
402    ///
403    /// If 0-RTT can be attempted, returns a [`OutgoingZeroRttConnection`], which represents
404    /// outgoing 0-RTT connection.
405    ///
406    /// If the 0-RTT cannot even be attempted, returns back the same [`Connecting`] without
407    /// changes. You can still `.await` this [`Connecting`] to get a normal [`Connection`].
408    ///
409    /// The [`OutgoingZeroRttConnection`] will attempt to resume a previous TLS session. However,
410    /// **the remote endpoint may actually _reject_ the 0-RTT data--yet still accept
411    /// the connection attempt in general**, once the handshake has completed.
412    ///
413    /// This possibility of whether the 0-RTT data was accepted or rejected is conveyed
414    /// through the [`ZeroRttStatus`] after calling [`OutgoingZeroRttConnection::handshake_completed`].
415    /// When the handshake completes, it returns [`ZeroRttStatus::Accepted`] if the 0-RTT data
416    /// was accepted and [`ZeroRttStatus::Rejected`] if it was rejected. If it was rejected, the
417    /// existence of any streams opened and application data sent prior to the handshake
418    /// completing will not be conveyed to the remote application, and local operations on them
419    /// will return `ZeroRttRejected` errors.
420    ///
421    /// A server may reject 0-RTT data at its discretion, but accepting 0-RTT data requires the
422    /// relevant resumption state to be stored in the server, which servers may limit or lose for
423    /// various reasons including not persisting resumption state across server restarts.
424    ///
425    /// ## Security
426    ///
427    /// This enables transmission of 0-RTT data, which is vulnerable to replay attacks, and
428    /// should therefore never invoke non-idempotent operations.
429    ///
430    /// You can use [`RecvStream::is_0rtt`] to check whether a stream has been opened in 0-RTT
431    /// and thus whether parts of the stream are operating under this reduced security level.
432    ///
433    /// See also documentation for [`Accepting::into_0rtt`].
434    ///
435    /// [`RecvStream::is_0rtt`]: quinn::RecvStream::is_0rtt
436    #[allow(clippy::result_large_err)]
437    pub fn into_0rtt(self) -> Result<OutgoingZeroRttConnection, Connecting> {
438        match self.inner.into_0rtt() {
439            Ok((quinn_conn, zrtt_accepted)) => {
440                let accepted: BoxFuture<_> = Box::pin({
441                    let quinn_conn = quinn_conn.clone();
442                    async move {
443                        let accepted = zrtt_accepted.await;
444                        let conn = conn_from_quinn_conn(quinn_conn, &self.ep)?.await?;
445                        Ok(match accepted {
446                            true => ZeroRttStatus::Accepted(conn),
447                            false => ZeroRttStatus::Rejected(conn),
448                        })
449                    }
450                });
451                let accepted = accepted.shared();
452                Ok(Connection {
453                    inner: quinn_conn,
454                    data: OutgoingZeroRttData { accepted },
455                })
456            }
457            Err(inner) => Err(Self { inner, ..self }),
458        }
459    }
460
461    /// Parameters negotiated during the handshake
462    pub async fn handshake_data(&mut self) -> Result<Box<dyn Any>, ConnectionError> {
463        self.inner.handshake_data().await
464    }
465
466    /// Extracts the ALPN protocol from the peer's handshake data.
467    pub async fn alpn(&mut self) -> Result<Vec<u8>, AlpnError> {
468        alpn_from_quinn_connecting(&mut self.inner).await
469    }
470
471    /// Returns the [`EndpointId`] of the endpoint that this connection attempt tries to connect to.
472    pub fn remote_id(&self) -> EndpointId {
473        self.remote_endpoint_id
474    }
475}

502    /// Converts this [`Accepting`] into a 0-RTT or 0.5-RTT connection at the cost of weakened
503    /// security.
504    ///
505    /// Returns a [`IncomingZeroRttConnection`], which represents an incoming 0-RTT or 0.5-RTT connection.
506    ///
507    /// If the connection was initiated with 0-RTT by the remote endpoint, the local endpoint
508    /// might accept the 0-RTT attempt, allowing the local endpoint to receive application streams
509    /// and data before the handshake finishes.
510    ///
511    /// Otherwise this will enable 0.5-RTT, allowing the [`IncomingZeroRttConnection`] to open streams and send
512    /// data before the handshake finishes.
513    ///
514    /// ## Security
515    ///
516    /// Transmitted 0-RTT data from the client is vulnerable to replay attacks, and should
517    /// therefore never invoke non-idempotent operations.
518    ///
519    /// Transmission of 0.5-RTT data from the server may be sent before TLS client authentication
520    /// has occurred, and should therefore not be used to send data for which client
521    /// authentication is required.
522    ///
523    /// You can use [`RecvStream::is_0rtt`] to check whether a stream has been opened in 0-RTT
524    /// and thus whether parts of the stream are operating under this reduced security level.
525    ///
526    /// See also documentation for [`Connecting::into_0rtt`].
527    ///
528    /// [`RecvStream::is_0rtt`]: quinn::RecvStream::is_0rtt
529    pub fn into_0rtt(self) -> IncomingZeroRttConnection {
530        let (quinn_conn, zrtt_accepted) = self
531            .inner
532            .into_0rtt()
533            .expect("incoming connections can always be converted to 0-RTT");
534        let accepted: BoxFuture<_> = Box::pin({
535            let quinn_conn = quinn_conn.clone();
536            async move {
537                zrtt_accepted.await;
538                let conn = conn_from_quinn_conn(quinn_conn, &self.ep)?.await?;
539                Ok(conn)
540            }
541        });
542        let accepted = accepted.shared();
543        IncomingZeroRttConnection {
544            inner: quinn_conn,
545            data: IncomingZeroRttData { accepted },
546        }
547    }
548
549    /// Parameters negotiated during the handshake
550    pub async fn handshake_data(&mut self) -> Result<Box<dyn Any>, ConnectionError> {
551        self.inner.handshake_data().await
552    }
553
554    /// Extracts the ALPN protocol from the peer's handshake data.
555    pub async fn alpn(&mut self) -> Result<Vec<u8>, AlpnError> {
556        alpn_from_quinn_connecting(&mut self.inner).await
557    }
558}

578/// The client side of a 0-RTT connection.
579///
580/// This is created using [`Connecting::into_0rtt`].
581///
582/// Creating a `OutgoingZeroRttConnection` means that the endpoint is capable
583/// of attempting a 0-RTT connection with the remote. The remote may still
584/// reject the 0-RTT connection. In which case, any data sent before the
585/// handshake has completed may need to be resent.
586///
587/// Look at the [`OutgoingZeroRttConnection::handshake_completed`] method for
588/// more details.
589pub type OutgoingZeroRttConnection = Connection<OutgoingZeroRtt>;
590
591/// Returned from [`OutgoingZeroRttConnection::handshake_completed`].
592#[derive(Debug, Clone)]
593pub enum ZeroRttStatus {
594    /// If the 0-RTT data was accepted, you can continue to use any streams
595    /// that were created before the handshake was completed.
596    Accepted(Connection),
597    /// If the 0-RTT data was rejected, any streams that were created before
598    /// the handshake was completed will error and any data that was
599    /// previously sent on those streams will need to be resent.
600    Rejected(Connection),
601}

603/// A QUIC connection on the server-side that can possibly accept 0-RTT data.
604///
605/// It is very similar to a `Connection`, but the `IncomingZeroRttConnection::remote_id`
606/// and `IncomingZeroRttConnection::alpn` may not be set yet, since the handshake has
607/// not necessarily occurred yet.
608///
609/// If the `IncomingZeroRttConnection` has rejected 0-RTT or does not have enough information
610/// to accept 0-RTT, any received 0-RTT packets will simply be dropped before
611/// reaching any receive streams.
612///
613/// Any streams that are created to send or receive data can continue to be used
614/// even after the handshake has completed and we are no longer in a 0-RTT
615/// situation.
616///
617/// Use the [`IncomingZeroRttConnection::handshake_completed`] method to get a [`Connection`] from a
618/// `IncomingZeroRttConnection`. This waits until 0-RTT connection has completed
619/// the handshake and can now confidently derive the ALPN and the
620/// [`EndpointId`] of the remote endpoint.
621pub type IncomingZeroRttConnection = Connection<IncomingZeroRtt>;

623/// A QUIC connection.
624///
625/// If all references to a connection (including every clone of the Connection handle,
626/// streams of incoming streams, and the various stream types) have been dropped, then the
627/// connection will be automatically closed with an error_code of 0 and an empty reason. You
628/// can also close the connection explicitly by calling [`Connection::close`].
629///
630/// Closing the connection immediately abandons efforts to deliver data to the peer. Upon
631/// receiving CONNECTION_CLOSE the peer may drop any stream data not yet delivered to the
632/// application. [`Connection::close`] describes in more detail how to gracefully close a
633/// connection without losing application data.
634///
635/// May be cloned to obtain another handle to the same connection.
636#[derive(Debug, Clone)]
637pub struct Connection<State: ConnectionState = HandshakeCompleted> {
638    inner: quinn::Connection,
639    /// State-specific information
640    data: State::Data,
641}

783    /// Closes the connection immediately.
784    ///
785    /// Pending operations will fail immediately with [`ConnectionError::LocallyClosed`]. No
786    /// more data is sent to the peer and the peer may drop buffered data upon receiving the
787    /// CONNECTION_CLOSE frame.
788    ///
789    /// `error_code` and `reason` are not interpreted, and are provided directly to the
790    /// peer.
791    ///
792    /// `reason` will be truncated to fit in a single packet with overhead; to improve odds
793    /// that it is preserved in full, it should be kept under 1KiB.
794    ///
795    /// # Gracefully closing a connection
796    ///
797    /// Only the peer last receiving application data can be certain that all data is
798    /// delivered. The only reliable action it can then take is to close the connection,
799    /// potentially with a custom error code. The delivery of the final CONNECTION_CLOSE
800    /// frame is very likely if both endpoints stay online long enough, calling
801    /// [`Endpoint::close`] will wait to provide sufficient time. Otherwise, the remote peer
802    /// will time out the connection, provided that the idle timeout is not disabled.
803    ///
804    /// The sending side can not guarantee all stream data is delivered to the remote
805    /// application. It only knows the data is delivered to the QUIC stack of the remote
806    /// endpoint. Once the local side sends a CONNECTION_CLOSE frame in response to calling
807    /// [`close`] the remote endpoint may drop any data it received but is as yet
808    /// undelivered to the application, including data that was acknowledged as received to
809    /// the local endpoint.

1361    #[tokio::test]
1362    #[traced_test]
1363    async fn test_paths_watcher() -> Result {
1364        const ALPN: &[u8] = b"test";
1365        let mut rng = rand_chacha::ChaCha8Rng::seed_from_u64(0u64);
1366        let (relay_map, _relay_map, _guard) = run_relay_server().await?;
1367        let server = Endpoint::empty_builder(RelayMode::Custom(relay_map.clone()))
1368            .secret_key(SecretKey::generate(&mut rng))
1369            .insecure_skip_relay_cert_verify(true)
1370            .alpns(vec![ALPN.to_vec()])
1371            .bind()
1372            .await?;
1373
1374        let client = Endpoint::empty_builder(RelayMode::Custom(relay_map.clone()))
1375            .secret_key(SecretKey::generate(&mut rng))
1376            .insecure_skip_relay_cert_verify(true)
1377            .bind()
1378            .await?;
1379
1380        server.online().await;
1381        let server_addr = server.addr();
1382        info!("server addr: {server_addr:?}");
1383
1384        let (conn_client, conn_server) = tokio::join!(
1385            async { client.connect(server_addr, ALPN).await.unwrap() },
1386            async { server.accept().await.unwrap().await.unwrap() }
1387        );
1388        info!("connected");
1389        let mut paths_client = conn_client.paths().stream();
1390        let mut paths_server = conn_server.paths().stream();
1391
1392        /// Advances the path stream until at least one IP and one relay paths are available.
1393        ///
1394        /// Panics if the path stream finishes before that happens.
1395        async fn wait_for_paths(
1396            stream: &mut n0_watcher::Stream<impl n0_watcher::Watcher<Value = PathInfoList> + Unpin>,
1397        ) {
1398            loop {
1399                let paths = stream.next().await.expect("paths stream ended");
1400                info!(?paths, "paths");
1401                if paths.len() >= 2
1402                    && paths.iter().any(PathInfo::is_relay)
1403                    && paths.iter().any(PathInfo::is_ip)
1404                {
1405                    info!("break");
1406                    return;
1407                }
1408            }
1409        }
1410
1411        // Verify that both connections are notified of path changes and get an IP and a relay path.
1412        tokio::join!(
1413            async {
1414                tokio::time::timeout(Duration::from_secs(1), wait_for_paths(&mut paths_server))
1415                    .instrument(error_span!("paths-server"))
1416                    .await
1417                    .unwrap()
1418            },
1419            async {
1420                tokio::time::timeout(Duration::from_secs(1), wait_for_paths(&mut paths_client))
1421                    .instrument(error_span!("paths-client"))
1422                    .await
1423                    .unwrap()
1424            }
1425        );
1426
1427        // Close the client connection.
1428        info!("close client conn");
1429        conn_client.close(0u32.into(), b"");
1430
1431        // Verify that the path watch streams close.
1432        assert_eq!(
1433            tokio::time::timeout(Duration::from_secs(1), paths_client.next())
1434                .await
1435                .unwrap(),
1436            None
1437        );
1438        assert_eq!(
1439            tokio::time::timeout(Duration::from_secs(1), paths_server.next())
1440                .await
1441                .unwrap(),
1442            None
1443        );
1444
1445        server.close().await;
1446        client.close().await;
1447
1448        Ok(())
1449    }
