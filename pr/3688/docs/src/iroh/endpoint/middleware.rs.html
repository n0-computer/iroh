10type BoxFuture<'a, T> = Pin<Box<dyn Future<Output = T> + Send + 'a>>;

12#[derive(Debug)]
13pub enum BeforeConnectOutcome {
14    Accept,
15    Reject,
16}
17
18#[derive(Debug)]
19pub enum AfterHandshakeOutcome {
20    Accept,
21    Reject { error_code: VarInt, reason: Vec<u8> },
22}
23
24impl AfterHandshakeOutcome {
25    pub fn accept() -> Self {
26        Self::Accept
27    }
28
29    pub fn close(&self, error_code: VarInt, reason: &[u8]) -> Self {
30        Self::Reject {
31            error_code,
32            reason: reason.to_vec(),
33        }
34    }
35}

37pub trait Middleware: std::fmt::Debug + Send + Sync {
38    fn before_connect(
39        &self,
40        _remote_addr: &EndpointAddr,
41        _alpn: &[u8],
42    ) -> impl Future<Output = BeforeConnectOutcome> + Send + '_ {
43        async { BeforeConnectOutcome::Accept }
44    }
45
46    fn handshake_completed(
47        &self,
48        _conn: &ConnectionInfo,
49    ) -> impl Future<Output = AfterHandshakeOutcome> + Send + '_ {
50        async { AfterHandshakeOutcome::accept() }
51    }
52}

54pub(crate) trait DynMiddleware: std::fmt::Debug + Send + Sync {
55    fn before_connect(
56        &self,
57        _remote_addr: &EndpointAddr,
58        _alpn: &[u8],
59    ) -> BoxFuture<'_, BeforeConnectOutcome>;
60    fn handshake_completed(&self, _conn: &ConnectionInfo) -> BoxFuture<'_, AfterHandshakeOutcome>;
61}
62
63impl<T: Middleware> DynMiddleware for T {
64    fn before_connect(
65        &self,
66        remote_addr: &EndpointAddr,
67        alpn: &[u8],
68    ) -> BoxFuture<'_, BeforeConnectOutcome> {
69        Box::pin(Middleware::before_connect(self, remote_addr, alpn))
70    }
71
72    fn handshake_completed(&self, conn: &ConnectionInfo) -> BoxFuture<'_, AfterHandshakeOutcome> {
73        Box::pin(Middleware::handshake_completed(self, conn))
74    }
75}

77#[derive(Debug, Default)]
78pub(crate) struct MiddlewareList {
79    inner: Vec<Box<dyn DynMiddleware>>,
80}
81
82impl MiddlewareList {
83    pub(super) fn push(&mut self, middleware: impl Middleware + 'static) {
84        let middleware: Box<dyn DynMiddleware> = Box::new(middleware);
85        self.inner.push(middleware);
86    }
87
88    pub(super) async fn before_connect(
89        &self,
90        remote_addr: &EndpointAddr,
91        alpn: &[u8],
92    ) -> BeforeConnectOutcome {
93        for middleware in self.inner.iter() {
94            match middleware.before_connect(remote_addr, alpn).await {
95                BeforeConnectOutcome::Accept => continue,
96                reject @ BeforeConnectOutcome::Reject => return reject,
97            }
98        }
99        BeforeConnectOutcome::Accept
100    }
101
102    pub(super) async fn handshake_completed(&self, conn: &ConnectionInfo) -> AfterHandshakeOutcome {
103        for middleware in self.inner.iter() {
104            match middleware.handshake_completed(conn).await {
105                AfterHandshakeOutcome::Accept => continue,
106                reject @ AfterHandshakeOutcome::Reject { .. } => return reject,
107            }
108        }
109        AfterHandshakeOutcome::Accept
110    }
