10/// Outcome of [`Middleware::before_connect`]
11#[derive(Debug)]
12pub enum BeforeConnectOutcome {
13    /// Accept the connect attempt.
14    Accept,
15    /// Reject the connect attempt.
16    Reject,
17}
18
19/// Outcome of [`Middleware::after_handshake`]
20#[derive(Debug)]
21pub enum AfterHandshakeOutcome {
22    /// Accept the connection.
23    Accept,
24    /// Reject and close the connection.
25    ///
26    /// See [`Connection::close`] for details on `error_code` and `reason`.
27    ///
28    /// [`Connection::close`]: crate::endpoint::Connection::close
29    Reject {
30        /// Error code to send with the connection close frame.
31        error_code: VarInt,
32        /// Close reason to send with the connection close frame.
33        reason: Vec<u8>,
34    },
35}
36
37impl AfterHandshakeOutcome {
38    /// Returns [`Self::Accept`].
39    pub fn accept() -> Self {
40        Self::Accept
41    }
42
43    /// Returns [`Self::Reject`].
44    pub fn reject(&self, error_code: VarInt, reason: &[u8]) -> Self {
45        Self::Reject {
46            error_code,
47            reason: reason.to_vec(),
48        }
49    }
50}

52/// Middlewares intercept the connection establishment process of an [`Endpoint`].
53///
54/// Use [`Builder::middleware`] to install middlewares onto an endpoint.
55///
56/// For each hook, all installed middlewares are invoked in the order they were installed on
57/// the endpoint builder. If a middleware returns `Accept`, processing continues with the next
58/// middleware. If a middleware returns `Reject`, processing is aborted and further middlewares
59/// are not invoked for this hook.
60///
61/// ## Notes to implementers
62///
63/// As middlewares are stored on the endpoint, you must make sure to never store an [`Endpoint`]
64/// on the middleware struct itself, as this would create reference counting loop and cause the
65/// endpoint to never be dropped, leaking memory.
66///
67/// [`Endpoint`]: crate::Endpoint
68/// [`Builder::middleware`]: crate::endpoint::Builder::middleware
69pub trait Middleware: std::fmt::Debug + Send + Sync {
70    /// Intercept outgoing connections before they are started.
71    ///
72    /// This is called whenever a new outgoing connection is initiated via [`Endpoint::connect`]
73    /// or [`Endpoint::connect_with_opts`].
74    ///
75    /// If any middleware returns [`BeforeConnectOutcome::Reject`], the connection attempt is aborted
76    /// before any packets are sent to the remote.
77    ///
78    /// [`Endpoint::connect`]: crate::Endpoint::connect
79    /// [`Endpoint::connect_with_opts`]: crate::Endpoint::connect_with_opts
80    fn before_connect<'a>(
81        &'a self,
82        _remote_addr: &'a EndpointAddr,
83        _alpn: &'a [u8],
84    ) -> impl Future<Output = BeforeConnectOutcome> + Send + 'a {
85        async { BeforeConnectOutcome::Accept }
86    }
87
88    /// Intercept both incoming and outgoing connections once the TLS handshake has completed.
89    ///
90    /// At this point in time, we know the remote's endpoint id and ALPN. If any middleware returns
91    /// [`AfterHandshakeOutcome::Reject`], the connection is closed with the provided error code
92    /// and reason.
93    fn after_handshake<'a>(
94        &'a self,
95        _conn: &'a ConnectionInfo,
96    ) -> impl Future<Output = AfterHandshakeOutcome> + Send + 'a {
97        async { AfterHandshakeOutcome::accept() }
98    }
99}

101pub(crate) trait DynMiddleware: std::fmt::Debug + Send + Sync {
102    fn before_connect<'a>(
103        &'a self,
104        remote_addr: &'a EndpointAddr,
105        alpn: &'a [u8],
106    ) -> BoxFuture<'a, BeforeConnectOutcome>;
107    fn after_handshake<'a>(
108        &'a self,
109        conn: &'a ConnectionInfo,
110    ) -> BoxFuture<'a, AfterHandshakeOutcome>;
111}
112
113impl<T: Middleware> DynMiddleware for T {
114    fn before_connect<'a>(
115        &'a self,
116        remote_addr: &'a EndpointAddr,
117        alpn: &'a [u8],
118    ) -> BoxFuture<'a, BeforeConnectOutcome> {
119        Box::pin(Middleware::before_connect(self, remote_addr, alpn))
120    }
121
122    fn after_handshake<'a>(
123        &'a self,
124        conn: &'a ConnectionInfo,
125    ) -> BoxFuture<'a, AfterHandshakeOutcome> {
126        Box::pin(Middleware::after_handshake(self, conn))
127    }
128}

130#[derive(Debug, Default)]
131pub(crate) struct MiddlewareList {
132    inner: Vec<Box<dyn DynMiddleware>>,
133}
134
135impl MiddlewareList {
136    pub(super) fn push(&mut self, middleware: impl Middleware + 'static) {
137        let middleware: Box<dyn DynMiddleware> = Box::new(middleware);
138        self.inner.push(middleware);
139    }
140
141    pub(super) async fn before_connect(
142        &self,
143        remote_addr: &EndpointAddr,
144        alpn: &[u8],
145    ) -> BeforeConnectOutcome {
146        for middleware in self.inner.iter() {
147            match middleware.before_connect(remote_addr, alpn).await {
148                BeforeConnectOutcome::Accept => continue,
149                reject @ BeforeConnectOutcome::Reject => {
150                    return reject;
151                }
152            }
153        }
154        BeforeConnectOutcome::Accept
155    }
156
157    pub(super) async fn after_handshake(&self, conn: &ConnectionInfo) -> AfterHandshakeOutcome {
158        for middleware in self.inner.iter() {
159            match middleware.after_handshake(conn).await {
160                AfterHandshakeOutcome::Accept => continue,
161                reject @ AfterHandshakeOutcome::Reject { .. } => {
162                    return reject;
163                }
164            }
165        }
166        AfterHandshakeOutcome::Accept
167    }
168}
