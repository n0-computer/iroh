1//! Implements a socket that can change its communication path while in use, actively searching for the best way to communicate.
2//!
3//! Based on tailscale/wgengine/magicsock
4//!
5//! ### `RelayOnly` path selection:
6//! When set this will force all packets to be sent over
7//! the relay connection, regardless of whether or
8//! not we have a direct UDP address for the given endpoint.
9//!
10//! The intended use is for testing the relay protocol inside the MagicSock
11//! to ensure that we can rely on the relay to send packets when two endpoints
12//! are unable to find direct UDP connections to each other.
13//!
14//! This also prevent this endpoint from attempting to hole punch and prevents it
15//! from responding to any hole punching attempts. This endpoint will still,
16//! however, read any packets that come off the UDP sockets.

267    /// Registers the connection in the `EndpointStateActor`.
268    ///
269    /// The actor is responsible for holepunching and opening additional paths to this
270    /// connection.
271    pub(crate) fn register_connection(
272        &self,
273        remote: EndpointId,
274        conn: &quinn::Connection,
275        paths_info: n0_watcher::Watchable<PathsInfo>,
276    ) {
277        // TODO: Spawning tasks like this is obviously bad.  But it is solvable:
278        //   - This is only called from inside Connection::new.
279        //   - Connection::new is called from:
280        //     - impl Future for IncomingFuture
281        //     - impl Future for Connecting
282        //     - Connecting::into_0rtt()
283        //
284        // The first two can keep returning Pending until this message is also sent.  It'll
285        // require storing the pinned future but it'll work.
286        //
287        // The last one is trickier.  But we can make that function async.  Or more likely
288        // we'll end up changing Connecting::into_0rtt() to return a ZrttConnection.  Then
289        // have a ZrttConnection::into_connection() function which can be async and actually
290        // send this.  Before the handshake has completed we don't have anything useful to
291        // do with this connection inside of the EndpointStateActor anyway.
292        let weak_handle = conn.weak_handle();
293        let endpoint_state = self.endpoint_map.endpoint_state_actor(remote);
294        let msg = EndpointStateMessage::AddConnection(weak_handle, paths_info);
295
296        task::spawn(async move {
297            endpoint_state.send(msg).await.ok();
298        });
299    }

431    /// Add potential addresses for a endpoint to the `EndpointStateActor`.
432    ///
433    /// This is used to add possible paths that the remote endpoint might be reachable on.  They
434    /// will be used when there is no active connection to the endpoint to attempt to establish
435    /// a connection.
436    #[instrument(skip_all)]
437    pub(crate) async fn add_endpoint_addr(
438        &self,
439        mut addr: EndpointAddr,
440        source: endpoint_map::Source,
441    ) -> Result<(), AddEndpointAddrError> {
442        let mut pruned: usize = 0;
443        for my_addr in self.direct_addrs.sockaddrs() {
444            if addr.addrs.remove(&TransportAddr::Ip(my_addr)) {
445                warn!( endpoint_id=%addr.id.fmt_short(), %my_addr, %source, "not adding our addr for endpoint");
446                pruned += 1;
447            }
448        }
449        if !addr.is_empty() {
450            // Add addr to the internal EndpointMap
451            self.endpoint_map
452                .add_endpoint_addr(addr.clone(), source)
453                .await;
454            Ok(())
455        } else if pruned != 0 {
456            Err(e!(AddEndpointAddrError::EmptyPruned { pruned }))
457        } else {
458            Err(e!(AddEndpointAddrError::Empty))
459        }
460    }

510    #[cfg_attr(windows, allow(dead_code))]
511    fn normalized_local_addr(&self) -> io::Result<SocketAddr> {
512        let addrs = self.local_addrs_watch.clone().get();
513
514        let mut ipv4_addr = None;
515        for addr in addrs {
516            let Some(addr) = addr.into_socket_addr() else {
517                continue;
518            };
519            if addr.is_ipv6() {
520                return Ok(addr);
521            }
522            if addr.is_ipv4() && ipv4_addr.is_none() {
523                ipv4_addr.replace(addr);
524            }
525        }
526        match ipv4_addr {
527            Some(addr) => Ok(addr),
528            None => Err(io::Error::other("no valid socket available")),

532    /// Process datagrams received from all the transports.
533    ///
534    /// All the `bufs` and `metas` should have initialized packets in them.
535    ///
536    /// This fixes up the datagrams to use the correct [`MultipathMappedAddr`] and extracts
537    /// DISCO packets, processing them inside the magic socket.
538    ///
539    /// [`MultipathMappedAddr`]: mapped_addrs::MultipathMappedAddr
540    fn process_datagrams(
541        &self,
542        bufs: &mut [io::IoSliceMut<'_>],
543        metas: &mut [quinn_udp::RecvMeta],
544        source_addrs: &[transports::Addr],
545    ) {
546        debug_assert_eq!(bufs.len(), metas.len(), "non matching bufs & metas");
547        debug_assert_eq!(
548            bufs.len(),
549            source_addrs.len(),
550            "non matching bufs & source_addrs"
551        );
552
553        // Adding the IP address we received something on results in Quinn using this
554        // address on the send path to send from.  However we let Quinn use a
555        // EndpointIdMappedAddress, not a real address.  So we used to substitute our bind address
556        // here so that Quinn would send on the right address.  But that would sometimes
557        // result in the wrong address family and Windows trips up on that.
558        //
559        // What should be done is that this dst_ip from the RecvMeta is stored in the
560        // EndpointState/PathState.  Then on the send path it should be retrieved from the
561        // EndpointState/PathSate together with the send address and substituted at send time.
562        // This is relevant for IPv6 link-local addresses where the OS otherwise does not
563        // know which interface to send from.
564        #[cfg(not(windows))]
565        let dst_ip = self.normalized_local_addr().ok().map(|addr| addr.ip());
566        // Reasoning for this here:
567        // https://github.com/n0-computer/iroh/pull/2595#issuecomment-2290947319
568        #[cfg(windows)]
569        let dst_ip = None;
570
571        let mut quic_packets_total = 0;
572
573        for ((quinn_meta, buf), source_addr) in metas
574            .iter_mut()
575            .zip(bufs.iter_mut())
576            .zip(source_addrs.iter())
577        {
578            let mut buf_contains_quic_datagrams = false;
579            let mut quic_datagram_count = 0;
580            if quinn_meta.len > quinn_meta.stride {
581                trace!(%quinn_meta.len, %quinn_meta.stride, "GRO datagram received");
582                self.metrics.magicsock.recv_gro_datagrams.inc();
583            }
584
585            // Chunk through the datagrams in this GRO payload to find disco
586            // packets and forward them to the actor
587            for datagram in buf[..quinn_meta.len].chunks_mut(quinn_meta.stride) {
588                if datagram.len() < quinn_meta.stride {
589                    trace!(
590                        len = %datagram.len(),
591                        %quinn_meta.stride,
592                        "Last GRO datagram smaller than stride",
593                    );
594                }
595
596                // Detect DISCO datagrams and process them.  Overwrite the first
597                // byte of those packets with zero to make Quinn ignore the packet.  This
598                // relies on quinn::EndpointConfig::grease_quic_bit being set to `false`,
599                // which we do in Endpoint::bind.
600                if let Some((sender, sealed_box)) = disco::source_and_box(datagram) {
601                    trace!(src = ?source_addr, len = datagram.len(), "UDP recv: DISCO packet");
602                    self.handle_disco_message(sender, sealed_box, source_addr);
603                    datagram[0] = 0u8;
604                } else {
605                    trace!(src = ?source_addr, len = datagram.len(), "UDP recv: QUIC packet");
606                    match source_addr {
607                        transports::Addr::Ip(SocketAddr::V4(..)) => {
608                            self.metrics
609                                .magicsock
610                                .recv_data_ipv4
611                                .inc_by(datagram.len() as _);
612                        }
613                        transports::Addr::Ip(SocketAddr::V6(..)) => {
614                            self.metrics
615                                .magicsock
616                                .recv_data_ipv6
617                                .inc_by(datagram.len() as _);
618                        }
619                        transports::Addr::Relay(..) => {
620                            self.metrics
621                                .magicsock
622                                .recv_data_relay
623                                .inc_by(datagram.len() as _);
624                        }
625                    }
626
627                    quic_datagram_count += 1;
628                    buf_contains_quic_datagrams = true;
629                }
630            }
631
632            if buf_contains_quic_datagrams {
633                match source_addr {
634                    #[cfg(wasm_browser)]
635                    transports::Addr::Ip(_addr) => {
636                        panic!("cannot use IP based addressing in the browser");
637                    }
638                    #[cfg(not(wasm_browser))]
639                    transports::Addr::Ip(_addr) => {
640                        quic_packets_total += quic_datagram_count;
641                    }
642                    transports::Addr::Relay(src_url, src_endpoint) => {
643                        let mapped_addr = self
644                            .endpoint_map
645                            .relay_mapped_addrs
646                            .get(&(src_url.clone(), *src_endpoint));
647                        quinn_meta.addr = mapped_addr.private_socket_addr();
648                    }
649                }
650            } else {
651                // If all datagrams in this buf are DISCO, set len to zero to make
652                // Quinn skip the buf completely.
653                quinn_meta.len = 0;
654            }
655            // Normalize local_ip
656            quinn_meta.dst_ip = dst_ip;
657        }
658
659        if quic_packets_total > 0 {
660            self.metrics
661                .magicsock
662                .recv_datagrams
663                .inc_by(quic_packets_total as _);
664            trace!("UDP recv: {} packets", quic_packets_total);
665        }
666    }

668    /// Handles a discovery message.
669    #[instrument("disco_in", skip_all, fields(endpoint = %sender.fmt_short(), ?src))]
670    fn handle_disco_message(&self, sender: PublicKey, sealed_box: &[u8], src: &transports::Addr) {
671        if self.is_closed() {
672            return;
673        }
674
675        if let transports::Addr::Relay(_, endpoint_id) = src {
676            if endpoint_id != &sender {
677                // TODO: return here?
678                warn!(
679                    "Received relay disco message from connection for {}, but with message from {}",
680                    endpoint_id.fmt_short(),
681                    sender.fmt_short()
682                );
683            }
684        }
685
686        // We're now reasonably sure we're expecting communication from
687        // this endpoint, do the heavy crypto lifting to see what they want.
688        let dm = match self.disco.unseal_and_decode(sender, sealed_box) {
689            Ok(dm) => dm,
690            Err(DiscoBoxError::Open { source, .. }) => {
691                warn!(?source, "failed to open disco box");
692                self.metrics.magicsock.recv_disco_bad_key.inc();
693                return;
694            }
695            Err(DiscoBoxError::Parse { source, .. }) => {
696                // Couldn't parse it, but it was inside a correctly
697                // signed box, so just ignore it, assuming it's from a
698                // newer version of Tailscale that we don't
699                // understand. Not even worth logging about, lest it
700                // be too spammy for old clients.
701
702                self.metrics.magicsock.recv_disco_bad_parse.inc();
703                debug!(?source, "failed to parse disco message");
704                return;
705            }
706        };
707
708        if src.is_relay() {
709            self.metrics.magicsock.recv_disco_relay.inc();
710        } else {
711            self.metrics.magicsock.recv_disco_udp.inc();
712        }
713
714        trace!(?dm, "receive disco message");
715        match dm {
716            disco::Message::Ping(ping) => {
717                self.metrics.magicsock.recv_disco_ping.inc();
718                self.endpoint_map.handle_ping(ping, sender, src.clone());
719            }
720            disco::Message::Pong(pong) => {
721                self.metrics.magicsock.recv_disco_pong.inc();
722                self.endpoint_map.handle_pong(pong, sender, src.clone());
723            }
724            disco::Message::CallMeMaybe(cm) => {
725                self.metrics.magicsock.recv_disco_call_me_maybe.inc();
726                self.endpoint_map
727                    .handle_call_me_maybe(cm, sender, src.clone());
728            }
729        }
730    }

732    /// Sends out a disco message.
733    async fn send_disco_message(
734        &self,
735        sender: &TransportsSender,
736        dst: SendAddr,
737        dst_key: PublicKey,
738        msg: disco::Message,
739    ) -> io::Result<()> {
740        let dst = match dst {
741            SendAddr::Udp(addr) => transports::Addr::Ip(addr),
742            SendAddr::Relay(url) => transports::Addr::Relay(url, dst_key),
743        };
744
745        trace!(?dst, %msg, "send disco message (UDP)");
746        if self.is_closed() {
747            return Err(io::Error::new(
748                io::ErrorKind::NotConnected,
749                "connection closed",
750            ));
751        }
752
753        let pkt = self.disco.encode_and_seal(dst_key, &msg);
754
755        let transmit = transports::Transmit {
756            contents: &pkt,
757            ecn: None,
758            segment_size: None,
759        };
760
761        let dst2 = dst.clone();
762        match sender.send(&dst2, None, &transmit).await {
763            Ok(()) => {
764                trace!(?dst, %msg, "sent disco message");
765                self.metrics.magicsock.sent_disco_udp.inc();
766                disco_message_sent(&msg, &self.metrics.magicsock);
767                Ok(())
768            }
769            Err(err) => {
770                warn!(?dst, ?msg, ?err, "failed to send disco message");
771                Err(err)
772            }
773        }
774    }

805/// Manages currently running direct addr discovery, aka net_report runs.
806///
807/// Invariants:
808/// - only one direct addr update must be running at a time
809/// - if an update is scheduled while another one is running, remember that
810///   and start a new one when the current one has finished
811#[derive(Debug)]
812struct DirectAddrUpdateState {
813    /// If set, start a new update as soon as the current one is finished.
814    want_update: Option<UpdateReason>,
815    msock: Arc<MagicSock>,
816    #[cfg(not(wasm_browser))]
817    port_mapper: portmapper::Client,
818    /// The prober that discovers local network conditions, including the closest relay relay and NAT mappings.
819    net_reporter: Arc<AsyncMutex<net_report::Client>>,
820    relay_map: RelayMap,
821    run_done: mpsc::Sender<()>,
822}
823
824#[derive(Default, Debug, PartialEq, Eq, Clone, Copy)]
825enum UpdateReason {
826    /// Initial state
827    #[default]
828    None,
829    Periodic,
830    PortmapUpdated,
831    LinkChangeMajor,
832    LinkChangeMinor,
833    RelayMapChange,
834}
835
836impl UpdateReason {
837    fn is_major(self) -> bool {
838        matches!(self, Self::LinkChangeMajor | Self::RelayMapChange)
839    }
840}
841
842impl DirectAddrUpdateState {
843    fn new(
844        msock: Arc<MagicSock>,
845        #[cfg(not(wasm_browser))] port_mapper: portmapper::Client,
846        net_reporter: Arc<AsyncMutex<net_report::Client>>,
847        relay_map: RelayMap,
848        run_done: mpsc::Sender<()>,
849    ) -> Self {
850        DirectAddrUpdateState {
851            want_update: Default::default(),
852            #[cfg(not(wasm_browser))]
853            port_mapper,
854            net_reporter,
855            msock,
856            relay_map,
857            run_done,
858        }
859    }
860
861    /// Schedules a new run, either starting it immediately if none is running or
862    /// scheduling it for later.
863    fn schedule_run(&mut self, why: UpdateReason, if_state: IfStateDetails) {
864        match self.net_reporter.clone().try_lock_owned() {
865            Ok(net_reporter) => {
866                self.run(why, if_state, net_reporter);
867            }
868            Err(_) => {
869                let _ = self.want_update.insert(why);
870            }
871        }
872    }
873
874    /// If another run is needed, triggers this run, otherwise does nothing.
875    fn try_run(&mut self, if_state: IfStateDetails) {
876        match self.net_reporter.clone().try_lock_owned() {
877            Ok(net_reporter) => {
878                if let Some(why) = self.want_update.take() {
879                    self.run(why, if_state, net_reporter);
880                }
881            }
882            Err(_) => {
883                // do nothing
884            }
885        }
886    }
887
888    /// Trigger a new run.
889    fn run(
890        &mut self,
891        why: UpdateReason,
892        if_state: IfStateDetails,
893        mut net_reporter: tokio::sync::OwnedMutexGuard<net_report::Client>,
894    ) {
895        debug!("starting direct addr update ({:?})", why);
896        #[cfg(not(wasm_browser))]
897        self.port_mapper.procure_mapping();
898        // Don't start a net report probe if we know
899        // we are shutting down
900        if self.msock.is_closing() || self.msock.is_closed() {
901            debug!("skipping net_report, socket is shutting down");
902            return;
903        }
904        if self.relay_map.is_empty() {
905            debug!("skipping net_report, empty RelayMap");
906            self.msock.net_report.set((None, why)).ok();
907            return;
908        }
909
910        debug!("requesting net_report report");
911        let msock = self.msock.clone();
912
913        let run_done = self.run_done.clone();
914        task::spawn(
915            async move {
916                let fut = time::timeout(
917                    NET_REPORT_TIMEOUT,
918                    net_reporter.get_report(if_state, why.is_major()),
919                );
920                match fut.await {
921                    Ok(report) => {
922                        msock.net_report.set((Some(report), why)).ok();
923                    }
924                    Err(time::Elapsed { .. }) => {
925                        warn!("net_report report timed out");
926                    }
927                }
928
929                // mark run as finished
930                debug!("direct addr update done ({:?})", why);
931                run_done.send(()).await.ok();
932            }
933            .instrument(tracing::Span::current()),
934        );
935    }
936}

1159    /// Closes the connection.
1160    ///
1161    /// Only the first close does anything. Any later closes return nil.  Polling the socket
1162    /// ([`quinn::AsyncUdpSocket::poll_recv`]) will return [`Poll::Pending`] indefinitely
1163    /// after this call.
1164    ///
1165    /// [`Poll::Pending`]: std::task::Poll::Pending
1166    #[instrument(skip_all)]
1167    pub(crate) async fn close(&self) {
1168        trace!(me = ?self.public_key, "magicsock closing...");
1169        // Initiate closing all connections, and refuse future connections.
1170        self.endpoint.close(0u16.into(), b"");
1171
1172        // In the history of this code, this call had been
1173        // - removed: https://github.com/n0-computer/iroh/pull/1753
1174        // - then added back in: https://github.com/n0-computer/iroh/pull/2227/files#diff-ba27e40e2986a3919b20f6b412ad4fe63154af648610ea5d9ed0b5d5b0e2d780R573
1175        // - then removed again: https://github.com/n0-computer/iroh/pull/3165
1176        // and finally added back in together with this comment.
1177        // So before removing this call, please consider carefully.
1178        // Among other things, this call tries its best to make sure that any queued close frames
1179        // (e.g. via the call to `endpoint.close(...)` above), are flushed out to the sockets
1180        // *and acknowledged* (or time out with the "probe timeout" of usually 3 seconds).
1181        // This allows the other endpoints for these connections to be notified to release
1182        // their resources, or - depending on the protocol - that all data was received.
1183        // With the current quinn API, this is the only way to ensure protocol code can use
1184        // connection close codes, and close the endpoint properly.
1185        // If this call is skipped, then connections that protocols close just shortly before the
1186        // call to `Endpoint::close` will in most cases cause connection time-outs on remote ends.
1187        self.endpoint.wait_idle().await;
1188
1189        if self.msock.is_closed() {
1190            return;
1191        }
1192        self.msock.closing.store(true, Ordering::Relaxed);
1193        self.shutdown_token.cancel();
1194
1195        // MutexGuard is not held across await points
1196        let task = self.actor_task.lock().expect("poisoned").take();
1197        if let Some(task) = task {
1198            // give the tasks a moment to shutdown cleanly
1199            let shutdown_done = time::timeout(Duration::from_millis(100), async move {
1200                if let Err(err) = task.await {
1201                    warn!("unexpected error in task shutdown: {:?}", err);
1202                }
1203            })
1204            .await;
1205            match shutdown_done {
1206                Ok(_) => trace!("tasks finished in time, shutdown complete"),
1207                Err(time::Elapsed { .. }) => {
1208                    // Dropping the task will abort itt
1209                    warn!("tasks didn't finish in time, aborting");
1210                }
1211            }
1212        }
1213
1214        self.msock.closed.store(true, Ordering::SeqCst);
1215
1216        trace!("magicsock closed");
1217    }

1234struct DiscoState {
1235    /// The EndpointId/PublikeKey of this endpoint.
1236    this_id: EndpointId,
1237    /// Encryption key for this endpoint.
1238    secret_encryption_key: Arc<crypto_box::SecretKey>,

1245impl DiscoState {
1246    fn new(
1247        secret_key: &SecretKey,
1248    ) -> (Self, mpsc::Receiver<(SendAddr, PublicKey, disco::Message)>) {
1249        let this_id = secret_key.public();
1250        let secret_encryption_key = secret_ed_box(secret_key);
1251        let (disco_sender, disco_receiver) = mpsc::channel(256);
1252
1253        (
1254            Self {
1255                this_id,
1256                secret_encryption_key: Arc::new(secret_encryption_key),
1257                secrets: Default::default(),
1258                sender: disco_sender,
1259            },
1260            disco_receiver,
1261        )
1262    }
1263
1264    fn try_send(&self, dst: SendAddr, dst_key: PublicKey, msg: disco::Message) -> bool {
1265        self.sender.try_send((dst, dst_key, msg)).is_ok()
1266    }
1267
1268    fn encode_and_seal(&self, other_key: PublicKey, msg: &disco::Message) -> Bytes {
1269        let mut seal = msg.as_bytes();
1270        self.get_secret(other_key, |secret| secret.seal(&mut seal));
1271        disco::encode_message(&self.this_id, seal).into()
1272    }
1273
1274    fn unseal_and_decode(
1275        &self,
1276        endpoint_key: PublicKey,
1277        sealed_box: &[u8],
1278    ) -> Result<disco::Message, DiscoBoxError> {
1279        let mut sealed_box = sealed_box.to_vec();
1280        self.get_secret(endpoint_key, |secret| secret.open(&mut sealed_box))
1281            .map_err(|source| e!(DiscoBoxError::Open { source }))?;
1282        disco::Message::from_bytes(&sealed_box)
1283            .map_err(|source| e!(DiscoBoxError::Parse { source }))
1284    }
1285
1286    fn get_secret<F, T>(&self, endpoint_id: PublicKey, cb: F) -> T
1287    where
1288        F: FnOnce(&mut SharedSecret) -> T,
1289    {
1290        let mut inner = self.secrets.lock().expect("poisoned");
1291        let x = inner.entry(endpoint_id).or_insert_with(|| {
1292            let public_key = public_ed_box(&endpoint_id);
1293            SharedSecret::new(&self.secret_encryption_key, &public_key)
1294        });
1295        cb(x)
1296    }
1297}

1337#[cfg(not(wasm_browser))]
1338fn bind_ip(
1339    addr_v4: SocketAddrV4,
1340    addr_v6: Option<SocketAddrV6>,
1341    metrics: &EndpointMetrics,
1342) -> io::Result<(Vec<IpTransport>, portmapper::Client)> {
1343    let port_mapper =
1344        portmapper::Client::with_metrics(Default::default(), metrics.portmapper.clone());
1345
1346    let v4 = Arc::new(bind_with_fallback(SocketAddr::V4(addr_v4))?);
1347    let ip4_port = v4.local_addr()?.port();
1348    let ip6_port = ip4_port.checked_add(1).unwrap_or(ip4_port - 1);
1349
1350    let addr_v6 =
1351        addr_v6.unwrap_or_else(|| SocketAddrV6::new(Ipv6Addr::UNSPECIFIED, ip6_port, 0, 0));
1352
1353    let v6 = match bind_with_fallback(SocketAddr::V6(addr_v6)) {
1354        Ok(sock) => Some(Arc::new(sock)),
1355        Err(err) => {
1356            info!("bind ignoring IPv6 bind failure: {:?}", err);
1357            None
1358        }
1359    };
1360
1361    let port = v4.local_addr().map_or(0, |p| p.port());
1362
1363    let mut ip = vec![IpTransport::new(
1364        addr_v4.into(),
1365        v4,
1366        metrics.magicsock.clone(),
1367    )];
1368    if let Some(v6) = v6 {
1369        ip.push(IpTransport::new(
1370            addr_v6.into(),
1371            v6,
1372            metrics.magicsock.clone(),
1373        ))
1374    }
1375
1376    // NOTE: we can end up with a zero port if `netwatch::UdpSocket::socket_addr` fails
1377    match port.try_into() {
1378        Ok(non_zero_port) => {
1379            port_mapper.update_local_port(non_zero_port);
1380        }
1381        Err(_zero_port) => debug!("Skipping port mapping with zero local port"),
1382    }
1383
1384    Ok((ip, port_mapper))
1385}

1388    async fn run(
1389        mut self,
1390        shutdown_token: CancellationToken,
1391        mut watcher: impl Watcher<Value = Vec<transports::Addr>> + Send + Sync,
1392        sender: TransportsSender,
1393    ) {
1394        // Setup network monitoring
1395        let mut current_netmon_state = self.netmon_watcher.get();
1396
1397        #[cfg(not(wasm_browser))]
1398        let mut portmap_watcher = self
1399            .direct_addr_update_state
1400            .port_mapper
1401            .watch_external_address();
1402
1403        let mut receiver_closed = false;
1404        #[cfg_attr(wasm_browser, allow(unused_mut))]
1405        let mut portmap_watcher_closed = false;
1406
1407        let mut net_report_watcher = self.msock.net_report.watch();
1408
1409        // ensure we are doing an initial publish of our addresses
1410        self.msock.publish_my_addr();
1411
1412        loop {
1413            self.msock.metrics.magicsock.actor_tick_main.inc();
1414            #[cfg(not(wasm_browser))]
1415            let portmap_watcher_changed = portmap_watcher.changed();
1416            #[cfg(wasm_browser)]
1417            let portmap_watcher_changed = n0_future::future::pending();
1418
1419            tokio::select! {
1420                _ = shutdown_token.cancelled() => {
1421                    debug!("shutting down");
1422                    return;
1423                }
1424                msg = self.msg_receiver.recv(), if !receiver_closed => {
1425                    let Some(msg) = msg else {
1426                        trace!("tick: magicsock receiver closed");
1427                        self.msock.metrics.magicsock.actor_tick_other.inc();
1428
1429                        receiver_closed = true;
1430                        continue;
1431                    };
1432
1433                    trace!(?msg, "tick: msg");
1434                    self.msock.metrics.magicsock.actor_tick_msg.inc();
1435                    self.handle_actor_message(msg).await;
1436                }
1437                tick = self.periodic_re_stun_timer.tick() => {
1438                    trace!("tick: re_stun {:?}", tick);
1439                    self.msock.metrics.magicsock.actor_tick_re_stun.inc();
1440                    self.re_stun(UpdateReason::Periodic);
1441                }
1442                new_addr = watcher.updated() => {
1443                    match new_addr {
1444                        Ok(addrs) => {
1445                            if !addrs.is_empty() {
1446                                trace!(?addrs, "local addrs");
1447                                self.msock.publish_my_addr();
1448                            }
1449                        }
1450                        Err(_) => {
1451                            warn!("local addr watcher stopped");
1452                        }
1453                    }
1454                }
1455                report = net_report_watcher.updated() => {
1456                    match report {
1457                        Ok((report, _)) => {
1458                            self.handle_net_report_report(report);
1459                            #[cfg(not(wasm_browser))]
1460                            {
1461                                self.periodic_re_stun_timer = new_re_stun_timer(true);
1462                            }
1463                        }
1464                        Err(_) => {
1465                            warn!("net report watcher stopped");
1466                        }
1467                    }
1468                }
1469                reason = self.direct_addr_done_rx.recv() => {
1470                    match reason {
1471                        Some(()) => {
1472                            // check if a new run needs to be scheduled
1473                            let state = self.netmon_watcher.get();
1474                            self.direct_addr_update_state.try_run(state.into());
1475                        }
1476                        None => {
1477                            warn!("direct addr watcher died");
1478                        }
1479                    }
1480                }
1481                change = portmap_watcher_changed, if !portmap_watcher_closed => {
1482                    #[cfg(not(wasm_browser))]
1483                    {
1484                        if change.is_err() {
1485                            trace!("tick: portmap watcher closed");
1486                            self.msock.metrics.magicsock.actor_tick_other.inc();
1487
1488                            portmap_watcher_closed = true;
1489                            continue;
1490                        }
1491
1492                        trace!("tick: portmap changed");
1493                        self.msock.metrics.magicsock.actor_tick_portmap_changed.inc();
1494                        let new_external_address = *portmap_watcher.borrow();
1495                        debug!("external address updated: {new_external_address:?}");
1496                        self.re_stun(UpdateReason::PortmapUpdated);
1497                    }
1498                    #[cfg(wasm_browser)]
1499                    let _unused_in_browsers = change;
1500                },
1501                state = self.netmon_watcher.updated() => {
1502                    let Ok(state) = state else {
1503                        trace!("tick: link change receiver closed");
1504                        self.msock.metrics.magicsock.actor_tick_other.inc();
1505                        continue;
1506                    };
1507                    let is_major = state.is_major_change(&current_netmon_state);
1508                    event!(
1509                        target: "iroh::_events::link_change",
1510                        Level::DEBUG,
1511                        ?state,
1512                        is_major
1513                    );
1514                    current_netmon_state = state;
1515                    self.msock.metrics.magicsock.actor_link_change.inc();
1516                    self.handle_network_change(is_major).await;
1517                }
1518                Some((dst, dst_key, msg)) = self.disco_receiver.recv() => {
1519                    if let Err(err) = self.msock.send_disco_message(&sender, dst.clone(), dst_key, msg).await {
1520                        warn!(%dst, endpoint = %dst_key.fmt_short(), ?err, "failed to send disco message (UDP)");
1521                    }
1522                }
1523            }
1524        }
1525    }

1571    /// Updates the direct addresses of this magic socket.
1572    ///
1573    /// Updates the [`DiscoveredDirectAddrs`] of this [`MagicSock`] with the current set of
1574    /// direct addresses from:
1575    ///
1576    /// - The portmapper.
1577    /// - A net_report report.
1578    /// - The local interfaces IP addresses.
1579    #[cfg(not(wasm_browser))]
1580    fn update_direct_addresses(&mut self, net_report_report: Option<&net_report::Report>) {
1581        let portmap_watcher = self
1582            .direct_addr_update_state
1583            .port_mapper
1584            .watch_external_address();
1585
1586        // We only want to have one DirectAddr for each SocketAddr we have.  So we store
1587        // this as a map of SocketAddr -> DirectAddrType.  At the end we will construct a
1588        // DirectAddr from each entry.
1589        let mut addrs: BTreeMap<SocketAddr, DirectAddrType> = BTreeMap::new();
1590
1591        // First add PortMapper provided addresses.
1592        let maybe_port_mapped = *portmap_watcher.borrow();
1593        if let Some(portmap_ext) = maybe_port_mapped.map(SocketAddr::V4) {
1594            addrs
1595                .entry(portmap_ext)
1596                .or_insert(DirectAddrType::Portmapped);
1597        }
1598
1599        // Next add STUN addresses from the net_report report.
1600        if let Some(net_report_report) = net_report_report {
1601            if let Some(global_v4) = net_report_report.global_v4 {
1602                addrs.entry(global_v4.into()).or_insert(DirectAddrType::Qad);
1603
1604                // If they're behind a hard NAT and are using a fixed
1605                // port locally, assume they might've added a static
1606                // port mapping on their router to the same explicit
1607                // port that we are running with. Worst case it's an invalid candidate mapping.
1608                let port = self.msock.ip_bind_addrs().iter().find_map(|addr| {
1609                    if addr.port() != 0 {
1610                        Some(addr.port())
1611                    } else {
1612                        None
1613                    }
1614                });
1615
1616                if let Some(port) = port {
1617                    if net_report_report
1618                        .mapping_varies_by_dest()
1619                        .unwrap_or_default()
1620                    {
1621                        let mut addr = global_v4;
1622                        addr.set_port(port);
1623                        addrs
1624                            .entry(addr.into())
1625                            .or_insert(DirectAddrType::Qad4LocalPort);
1626                    }
1627                }
1628            }
1629            if let Some(global_v6) = net_report_report.global_v6 {
1630                addrs.entry(global_v6.into()).or_insert(DirectAddrType::Qad);
1631            }
1632        }
1633
1634        self.collect_local_addresses(&mut addrs);
1635
1636        // Finally create and store store all these direct addresses and send any
1637        // queued call-me-maybe messages.
1638        self.msock.store_direct_addresses(
1639            addrs
1640                .iter()
1641                .map(|(addr, typ)| DirectAddr {
1642                    addr: *addr,
1643                    typ: *typ,
1644                })
1645                .collect(),
1646        );
1647        self.send_queued_call_me_maybes();
1648    }

1650    #[cfg(not(wasm_browser))]
1651    fn collect_local_addresses(&mut self, addrs: &mut BTreeMap<SocketAddr, DirectAddrType>) {
1652        // Matches the addresses that have been bound vs the requested ones.
1653        let local_addrs: Vec<(SocketAddr, SocketAddr)> = self
1654            .msock
1655            .ip_bind_addrs()
1656            .iter()
1657            .copied()
1658            .zip(self.msock.ip_local_addrs())
1659            .collect();
1660
1661        // Do we listen on any IPv4 unspecified address?
1662        let has_ipv4_unspecified = local_addrs.iter().find_map(|(_, a)| {
1663            if a.is_ipv4() && a.ip().is_unspecified() {
1664                Some(a.port())
1665            } else {
1666                None
1667            }
1668        });
1669        // Do we listen on any IPv6 unspecified address?
1670        let has_ipv6_unspecified = local_addrs.iter().find_map(|(_, a)| {
1671            if a.is_ipv6() && a.ip().is_unspecified() {
1672                Some(a.port())
1673            } else {
1674                None
1675            }
1676        });
1677
1678        // If a socket is bound to the unspecified address, create SocketAddrs for
1679        // each local IP address by pairing it with the port the socket is bound on.
1680        if local_addrs
1681            .iter()
1682            .any(|(_, local)| local.ip().is_unspecified())
1683        {
1684            let LocalAddresses {
1685                regular: mut ips,
1686                loopback,
1687            } = self.netmon_watcher.get().local_addresses;
1688            if ips.is_empty() && addrs.is_empty() {
1689                // Include loopback addresses only if there are no other interfaces
1690                // or public addresses, this allows testing offline.
1691                ips = loopback;
1692            }
1693
1694            for ip in ips {
1695                let port_if_unspecified = match ip {
1696                    IpAddr::V4(_) => has_ipv4_unspecified,
1697                    IpAddr::V6(_) => has_ipv6_unspecified,
1698                };
1699                if let Some(port) = port_if_unspecified {
1700                    let addr = SocketAddr::new(ip, port);
1701                    addrs.entry(addr).or_insert(DirectAddrType::Local);
1702                }
1703            }
1704        }
1705
1706        // If a socket is bound to a specific address, add it.
1707        for (bound, local) in local_addrs {
1708            if !bound.ip().is_unspecified() {
1709                addrs.entry(local).or_insert(DirectAddrType::Local);
1710            }
1711        }
1712    }

1714    fn send_queued_call_me_maybes(&mut self) {
1715        let msg = self.msock.direct_addrs.to_call_me_maybe_message();
1716        let msg = disco::Message::CallMeMaybe(msg);
1717        // allocate, to minimize locking duration
1718
1719        for (public_key, url) in self.pending_call_me_maybes.drain() {
1720            if !self
1721                .msock
1722                .disco
1723                .try_send(SendAddr::Relay(url), public_key, msg.clone())
1724            {
1725                warn!(endpoint = %public_key.fmt_short(), "relay channel full, dropping call-me-maybe");
1726            }
1727        }
1728    }
1729
1730    fn handle_net_report_report(&mut self, mut report: Option<net_report::Report>) {
1731        if let Some(ref mut r) = report {
1732            self.msock.ipv6_reported.store(r.udp_v6, Ordering::Relaxed);
1733            if r.preferred_relay.is_none() {
1734                if let Some(my_relay) = self.msock.my_relay() {
1735                    r.preferred_relay.replace(my_relay);
1736                }
1737            }
1738
1739            // Notify all transports
1740            self.network_change_sender.on_network_change(r);
1741        }
1742
1743        #[cfg(not(wasm_browser))]
1744        self.update_direct_addresses(report.as_ref());
1745    }
1746}

1748fn new_re_stun_timer(initial_delay: bool) -> time::Interval {
1749    // Pick a random duration between 20 and 26 seconds (just under 30s,
1750    // a common UDP NAT timeout on Linux,etc)
1751    let mut rng = rand::rng();
1752    let d: Duration = rng.random_range(Duration::from_secs(20)..=Duration::from_secs(26));
1753    if initial_delay {
1754        debug!("scheduling periodic_stun to run in {}s", d.as_secs());
1755        time::interval_at(time::Instant::now() + d, d)
1756    } else {
1757        debug!(
1758            "scheduling periodic_stun to run immediately and in {}s",
1759            d.as_secs()
1760        );
1761        time::interval(d)
1762    }
1763}

1766fn bind_with_fallback(mut addr: SocketAddr) -> io::Result<UdpSocket> {
1767    debug!(%addr, "binding");
1768
1769    // First try binding a preferred port, if specified
1770    match UdpSocket::bind_full(addr) {
1771        Ok(socket) => {
1772            let local_addr = socket.local_addr()?;
1773            debug!(%addr, %local_addr, "successfully bound");
1774            return Ok(socket);
1775        }
1776        Err(err) => {
1777            debug!(%addr, "failed to bind: {err:#}");
1778            // If that was already the fallback port, then error out
1779            if addr.port() == 0 {
1780                return Err(err);
1781            }
1782        }
1783    }
1784
1785    // Otherwise, try binding with port 0
1786    addr.set_port(0);
1787    UdpSocket::bind_full(addr)
1788}

1845/// A *direct address* on which an iroh-endpoint might be contactable.
1846///
1847/// Direct addresses are UDP socket addresses on which an iroh endpoint could potentially be
1848/// contacted.  These can come from various sources depending on the network topology of the
1849/// iroh endpoint, see [`DirectAddrType`] for the several kinds of sources.
1850///
1851/// This is essentially a combination of our local addresses combined with any reflexive
1852/// transport addresses we discovered using QAD.
1853#[derive(Debug, Clone, PartialEq, Eq, Hash, PartialOrd, Ord)]
1854pub struct DirectAddr {
1855    /// The address.

1864/// possible [`DirectAddr`].

2065    /// Runs a roundtrip between the [`echo_sender`] and [`echo_receiver`].
2066    async fn run_roundtrip(
2067        sender: Endpoint,
2068        receiver: Endpoint,
2069        payload: &[u8],
2070        loss: ExpectedLoss,
2071    ) {
2072        let send_endpoint_id = sender.id();
2073        let recv_endpoint_id = receiver.id();
2074        info!("\nroundtrip: {send_endpoint_id:#} -> {recv_endpoint_id:#}");
2075
2076        let receiver_task = tokio::spawn(echo_receiver(receiver, loss));
2077        let sender_res = echo_sender(sender, recv_endpoint_id, payload, loss).await;
2078        let sender_is_err = match sender_res {
2079            Ok(()) => false,
2080            Err(err) => {
2081                eprintln!("[sender] Error:\n{err:#?}");
2082                true
2083            }
2084        };
2085        let receiver_is_err = match receiver_task.await {
2086            Ok(Ok(())) => false,
2087            Ok(Err(err)) => {
2088                eprintln!("[receiver] Error:\n{err:#?}");
2089                true
2090            }
2091            Err(joinerr) => {
2092                if joinerr.is_panic() {
2093                    std::panic::resume_unwind(joinerr.into_panic());
2094                } else {
2095                    eprintln!("[receiver] Error:\n{joinerr:#?}");
2096                }
2097                true
2098            }
2099        };
2100        if sender_is_err || receiver_is_err {
2101            panic!("Sender or receiver errored");
2102        }
2103    }
