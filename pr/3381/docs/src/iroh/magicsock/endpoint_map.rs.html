66/// The origin or *source* through which an address associated with a remote endpoint
67/// was discovered.
68///
69/// An aggregate of the [`Source`]s of all the addresses of an endpoint describe the
70/// [`Source`]s of the endpoint itself.
71///
72/// A [`Source`] helps track how and where an address was learned. Multiple
73/// sources can be associated with a single address, if we have discovered this
74/// address through multiple means.
75#[derive(Serialize, Deserialize, strum::Display, Debug, Clone, Eq, PartialEq, Hash)]
76#[strum(serialize_all = "kebab-case")]
77pub enum Source {
78    /// Address was loaded from the fs.
79    Saved,
80    /// An endpoint communicated with us first via UDP.
81    Udp,
82    /// An endpoint communicated with us first via relay.
83    Relay,
84    /// Application layer added the address directly.
85    App,
86    /// The address was discovered by a discovery service.
87    #[strum(serialize = "{name}")]
88    Discovery {
89        /// The name of the discovery service that discovered the address.
90        name: String,
91    },
92    /// Application layer with a specific name added the endpoint directly.
93    #[strum(serialize = "{name}")]
94    NamedApp {
95        /// The name of the application that added the endpoint
96        name: String,
97    },
98    /// The address was advertised by a call-me-maybe DISCO message.
99    CallMeMaybe,
100    /// We received a ping on the path.
101    Ping,
102    /// We established a connection on this address.
103    ///
104    /// Currently this means the path was in uses as [`PathId::ZERO`] when the a connection
105    /// was added to the `EndpointStateActor`.
106    ///
107    /// [`PathId::ZERO`]: quinn_proto::PathId::ZERO
108    Connection,
109}

139    /// Adds addresses where a endpoint might be contactable.
140    pub(super) async fn add_endpoint_addr(&self, endpoint_addr: EndpointAddr, source: Source) {
141        for url in endpoint_addr.relay_urls() {
142            // Ensure we have a RelayMappedAddress.
143            self.relay_mapped_addrs
144                .get(&(url.clone(), endpoint_addr.id));
145        }
146        let actor = self.endpoint_state_actor(endpoint_addr.id);
147
148        // This only fails if the sender is closed.  That means the EndpointStateActor has
149        // stopped, which only happens during shutdown.
150        actor
151            .send(EndpointStateMessage::AddEndpointAddr(endpoint_addr, source))
152            .await
153            .ok();
154    }

156    pub(super) fn endpoint_mapped_addr(&self, eid: EndpointId) -> EndpointIdMappedAddr {

158    }

160    /// Converts a mapped address as we use them inside Quinn.
161    pub(crate) fn transport_addr_from_mapped(&self, mapped: SocketAddr) -> Option<TransportAddr> {
162        match MultipathMappedAddr::from(mapped) {
163            MultipathMappedAddr::Mixed(_) => None,
164            MultipathMappedAddr::Relay(addr) => match self.relay_mapped_addrs.lookup(&addr) {
165                Some((url, _)) => Some(TransportAddr::Relay(url)),
166                None => {
167                    error!("Unknown RelayMappedAddr");
168                    None
169                }
170            },
171            MultipathMappedAddr::Ip(addr) => Some(TransportAddr::Ip(addr)),
172        }
173    }

188    /// Returns the sender for the [`EndpointStateActor`].
189    ///
190    /// If needed a new actor is started on demand.
191    ///
192    /// [`EndpointStateActor`]: endpoint_state::EndpointStateActor
193    pub(super) fn endpoint_state_actor(
194        &self,
195        eid: EndpointId,
196    ) -> mpsc::Sender<EndpointStateMessage> {
197        let mut inner = self.inner.lock().expect("poisoned");
198        match inner.endpoint_states.get(&eid) {
199            Some(handle) => handle.sender.clone(),
200            None => {
201                // Create a new EndpointStateActor and insert it into the endpoint map.
202                let sender = inner.transports_handle.inbox.clone();
203                let local_addrs = inner.local_addrs.clone();
204                let disco = inner.disco.clone();
205                let metrics = inner.metrics.clone();
206                let actor = EndpointStateActor::new(
207                    eid,
208                    self.local_endpoint_id,
209                    sender,
210                    local_addrs,
211                    disco,
212                    self.relay_mapped_addrs.clone(),
213                    metrics,
214                );
215                let handle = actor.start();
216                let sender = handle.sender.clone();
217                inner.endpoint_states.insert(eid, handle);
218
219                // Ensure there is a EndpointMappedAddr for this EndpointId.
220                self.endpoint_mapped_addrs.get(&eid);
221                sender
222            }
223        }
224    }

232        if let Err(err) = endpoint_state.try_send(EndpointStateMessage::PingReceived(msg, src)) {
233            // TODO: This is really, really bad and will drop pings under load.  But
234            //    DISCO pings are going away with QUIC-NAT-TRAVERSAL so I don't care.
235            warn!("DISCO Ping dropped: {err:#}");
236        }
237    }

263        }

342/// An actor that can send datagrams onto iroh transports.
343///
344/// The [`EndpointStateActor`]s want to be able to send datagrams.  Because we can not create
345/// [`TransportsSender`]s on demand we must share one for the entire [`EndpointMap`], which
346/// lives in this actor.
347///
348/// [`EndpointStateActor`]: endpoint_state::EndpointStateActor
349#[derive(Debug)]
350struct TransportsSenderActor {
351    sender: TransportsSender,
352}
353
354impl TransportsSenderActor {
355    fn new(sender: TransportsSender) -> Self {
356        Self { sender }
357    }
358
359    fn start(self) -> TransportsSenderHandle {
360        // This actor gets an inbox size of exactly 1.  This is the same as if they had the
361        // underlying sender directly: either you can send or not, or you await until you
362        // can.  No need to introduce extra buffering.
363        let (tx, rx) = mpsc::channel(1);
364
365        let task = task::spawn(
366            async move {
367                self.run(rx).await;
368            }
369            .instrument(info_span!("TransportsSenderActor")),
370        );
371        TransportsSenderHandle {
372            inbox: tx,
373            _task: AbortOnDropHandle::new(task),
374        }
375    }
376
377    async fn run(self, mut inbox: mpsc::Receiver<TransportsSenderMessage>) {
378        use TransportsSenderMessage::SendDatagram;
379
380        while let Some(SendDatagram(dst, owned_transmit)) = inbox.recv().await {
381            let transmit = transports::Transmit {
382                ecn: owned_transmit.ecn,
383                contents: owned_transmit.contents.as_ref(),
384                segment_size: owned_transmit.segment_size,
385            };
386            let len = transmit.contents.len();
387            match self.sender.send(&dst, None, &transmit).await {
388                Ok(()) => {}
389                Err(err) => {
390                    trace!(?dst, %len, "transmit failed to send: {err:#}");
391                }
392            };
393        }
394        trace!("actor terminating");
395    }
396}
397
398#[derive(Debug)]
399struct TransportsSenderHandle {
400    inbox: mpsc::Sender<TransportsSenderMessage>,
401    _task: AbortOnDropHandle<()>,
402}
403
404#[derive(Debug)]
405enum TransportsSenderMessage {
406    SendDatagram(transports::Addr, OwnedTransmit),
407}
408
409impl From<(transports::Addr, OwnedTransmit)> for TransportsSenderMessage {
410    fn from(source: (transports::Addr, OwnedTransmit)) -> Self {
411        Self::SendDatagram(source.0, source.1)
412    }
413}
