36/// Map containing all the state for endpoints.
37///
38/// - Has actors which each manage all the connection state for a remote endpoint.
39///
40/// - Has the mapped addresses we use to refer to non-IP transports destinations into IPv6
41///   addressing space that is used by Quinn.
42#[derive(Debug)]
43pub(crate) struct EndpointMap {
44    //
45    // State we keep about remote endpoints.
46    //
47    /// The actors tracking each remote endpoint.
48    actor_handles: Mutex<FxHashMap<EndpointId, EndpointStateHandle>>,
49    /// The mapping between [`EndpointId`]s and [`EndpointIdMappedAddr`]s.
50    pub(super) endpoint_mapped_addrs: AddrMap<EndpointId, EndpointIdMappedAddr>,
51    /// The mapping between endpoints via a relay and their [`RelayMappedAddr`]s.
52    pub(super) relay_mapped_addrs: AddrMap<(RelayUrl, EndpointId), RelayMappedAddr>,
53
54    //
55    // State needed to start a new EndpointStateHandle.
56    //
57    /// The endpoint ID of the local endpoint.
58    local_endpoint_id: EndpointId,
59    metrics: Arc<MagicsockMetrics>,
60    local_addrs: n0_watcher::Direct<BTreeSet<DirectAddr>>,
61    disco: DiscoState,
62    sender: TransportsSender,
63}

89    /// Adds addresses where an endpoint might be contactable.
90    pub(super) async fn add_endpoint_addr(&self, endpoint_addr: EndpointAddr, source: Source) {
91        for url in endpoint_addr.relay_urls() {
92            // Ensure we have a RelayMappedAddress.
93            self.relay_mapped_addrs
94                .get(&(url.clone(), endpoint_addr.id));
95        }
96        let actor = self.endpoint_state_actor(endpoint_addr.id);
97
98        // This only fails if the sender is closed.  That means the EndpointStateActor has
99        // stopped, which only happens during shutdown.
100        actor
101            .send(EndpointStateMessage::AddEndpointAddr(endpoint_addr, source))
102            .await
103            .ok();
104    }

123    /// Returns the sender for the [`EndpointStateActor`].
124    ///
125    /// If needed a new actor is started on demand.
126    ///
127    /// [`EndpointStateActor`]: endpoint_state::EndpointStateActor
128    pub(super) fn endpoint_state_actor(
129        &self,
130        eid: EndpointId,
131    ) -> mpsc::Sender<EndpointStateMessage> {
132        let mut handles = self.actor_handles.lock().expect("poisoned");
133        match handles.get(&eid) {
134            Some(handle) => handle.sender.clone(),
135            None => {
136                // Create a new EndpointStateActor and insert it into the endpoint map.
137                let local_addrs = self.local_addrs.clone();
138                let disco = self.disco.clone();
139                let metrics = self.metrics.clone();
140                let actor = EndpointStateActor::new(
141                    eid,
142                    self.local_endpoint_id,
143                    local_addrs,
144                    disco,
145                    self.relay_mapped_addrs.clone(),
146                    metrics,
147                    self.sender.clone(),
148                );
149                let handle = actor.start();
150                let sender = handle.sender.clone();
151                handles.insert(eid, handle);
152
153                // Ensure there is a EndpointMappedAddr for this EndpointId.
154                self.endpoint_mapped_addrs.get(&eid);
155                sender
156            }
157        }
158    }

160    pub(super) fn handle_ping(&self, msg: disco::Ping, sender: EndpointId, src: transports::Addr) {
161        if msg.endpoint_key != sender {
162            warn!("DISCO Ping EndpointId mismatch, ignoring ping");
163            return;
164        }
165        let endpoint_state = self.endpoint_state_actor(sender);
166        if let Err(err) = endpoint_state.try_send(EndpointStateMessage::PingReceived(msg, src)) {
167            // TODO: This is really, really bad and will drop pings under load.  But
168            //    DISCO pings are going away with QUIC-NAT-TRAVERSAL so I don't care.
169            warn!("DISCO Ping dropped: {err:#}");
170        }
171    }

182    pub(super) fn handle_call_me_maybe(
183        &self,
184        msg: disco::CallMeMaybe,
185        sender: EndpointId,
186        src: transports::Addr,
187    ) {
188        if !src.is_relay() {
189            warn!("DISCO CallMeMaybe packets should only come via relay");
190            return;
191        }
192        let actor = self.endpoint_state_actor(sender);
193        if let Err(err) = actor.try_send(EndpointStateMessage::CallMeMaybeReceived(msg)) {
194            // TODO: This is bad and will drop call-me-maybe's under load.  But
195            //    DISCO CallMeMaybe going away with QUIC-NAT-TRAVERSAL so I don't care.
196            warn!("DISCO CallMeMaybe dropped: {err:#}");
197        }
198    }

201/// The origin or *source* through which an address associated with a remote endpoint
202/// was discovered.
203///
204/// An aggregate of the [`Source`]s of all the addresses of an endpoint describe the
205/// [`Source`]s of the endpoint itself.
206///
207/// A [`Source`] helps track how and where an address was learned. Multiple
208/// sources can be associated with a single address, if we have discovered this
209/// address through multiple means.
210#[derive(Serialize, Deserialize, strum::Display, Debug, Clone, Eq, PartialEq, Hash)]
211#[strum(serialize_all = "kebab-case")]
212pub enum Source {
213    /// Address was loaded from the fs.
214    Saved,
215    /// An endpoint communicated with us first via UDP.
216    Udp,
217    /// An endpoint communicated with us first via relay.
218    Relay,
219    /// Application layer added the address directly.
220    App,
221    /// The address was discovered by a discovery service.
222    #[strum(serialize = "{name}")]
223    Discovery {
224        /// The name of the discovery service that discovered the address.
225        name: String,
226    },
227    /// Application layer with a specific name added the endpoint directly.
228    #[strum(serialize = "{name}")]
229    NamedApp {
230        /// The name of the application that added the endpoint
231        name: String,
232    },
233    /// The address was advertised by a call-me-maybe DISCO message.
234    CallMeMaybe,
235    /// We received a ping on the path.
236    Ping,
237    /// We established a connection on this address.
238    ///
239    /// Currently this means the path was in uses as [`PathId::ZERO`] when the a connection
240    /// was added to the `EndpointStateActor`.
241    ///
242    /// [`PathId::ZERO`]: quinn_proto::PathId::ZERO
243    Connection,
244}
245
246/// An (Ip, Port) pair.
247///
248/// NOTE: storing an [`IpPort`] is safer than storing a [`SocketAddr`] because for IPv6 socket
249/// addresses include fields that can't be assumed consistent even within a single connection.
250#[derive(Debug, derive_more::Display, Clone, Copy, Hash, PartialEq, Eq, PartialOrd, Ord)]
251#[display("{}", SocketAddr::from(*self))]
252pub struct IpPort {
253    ip: IpAddr,
254    port: u16,
255}
256
257impl From<SocketAddr> for IpPort {
258    fn from(socket_addr: SocketAddr) -> Self {
259        Self {
260            ip: socket_addr.ip(),
261            port: socket_addr.port(),
262        }
263    }
264}
265
266impl From<IpPort> for SocketAddr {
267    fn from(ip_port: IpPort) -> Self {
268        let IpPort { ip, port } = ip_port;
269        (ip, port).into()
270    }
271}
272
273impl IpPort {
274    pub fn ip(&self) -> &IpAddr {
275        &self.ip
276    }
277
278    pub fn port(&self) -> u16 {
279        self.port
280    }
