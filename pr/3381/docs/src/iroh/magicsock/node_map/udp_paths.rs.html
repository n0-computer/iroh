3//! This started as simply moving the [`NodeState`]'s `direct_addresses` and `best_addr`

19/// The address on which to send datagrams over UDP.
20///
21/// The [`MagicSock`] sends packets to zero or one UDP address, depending on the known paths
22/// to the remote node.  This conveys the UDP address to send on from the [`NodeUdpPaths`]
23/// to the [`NodeState`].
24///
25/// [`NodeUdpPaths`] contains all the UDP path states, while [`NodeState`] has to decide the
26/// bigger picture including the relay server.
27///
28/// See [`NodeUdpPaths::send_addr`].
29///
30/// [`MagicSock`]: crate::magicsock::MagicSock
31/// [`NodeState`]: super::node_state::NodeState
32#[derive(Debug)]
33pub(super) enum UdpSendAddr {
34    /// The UDP address can be relied on to deliver data to the remote node.
35    ///
36    /// This means this path is usable with a reasonable latency and can be fully trusted to
37    /// transport payload data to the remote node.
38    Valid(SocketAddr),
39    /// The UDP address is highly likely to work, but has not been used for a while.
40    ///
41    /// The path should be usable but has not carried DISCO or payload data for a little too
42    /// long.  It is best to also use a backup, i.e. relay, path if possible.
43    Outdated(SocketAddr),
44    /// The UDP address is not known to work, but it might.
45    ///
46    /// We know this UDP address belongs to the remote node, but we do not know if the path
47    /// already works or may need holepunching before it will start to work.  It might even
48    /// never work.  It is still useful to send to this together with backup path,
49    /// i.e. relay, in case the path works: if the path does not need holepunching it might
50    /// be much faster.  And if there is no relay path at all it might be the only way to
51    /// establish a connection.
52    Unconfirmed(SocketAddr),
53    /// No known UDP path exists to the remote node.
54    None,
55}

57/// The UDP paths for a single node.
58///
59/// Paths are identified by the [`IpPort`] of their UDP address.
60///
61/// Initially this collects two structs directly from the [`NodeState`] into one place,
62/// leaving the APIs and astractions the same.  The goal is that this slowly migrates
63/// directly interacting with this data into only receiving [`PathState`] updates.  This
64/// will consolidate the logic of direct path selection and make this simpler to reason
65/// about.  However doing that all at once is too large a refactor.
66///
67/// [`NodeState`]: super::node_state::NodeState
68#[derive(Debug, Default)]
69pub(super) struct NodeUdpPaths {
70    /// The state for each of this node's direct paths.
71    pub(super) paths: BTreeMap<IpPort, PathState>,
72    /// Best UDP path currently selected.
73    pub(super) best_addr: BestAddr,
74    /// If we had to choose a path because we had no `best_addr` it is stored here.
75    chosen_candidate: Option<IpPort>,
76}

92    /// Returns the current UDP address to send on.
93    ///
94    /// TODO: The goal here is for this to simply return the already known send address, so
95    /// it should be `&self` and not `&mut self`.  This is only possible once the state from
96    /// [`NodeUdpPaths`] is no longer modified from outside.
97    pub(super) fn send_addr(&mut self, now: Instant, have_ipv6: bool) -> UdpSendAddr {
98        self.assign_best_addr_from_candidates_if_empty();
99        match self.best_addr.state(now) {
100            best_addr::State::Valid(addr) => UdpSendAddr::Valid(addr.addr),
101            best_addr::State::Outdated(addr) => UdpSendAddr::Outdated(addr.addr),
102            best_addr::State::Empty => {
103                // No direct connection has been used before.  If we know of any possible
104                // candidate addresses, randomly try to use one.  This path is most
105                // effective when folks use a NodeAddr with exactly one direct address which
106                // they know to work, effectively like using a traditional socket or QUIC
107                // endpoint.
108                let addr = self
109                    .chosen_candidate
110                    .and_then(|ipp| self.paths.get(&ipp))
111                    .and_then(|path| path.udp_addr())
112                    .filter(|addr| addr.is_ipv4() || have_ipv6)
113                    .or_else(|| {
114                        // Look for a new candidate in all the known paths.  This may look
115                        // like a RNG use on the hot-path but this is normally invoked at
116                        // most most once at startup.
117                        let addr = self
118                            .paths
119                            .values()
120                            .filter_map(|path| path.udp_addr())
121                            .filter(|addr| addr.is_ipv4() || have_ipv6)
122                            .choose(&mut rand::thread_rng());
123                        self.chosen_candidate = addr.map(IpPort::from);
124                        addr
125                    });
126                match addr {
127                    Some(addr) => UdpSendAddr::Unconfirmed(addr),
128                    None => UdpSendAddr::None,
129                }
130            }
131        }
132    }
