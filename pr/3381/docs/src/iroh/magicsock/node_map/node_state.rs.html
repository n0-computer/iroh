51/// An iroh node, which we can have connections with.
52///
53/// The whole point of the magicsock is that we can have multiple **paths** to a particular
54/// node.  One of these paths is via the endpoint's home relay node but as we establish a
55/// connection we'll hopefully discover more direct paths.
56#[derive(Debug)]
57pub(super) struct NodeState {
58    /// The ID used as index in the [`NodeMap`].
59    ///
60    /// [`NodeMap`]: super::NodeMap
61    id: usize,
62    /// The UDP address used on the QUIC-layer to address this node.
63    quic_mapped_addr: NodeIdMappedAddr,
64    /// The global identifier for this endpoint.
65    node_id: NodeId,
66    /// The url of relay node that we can relay over to communicate.
67    ///
68    /// The fallback/bootstrap path, if non-zero (non-zero for well-behaved clients).
69    relay_url: Option<(RelayUrl, PathState)>,
70    udp_paths: NodeUdpPaths,
71    /// Last time this node was used.
72    ///
73    /// A node is marked as in use when sending datagrams to them, or when having received
74    /// datagrams from it. Regardless of whether the datagrams are payload or DISCO, and whether
75    /// they go via UDP or the relay.
76    ///
77    /// Note that sending datagrams to a node does not mean the node receives them.
78    last_used: Option<Instant>,
79    /// Last time we sent a call-me-maybe.
80    ///
81    /// When we do not have a direct connection and we try to send some data, we will try to
82    /// do a full ping + call-me-maybe.  Usually each side only needs to send one
83    /// call-me-maybe to the other for holes to be punched in both directions however.  So
84    /// we only try and send one per [`HEARTBEAT_INTERVAL`].  Each [`HEARTBEAT_INTERVAL`]
85    /// the [`NodeState::stayin_alive`] function is called, which will trigger new
86    /// call-me-maybe messages as backup.
87    last_call_me_maybe: Option<Instant>,
88    /// The type of connection we have to the node, either direct, relay, mixed, or none.
89    conn_type: Watchable<ConnectionType>,
90    /// Whether the conn_type was ever observed to be `Direct` at some point.
91    ///
92    /// Used for metric reporting.
93    has_been_direct: bool,
94    /// Configuration for what path selection to use
95    #[cfg(any(test, feature = "test-utils"))]
96    path_selection: PathSelection,
97}

203    /// Returns the address(es) that should be used for sending the next packet.
204    ///
205    /// This may return to send on one, both or no paths.
206    fn addr_for_send(
207        &mut self,
208        now: &Instant,
209        have_ipv6: bool,
210        metrics: &MagicsockMetrics,
211    ) -> (Option<SocketAddr>, Option<RelayUrl>) {
212        #[cfg(any(test, feature = "test-utils"))]
213        if self.path_selection == PathSelection::RelayOnly {
214            debug!(
215                "in `RelayOnly` mode, giving the relay address as the only viable address for this endpoint"
216            );
217            return (None, self.relay_url());
218        }
219        let (best_addr, relay_url) = match self.udp_paths.send_addr(*now, have_ipv6) {
220            UdpSendAddr::Valid(addr) => {
221                // If we have a valid address we use it.
222                trace!(%addr, "UdpSendAddr is valid, use it");
223                (Some(addr), None)
224            }
225            UdpSendAddr::Outdated(addr) => {
226                // If the address is outdated we use it, but send via relay at the same time.
227                // We also send disco pings so that it will become valid again if it still
228                // works (i.e. we don't need to holepunch again).
229                trace!(%addr, "UdpSendAddr is outdated, use it together with relay");
230                (Some(addr), self.relay_url())
231            }
232            UdpSendAddr::Unconfirmed(addr) => {
233                trace!(%addr, "UdpSendAddr is unconfirmed, use it together with relay");
234                (Some(addr), self.relay_url())
235            }
236            UdpSendAddr::None => {
237                trace!("No UdpSendAddr, use relay");
238                (None, self.relay_url())
239            }
240        };
241        let typ = match (best_addr, relay_url.clone()) {
242            (Some(best_addr), Some(relay_url)) => ConnectionType::Mixed(best_addr, relay_url),
243            (Some(best_addr), None) => ConnectionType::Direct(best_addr),
244            (None, Some(relay_url)) => ConnectionType::Relay(relay_url),
245            (None, None) => ConnectionType::None,
246        };
247        if !self.has_been_direct && matches!(&typ, ConnectionType::Direct(_)) {
248            self.has_been_direct = true;
249            metrics.nodes_contacted_directly.inc();
250        }
251        if let Ok(prev_typ) = self.conn_type.set(typ.clone()) {
252            // The connection type has changed.
253            event!(
254                target: "iroh::_events::conn_type::changed",
255                Level::DEBUG,
256                remote_node = %self.node_id.fmt_short(),
257                conn_type = ?typ,
258            );
259            info!(%typ, "new connection type");
260
261            // Update some metrics
262            match (prev_typ, typ) {
263                (ConnectionType::Relay(_), ConnectionType::Direct(_))
264                | (ConnectionType::Mixed(_, _), ConnectionType::Direct(_)) => {
265                    metrics.num_direct_conns_added.inc();
266                    metrics.num_relay_conns_removed.inc();
267                }
268                (ConnectionType::Direct(_), ConnectionType::Relay(_))
269                | (ConnectionType::Direct(_), ConnectionType::Mixed(_, _)) => {
270                    metrics.num_direct_conns_removed.inc();
271                    metrics.num_relay_conns_added.inc();
272                }
273                (ConnectionType::None, ConnectionType::Direct(_)) => {
274                    metrics.num_direct_conns_added.inc();
275                }
276                (ConnectionType::Direct(_), ConnectionType::None) => {
277                    metrics.num_direct_conns_removed.inc();
278                }
279                (ConnectionType::None, ConnectionType::Relay(_))
280                | (ConnectionType::None, ConnectionType::Mixed(_, _)) => {
281                    metrics.num_relay_conns_added.inc();
282                }
283                (ConnectionType::Relay(_), ConnectionType::None)
284                | (ConnectionType::Mixed(_, _), ConnectionType::None) => {
285                    metrics.num_relay_conns_removed.inc();
286                }
287                _ => (),
288            }
289        }
290        (best_addr, relay_url)
291    }

293    /// Removes a direct address for this node.
294    ///
295    /// If this is also the best address, it will be cleared as well.
296    pub(super) fn remove_direct_addr(&mut self, ip_port: &IpPort, reason: ClearReason) {
297        let Some(state) = self.udp_paths.paths.remove(ip_port) else {
298            return;
299        };
300
301        match state.last_alive().map(|instant| instant.elapsed()) {
302            Some(last_alive) => debug!(%ip_port, ?last_alive, ?reason, "pruning address"),
303            None => debug!(%ip_port, last_seen=%"never", ?reason, "pruning address"),
304        }
305
306        self.udp_paths.best_addr.clear_if_equals(
307            (*ip_port).into(),
308            reason,
309            self.relay_url.is_some(),
310        );
311    }

313    /// Whether we need to send another call-me-maybe to the endpoint.
314    ///
315    /// Basically we need to send a call-me-maybe if we need to find a better path.  Maybe
316    /// we only have a relay path, or our path is expired.
317    ///
318    /// When a call-me-maybe message is sent we also need to send pings to all known paths
319    /// of the endpoint.  The [`NodeState::send_call_me_maybe`] function takes care of this.
320    #[cfg(not(wasm_browser))]
321    #[instrument("want_call_me_maybe", skip_all)]
322    fn want_call_me_maybe(&self, now: &Instant) -> bool {
323        trace!("full ping: wanted?");
324        match self.udp_paths.best_addr.state(*now) {
325            best_addr::State::Empty => {
326                debug!("best addr not set: need full ping");
327                true
328            }
329            best_addr::State::Outdated(_) => {
330                debug!("best addr expired: need full ping");
331                true
332            }
333            best_addr::State::Valid(addr) => {
334                if addr.latency > GOOD_ENOUGH_LATENCY {
335                    debug!(
336                        "full ping interval expired and latency is only {}ms: need full ping",
337                        addr.latency.as_millis()
338                    );
339                    true
340                } else {
341                    trace!(?now, "best_addr valid: not needed");
342                    false
343                }
344            }
345        }
346    }

349    fn want_call_me_maybe(&self, _now: &Instant) -> bool {
350        trace!("full ping: skipped in browser");
351        false

354    /// Send a DISCO call-me-maybe message to the peer.
355    ///
356    /// This takes care of sending the needed pings beforehand.  This ensures that we open
357    /// our firewall's port so that when the receiver sends us DISCO pings in response to
358    /// our call-me-maybe they will reach us and the other side establishes a direct
359    /// connection upon our subsequent pong response.
360    ///
361    /// For [`SendCallMeMaybe::IfNoRecent`], **no** paths will be pinged if there already
362    /// was a recent call-me-maybe sent.
363    ///
364    /// The caller is responsible for sending the messages.
365    #[must_use = "actions must be handled"]
366    fn send_call_me_maybe(&mut self, now: Instant, always: SendCallMeMaybe) -> Vec<PingAction> {
367        match always {
368            SendCallMeMaybe::Always => (),
369            SendCallMeMaybe::IfNoRecent => {
370                let had_recent_call_me_maybe = self
371                    .last_call_me_maybe
372                    .map(|when| when.elapsed() < HEARTBEAT_INTERVAL)
373                    .unwrap_or(false);
374                if had_recent_call_me_maybe {
375                    trace!("skipping call-me-maybe, still recent");
376                    return Vec::new();
377                }
378            }
379        }
380        let mut msgs = Vec::new();
381
382        if let Some(url) = self.relay_url() {
383            debug!(%url, "queue call-me-maybe");
384            msgs.push(PingAction::SendCallMeMaybe {
385                relay_url: url,
386                dst_node: self.node_id,
387            });
388            self.last_call_me_maybe = Some(now);
389        } else {
390            debug!("can not send call-me-maybe, no relay URL");
391        }
392
393        msgs
394    }

396    pub(super) fn update_from_node_addr(
397        &mut self,
398        new_relay_url: Option<&RelayUrl>,
399        new_addrs: &BTreeSet<SocketAddr>,
400        source: super::Source,
401        metrics: &MagicsockMetrics,
402    ) {
403        if self.udp_paths.best_addr.is_empty() {
404            // we do not have a direct connection, so changing the relay information may
405            // have an effect on our connection status
406            if self.relay_url.is_none() && new_relay_url.is_some() {
407                // we did not have a relay connection before, but now we do
408                metrics.num_relay_conns_added.inc();
409            } else if self.relay_url.is_some() && new_relay_url.is_none() {
410                // we had a relay connection before but do not have one now
411                metrics.num_relay_conns_removed.inc();
412            }
413        }
414
415        let now = Instant::now();
416
417        if new_relay_url.is_some() && new_relay_url != self.relay_url().as_ref() {
418            debug!(
419                "Changing relay node from {:?} to {:?}",
420                self.relay_url, new_relay_url
421            );
422            self.relay_url = new_relay_url.map(|url| {
423                (
424                    url.clone(),
425                    PathState::new(self.node_id, url.clone().into(), source.clone(), now),
426                )
427            });
428        }
429
430        for &addr in new_addrs.iter() {
431            self.udp_paths
432                .paths
433                .entry(addr.into())
434                .and_modify(|path_state| {
435                    path_state.add_source(source.clone(), now);
436                })
437                .or_insert_with(|| {
438                    PathState::new(self.node_id, SendAddr::from(addr), source.clone(), now)
439                });
440        }
441        let paths = summarize_node_paths(&self.udp_paths.paths);
442        debug!(new = ?new_addrs , %paths, "added new direct paths for endpoint");
443    }

445    /// Prune inactive paths.
446    ///
447    /// This trims the list of inactive paths for an endpoint.  At most
448    /// [`MAX_INACTIVE_DIRECT_ADDRESSES`] are kept.
449    pub(super) fn prune_direct_addresses(&mut self) {
450        // prune candidates are addresses that are not active
451        let mut prune_candidates: Vec<_> = self
452            .udp_paths
453            .paths
454            .iter()
455            .filter(|(_ip_port, state)| !state.is_active())
456            .map(|(ip_port, state)| (*ip_port, state.last_alive()))
457            .filter(|(_ipp, last_alive)| match last_alive {
458                Some(last_seen) => last_seen.elapsed() > LAST_ALIVE_PRUNE_DURATION,
459                None => true,
460            })
461            .collect();
462        let prune_count = prune_candidates
463            .len()
464            .saturating_sub(MAX_INACTIVE_DIRECT_ADDRESSES);
465        if prune_count == 0 {
466            // nothing to do, within limits
467            debug!(
468                paths = %summarize_node_paths(&self.udp_paths.paths),
469                "prune addresses: {prune_count} pruned",
470            );
471            return;
472        }
473
474        // sort leaving the worst addresses first (never contacted) and better ones (most recently
475        // used ones) last
476        prune_candidates.sort_unstable_by_key(|(_ip_port, last_alive)| *last_alive);
477        prune_candidates.truncate(prune_count);
478        for (ip_port, _last_alive) in prune_candidates.into_iter() {
479            self.remove_direct_addr(&ip_port, ClearReason::Inactive)
480        }
481        debug!(
482            paths = %summarize_node_paths(&self.udp_paths.paths),
483            "prune addresses: {prune_count} pruned",
484        );
485    }

497    /// Handles a DISCO CallMeMaybe discovery message.
498    ///
499    /// The contract for use of this message is that the node has already pinged to us via
500    /// UDP, so their stateful firewall should be open. Now we can Ping back and make it
501    /// through.
502    ///
503    /// However if the remote side has no direct path information to us, they would not have
504    /// had any [`IpPort`]s to send pings to and our pings might end up blocked.  But at
505    /// least open the firewalls on our side, giving the other side another change of making
506    /// it through when it pings in response.
507    pub(super) fn handle_call_me_maybe(&mut self, m: disco::CallMeMaybe) {
508        let now = Instant::now();
509        let mut call_me_maybe_ipps = BTreeSet::new();
510
511        for peer_sockaddr in &m.my_numbers {
512            if let IpAddr::V6(ip) = peer_sockaddr.ip() {
513                if netwatch::ip::is_unicast_link_local(ip) {
514                    // We send these out, but ignore them for now.
515                    // TODO: teach the ping code to ping on all interfaces for these.
516                    continue;
517                }
518            }
519            let ipp = IpPort::from(*peer_sockaddr);
520            call_me_maybe_ipps.insert(ipp);
521            self.udp_paths
522                .paths
523                .entry(ipp)
524                .or_insert_with(|| {
525                    PathState::new(
526                        self.node_id,
527                        SendAddr::from(*peer_sockaddr),
528                        Source::Relay,
529                        now,
530                    )
531                })
532                .call_me_maybe_time
533                .replace(now);
534        }
535
536        // Zero out all the last_ping times to force send_pings to send new ones, even if
537        // it's been less than 5 seconds ago.  Also clear pongs for direct addresses not
538        // included in the updated set.
539        for (_ipp, st) in self.udp_paths.paths.iter_mut() {
540            st.last_ping = None;
541        }
542        // Clear trust on our best_addr if it is not included in the updated set.  Also
543        // clear the last call-me-maybe send time so we will send one again.
544        if let Some(addr) = self.udp_paths.best_addr.addr() {
545            let ipp: IpPort = addr.into();
546            if !call_me_maybe_ipps.contains(&ipp) {
547                self.udp_paths
548                    .best_addr
549                    .clear_trust("best_addr not in new call-me-maybe");
550                self.last_call_me_maybe = None;
551            }
552        }
553        debug!(
554            paths = %summarize_node_paths(&self.udp_paths.paths),
555            "updated endpoint paths from call-me-maybe",
556        );
557    }

559    /// Marks this node as having received a UDP payload message.

561    pub(super) fn receive_udp(&mut self, addr: IpPort, now: Instant) {
562        let Some(state) = self.udp_paths.paths.get_mut(&addr) else {
563            debug_assert!(false, "node map inconsistency by_ip_port <-> direct addr");
564            return;
565        };
566        state.last_payload_msg = Some(now);
567        self.last_used = Some(now);

570            .reconfirm_if_used(addr.into(), BestAddrSource::Udp, now);

573    pub(super) fn receive_relay(&mut self, url: &RelayUrl, src: NodeId, now: Instant) {
574        match self.relay_url.as_mut() {
575            Some((current_home, state)) if current_home == url => {
576                // We received on the expected url. update state.
577                state.last_payload_msg = Some(now);
578            }
579            Some((_current_home, _state)) => {
580                // we have a different url. we only update on ping, not on receive_relay.
581            }
582            None => {
583                self.relay_url = Some((
584                    url.clone(),
585                    PathState::with_last_payload(
586                        src,
587                        SendAddr::from(url.clone()),
588                        Source::Relay,
589                        now,
590                    ),
591                ));
592            }
593        }
594        self.last_used = Some(now);
595    }

597    pub(super) fn last_ping(&self, addr: &SendAddr) -> Option<Instant> {
598        match addr {
599            SendAddr::Udp(addr) => self
600                .udp_paths
601                .paths
602                .get(&(*addr).into())
603                .and_then(|ep| ep.last_ping),
604            SendAddr::Relay(url) => self
605                .relay_url
606                .as_ref()
607                .filter(|(home_url, _state)| home_url == url)
608                .and_then(|(_home_url, state)| state.last_ping),

612    /// Checks if this `Endpoint` is currently actively being used.
613    pub(super) fn is_active(&self, now: &Instant) -> bool {
614        match self.last_used {
615            Some(last_active) => now.duration_since(last_active) <= SESSION_ACTIVE_TIMEOUT,
616            None => false,
617        }
618    }
619
620    /// Send a heartbeat to the node to keep the connection alive, or trigger a full ping
621    /// if necessary.
622    #[instrument("stayin_alive", skip_all, fields(node = %self.node_id.fmt_short()))]
623    pub(super) fn stayin_alive(&mut self) -> Vec<PingAction> {
624        trace!("stayin_alive");
625        let now = Instant::now();
626        if !self.is_active(&now) {
627            trace!("skipping stayin alive: session is inactive");
628            return Vec::new();
629        }
630
631        // If we do not have an optimal addr, send pings to all known places.
632        if self.want_call_me_maybe(&now) {
633            debug!("sending a call-me-maybe");
634            return self.send_call_me_maybe(now, SendCallMeMaybe::Always);
635        }
636
637        Vec::new()
638    }
639
640    /// Returns the addresses on which a payload should be sent right now.
641    ///
642    /// This is in the hot path of `.poll_send()`.
643    #[instrument("get_send_addrs", skip_all, fields(node = %self.node_id.fmt_short()))]
644    pub(crate) fn get_send_addrs(
645        &mut self,
646        have_ipv6: bool,
647        metrics: &MagicsockMetrics,
648    ) -> (Option<SocketAddr>, Option<RelayUrl>, Vec<PingAction>) {
649        let now = Instant::now();
650        let prev = self.last_used.replace(now);
651        if prev.is_none() {
652            // this is the first time we are trying to connect to this node
653            metrics.nodes_contacted.inc();
654        }
655        let (udp_addr, relay_url) = self.addr_for_send(&now, have_ipv6, metrics);
656
657        let ping_msgs = if self.want_call_me_maybe(&now) {
658            self.send_call_me_maybe(now, SendCallMeMaybe::IfNoRecent)
659        } else {
660            Vec::new()
661        };
662        trace!(
663            ?udp_addr,
664            ?relay_url,
665            pings = %ping_msgs.len(),
666            "found send address",
667        );
668        (udp_addr, relay_url, ping_msgs)
669    }

671    /// Get the direct addresses for this endpoint.
672    pub(super) fn direct_addresses(&self) -> impl Iterator<Item = IpPort> + '_ {
673        self.udp_paths.paths.keys().copied()
674    }

677    pub(super) fn direct_address_states(&self) -> impl Iterator<Item = (&IpPort, &PathState)> + '_ {
678        self.udp_paths.paths.iter()
679    }

681    pub(super) fn last_used(&self) -> Option<Instant> {
682        self.last_used
683    }

740/// Information about a *direct address*.
741///
742/// The *direct addresses* of an iroh node are those that could be used by other nodes to
743/// establish direct connectivity, depending on the network situation. Due to NAT configurations,
744/// for example, not all direct addresses of a node are usable by all peers.
745#[derive(Debug, Clone, Eq, PartialEq, Serialize, Deserialize)]
746pub struct DirectAddrInfo {
747    /// The UDP address reported by the remote node.
748    pub addr: SocketAddr,
749    /// The latency to the remote node over this network path.
750    ///
751    /// If there has never been any connectivity via this address no latency will be known.
752    pub latency: Option<Duration>,
753    /// Last control message received by this node about this address.
754    ///
755    /// This contains the elapsed duration since the control message was received and the
756    /// kind of control message received at that time.  Only the most recent control message
757    /// is returned.
758    ///
759    /// Note that [`ControlMsg::CallMeMaybe`] is received via a relay path, while
760    /// [`ControlMsg::Ping`] and [`ControlMsg::Pong`] are received on the path to
761    /// [`DirectAddrInfo::addr`] itself and thus convey very different information.
762    pub last_control: Option<(Duration, ControlMsg)>,
763    /// Elapsed time since the last payload message was received on this network path.
764    ///
765    /// This indicates how long ago a QUIC datagram was received from the remote node sent
766    /// from this [`DirectAddrInfo::addr`].  It indicates the network path was in use to
767    /// transport payload data.
768    pub last_payload: Option<Duration>,
769    /// Elapsed time since this network path was known to exist.
770    ///
771    /// A network path is considered to exist only because the remote node advertised it.
772    /// It may not mean the path is usable.  However, if there was any communication with
773    /// the remote node over this network path it also means the path exists.
774    ///
775    /// The elapsed time since *any* confirmation of the path's existence was received is
776    /// returned.  If the remote node moved networks and no longer has this path, this could
777    /// be a long duration.  If the path was added via [`Endpoint::add_node_addr`] or some
778    /// node discovery the path may never have been known to exist.
779    ///
780    /// [`Endpoint::add_node_addr`]: crate::endpoint::Endpoint::add_node_addr
781    pub last_alive: Option<Duration>,
782    /// A [`HashMap`] of [`Source`]s to [`Duration`]s.
783    ///
784    /// The [`Duration`] indicates the elapsed time since this source last
785    /// recorded this address.
786    ///
787    /// The [`Duration`] will always indicate the most recent time the source
788    /// recorded this address.
789    pub sources: HashMap<Source, Duration>,
790}

819/// Details about a remote iroh node which is known to this node.
820///
821/// Having details of a node does not mean it can be connected to, nor that it has ever been
822/// connected to in the past. There are various reasons a node might be known: it could have
823/// been manually added via [`Endpoint::add_node_addr`], it could have been added by some
824/// discovery mechanism, the node could have contacted this node, etc.
825///
826/// [`Endpoint::add_node_addr`]: crate::endpoint::Endpoint::add_node_addr
827#[derive(Debug, Clone, Eq, PartialEq, Serialize, Deserialize)]
828pub struct RemoteInfo {
829    /// The globally unique identifier for this node.
830    pub node_id: NodeId,
831    /// Relay server information, if available.
832    pub relay_url: Option<RelayUrlInfo>,
833    /// The addresses at which this node might be reachable.
834    ///
835    /// Some of these addresses might only be valid for networks we are not part of, but the remote
836    /// node might be a part of.
837    pub addrs: Vec<DirectAddrInfo>,
838    /// The type of connection we have to the node, either direct or over relay.
839    pub conn_type: ConnectionType,
840    /// The latency of the current network path to the remote node.
841    pub latency: Option<Duration>,
842    /// Time elapsed time since last we have sent to or received from the node.
843    ///
844    /// This is the duration since *any* data (payload or control messages) was sent or receive
845    /// from the remote node. Note that sending to the remote node does not imply
846    /// the remote node received anything.
847    pub last_used: Option<Duration>,
848}
849
850impl RemoteInfo {
851    /// Get the duration since the last activity we received from this endpoint
852    /// on any of its direct addresses.
853    pub fn last_received(&self) -> Option<Duration> {
854        self.addrs
855            .iter()
856            .filter_map(|addr| addr.last_control.map(|x| x.0).min(addr.last_payload))
857            .min()
858    }
859
860    /// Whether there is a possible known network path to the remote node.
861    ///
862    /// Note that this does not provide any guarantees of whether any network path is
863    /// usable.
864    pub fn has_send_address(&self) -> bool {
865        self.relay_url.is_some() || !self.addrs.is_empty()
866    }
867
868    /// Returns a deduplicated list of [`Source`]s merged from all address in the [`RemoteInfo`].
869    ///
870    /// Deduplication is on the (`Source`, `Duration`) tuple, so you will get multiple [`Source`]s
871    /// for each `Source` variant, if different addresses were discovered from the same [`Source`]
872    /// at different times.
873    ///
874    /// The list is sorted from least to most recent [`Source`].
875    pub fn sources(&self) -> Vec<(Source, Duration)> {
876        let mut sources = vec![];
877        for addr in &self.addrs {
878            for source in &addr.sources {
879                let source = (source.0.clone(), *source.1);
880                if !sources.contains(&source) {
881                    sources.push(source)
882                }
883            }
884        }
885        sources.sort_by(|a, b| b.1.cmp(&a.1));
886        sources
887    }
888}

890/// The type of connection we have to the endpoint.
891#[derive(derive_more::Display, Default, Debug, Clone, Eq, PartialEq, Serialize, Deserialize)]
892pub enum ConnectionType {
893    /// Direct UDP connection
894    #[display("direct({_0})")]
895    Direct(SocketAddr),
896    /// Relay connection over relay
897    #[display("relay({_0})")]
898    Relay(RelayUrl),
899    /// Both a UDP and a relay connection are used.
900    ///
901    /// This is the case if we do have a UDP address, but are missing a recent confirmation that
902    /// the address works.
903    #[display("mixed(udp: {_0}, relay: {_1})")]
904    Mixed(SocketAddr, RelayUrl),
905    /// We have no verified connection to this PublicKey
906    #[default]
907    #[display("none")]
908    None,
909}
