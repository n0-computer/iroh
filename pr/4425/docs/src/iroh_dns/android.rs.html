1//! Android system DNS reader.
2//!
3//! Forwards to [`hickory_resolver::system_conf::read_system_conf`], which
4//! reads `LinkProperties.getDnsServers()` through [`ndk_context`]. iroh on
5//! Android therefore requires [`ndk_context`] to be initialized before any
6//! [`DnsResolver`] is constructed, either by ndk-glue or android-activity
7//! (both do this before `main`) or by an explicit
8//! [`install_android_jni_context`] call.
9//!
10//! Without an initialized [`ndk_context`], the JNI lookup panics internally.
11//! We catch the panic via [`catch_unwind`] and convert the panic into a regular
12//! error, which will then make the [`DnsResolver`] use fallback nameservers.
13//!
14//! This depends on panic unwinding though, so if you set `panic = abort` in
15//! your compilation profile, and don't install the JNI context, your app
16//! would panic and abort on the first DNS lookup.
17//!
18//! [`DnsResolver`]: crate::dns::DnsResolver
19//! [`ndk_context`]: https://docs.rs/ndk-context
20//! [`catch_unwind`]: std::panic::catch_unwind
21
22use std::{
23    ffi::c_void,
24    panic::{AssertUnwindSafe, catch_unwind},
25};
26
27use hickory_resolver::{
28    config::{ResolverConfig, ResolverOpts},
29    net::NetError,
30};
31
32/// Reads the active network's DNS configuration via JNI.
33pub(crate) fn read_system_conf() -> Result<(ResolverConfig, ResolverOpts), NetError> {
34    match catch_unwind(AssertUnwindSafe(
35        hickory_resolver::system_conf::read_system_conf,
36    )) {
37        Ok(Ok(conf)) => Ok(conf),
38        Ok(Err(err)) => Err(NetError::from(err)),
39        Err(_) => Err(NetError::Msg(
40            "ndk_context not initialized; call install_android_jni_context".to_string(),
41        )),
42    }
43}
44
45/// Exposes a JVM to iroh so that we can read the system's DNS configuration.
46///
47/// This calls [`ndk_context::initialize_android_context`] to expose a
48/// `JavaVM` and Application Context to Rust code so that we can use JNI.
49/// This is required to get the configured nameservers on Android.
50///
51/// If this function is not called, fetching the configured nameservers will fail
52/// and the default [`DnsResolver`] will use fallback nameservers instead.
53///
54/// If you call [`ndk_context::initialize_android_context`] already somewhere
55/// up the stack in your app, or use a crate like `ndk-glue` or `android-activity`
56/// that do this for you, then there's no need to call this function.
57///
58/// If you don't use a glue crate, a typical way to initialize the context is
59/// via `JNI_OnLoad`:
60///
61/// *Note: `install_android_jni_context` is reexported from `iroh`, so you can substitute
62/// `iroh_dns` for `iroh` below.*
63///
64/// ```ignore
65/// #[cfg(target_os = "android")]
66/// #[no_mangle]
67/// pub extern "C" fn JNI_OnLoad(
68///     vm: jni::JavaVM,
69///     res: *mut std::os::raw::c_void,
70/// ) -> jni::sys::jint {
71///     use std::ffi::c_void;
72///
73///     let vm = vm.get_java_vm_pointer() as *mut c_void;
74///     unsafe {
75///         iroh_dns::install_android_jni_context(vm, res);
76///     }
77///     jni::JNIVersion::V6.into()
78/// }
79/// ```
80///
81/// # Safety
82///
83/// Both the `java_vm` and `context_jobject` pointers must remain valid until the process exits.
84/// See also [`ndk_context::initialize_android_context`].
85///
86/// [`DnsResolver`]: crate::dns::DnsResolver
87/// [`ndk_context`]: https://docs.rs/ndk-context
88/// [`ndk_context::initialize_android_context`]: https://docs.rs/ndk-context/latest/ndk_context/fn.initialize_android_context.html
89pub unsafe fn install_android_jni_context(java_vm: *mut c_void, context_jobject: *mut c_void) {
90    unsafe {
91        ndk_context::initialize_android_context(java_vm, context_jobject);
92    }
93}
