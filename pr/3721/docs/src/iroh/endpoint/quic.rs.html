30/// Parameters governing the core QUIC state machine
31///
32/// Default values should be suitable for most internet applications. Applications protocols which
33/// forbid remotely-initiated streams should set `max_concurrent_bidi_streams` and
34/// `max_concurrent_uni_streams` to zero.
35///
36/// In some cases, performance or resource requirements can be improved by tuning these values to
37/// suit a particular application and/or network connection. In particular, data window sizes can be
38/// tuned for a particular expected round trip time, link capacity, and memory availability. Tuning
39/// for higher bandwidths and latencies increases worst-case memory consumption, but does not impair
40/// performance at lower bandwidths and latencies. The default configuration is tuned for a 100Mbps
41/// link with a 100ms round trip time.
42#[derive(Debug)]
43pub struct TransportConfigBuilder(pub(crate) quinn::TransportConfig);
44
45impl TransportConfigBuilder {}
46
47impl Default for TransportConfigBuilder {
48    fn default() -> Self {
49        let cfg = quinn::TransportConfig::default();
50        // TODO: adjust to use multipath defaults
51        Self(cfg)
52    }
53}

84    /// The true idle timeout is the minimum of this and the peer's own max idle timeout. `None`

179    /// The initial value to be used as the maximum UDP payload size before running MTU discovery
180    /// (see [`TransportConfigBuilder::mtu_discovery_config`]).
181    ///
182    /// Must be at least 1200, which is the default, and known to be safe for typical internet
183    /// applications. Larger values are more efficient, but increase the risk of packet loss due to
184    /// exceeding the network path's IP MTU. If the provided value is higher than what the network
185    /// path actually supports, packet loss will eventually trigger black hole detection and bring
186    /// it down to [`TransportConfigBuilder::min_mtu`].

188        self.0.initial_mtu(value);
189        self

192    /// The maximum UDP payload size guaranteed to be supported by the network.
193    ///
194    /// Must be at least 1200, which is the default, and lower than or equal to
195    /// [`TransportConfigBuilder::initial_mtu`].
196    ///
197    /// Real-world MTUs can vary according to ISP, VPN, and properties of intermediate network links
198    /// outside of either endpoint's control. Extreme care should be used when raising this value
199    /// outside of private networks where these factors are fully controlled. If the provided value
200    /// is higher than what the network path actually supports, the result will be unpredictable and
201    /// catastrophic packet loss, without a possibility of repair. Prefer
202    /// [`TransportConfigBuilder::initial_mtu`] together with
203    /// [`TransportConfigBuilder::mtu_discovery_config`] to set a maximum UDP payload size that robustly
204    /// adapts to the network.
