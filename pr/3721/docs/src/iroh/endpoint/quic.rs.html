32/// Parameters governing the core QUIC state machine
33///
34/// Default values should be suitable for most internet applications. Applications protocols which
35/// forbid remotely-initiated streams should set `max_concurrent_bidi_streams` and
36/// `max_concurrent_uni_streams` to zero.
37///
38/// In some cases, performance or resource requirements can be improved by tuning these values to
39/// suit a particular application and/or network connection. In particular, data window sizes can be
40/// tuned for a particular expected round trip time, link capacity, and memory availability. Tuning
41/// for higher bandwidths and latencies increases worst-case memory consumption, but does not impair
42/// performance at lower bandwidths and latencies. The default configuration is tuned for a 100Mbps
43/// link with a 100ms round trip time.
44///
45/// In iroh, the config has some specific default values that make iroh's holepunching work
46/// well with QUIC multipath. Adjusting those settings may error if the given values
47/// are out of scope with what is optimal for iroh.
48///
49/// Look at the following methods for more details:
50/// - [`QuicTransportConfig::keep_alive_interval`]
51/// - [`QuicTransportConfig::default_path_keep_alive_interval`]
52/// - [`QuicTransportConfig::default_path_max_idle_timeout`]
53/// - [`QuicTransportConfig::max_concurrent_multipath_paths`]
54/// - [`QuicTransportConfig::set_max_remote_nat_traversal_addresses`]
55#[derive(Debug)]
56pub struct QuicTransportConfig(pub(crate) TransportConfig);
57
58impl Default for QuicTransportConfig {
59    fn default() -> Self {
60        let mut cfg = TransportConfig::default();
61        // Override some transport config settings.
62        cfg.keep_alive_interval(Some(Duration::from_millis(HEARTBEAT_INTERVAL)));
63        cfg.default_path_keep_alive_interval(Some(Duration::from_millis(HEARTBEAT_INTERVAL)));
64        cfg.default_path_max_idle_timeout(Some(Duration::from_millis(PATH_MAX_IDLE_TIMEOUT)));
65        cfg.max_concurrent_multipath_paths(MAX_MULTIPATH_PATHS + 1);
66        cfg.set_max_remote_nat_traversal_addresses(MAX_MULTIPATH_PATHS as u8);
67        Self(cfg)
68    }
69}

75pub enum QuicTransportConfigError {
76    #[error(
77        "keep alive value too high, must be at minimum {}ms",
78        HEARTBEAT_INTERVAL

81    #[error(
82        "path keep alive value too high, must be at minimum {}ms",
83        HEARTBEAT_INTERVAL
84    )]
85    PathKeepAliveInterval,
86    #[error(
87        "path max idle timeout value too low, must be at least {}ms",
88        PATH_MAX_IDLE_TIMEOUT

90    PathMaxIdleTimeout,
91    #[error("max concurrent multipath paths too low, must be at least {}", MAX_MULTIPATH_PATHS + 1)]
92    MaxConcurrentMultipathPaths,
93    #[error(
94        "max remote nat traversal addresses value too low, must be at least {}",
95        MAX_MULTIPATH_PATHS
96    )]
97    MaxRemoteNatTraversalAddresses,
98}

118    /// Maximum duration of inactivity to accept before timing out the connection.
119    ///
120    /// The true idle timeout is the minimum of this and the peer's own max idle timeout. `None`
121    /// represents an infinite timeout. Defaults to 30 seconds.
122    ///
123    /// **WARNING**: If a peer or its network path malfunctions or acts maliciously, an infinite
124    /// idle timeout can result in permanently hung futures!
125    ///
126    /// ```
127    /// # use std::{convert::TryInto, time::Duration};
128    /// # use iroh::endpoint::{QuicTransportConfig, VarInt, VarIntBoundsExceeded};
129    /// # fn main() -> Result<(), VarIntBoundsExceeded> {
130    /// let mut config = QuicTransportConfig::default();
131    ///
132    /// // Set the idle timeout as `VarInt`-encoded milliseconds
133    /// config.max_idle_timeout(Some(VarInt::from_u32(10_000).into()));
134    ///
135    /// // Set the idle timeout as a `Duration`
136    /// config.max_idle_timeout(Some(Duration::from_secs(10).try_into()?));
137    /// # Ok(())
138    /// # }
139    /// ```
140    pub fn max_idle_timeout(&mut self, value: Option<IdleTimeout>) -> &mut Self {
141        self.0.max_idle_timeout(value);
142        self
143    }

145    /// Maximum number of bytes the peer may transmit without acknowledgement on any one stream
146    /// before becoming blocked.
147    ///
148    /// This should be set to at least the expected connection latency multiplied by the maximum
149    /// desired throughput. Setting this smaller than `receive_window` helps ensure that a single
150    /// stream doesn't monopolize receive buffers, which may otherwise occur if the application
151    /// chooses not to read from a large stream for a time while still requiring data on other
152    /// streams.

155        self

180    /// Whether to implement fair queuing for send streams having the same priority.

215    /// The initial value to be used as the maximum UDP payload size before running MTU discovery
216    /// (see [`QuicTransportConfig::mtu_discovery_config`]).
217    ///
218    /// Must be at least 1200, which is the default, and known to be safe for typical internet
219    /// applications. Larger values are more efficient, but increase the risk of packet loss due to
220    /// exceeding the network path's IP MTU. If the provided value is higher than what the network
221    /// path actually supports, packet loss will eventually trigger black hole detection and bring
222    /// it down to [`QuicTransportConfig::min_mtu`].
223    pub fn initial_mtu(&mut self, value: u16) -> &mut Self {
224        self.0.initial_mtu(value);
225        self
226    }

228    /// The maximum UDP payload size guaranteed to be supported by the network.
229    ///
230    /// Must be at least 1200, which is the default, and lower than or equal to
231    /// [`QuicTransportConfig::initial_mtu`].
232    ///
233    /// Real-world MTUs can vary according to ISP, VPN, and properties of intermediate network links
234    /// outside of either endpoint's control. Extreme care should be used when raising this value
235    /// outside of private networks where these factors are fully controlled. If the provided value
236    /// is higher than what the network path actually supports, the result will be unpredictable and
237    /// catastrophic packet loss, without a possibility of repair. Prefer
238    /// [`QuicTransportConfig::initial_mtu`] together with
239    /// [`QuicTransportConfig::mtu_discovery_config`] to set a maximum UDP payload size that robustly
240    /// adapts to the network.
241    pub fn min_mtu(&mut self, value: u16) -> &mut Self {
242        self.0.min_mtu(value);
243        self
244    }

287    /// Period of inactivity before sending a keep-alive packet
288    ///
289    /// Keep-alive packets prevent an inactive but otherwise healthy connection from timing out.
290    ///
291    /// `None` to disable, which is the default. Only one side of any given connection needs keep-alive
292    /// enabled for the connection to be preserved. Must be set lower than the idle_timeout of both
293    /// peers to be effective.
294    ///
295    /// # Errors
296    ///
297    /// Will return a [`QuicTransportConfigError::KeepAliveInterval`] if the given value is more than [`HEARTBEAT_INTERVAL`].
298    pub fn keep_alive_interval(
299        &mut self,
300        value: Duration,
301    ) -> Result<&mut Self, QuicTransportConfigError> {
302        if value > Duration::from_millis(HEARTBEAT_INTERVAL) {
303            return Err(e!(QuicTransportConfigError::KeepAliveInterval));
304        }
305        self.0.keep_alive_interval(Some(value));
306        Ok(self)
307    }

400    /// Enables the Multipath Extension for QUIC.
401    ///
402    /// Setting this to any nonzero value will enable the Multipath Extension for QUIC,
403    /// <https://datatracker.ietf.org/doc/draft-ietf-quic-multipath/>.
404    ///
405    /// The value provided specifies the number maximum number of paths this endpoint may open
406    /// concurrently when multipath is negotiated. For any path to be opened, the remote must
407    /// enable multipath as well.
408    ///
409    /// # Errors
410    ///
411    /// Will return a [`QuicTransportConfigError::MaxConcurrentMultipathPaths`] error if the
412    /// given value is less than [`MAX_MULTIPATH_PATHS`] + 1.
413    pub fn max_concurrent_multipath_paths(
414        &mut self,
415        max_concurrent: u32,
416    ) -> Result<&mut Self, QuicTransportConfigError> {
417        if max_concurrent < MAX_MULTIPATH_PATHS + 1 {
418            return Err(e!(QuicTransportConfigError::MaxConcurrentMultipathPaths));
419        }
420        self.0.max_concurrent_multipath_paths(max_concurrent);
421        Ok(self)
422    }

487        }
488        self.0.set_max_remote_nat_traversal_addresses(max_addresses);
489        Ok(self)
490    }
