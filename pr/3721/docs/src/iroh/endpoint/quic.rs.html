31/// Parameters governing the core QUIC state machine
32///
33/// Default values should be suitable for most internet applications. Applications protocols which
34/// forbid remotely-initiated streams should set `max_concurrent_bidi_streams` and
35/// `max_concurrent_uni_streams` to zero.
36///
37/// In some cases, performance or resource requirements can be improved by tuning these values to
38/// suit a particular application and/or network connection. In particular, data window sizes can be
39/// tuned for a particular expected round trip time, link capacity, and memory availability. Tuning
40/// for higher bandwidths and latencies increases worst-case memory consumption, but does not impair
41/// performance at lower bandwidths and latencies. The default configuration is tuned for a 100Mbps
42/// link with a 100ms round trip time.
43///
44/// In iroh, the config has some specific default values that make iroh's holepunching work
45/// well with QUIC multipath. Adjusting those settings may cause suboptimal usage.
46///
47/// Look at the following methods for more details:
48/// - [`QuicTransportConfig::default_path_keep_alive_interval`]
49/// - [`QuicTransportConfig::default_path_max_idle_timeout`]
50/// - [`QuicTransportConfig::max_concurrent_multipath_paths`]
51/// - [`QuicTransportConfig::set_max_remote_nat_traversal_addresses`]
52#[derive(Debug, Clone)]
53pub struct QuicTransportConfig(pub(crate) quinn::TransportConfig);
54
55impl Default for QuicTransportConfig {
56    fn default() -> Self {
57        let mut cfg = quinn::TransportConfig::default();
58        // Override some transport config settings.
59        cfg.keep_alive_interval(Some(HEARTBEAT_INTERVAL));
60        cfg.default_path_keep_alive_interval(Some(HEARTBEAT_INTERVAL));
61        cfg.default_path_max_idle_timeout(Some(PATH_MAX_IDLE_TIMEOUT));
62        cfg.max_concurrent_multipath_paths(MAX_MULTIPATH_PATHS + 1);
63        cfg.set_max_remote_nat_traversal_addresses(MAX_MULTIPATH_PATHS as u8);
64        Self(cfg)
65    }
66}

91    /// Maximum duration of inactivity to accept before timing out the connection.
92    ///
93    /// The true idle timeout is the minimum of this and the peer's own max idle timeout. `None`

103    /// let mut config = QuicTransportConfig::default();

105    /// // Set the idle timeout as `VarInt`-encoded milliseconds

188    /// The initial value to be used as the maximum UDP payload size before running MTU discovery
189    /// (see [`QuicTransportConfig::mtu_discovery_config`]).

201    /// The maximum UDP payload size guaranteed to be supported by the network.
202    ///
203    /// Must be at least 1200, which is the default, and lower than or equal to
204    /// [`QuicTransportConfig::initial_mtu`].
205    ///
206    /// Real-world MTUs can vary according to ISP, VPN, and properties of intermediate network links
207    /// outside of either endpoint's control. Extreme care should be used when raising this value
208    /// outside of private networks where these factors are fully controlled. If the provided value
209    /// is higher than what the network path actually supports, the result will be unpredictable and
210    /// catastrophic packet loss, without a possibility of repair. Prefer
211    /// [`QuicTransportConfig::initial_mtu`] together with
212    /// [`QuicTransportConfig::mtu_discovery_config`] to set a maximum UDP payload size that robustly
213    /// adapts to the network.

363    /// Enables the Multipath Extension for QUIC.
364    ///
365    /// Setting this to any nonzero value will enable the Multipath Extension for QUIC,
366    /// <https://datatracker.ietf.org/doc/draft-ietf-quic-multipath/>.
367    ///
368    /// The value provided specifies the number maximum number of paths this endpoint may open
369    /// concurrently when multipath is negotiated. For any path to be opened, the remote must
370    /// enable multipath as well.
371    ///
372    /// Note: this method will ignore values less than the recommended 13 and will log a warning.
373    pub fn max_concurrent_multipath_paths(&mut self, max_concurrent: u32) -> &mut Self {
374        if max_concurrent < MAX_MULTIPATH_PATHS + 1 {
375            warn!(
376                "QuicTransportConfig::max_concurrent_multipath_paths must be at minimum {}, ignoring user supplied value",
377                MAX_MULTIPATH_PATHS + 1
378            );
379            return self;
380        }
381        self.0.max_concurrent_multipath_paths(max_concurrent);
382        self
383    }
