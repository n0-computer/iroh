43    /// NOTE: Receiving on a closed socket will return [`Poll::Pending`] indefinitely.
44    pub(super) fn poll_recv(
45        &mut self,
46        cx: &mut Context,
47        bufs: &mut [io::IoSliceMut<'_>],
48        metas: &mut [quinn_udp::RecvMeta],
49        source_addrs: &mut [Addr],
50    ) -> Poll<io::Result<usize>> {
51        match self.socket.poll_recv_quinn(cx, bufs, metas) {
52            Poll::Pending => Poll::Pending,
53            Poll::Ready(Ok(n)) => {
54                for (source_addr, meta) in source_addrs.iter_mut().zip(metas.iter_mut()).take(n) {
55                    if meta.addr.is_ipv4() {
56                        // The AsyncUdpSocket is an AF_INET6 socket and needs to show this
57                        // as coming from an IPv4-mapped IPv6 addresses, since Quinn will
58                        // use those when sending on an INET6 socket.
59                        let v6_ip = match meta.addr.ip() {
60                            IpAddr::V4(ipv4_addr) => ipv4_addr.to_ipv6_mapped(),
61                            IpAddr::V6(ipv6_addr) => ipv6_addr,
62                        };
63                        meta.addr = SocketAddr::new(v6_ip.into(), meta.addr.port());
64                    }
65                    // The transport addresses are internal to iroh and we always want those
66                    // to remain the canonical address.
67                    *source_addr =
68                        SocketAddr::new(meta.addr.ip().to_canonical(), meta.addr.port()).into();
69                }
70                Poll::Ready(Ok(n))
71            }
72            Poll::Ready(Err(err)) => Poll::Ready(Err(err)),
73        }
74    }

120    pub(super) fn rebind(&self) -> io::Result<()> {
121        let old_addr = self.local_addr.get();
122        self.socket.rebind()?;
123        let addr = self.socket.local_addr()?;
124        self.local_addr.set(addr).ok();
125        trace!("rebound from {} to {}", old_addr, addr);
126
127        Ok(())
128    }

144impl IpSender {
145    pub(super) fn is_valid_send_addr(&self, dst: &SocketAddr) -> bool {
146        // Our net-tools crate binds sockets to their specific family.  This means an IPv6
147        // socket can not sent to IPv4, on any platform.  So we need to convert and
148        // IPv4-mapped IPv6 address back to it's canonical IPv4 address.
149        let dst_ip = dst.ip().to_canonical();
150
151        #[allow(clippy::match_like_matches_macro)]
152        match (self.bind_addr.ip(), dst_ip) {
153            (IpAddr::V4(_), IpAddr::V4(_)) => true,
154            (IpAddr::V6(_), IpAddr::V6(_)) => true,
155            _ => false,
156        }
157    }
158
159    /// Creates a canonical socket address.
160    ///
161    /// We may be asked to send IPv4-mapped IPv6 addresses.  But our sockets are configured
162    /// to only send their actual family.  So we need to map those back to the canonical
163    /// addresses.
164    #[inline]
165    fn canonical_addr(addr: SocketAddr) -> SocketAddr {
166        SocketAddr::new(addr.ip().to_canonical(), addr.port())
167    }
168
169    pub(super) async fn send(
170        &self,
171        dst: SocketAddr,
172        src: Option<IpAddr>,
173        transmit: &Transmit<'_>,
174    ) -> io::Result<()> {
175        let total_bytes = transmit.contents.len() as u64;
176        let res = self
177            .sender
178            .send(&quinn_udp::Transmit {
179                destination: Self::canonical_addr(dst),
180                ecn: transmit.ecn,
181                contents: transmit.contents,
182                segment_size: transmit.segment_size,
183                src_ip: src,
184            })
185            .await;
186
187        match res {
188            Ok(res) => {
189                match dst {
190                    SocketAddr::V4(_) => {
191                        self.metrics.send_ipv4.inc_by(total_bytes);
192                    }
193                    SocketAddr::V6(_) => {
194                        self.metrics.send_ipv6.inc_by(total_bytes);
195                    }
196                }
197                Ok(res)
198            }
199            Err(err) => Err(err),
200        }
201    }
202
203    pub(super) fn poll_send(
204        mut self: Pin<&mut Self>,
205        cx: &mut std::task::Context,
206        dst: SocketAddr,
207        src: Option<IpAddr>,
208        transmit: &Transmit<'_>,
209    ) -> Poll<io::Result<()>> {
210        let total_bytes = transmit.contents.len() as u64;
211        let res = Pin::new(&mut self.sender).poll_send(
212            &quinn_udp::Transmit {
213                destination: Self::canonical_addr(dst),
214                ecn: transmit.ecn,
215                contents: transmit.contents,
216                segment_size: transmit.segment_size,
217                src_ip: src,
218            },
219            cx,
220        );
221
222        match res {
223            Poll::Ready(Ok(res)) => {
224                match dst {
225                    SocketAddr::V4(_) => {
226                        self.metrics.send_ipv4.inc_by(total_bytes);
227                    }
228                    SocketAddr::V6(_) => {
229                        self.metrics.send_ipv6.inc_by(total_bytes);
230                    }
231                }
232                Poll::Ready(Ok(res))
233            }
234            Poll::Ready(Err(err)) => Poll::Ready(Err(err)),
235            Poll::Pending => Poll::Pending,
236        }
237    }
