1//! A discovery service that uses an mdns-like service to discover local nodes.
2//!
3//! This allows you to use an mdns-like swarm discovery service to find address information about nodes that are on your local network, no relay or outside internet needed.
4//! See the [`swarm-discovery`](https://crates.io/crates/swarm-discovery) crate for more details.
5//!
6//! When [`MdnsDiscovery`] is enabled, it's possible to get a list of the locally discovered nodes by filtering a list of `RemoteInfo`s.
7//!
8//! ```no_run
9//! use std::time::Duration;
10//!
11//! use iroh::{
12//!     SecretKey,
13//!     discovery::mdns::{DiscoveryEvent, MdnsDiscovery},
14//!     endpoint::{Endpoint, Source},
15//! };
16//! use n0_future::StreamExt;
17//!
18//! #[tokio::main]
19//! async fn main() {
20//!     let recent = Duration::from_secs(600); // 10 minutes in seconds
21//!     let node_key = SecretKey::generate(rand::thread_rng());
22//!     let node_id = node_key.public();
23//!     let mdns = MdnsDiscovery::builder().build(node_id).unwrap();
24//!     let endpoint = Endpoint::builder()
25//!         .secret_key(node_key)
26//!         .add_discovery(mdns.clone())
27//!         .bind()
28//!         .await
29//!         .unwrap();
30//!     let mut events = mdns.subscribe().await;
31//!     while let Some(event) = events.next().await {
32//!         match event {
33//!             DiscoveryEvent::Discovered { node_info, .. } => {
34//!                 println!("locally discovered: {:?}", node_info);
35//!             }
36//!             _ => {}
37//!         }
38//!     }
39//! }
40//! ```

80/// Discovery using `swarm-discovery`, a variation on mdns
81#[derive(Debug, Clone)]
82pub struct MdnsDiscovery {
83    #[allow(dead_code)]
84    handle: Arc<AbortOnDropHandle<()>>,
85    sender: mpsc::Sender<Message>,
86    advertise: bool,
87    /// When `local_addrs` changes, we re-publish our info.
88    local_addrs: Watchable<Option<NodeData>>,

91#[derive(Debug)]
92enum Message {
93    Discovery(String, Peer),
94    Resolve(NodeId, mpsc::Sender<Result<DiscoveryItem, DiscoveryError>>),
95    Timeout(NodeId, usize),
96    Subscribe(mpsc::Sender<DiscoveryEvent>),
97}

99/// Manages the list of subscribers that are subscribed to this discovery service.
100#[derive(Debug)]
101struct Subscribers(Vec<mpsc::Sender<DiscoveryEvent>>);
102
103impl Subscribers {
104    fn new() -> Self {
105        Self(vec![])
106    }
107
108    /// Add the subscriber to the list of subscribers
109    fn push(&mut self, subscriber: mpsc::Sender<DiscoveryEvent>) {
110        self.0.push(subscriber);
111    }
112
113    /// Sends the `node_id` and `item` to each subscriber.
114    ///
115    /// Cleans up any subscribers that have been dropped.
116    fn send(&mut self, item: DiscoveryEvent) {
117        let mut clean_up = vec![];
118        for (i, subscriber) in self.0.iter().enumerate() {
119            // assume subscriber was dropped
120            if let Err(err) = subscriber.try_send(item.clone()) {
121                match err {
122                    TrySendError::Full(_) => {
123                        warn!(
124                            ?item,
125                            idx = i,
126                            "local swarm discovery subscriber is blocked, dropping item"
127                        )
128                    }
129                    TrySendError::Closed(_) => clean_up.push(i),
130                }
131            }
132        }
133        for i in clean_up.into_iter().rev() {
134            self.0.swap_remove(i);
135        }
136    }
137}

203/// An event emitted from the [`MdnsDiscovery`] service.
204#[derive(Debug, Clone, Eq, PartialEq)]
205pub enum DiscoveryEvent {
206    /// A peer was discovered or it's information was updated.
207    Discovered {
208        /// The node info for the node, as discovered.
209        node_info: NodeInfo,
210        /// Optional timestamp when this node address info was last updated.
211        last_updated: Option<u64>,
212    },
213    /// A peer was expired due to being inactive, unreachable, or otherwise
214    /// unavailable.
215    Expired {
216        /// The id of the node that expired.
217        node_id: NodeId,
218    },
219}

227    /// Create a new [`MdnsDiscovery`] Service.
228    ///
229    /// This starts a [`Discoverer`] that broadcasts your addresses (if advertise is set to true)
230    /// and receives addresses from other nodes in your local network.
231    ///
232    /// # Errors
233    /// Returns an error if the network does not allow ipv4 OR ipv6.
234    ///
235    /// # Panics
236    /// This relies on [`tokio::runtime::Handle::current`] and will panic if called outside of the context of a tokio runtime.
237    fn new(
238        node_id: NodeId,
239        advertise: bool,
240        service_name: String,
241    ) -> Result<Self, IntoDiscoveryError> {
242        debug!("Creating new MdnsDiscovery service");
243        let (send, mut recv) = mpsc::channel(64);
244        let task_sender = send.clone();
245        let rt = tokio::runtime::Handle::current();
246        let discovery = MdnsDiscovery::spawn_discoverer(
247            node_id,
248            advertise,
249            task_sender.clone(),
250            BTreeSet::new(),
251            service_name,
252            &rt,
253        )?;
254
255        let local_addrs: Watchable<Option<NodeData>> = Watchable::default();
256        let mut addrs_change = local_addrs.watch();
257        let discovery_fut = async move {
258            let mut node_addrs: HashMap<PublicKey, Peer> = HashMap::default();
259            let mut subscribers = Subscribers::new();
260            let mut last_id = 0;
261            let mut senders: HashMap<
262                PublicKey,
263                HashMap<usize, mpsc::Sender<Result<DiscoveryItem, DiscoveryError>>>,
264            > = HashMap::default();
265            let mut timeouts = JoinSet::new();
266            loop {
267                trace!(?node_addrs, "MdnsDiscovery Service loop tick");
268                let msg = tokio::select! {
269                    msg = recv.recv() => {
270                        msg
271                    }
272                    Ok(Some(data)) = addrs_change.updated() => {
273                        tracing::trace!(?data, "MdnsDiscovery address changed");
274                        discovery.remove_all();
275                        let addrs =
276                            MdnsDiscovery::socketaddrs_to_addrs(data.direct_addresses());
277                        for addr in addrs {
278                            discovery.add(addr.0, addr.1)
279                        }
280                        if let Some(user_data) = data.user_data() {
281                            if let Err(err) = discovery.set_txt_attribute(USER_DATA_ATTRIBUTE.to_string(), Some(user_data.to_string())) {
282                                warn!("Failed to set the user-defined data in local swarm discovery: {err:?}");
283                            }
284                        }
285                        continue;
286                    }
287                };
288                let msg = match msg {
289                    None => {
290                        error!("MdnsDiscovery channel closed");
291                        error!("closing MdnsDiscovery");
292                        timeouts.abort_all();
293                        discovery.remove_all();
294                        return;
295                    }
296                    Some(msg) => msg,
297                };
298                match msg {
299                    Message::Discovery(discovered_node_id, peer_info) => {
300                        trace!(
301                            ?discovered_node_id,
302                            ?peer_info,
303                            "MdnsDiscovery Message::Discovery"
304                        );
305                        let discovered_node_id = match PublicKey::from_str(&discovered_node_id) {
306                            Ok(node_id) => node_id,
307                            Err(e) => {
308                                warn!(
309                                    discovered_node_id,
310                                    "couldn't parse node_id from mdns discovery service: {e:?}"
311                                );
312                                continue;
313                            }
314                        };
315
316                        if discovered_node_id == node_id {
317                            continue;
318                        }
319
320                        if peer_info.is_expiry() {
321                            trace!(
322                                ?discovered_node_id,
323                                "removing node from MdnsDiscovery address book"
324                            );
325                            node_addrs.remove(&discovered_node_id);
326                            subscribers.send(DiscoveryEvent::Expired {
327                                node_id: discovered_node_id,
328                            });
329                            continue;
330                        }
331
332                        let entry = node_addrs.entry(discovered_node_id);
333                        if let std::collections::hash_map::Entry::Occupied(ref entry) = entry {
334                            if entry.get() == &peer_info {
335                                // this is a republish we already know about
336                                continue;
337                            }
338                        }
339
340                        debug!(
341                            ?discovered_node_id,
342                            ?peer_info,
343                            "adding node to MdnsDiscovery address book"
344                        );
345
346                        let mut resolved = false;
347                        let item = peer_to_discovery_item(&peer_info, &discovered_node_id);
348                        if let Some(senders) = senders.get(&discovered_node_id) {
349                            trace!(?item, senders = senders.len(), "sending DiscoveryItem");
350                            resolved = true;
351                            for sender in senders.values() {
352                                sender.send(Ok(item.clone())).await.ok();
353                            }
354                        }
355                        entry.or_insert(peer_info);
356
357                        // only send nodes to the `subscriber` if they weren't explicitly resolved
358                        // in other words, nodes sent to the `subscribers` should only be the ones that
359                        // have been "passively" discovered
360                        if !resolved {
361                            subscribers.send(DiscoveryEvent::Discovered {
362                                node_info: item.node_info,
363                                last_updated: item.last_updated,
364                            });
365                        }
366                    }
367                    Message::Resolve(node_id, sender) => {
368                        let id = last_id + 1;
369                        last_id = id;
370                        trace!(?node_id, "MdnsDiscovery Message::SendAddrs");
371                        if let Some(peer_info) = node_addrs.get(&node_id) {
372                            let item = peer_to_discovery_item(peer_info, &node_id);
373                            debug!(?item, "sending DiscoveryItem");
374                            sender.send(Ok(item)).await.ok();
375                        }
376                        if let Some(senders_for_node_id) = senders.get_mut(&node_id) {
377                            senders_for_node_id.insert(id, sender);
378                        } else {
379                            let mut senders_for_node_id = HashMap::new();
380                            senders_for_node_id.insert(id, sender);
381                            senders.insert(node_id, senders_for_node_id);
382                        }
383                        let timeout_sender = task_sender.clone();
384                        timeouts.spawn(async move {
385                            time::sleep(DISCOVERY_DURATION).await;
386                            trace!(?node_id, "discovery timeout");
387                            timeout_sender
388                                .send(Message::Timeout(node_id, id))
389                                .await
390                                .ok();
391                        });
392                    }
393                    Message::Timeout(node_id, id) => {
394                        trace!(?node_id, "MdnsDiscovery Message::Timeout");
395                        if let Some(senders_for_node_id) = senders.get_mut(&node_id) {
396                            senders_for_node_id.remove(&id);
397                            if senders_for_node_id.is_empty() {
398                                senders.remove(&node_id);
399                            }
400                        }
401                    }
402                    Message::Subscribe(subscriber) => {
403                        trace!("MdnsDiscovery Message::Subscribe");
404                        subscribers.push(subscriber);
405                    }
406                }
407            }
408        };
409        let handle = task::spawn(discovery_fut.instrument(info_span!("swarm-discovery.actor")));
410        Ok(Self {
411            handle: Arc::new(AbortOnDropHandle::new(handle)),
412            sender: send,
413            advertise,
414            local_addrs,
415        })
416    }

418    /// Subscribe to discovered nodes
419    pub async fn subscribe(&self) -> impl Stream<Item = DiscoveryEvent> + Unpin + use<> {
420        let (sender, recv) = mpsc::channel(20);
421        let discovery_sender = self.sender.clone();
422        discovery_sender.send(Message::Subscribe(sender)).await.ok();
423        tokio_stream::wrappers::ReceiverStream::new(recv)
424    }

426    fn spawn_discoverer(
427        node_id: PublicKey,
428        advertise: bool,
429        sender: mpsc::Sender<Message>,
430        socketaddrs: BTreeSet<SocketAddr>,
431        service_name: String,
432        rt: &tokio::runtime::Handle,
433    ) -> Result<DropGuard, IntoDiscoveryError> {
434        let spawn_rt = rt.clone();
435        let callback = move |node_id: &str, peer: &Peer| {
436            trace!(
437                node_id,
438                ?peer,
439                "Received peer information from MdnsDiscovery"
440            );
441
442            let sender = sender.clone();
443            let node_id = node_id.to_string();
444            let peer = peer.clone();
445            spawn_rt.spawn(async move {
446                sender.send(Message::Discovery(node_id, peer)).await.ok();
447            });
448        };
449        let node_id_str = data_encoding::BASE32_NOPAD
450            .encode(node_id.as_bytes())
451            .to_ascii_lowercase();
452        let mut discoverer = Discoverer::new_interactive(service_name, node_id_str)
453            .with_callback(callback)
454            .with_ip_class(IpClass::Auto);
455        if advertise {
456            let addrs = MdnsDiscovery::socketaddrs_to_addrs(&socketaddrs);
457            for addr in addrs {
458                discoverer = discoverer.with_addrs(addr.0, addr.1);
459            }
460        }
461        discoverer
462            .spawn(rt)
463            .map_err(|e| IntoDiscoveryError::from_err("mdns", e))
464    }
465
466    fn socketaddrs_to_addrs(socketaddrs: &BTreeSet<SocketAddr>) -> HashMap<u16, Vec<IpAddr>> {
467        let mut addrs: HashMap<u16, Vec<IpAddr>> = HashMap::default();
468        for socketaddr in socketaddrs {
469            addrs
470                .entry(socketaddr.port())
471                .and_modify(|a| a.push(socketaddr.ip()))
472                .or_insert(vec![socketaddr.ip()]);
473        }
474        addrs
475    }
476}

478fn peer_to_discovery_item(peer: &Peer, node_id: &NodeId) -> DiscoveryItem {
479    let direct_addresses: BTreeSet<SocketAddr> = peer
480        .addrs()
481        .iter()
482        .map(|(ip, port)| SocketAddr::new(*ip, *port))
483        .collect();
484    // Get the user-defined data from the resolved peer info. We expect an attribute with a value
485    // that parses as `UserData`. Otherwise, omit.
486    let user_data = if let Some(Some(user_data)) = peer.txt_attribute(USER_DATA_ATTRIBUTE) {
487        match user_data.parse() {
488            Err(err) => {
489                debug!("failed to parse user data from TXT attribute: {err}");
490                None
491            }
492            Ok(data) => Some(data),
493        }
494    } else {
495        None
496    };
497    let node_info = NodeInfo::new(*node_id)
498        .with_direct_addresses(direct_addresses)
499        .with_user_data(user_data);
500    DiscoveryItem::new(node_info, NAME, None)
501}

503impl Discovery for MdnsDiscovery {
504    fn resolve(&self, node_id: NodeId) -> Option<BoxStream<Result<DiscoveryItem, DiscoveryError>>> {
505        use futures_util::FutureExt;
506
507        let (send, recv) = mpsc::channel(20);
508        let discovery_sender = self.sender.clone();
509        let stream = async move {
510            discovery_sender
511                .send(Message::Resolve(node_id, send))
512                .await
513                .ok();
514            tokio_stream::wrappers::ReceiverStream::new(recv)
515        };
516        Some(Box::pin(stream.flatten_stream()))
517    }
518
519    fn publish(&self, data: &NodeData) {
520        if self.advertise {
521            self.local_addrs.set(Some(data.clone())).ok();
522        }
523    }
524}

662                node_ids.insert((node_id, Some(user_data)));
663                discovery.publish(&node_data);
664                discoverers.push(discovery);
665            }
666
667            let mut events = discovery.subscribe().await;
668
669            let test = async move {
670                let mut got_ids = BTreeSet::new();
671                while got_ids.len() != num_nodes {
672                    if let Some(DiscoveryEvent::Discovered { node_info, .. }) = events.next().await
673                    {
674                        let data = node_info.data.user_data().cloned();
675                        if node_ids.contains(&(node_info.node_id, data.clone())) {
676                            got_ids.insert((node_info.node_id, data));
677                        }
678                    } else {
679                        whatever!(
680                            "no more events, only got {} ids, expected {num_nodes}\n",
681                            got_ids.len()
682                        );
683                    }
684                }
685                assert_eq!(got_ids, node_ids);
686                Ok::<_, Error>(())
687            };
688            tokio::time::timeout(Duration::from_secs(5), test)
689                .await
690                .context("timeout")?
691        }

707

743            let discovery_b = MdnsDiscovery::builder()
744                .service_name("different.name")
745                .build(id_b)?;
746
747            // Create a discovery service using the same
748            // custom service name
749            let id_c = SecretKey::generate(rand::thread_rng()).public();
750            let discovery_c = MdnsDiscovery::builder()
751                .service_name("different.name")
752                .build(id_c)?;
753
754            let node_data_a =
755                NodeData::new(None, BTreeSet::from(["0.0.0.0:11111".parse().unwrap()]));
756            discovery_a.publish(&node_data_a);
757
758            let node_data_b =
759                NodeData::new(None, BTreeSet::from(["0.0.0.0:22222".parse().unwrap()]));
760            discovery_b.publish(&node_data_b);
761
762            let node_data_c =
763                NodeData::new(None, BTreeSet::from(["0.0.0.0:33333".parse().unwrap()]));
764            discovery_c.publish(&node_data_c);
765
766            let mut stream_a = discovery_a.resolve(id_b).unwrap();
767            let result_a = tokio::time::timeout(Duration::from_secs(2), stream_a.next()).await;
