42#[derive(derive_more::Debug)]
43enum Message {
44    Upsert {
45        packet: SignedPacket,
46        res: oneshot::Sender<bool>,
47    },
48    Get {
49        key: PublicKeyBytes,
50        res: oneshot::Sender<Option<SignedPacket>>,
51    },
52    Remove {
53        key: PublicKeyBytes,
54        res: oneshot::Sender<bool>,
55    },
56    Snapshot {
57        #[debug(skip)]
58        res: oneshot::Sender<Snapshot>,
59    },
60    CheckExpired {
61        time: Timestamp,
62        key: PublicKeyBytes,
63    },
64}

74#[derive(Debug, Clone, Copy)]
75pub struct Options {
76    /// Maximum number of packets to process in a single write transaction.
77    pub max_batch_size: usize,
78    /// Maximum time to keep a write transaction open.
79    pub max_batch_time: Duration,
80    /// Time to keep packets in the store before eviction.
81    pub eviction: Duration,
82    /// Pause between eviction checks.
83    pub eviction_interval: Duration,
84}
85
86impl Default for Options {
87    fn default() -> Self {
88        Self {
89            // 64k packets
90            max_batch_size: 1024 * 64,
91            // this means we lose at most 1 second of data in case of a crash
92            max_batch_time: Duration::from_secs(1),
93            // 7 days
94            eviction: Duration::from_secs(3600 * 24 * 7),
95            // eviction can run frequently since it does not do a full scan
96            eviction_interval: Duration::from_secs(10),
97        }
98    }

101impl Actor {
102    async fn run(mut self) {
103        match self.run0().await {
104            Ok(()) => {}
105            Err(e) => {
106                tracing::error!("packet store actor failed: {:?}", e);
107                self.cancel.cancel();
108            }
109        }
110    }
111
112    async fn run0(&mut self) -> Result<()> {
113        let expiry_us = self.options.eviction.as_micros() as u64;
114        while let Some(msg) = self.recv.recv().await {
115            // if we get a snapshot message here we don't need to do a write transaction
116            let msg = if let Message::Snapshot { res } = msg {
117                let snapshot = Snapshot::new(&self.db)?;
118                res.send(snapshot).ok();
119                continue;
120            } else {
121                msg
122            };
123            trace!("batch");
124            self.recv.push_back(msg).unwrap();
125            let transaction = self.db.begin_write().e()?;
126            let mut tables = Tables::new(&transaction).e()?;
127            let timeout = tokio::time::sleep(self.options.max_batch_time);
128            tokio::pin!(timeout);
129            for _ in 0..self.options.max_batch_size {
130                tokio::select! {
131                    _ = self.cancel.cancelled() => {
132                        drop(tables);
133                        transaction.commit().e()?;
134                        return Ok(());
135                    }
136                    _ = &mut timeout => break,
137                    Some(msg) = self.recv.recv() => {
138                        match msg {
139                            Message::Get { key, res } => {
140                                match get_packet(&tables.signed_packets, &key) {
141                                    Ok(packet) => {
142                                        trace!("get {key}: {}", packet.is_some());
143                                        res.send(packet).ok();
144                                    },
145                                    Err(err) => {
146                                        warn!("get {key} failed: {err:#}");
147                                        return Err(err).with_context(|| format!("get packet for {key} failed"))
148                                    }
149                                }
150                            }
151                            Message::Upsert { packet, res } => {
152                                let key = PublicKeyBytes::from_signed_packet(&packet);
153                                trace!("upsert {}", key);
154                                let replaced = match get_packet(&tables.signed_packets, &key)? { Some(existing) => {
155                                    if existing.more_recent_than(&packet) {
156                                        res.send(false).ok();
157                                        continue;
158                                    } else {
159                                        // remove the old packet from the update time index
160                                        tables.update_time.remove(&existing.timestamp().to_bytes(), key.as_bytes()).e()?;
161                                        true
162                                    }
163                                } _ => {
164                                    false
165                                }};
166                                let value = packet.serialize();
167                                tables.signed_packets
168                                    .insert(key.as_bytes(), &value[..]).e()?;
169                                tables.update_time
170                                     .insert(&packet.timestamp().to_bytes(), key.as_bytes()).e()?;
171                                if replaced {
172                                    self.metrics.store_packets_updated.inc();
173                                } else {
174                                    self.metrics.store_packets_inserted.inc();
175                                }
176                                res.send(true).ok();
177                            }
178                            Message::Remove { key, res } => {
179                                trace!("remove {}", key);
180                                let updated = match tables.signed_packets.remove(key.as_bytes()).e()? { Some(row) => {
181                                    let packet = SignedPacket::deserialize(row.value()).e()?;
182                                    tables.update_time.remove(&packet.timestamp().to_bytes(), key.as_bytes()).e()?;
183                                    self.metrics.store_packets_removed.inc();
184                                    true
185                                } _ => {
186                                    false
187                                }};
188                                res.send(updated).ok();
189                            }
190                            Message::Snapshot { res } => {
191                                trace!("snapshot");
192                                res.send(Snapshot::new(&self.db)?).ok();
193                            }
194                            Message::CheckExpired { key, time } => {
195                                trace!("check expired {} at {}", key, fmt_time(time));
196                                match get_packet(&tables.signed_packets, &key)? { Some(packet) => {
197                                    let expired = Timestamp::now() - expiry_us;
198                                    if packet.timestamp() < expired {
199                                        tables.update_time.remove(&time.to_bytes(), key.as_bytes()).e()?;
200                                        let _ = tables.signed_packets.remove(key.as_bytes()).e()?;
201                                        self.metrics.store_packets_expired.inc();
202                                        debug!("removed expired packet {key}");
203                                    } else {
204                                        debug!("packet {key} is no longer expired, removing obsolete expiry entry");
205                                        tables.update_time.remove(&time.to_bytes(), key.as_bytes()).e()?;
206                                    }
207                                } _ => {
208                                    debug!("expired packet {key} not found, remove from expiry table");
209                                    tables.update_time.remove(&time.to_bytes(), key.as_bytes()).e()?;
210                                }}
211                            }
212                        }
213                    }
214                }
215            }
216            drop(tables);
217            transaction.commit().e()?;
218        }
219        Ok(())
220    }
221}

259impl SignedPacketStore {
260    pub fn persistent(
261        path: impl AsRef<Path>,
262        options: Options,
263        metrics: Arc<Metrics>,
264    ) -> Result<Self> {
265        let path = path.as_ref();
266        info!("loading packet database from {}", path.to_string_lossy());
267        if let Some(parent) = path.parent() {
268            std::fs::create_dir_all(parent).with_context(|| {
269                format!(
270                    "failed to create database directory at {}",
271                    path.to_string_lossy()
272                )
273            })?;
274        }
275        let db = Database::builder()
276            .create(path)
277            .context("failed to open packet database")?;
278        Self::open(db, options, metrics)
279    }
280
281    pub fn in_memory(options: Options, metrics: Arc<Metrics>) -> Result<Self> {
282        info!("using in-memory packet database");
283        let db = Database::builder()
284            .create_with_backend(InMemoryBackend::new())
285            .e()?;
286        Self::open(db, options, metrics)
287    }
288
289    pub fn open(db: Database, options: Options, metrics: Arc<Metrics>) -> Result<Self> {
290        // create tables
291        let write_tx = db.begin_write().e()?;
292        let _ = Tables::new(&write_tx).e()?;
293        write_tx.commit().e()?;
294        let (send, recv) = mpsc::channel(1024);
295        let send2 = send.clone();
296        let cancel = CancellationToken::new();
297        let cancel2 = cancel.clone();
298        let cancel3 = cancel.clone();
299        let actor = Actor {
300            db,
301            recv: PeekableReceiver::new(recv),
302            cancel: cancel2,
303            options,
304            metrics,
305        };
306        // start an io thread and donate it to the tokio runtime so we can do blocking IO
307        // inside the thread despite being in a tokio runtime
308        let _write_thread = IoThread::new("packet-store-actor", move || actor.run())?;
309        let _evict_thread = IoThread::new("packet-store-evict", move || {
310            evict_task(send2, options, cancel3)
311        })?;
312        Ok(Self {
313            send,
314            cancel,
315            _write_thread,
316            _evict_thread,
317        })
318    }

348fn get_packet(
349    table: &impl ReadableTable<&'static SignedPacketsKey, &'static [u8]>,
350    key: &PublicKeyBytes,
351) -> Result<Option<SignedPacket>> {
352    let Some(row) = table.get(key.as_ref()).context("database fetch failed")? else {
353        return Ok(None);
354    };
355    match SignedPacket::deserialize(row.value()) {
356        Ok(packet) => Ok(Some(packet)),
357        Err(err) => {
358            // Prior to iroh-dns-server v0.35, we stored packets in the default `SignedPacket::as_bytes` serialization from pkarr v2,
359            // which did not include the `last_seen` timestamp added as a prefix in `SignedPacket::serialize` from pkarr v3.
360            // If decoding the packet as a serialized pkarr v3 packet fails, we assume it was stored with iroh-dns-server before v0.35,
361            // and prepend an empty timestamp.
362            let data = row.value();
363            let mut buf = Vec::with_capacity(data.len() + 8);
364            buf.extend(&[0u8; 8]);
365            buf.extend(data);
366            match SignedPacket::deserialize(&buf) {
367                Ok(packet) => Ok(Some(packet)),
368                Err(err2) => Err(format_err!(
369                    "Failed to decode as pkarr v3: {err:#}. Also failed to decode as pkarr v2: {err2:#}"
370                )),
371            }
372        }
373    }
374}

376async fn evict_task(send: mpsc::Sender<Message>, options: Options, cancel: CancellationToken) {
377    let cancel2 = cancel.clone();
378    let _ = cancel2
379        .run_until_cancelled(async move {
380            info!("starting evict task");
381            if let Err(cause) = evict_task_inner(send, options).await {
382                error!("evict task failed: {:?}", cause);
383            }
384            // when we are done for whatever reason we want to shut down the actor
385            cancel.cancel();
386        })
387        .await;
388}
389
390/// Periodically check for expired packets and remove them.
391async fn evict_task_inner(send: mpsc::Sender<Message>, options: Options) -> Result<()> {
392    let expiry_us = options.eviction.as_micros() as u64;
393    loop {
394        let (tx, rx) = oneshot::channel();
395        let _ = send.send(Message::Snapshot { res: tx }).await.ok();
396        // if we can't get the snapshot we exit the loop, main actor dead
397        let snapshot = rx.await.context("failed to get snapshot")?;
398
399        let expired = Timestamp::now() - expiry_us;
400        trace!("evicting packets older than {}", fmt_time(expired));
401        // if getting the range fails we exit the loop and shut down
402        // if individual reads fail we log the error and limp on
403        for item in snapshot.update_time.range(..expired.to_bytes()).e()? {
404            let (time, keys) = match item {
405                Ok(v) => v,
406                Err(e) => {
407                    error!("failed to read update_time row {:?}", e);
408                    continue;
409                }
410            };
411            let time = Timestamp::from(time.value());
412            trace!("evicting expired packets at {}", fmt_time(time));
413            for item in keys {
414                let key = match item {
415                    Ok(v) => v,
416                    Err(e) => {
417                        error!(
418                            "failed to read update_time item at {}: {:?}",
419                            fmt_time(time),
420                            e
421                        );
422                        continue;
423                    }
424                };
425                let key = PublicKeyBytes::new(key.value());
426
427                debug!("evicting expired packet {} {}", fmt_time(time), key);
428                send.send(Message::CheckExpired { time, key }).await.e()?;
429            }
430        }
431        // sleep for the eviction interval so we don't constantly check
432        tokio::time::sleep(options.eviction_interval).await;
433    }
434}

436/// An io thread that drives a future to completion on the current tokio runtime
437///
438/// Inside the future, blocking IO can be done without blocking one of the tokio
439/// pool threads.
440#[derive(Debug)]
441struct IoThread {
442    handle: Option<std::thread::JoinHandle<()>>,
443}
444
445impl IoThread {
446    /// Spawn a new io thread.
447    ///
448    /// Calling this function requires that the current thread is running in a
449    /// tokio runtime. It is up to the caller to make sure the future exits,
450    /// e.g. by using a cancellation token. Otherwise, drop will block.
451    fn new<F, Fut>(name: &str, f: F) -> Result<Self>
452    where
453        F: FnOnce() -> Fut + Send + 'static,
454        Fut: Future<Output = ()>,
455    {
456        let rt = tokio::runtime::Handle::try_current().context("get tokio handle")?;
457        let handle = std::thread::Builder::new()
458            .name(name.into())
459            .spawn(move || rt.block_on(f()))
460            .context("failed to spawn thread")?;
461        Ok(Self {
462            handle: Some(handle),
463        })
464    }
465}
466
467impl Drop for IoThread {
468    fn drop(&mut self) {
469        if let Some(handle) = self.handle.take() {
470            let _ = handle.join();
471        }
472    }
473}

475/// A wrapper for a tokio mpsc receiver that allows peeking at the next message.
476#[derive(Debug)]
477pub(super) struct PeekableReceiver<T> {
478    msg: Option<T>,
479    recv: tokio::sync::mpsc::Receiver<T>,
480}
481
482#[allow(dead_code)]
483impl<T> PeekableReceiver<T> {
484    pub fn new(recv: tokio::sync::mpsc::Receiver<T>) -> Self {
485        Self { msg: None, recv }
486    }
487
488    /// Receive the next message.
489    ///
490    /// Will block if there are no messages.
491    /// Returns None only if there are no more messages (sender is dropped).
492    pub async fn recv(&mut self) -> Option<T> {
493        if let Some(msg) = self.msg.take() {
494            return Some(msg);
495        }
496        self.recv.recv().await
497    }
498
499    /// Push back a message. This will only work if there is room for it.
500    /// Otherwise, it will fail and return the message.
501    pub fn push_back(&mut self, msg: T) -> std::result::Result<(), T> {
502        if self.msg.is_none() {
503            self.msg = Some(msg);
504            Ok(())
505        } else {
506            Err(msg)
507        }
508    }
