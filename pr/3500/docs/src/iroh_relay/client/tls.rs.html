62    pub async fn connect(self) -> Result<MaybeTlsStream<ProxyStream>, ConnectError> {
63        let roots = rustls::RootCertStore {
64            roots: webpki_roots::TLS_SERVER_ROOTS.to_vec(),
65        };
66        let mut config = rustls::client::ClientConfig::builder_with_provider(Arc::new(
67            rustls::crypto::ring::default_provider(),
68        ))
69        .with_safe_default_protocol_versions()
70        .expect("protocols supported by ring")
71        .with_root_certificates(roots)
72        .with_no_client_auth();
73        #[cfg(any(test, feature = "test-utils"))]
74        if self.insecure_skip_cert_verify {
75            warn!("Insecure config: SSL certificates from relay servers not verified");
76            config
77                .dangerous()
78                .set_certificate_verifier(Arc::new(NoCertVerifier));
79        }
80        config.resumption = Resumption::default();
81        let tls_connector: tokio_rustls::TlsConnector = Arc::new(config).into();
82
83        let tcp_stream = self.dial_url(&tls_connector).await?;
84
85        let local_addr = tcp_stream
86            .local_addr()
87            .map_err(|_| NoLocalAddrSnafu.build())?;
88
89        debug!(server_addr = ?tcp_stream.peer_addr(), %local_addr, "TCP stream connected");
90
91        if self.use_tls() {
92            debug!("Starting TLS handshake");
93            let hostname = self
94                .tls_servername()
95                .ok_or_else(|| InvalidTlsServernameSnafu.build())?;
96
97            let hostname = hostname.to_owned();
98            let tls_stream = tls_connector
99                .connect(hostname, tcp_stream)
100                .await
101                .context(TlsSnafu)?;
102            debug!("tls_connector connect success");
103            Ok(MaybeTlsStream::Tls(tls_stream))
104        } else {
105            debug!("Starting handshake");
106            Ok(MaybeTlsStream::Raw(tcp_stream))
107        }
108    }

110    fn use_tls(&self) -> bool {
111        // only disable tls if we are explicitly dialing a http url
112        #[allow(clippy::match_like_matches_macro)]
113        match self.url.scheme() {
114            "http" => false,
115            "ws" => false,
116            _ => true,
117        }
118    }
119
120    fn tls_servername(&self) -> Option<rustls::pki_types::ServerName<'_>> {
121        let host_str = self.url.host_str()?;
122        let servername = rustls::pki_types::ServerName::try_from(host_str).ok()?;
123        Some(servername)
124    }

126    async fn dial_url(
127        &self,
128        tls_connector: &tokio_rustls::TlsConnector,
129    ) -> Result<ProxyStream, DialError> {
130        if let Some(ref proxy) = self.proxy_url {
131            let stream = self.dial_url_proxy(proxy.clone(), tls_connector).await?;
132            Ok(ProxyStream::Proxied(stream))
133        } else {
134            let stream = self.dial_url_direct().await?;
135            Ok(ProxyStream::Raw(stream))
136        }
137    }
138
139    async fn dial_url_direct(&self) -> Result<tokio::net::TcpStream, DialError> {
140        use tokio::net::TcpStream;
141        debug!(%self.url, "dial url");
142        let dst_ip = self
143            .dns_resolver
144            .resolve_host(&self.url, self.prefer_ipv6, DNS_TIMEOUT)
145            .await?;
146
147        let port = url_port(&self.url).context(InvalidTargetPortSnafu)?;
148        let addr = SocketAddr::new(dst_ip, port);
149
150        debug!("connecting to {}", addr);
151        let tcp_stream = time::timeout(
152            DIAL_NODE_TIMEOUT,
153            async move { TcpStream::connect(addr).await },
154        )
155        .await??;
156
157        tcp_stream.set_nodelay(true)?;
158
159        Ok(tcp_stream)
160    }

162    async fn dial_url_proxy(
163        &self,
164        proxy_url: Url,
165        tls_connector: &tokio_rustls::TlsConnector,
166    ) -> Result<util::Chain<std::io::Cursor<Bytes>, MaybeTlsStream<tokio::net::TcpStream>>, DialError>
167    {
168        use hyper_util::rt::TokioIo;
169        use tokio::net::TcpStream;
170        debug!(%self.url, %proxy_url, "dial url via proxy");
171
172        // Resolve proxy DNS
173        let proxy_ip = self
174            .dns_resolver
175            .resolve_host(&proxy_url, self.prefer_ipv6, DNS_TIMEOUT)
176            .await?;
177
178        let proxy_port = url_port(&proxy_url).context(ProxyInvalidTargetPortSnafu)?;
179        let proxy_addr = SocketAddr::new(proxy_ip, proxy_port);
180
181        debug!(%proxy_addr, "connecting to proxy");
182
183        let tcp_stream = time::timeout(DIAL_NODE_TIMEOUT, async move {
184            TcpStream::connect(proxy_addr).await
185        })
186        .await??;
187
188        tcp_stream.set_nodelay(true)?;
189
190        // Setup TLS if necessary
191        let io = if proxy_url.scheme() == "http" {
192            MaybeTlsStream::Raw(tcp_stream)
193        } else {
194            let hostname = proxy_url.host_str().context(ProxyInvalidUrlSnafu {
195                proxy_url: proxy_url.clone(),
196            })?;
197            let hostname =
198                rustls::pki_types::ServerName::try_from(hostname.to_string()).map_err(|_| {
199                    ProxyInvalidTlsServernameSnafu {
200                        proxy_hostname: hostname.to_string(),
201                    }
202                    .build()
203                })?;
204            let tls_stream = tls_connector.connect(hostname, tcp_stream).await?;
205            MaybeTlsStream::Tls(tls_stream)
206        };
207        let io = TokioIo::new(io);
208
209        let target_host = self.url.host_str().context(InvalidUrlSnafu {
210            url: self.url.clone(),
211        })?;
212
213        let port = url_port(&self.url).context(InvalidTargetPortSnafu)?;
214
215        // Establish Proxy Tunnel
216        let mut req_builder = Request::builder()
217            .uri(format!("{target_host}:{port}"))
218            .method("CONNECT")
219            .header("Host", target_host)
220            .header("Proxy-Connection", "Keep-Alive");
221        if !proxy_url.username().is_empty() {
222            // Passthrough authorization
223            // https://developer.mozilla.org/en-US/docs/Web/HTTP/Headers/Proxy-Authorization
224            debug!(
225                "setting proxy-authorization: username={}",
226                proxy_url.username()
227            );
228            let to_encode = format!(
229                "{}:{}",
230                proxy_url.username(),
231                proxy_url.password().unwrap_or_default()
232            );
233            let encoded = BASE64URL.encode(to_encode.as_bytes());
234            req_builder = req_builder.header("Proxy-Authorization", format!("Basic {encoded}"));
235        }
236        let req = req_builder
237            .body(Empty::<Bytes>::new())
238            .expect("fixed config");
239
240        debug!("Sending proxy request: {:?}", req);
241
242        let (mut sender, conn) = hyper::client::conn::http1::handshake(io)
243            .await
244            .context(ProxyConnectSnafu)?;
245        task::spawn(async move {
246            if let Err(err) = conn.with_upgrades().await {
247                error!("Proxy connection failed: {:?}", err);
248            }
249        });
250
251        let res = sender.send_request(req).await.context(ProxyConnectSnafu)?;
252        if !res.status().is_success() {
253            return Err(ProxyConnectInvalidStatusSnafu {
254                status: res.status(),
255            }
256            .build());
257        }
258
259        let upgraded = hyper::upgrade::on(res).await.context(ProxyConnectSnafu)?;
260        let Parts { io, read_buf, .. } = upgraded
261            .downcast::<TokioIo<MaybeTlsStream<tokio::net::TcpStream>>>()
262            .expect("only this upgrade used");
263
264        let res = util::chain(std::io::Cursor::new(read_buf), io.into_inner());
265
266        Ok(res)
267    }
268}

270fn url_port(url: &Url) -> Option<u16> {
271    if let Some(port) = url.port() {
272        return Some(port);
273    }
274
275    match url.scheme() {
276        "http" | "ws" => Some(80),
277        "https" | "wss" => Some(443),
278        _ => None,
279    }
