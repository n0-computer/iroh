1//! A discovery service that uses an mdns-like service to discover local nodes.
2//!
3//! This allows you to use an mdns-like swarm discovery service to find address information about nodes that are on your local network, no relay or outside internet needed.
4//! See the [`swarm-discovery`](https://crates.io/crates/swarm-discovery) crate for more details.
5//!
6//! When [`MdnsDiscovery`] is enabled, it's possible to get a list of the locally discovered nodes by filtering a list of `RemoteInfo`s.
7//!
8//! ```
9//! use std::time::Duration;
10//!
11//! use iroh::endpoint::{Endpoint, Source};
12//!
13//! #[tokio::main]
14//! async fn main() {
15//!     let recent = Duration::from_secs(600); // 10 minutes in seconds
16//!
17//!     let endpoint = Endpoint::builder().bind().await.unwrap();
18//!     let remotes = endpoint.remote_info_iter();
19//!     let locally_discovered: Vec<_> = remotes
20//!         .filter(|remote| {
21//!             remote.sources().iter().any(|(source, duration)| {
22//!                 if let Source::Discovery { name } = source {
23//!                     name == iroh::discovery::mdns::NAME && *duration <= recent
24//!                 } else {
25//!                     false
26//!                 }
27//!             })
28//!         })
29//!         .collect();
30//!     println!("locally discovered nodes: {locally_discovered:?}");
31//! }
32//! ```

70/// Discovery using `swarm-discovery`, a variation on mdns
71#[derive(Debug)]
72pub struct MdnsDiscovery {
73    #[allow(dead_code)]
74    handle: AbortOnDropHandle<()>,
75    sender: mpsc::Sender<Message>,
76    advertise: bool,
77    /// When `local_addrs` changes, we re-publish our info.
78    local_addrs: Watchable<Option<NodeData>>,
79}

81#[derive(Debug)]
82enum Message {
83    Discovery(String, Peer),
84    Resolve(NodeId, mpsc::Sender<Result<DiscoveryItem, DiscoveryError>>),
85    Timeout(NodeId, usize),
86    Subscribe(mpsc::Sender<DiscoveryEvent>),
87}
88
89/// Manages the list of subscribers that are subscribed to this discovery service.
90#[derive(Debug)]
91struct Subscribers(Vec<mpsc::Sender<DiscoveryEvent>>);
92
93impl Subscribers {
94    fn new() -> Self {
95        Self(vec![])
96    }
97
98    /// Add the subscriber to the list of subscribers
99    fn push(&mut self, subscriber: mpsc::Sender<DiscoveryEvent>) {
100        self.0.push(subscriber);
101    }
102
103    /// Sends the `node_id` and `item` to each subscriber.
104    ///
105    /// Cleans up any subscribers that have been dropped.
106    fn send(&mut self, item: DiscoveryEvent) {
107        let mut clean_up = vec![];
108        for (i, subscriber) in self.0.iter().enumerate() {
109            // assume subscriber was dropped
110            if let Err(err) = subscriber.try_send(item.clone()) {
111                match err {
112                    TrySendError::Full(_) => {
113                        warn!(
114                            ?item,
115                            idx = i,
116                            "local swarm discovery subscriber is blocked, dropping item"
117                        )
118                    }
119                    TrySendError::Closed(_) => clean_up.push(i),
120                }
121            }
122        }
123        for i in clean_up.into_iter().rev() {
124            self.0.swap_remove(i);
125        }
126    }
127}

176    /// Create a new [`MdnsDiscovery`] Service.
177    ///
178    /// This starts a [`Discoverer`] that broadcasts your addresses (if advertise is set to true)
179    /// and receives addresses from other nodes in your local network.
180    ///
181    /// # Errors
182    /// Returns an error if the network does not allow ipv4 OR ipv6.
183    ///
184    /// # Panics
185    /// This relies on [`tokio::runtime::Handle::current`] and will panic if called outside of the context of a tokio runtime.
186    pub fn new(node_id: NodeId, advertise: bool) -> Result<Self, IntoDiscoveryError> {
187        debug!("Creating new MdnsDiscovery service");
188        let (send, mut recv) = mpsc::channel(64);
189        let task_sender = send.clone();
190        let rt = tokio::runtime::Handle::current();
191        let discovery = MdnsDiscovery::spawn_discoverer(
192            node_id,
193            advertise,
194            task_sender.clone(),
195            BTreeSet::new(),
196            &rt,
197        )?;
198
199        let local_addrs: Watchable<Option<NodeData>> = Watchable::default();
200        let mut addrs_change = local_addrs.watch();
201        let discovery_fut = async move {
202            let mut node_addrs: HashMap<PublicKey, Peer> = HashMap::default();
203            let mut subscribers = Subscribers::new();
204            let mut last_id = 0;
205            let mut senders: HashMap<
206                PublicKey,
207                HashMap<usize, mpsc::Sender<Result<DiscoveryItem, DiscoveryError>>>,
208            > = HashMap::default();
209            let mut timeouts = JoinSet::new();
210            loop {
211                trace!(?node_addrs, "MdnsDiscovery Service loop tick");
212                let msg = tokio::select! {
213                    msg = recv.recv() => {
214                        msg
215                    }
216                    Ok(Some(data)) = addrs_change.updated() => {
217                        tracing::trace!(?data, "MdnsDiscovery address changed");
218                        discovery.remove_all();
219                        let addrs =
220                            MdnsDiscovery::socketaddrs_to_addrs(data.direct_addresses());
221                        for addr in addrs {
222                            discovery.add(addr.0, addr.1)
223                        }
224                        if let Some(user_data) = data.user_data() {
225                            if let Err(err) = discovery.set_txt_attribute(USER_DATA_ATTRIBUTE.to_string(), Some(user_data.to_string())) {
226                                warn!("Failed to set the user-defined data in local swarm discovery: {err:?}");
227                            }
228                        }
229                        continue;
230                    }
231                };
232                let msg = match msg {
233                    None => {
234                        error!("MdnsDiscovery channel closed");
235                        error!("closing MdnsDiscovery");
236                        timeouts.abort_all();
237                        discovery.remove_all();
238                        return;
239                    }
240                    Some(msg) => msg,
241                };
242                match msg {
243                    Message::Discovery(discovered_node_id, peer_info) => {
244                        trace!(
245                            ?discovered_node_id,
246                            ?peer_info,
247                            "MdnsDiscovery Message::Discovery"
248                        );
249                        let discovered_node_id = match PublicKey::from_str(&discovered_node_id) {
250                            Ok(node_id) => node_id,
251                            Err(e) => {
252                                warn!(
253                                    discovered_node_id,
254                                    "couldn't parse node_id from mdns discovery service: {e:?}"
255                                );
256                                continue;
257                            }
258                        };
259
260                        if discovered_node_id == node_id {
261                            continue;
262                        }
263
264                        if peer_info.is_expiry() {
265                            trace!(
266                                ?discovered_node_id,
267                                "removing node from MdnsDiscovery address book"
268                            );
269                            node_addrs.remove(&discovered_node_id);
270                            subscribers.send(DiscoveryEvent::Expired(discovered_node_id));
271                            continue;
272                        }
273
274                        let entry = node_addrs.entry(discovered_node_id);
275                        if let std::collections::hash_map::Entry::Occupied(ref entry) = entry {
276                            if entry.get() == &peer_info {
277                                // this is a republish we already know about
278                                continue;
279                            }
280                        }
281
282                        debug!(
283                            ?discovered_node_id,
284                            ?peer_info,
285                            "adding node to MdnsDiscovery address book"
286                        );
287
288                        let mut resolved = false;
289                        let item = peer_to_discovery_item(&peer_info, &discovered_node_id);
290                        if let Some(senders) = senders.get(&discovered_node_id) {
291                            trace!(?item, senders = senders.len(), "sending DiscoveryItem");
292                            resolved = true;
293                            for sender in senders.values() {
294                                sender.send(Ok(item.clone())).await.ok();
295                            }
296                        }
297                        entry.or_insert(peer_info);
298
299                        // only send nodes to the `subscriber` if they weren't explicitly resolved
300                        // in other words, nodes sent to the `subscribers` should only be the ones that
301                        // have been "passively" discovered
302                        if !resolved {
303                            subscribers.send(DiscoveryEvent::Discovered(item));
304                        }
305                    }
306                    Message::Resolve(node_id, sender) => {
307                        let id = last_id + 1;
308                        last_id = id;
309                        trace!(?node_id, "MdnsDiscovery Message::SendAddrs");
310                        if let Some(peer_info) = node_addrs.get(&node_id) {
311                            let item = peer_to_discovery_item(peer_info, &node_id);
312                            debug!(?item, "sending DiscoveryItem");
313                            sender.send(Ok(item)).await.ok();
314                        }
315                        if let Some(senders_for_node_id) = senders.get_mut(&node_id) {
316                            senders_for_node_id.insert(id, sender);
317                        } else {
318                            let mut senders_for_node_id = HashMap::new();
319                            senders_for_node_id.insert(id, sender);
320                            senders.insert(node_id, senders_for_node_id);
321                        }
322                        let timeout_sender = task_sender.clone();
323                        timeouts.spawn(async move {
324                            time::sleep(DISCOVERY_DURATION).await;
325                            trace!(?node_id, "discovery timeout");
326                            timeout_sender
327                                .send(Message::Timeout(node_id, id))
328                                .await
329                                .ok();
330                        });
331                    }
332                    Message::Timeout(node_id, id) => {
333                        trace!(?node_id, "MdnsDiscovery Message::Timeout");
334                        if let Some(senders_for_node_id) = senders.get_mut(&node_id) {
335                            senders_for_node_id.remove(&id);
336                            if senders_for_node_id.is_empty() {
337                                senders.remove(&node_id);
338                            }
339                        }
340                    }
341                    Message::Subscribe(subscriber) => {
342                        trace!("MdnsDiscovery Message::Subscribe");
343                        subscribers.push(subscriber);
344                    }
345                }
346            }
347        };
348        let handle = task::spawn(discovery_fut.instrument(info_span!("swarm-discovery.actor")));
349        Ok(Self {
350            handle: AbortOnDropHandle::new(handle),
351            sender: send,
352            advertise,
353            local_addrs,
354        })
355    }

357    fn spawn_discoverer(
358        node_id: PublicKey,
359        advertise: bool,
360        sender: mpsc::Sender<Message>,
361        socketaddrs: BTreeSet<SocketAddr>,
362        rt: &tokio::runtime::Handle,
363    ) -> Result<DropGuard, IntoDiscoveryError> {
364        let spawn_rt = rt.clone();
365        let callback = move |node_id: &str, peer: &Peer| {
366            trace!(
367                node_id,
368                ?peer,
369                "Received peer information from MdnsDiscovery"
370            );
371
372            let sender = sender.clone();
373            let node_id = node_id.to_string();
374            let peer = peer.clone();
375            spawn_rt.spawn(async move {
376                sender.send(Message::Discovery(node_id, peer)).await.ok();
377            });
378        };
379        let node_id_str = data_encoding::BASE32_NOPAD
380            .encode(node_id.as_bytes())
381            .to_ascii_lowercase();
382        let mut discoverer = Discoverer::new_interactive(N0_LOCAL_SWARM.to_string(), node_id_str)
383            .with_callback(callback)
384            .with_ip_class(IpClass::Auto);
385        if advertise {
386            let addrs = MdnsDiscovery::socketaddrs_to_addrs(&socketaddrs);
387            for addr in addrs {
388                discoverer = discoverer.with_addrs(addr.0, addr.1);
389            }
390        }
391        discoverer
392            .spawn(rt)
393            .map_err(|e| IntoDiscoveryError::from_err("mdns", e))
394    }
395
396    fn socketaddrs_to_addrs(socketaddrs: &BTreeSet<SocketAddr>) -> HashMap<u16, Vec<IpAddr>> {
397        let mut addrs: HashMap<u16, Vec<IpAddr>> = HashMap::default();
398        for socketaddr in socketaddrs {
399            addrs
400                .entry(socketaddr.port())
401                .and_modify(|a| a.push(socketaddr.ip()))
402                .or_insert(vec![socketaddr.ip()]);
403        }
404        addrs
405    }
406}

408fn peer_to_discovery_item(peer: &Peer, node_id: &NodeId) -> DiscoveryItem {
409    let direct_addresses: BTreeSet<SocketAddr> = peer
410        .addrs()
411        .iter()
412        .map(|(ip, port)| SocketAddr::new(*ip, *port))
413        .collect();
414    // Get the user-defined data from the resolved peer info. We expect an attribute with a value
415    // that parses as `UserData`. Otherwise, omit.
416    let user_data = if let Some(Some(user_data)) = peer.txt_attribute(USER_DATA_ATTRIBUTE) {
417        match user_data.parse() {
418            Err(err) => {
419                debug!("failed to parse user data from TXT attribute: {err}");
420                None
421            }
422            Ok(data) => Some(data),
423        }
424    } else {
425        None
426    };
427    let node_info = NodeInfo::new(*node_id)
428        .with_direct_addresses(direct_addresses)
429        .with_user_data(user_data);
430    DiscoveryItem::new(node_info, NAME, None)
431}

433impl Discovery for MdnsDiscovery {
434    fn resolve(&self, node_id: NodeId) -> Option<BoxStream<Result<DiscoveryItem, DiscoveryError>>> {
435        use futures_util::FutureExt;
436
437        let (send, recv) = mpsc::channel(20);
438        let discovery_sender = self.sender.clone();
439        let stream = async move {
440            discovery_sender
441                .send(Message::Resolve(node_id, send))
442                .await
443                .ok();
444            tokio_stream::wrappers::ReceiverStream::new(recv)
445        };
446        Some(Box::pin(stream.flatten_stream()))
447    }
448
449    fn publish(&self, data: &NodeData) {
450        if self.advertise {
451            self.local_addrs.set(Some(data.clone())).ok();
452        }
453    }
454
455    fn subscribe(&self) -> Option<BoxStream<DiscoveryEvent>> {
456        use futures_util::FutureExt;
457
458        let (sender, recv) = mpsc::channel(20);
459        let discovery_sender = self.sender.clone();
460        let stream = async move {
461            discovery_sender.send(Message::Subscribe(sender)).await.ok();
462            tokio_stream::wrappers::ReceiverStream::new(recv)
463        };
464        Some(Box::pin(stream.flatten_stream()))
465    }
466}

654                "Expected timeout since node b isn't advertising"
655            );
656
657            Ok(())
658        }
659
660        fn make_discoverer(advertise: bool) -> Result<(PublicKey, MdnsDiscovery)> {
661            let node_id = SecretKey::generate(rand::thread_rng()).public();
662            Ok((node_id, MdnsDiscovery::new(node_id, advertise)?))
663        }
664    }
