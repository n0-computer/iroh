1//! Implements the handshake protocol that authenticates and authorizes clients connecting to the relays.
2//!
3//! The purpose of the handshake is to
4//! 1. Inform the relay of the client's NodeId
5//! 2. Check that the connecting client owns the secret key for its NodeId ("is authentic"/"authentication")
6//! 3. Possibly check that the client has access to this relay, if the relay requires authorization.
7//!
8//! Additional complexity comes from the fact that there's two ways that clients can authenticate with
9//! relays.
10//!
11//! One way is via an explicitly sent challenge:
12//!
13//! 1. Once a websocket connection is opened, a client receives a challenge (the `ServerChallenge` frame)
14//! 2. The client sends back what is essentially a signature of that challenge with their secret key
15//!    that matches the NodeId they have, as well as the NodeId (the `ClientAuth` frame)
16//!
17//! The second way is very similar to the [Concealed HTTP Auth RFC], and involves send a header that
18//! contains a signature of some shared keying material extracted from TLS ([RFC 5705]).
19//!
20//! The second way can save a full round trip, because the challenge doesn't have to be sent to the client
21//! first, however, it won't always work, as it relies on the keying material extraction feature of TLS,
22//! which is not available in browsers (but might be in the future?) and might break when there's an
23//! HTTPS proxy that doesn't properly deal with this TLS feature.
24//!
25//! [Concealed HTTP Auth RFC]: https://datatracker.ietf.org/doc/rfc9729/
26//! [RFC 5705]: https://datatracker.ietf.org/doc/html/rfc5705

173#[cfg(feature = "server")]

175pub(crate) enum VerificationError {
176    #[snafu(display("Couldn't export TLS keying material on our end"))]
177    NoKeyingMaterial,

182        expected: [u8; 16],
183        actual: [u8; 16],
184    },

186        "Client signature {signature:X?} for message {message:X?} invalid for public key {public_key}"
187    ))]

189        source: iroh_base::SignatureError,
190        message: Vec<u8>,
191        signature: [u8; 64],
192        public_key: PublicKey,
193    },

196impl ServerChallenge {
197    /// Generates a new challenge.
198    #[cfg(feature = "server")]
199    pub(crate) fn new(mut rng: impl RngCore + CryptoRng) -> Self {
200        let mut challenge = [0u8; 16];
201        rng.fill_bytes(&mut challenge);
202        Self { challenge }
203    }
204
205    /// The actual message bytes to sign (and verify against) for this challenge.
206    fn message_to_sign(&self) -> [u8; 32] {
207        // We're signing a key instead of the direct challenge.
208        // This gives us domain separation protecting from multiple possible attacks,
209        // but especially this one:
210        // Assume a malicious relay. If the protocol required the client to sign the
211        // challenge directly, this would allow the relay to obtain an arbitrary 16-byte
212        // signature, if it maliciously choses the challenge instead of generating it
213        // randomly.
214        // Deriving a key to sign instead mitigates this attack.
215        blake3::derive_key(DOMAIN_SEP_CHALLENGE, &self.challenge)
216    }
217}
218
219impl ClientAuth {
220    /// Generates a signature for the given challenge from the server.
221    pub(crate) fn new(secret_key: &SecretKey, challenge: &ServerChallenge) -> Self {
222        Self {
223            public_key: secret_key.public(),
224            signature: secret_key.sign(&challenge.message_to_sign()).to_bytes(),
225        }
226    }
227
228    /// Verifies this client's authentication given the challenge this was sent in response to.
229    #[cfg(feature = "server")]
230    pub(crate) fn verify(&self, challenge: &ServerChallenge) -> Result<(), Box<VerificationError>> {
231        let message = challenge.message_to_sign();
232        self.public_key
233            .verify(&message, &Signature::from_bytes(&self.signature))
234            .with_context(|_| SignatureInvalidSnafu {
235                message: message.to_vec(),
236                signature: self.signature,
237                public_key: self.public_key,
238            })
239            .map_err(Box::new)
240    }
241}

243#[cfg(not(wasm_browser))]
244impl KeyMaterialClientAuth {
245    /// Generates a client's authentication, similar to [`ClientAuth`], but by using TLS keying material
246    /// instead of a received challenge.
247    pub(crate) fn new(secret_key: &SecretKey, io: &impl ExportKeyingMaterial) -> Option<Self> {
248        let public_key = secret_key.public();
249        let key_material = io.export_keying_material(
250            [0u8; 32],
251            DOMAIN_SEP_TLS_EXPORT_LABEL,
252            Some(secret_key.public().as_bytes()),
253        )?;
254        // We split the export and only sign the first 16 bytes, and
255        // pass through the last 16 bytes. See also the note in [Self::verify].
256        let (message, suffix) = key_material.split_at(16);
257        Some(Self {
258            public_key,
259            signature: secret_key.sign(message).to_bytes(),
260            key_material_suffix: suffix.try_into().expect("hardcoded length"),
261        })
262    }
263
264    /// Generate the base64url-nopad-encoded header value.
265    pub(crate) fn into_header_value(self) -> HeaderValue {
266        HeaderValue::from_str(
267            &data_encoding::BASE64URL_NOPAD
268                .encode(&postcard::to_allocvec(&self).expect("encoding never fails")),
269        )
270        .expect("BASE64URL_NOPAD encoding contained invisible ascii characters")
271    }
272
273    /// Verifies this client auth on the server side using the same key material.
274    ///
275    /// This might return false for a couple of reasons:
276    /// 1. The exported keying material might not be the same between both ends of the TLS session
277    ///    (e.g. there's an HTTPS proxy in between that doesn't think/care about the TLS keying material exporter).
278    ///    This situation is detected when the key material suffix mismatches.
279    /// 2. The signature itself doesn't verify.
280    #[cfg(feature = "server")]
281    pub(crate) fn verify(
282        &self,
283        io: &impl ExportKeyingMaterial,
284    ) -> Result<(), Box<VerificationError>> {
285        use snafu::OptionExt;
286
287        let key_material = io
288            .export_keying_material(
289                [0u8; 32],
290                DOMAIN_SEP_TLS_EXPORT_LABEL,
291                Some(self.public_key.as_bytes()),
292            )
293            .context(NoKeyingMaterialSnafu)?;
294        // We split the export and only sign the first 16 bytes, and
295        // pass through the last 16 bytes.
296        // Passing on the suffix helps the verifying end figure out what
297        // went wrong: If there's a suffix mismatch, then the exported keying
298        // material on both ends wasn't the same - so perhaps there was a
299        // TLS proxy in between or similar.
300        // If the suffix does match, but the signature doesn't verify, then
301        // there must be something wrong with the client's secret key or signature.
302        let (message, suffix) = key_material.split_at(16);
303        let suffix: [u8; 16] = suffix.try_into().expect("hardcoded length");
304        snafu::ensure!(
305            suffix == self.key_material_suffix,
306            MismatchedSuffixSnafu {
307                expected: self.key_material_suffix,
308                actual: suffix
309            }
310        );
311        // NOTE: We don't blake3-hash here as we do it in [`ServerChallenge::message_to_sign`],
312        // because we already have a domain separation string and keyed hashing step in
313        // the TLS export keying material above.
314        self.public_key
315            .verify(message, &Signature::from_bytes(&self.signature))
316            .with_context(|_| SignatureInvalidSnafu {
317                message: message.to_vec(),
318                public_key: self.public_key,
319                signature: self.signature,
320            })
321            .map_err(Box::new)
322    }
323}

325/// Runs the client side of the handshake protocol.
326///
327/// See the module docs for details on the protocol.
328/// This is already after having potentially transferred a [`KeyMaterialClientAuth`],
329/// but before having received a response for whether that worked or not.
330///
331/// This requires access to the client's secret key to sign a challenge.
332pub(crate) async fn clientside(
333    io: &mut (impl BytesStreamSink + ExportKeyingMaterial),
334    secret_key: &SecretKey,
335) -> Result<ServerConfirmsAuth, Error> {
336    let (tag, frame) = read_frame(io, &[ServerChallenge::TAG, ServerConfirmsAuth::TAG]).await?;
337
338    let (tag, frame) = if tag == ServerChallenge::TAG {
339        let challenge: ServerChallenge = deserialize_frame(frame)?;
340
341        let client_info = ClientAuth::new(secret_key, &challenge);
342        write_frame(io, client_info).await?;
343
344        read_frame(io, &[ServerConfirmsAuth::TAG, ServerDeniesAuth::TAG]).await?
345    } else {
346        (tag, frame)
347    };
348
349    match tag {
350        FrameType::ServerConfirmsAuth => {
351            let confirmation: ServerConfirmsAuth = deserialize_frame(frame)?;
352            Ok(confirmation)
353        }
354        FrameType::ServerDeniesAuth => {
355            let denial: ServerDeniesAuth = deserialize_frame(frame)?;
356            Err(ServerDeniedAuthSnafu {
357                reason: denial.reason,
358            }
359            .build())
360        }
361        _ => unreachable!(),
362    }
363}

365/// This represents successful authentication for the client with the `client_key` public key
366/// via the authentication [`Mechanism`] `mechanism`.
367///
368/// You must call [`SuccessfulAuthentication::authorize_if`] to finish the protocol.
369#[cfg(feature = "server")]
370#[derive(Debug)]
371#[must_use = "the protocol is not finished unless `authorize_if` is called"]
372pub(crate) struct SuccessfulAuthentication {
373    pub(crate) client_key: PublicKey,
374    pub(crate) mechanism: Mechanism,
375}
376
377/// The mechanism that was used for authentication.
378#[cfg(feature = "server")]
379#[derive(Debug, Clone, Copy, PartialEq, Eq)]
380pub(crate) enum Mechanism {
381    /// Authentication was performed by verifying a signature of a challenge we sent
382    SignedChallenge,
383    /// Authentication was performed by verifying a signature of shared extracted TLS keying material
384    SignedKeyMaterial,
385}
386
387/// Runs the server side of the handshaking protocol.
388///
389/// See the module documentation for an overview of the handshaking protocol.
390///
391/// This takes `rng` to generate cryptographic randomness for the authentication challenge.
392///
393/// This also takes the `client_auth_header`, if present, to perform authentication without
394/// requiring sending a challenge, saving a round-trip, if possible.
395///
396/// If this fails, the protocol falls back to doing a normal extra round trip with a challenge.
397///
398/// The return value [`SuccessfulAuthentication`] still needs to be resolved by calling
399/// [`SuccessfulAuthentication::authorize_if`] to finish the whole authorization protocol
400/// (otherwise the client won't be notified about auth success or failure).
401#[cfg(feature = "server")]
402pub(crate) async fn serverside(
403    io: &mut (impl BytesStreamSink + ExportKeyingMaterial),
404    client_auth_header: Option<HeaderValue>,
405    rng: impl RngCore + CryptoRng,
406) -> Result<SuccessfulAuthentication, Error> {
407    if let Some(client_auth_header) = client_auth_header {
408        let client_auth_bytes = data_encoding::BASE64URL_NOPAD
409            .decode(client_auth_header.as_ref())
410            .map_err(|_| {
411                ClientAuthHeaderInvalidSnafu {
412                    value: client_auth_header.clone(),
413                }
414                .build()
415            })?;
416
417        let client_auth: KeyMaterialClientAuth =
418            postcard::from_bytes(&client_auth_bytes).map_err(|_| {
419                ClientAuthHeaderInvalidSnafu {
420                    value: client_auth_header.clone(),
421                }
422                .build()
423            })?;
424
425        if client_auth.verify(io).is_ok() {
426            trace!(?client_auth.public_key, "authentication succeeded via keying material");
427            return Ok(SuccessfulAuthentication {
428                client_key: client_auth.public_key,
429                mechanism: Mechanism::SignedKeyMaterial,
430            });
431        }
432        // Verification not succeeding is part of normal operation: The TLS exporter isn't required to match.
433        // We'll fall back to verification that takes another round trip more time.
434    }
435
436    let challenge = ServerChallenge::new(rng);
437    write_frame(io, &challenge).await?;
438
439    let (_, frame) = read_frame(io, &[ClientAuth::TAG]).await?;
440    let client_auth: ClientAuth = deserialize_frame(frame)?;
441
442    if let Err(err) = client_auth.verify(&challenge) {
443        trace!(?client_auth.public_key, ?err, "authentication failed");
444        let denial = ServerDeniesAuth {
445            reason: "signature invalid".into(),
446        };
447        write_frame(io, denial.clone()).await?;
448        ServerDeniedAuthSnafu {
449            reason: denial.reason,
450        }
451        .fail()
452    } else {
453        trace!(?client_auth.public_key, "authentication succeeded via challenge");
454        Ok(SuccessfulAuthentication {
455            client_key: client_auth.public_key,
456            mechanism: Mechanism::SignedChallenge,
457        })
458    }
459}

461#[cfg(feature = "server")]
462impl SuccessfulAuthentication {
463    pub async fn authorize_if(
464        self,
465        is_authorized: bool,
466        io: &mut (impl BytesStreamSink + ExportKeyingMaterial),
467    ) -> Result<PublicKey, Error> {
468        if is_authorized {
469            trace!("authorizing client");
470            write_frame(io, ServerConfirmsAuth).await?;
471            Ok(self.client_key)
472        } else {
473            trace!("denying client auth");
474            let denial = ServerDeniesAuth {
475                reason: "not authorized".into(),
476            };
477            write_frame(io, denial.clone()).await?;
478            ServerDeniedAuthSnafu {
479                reason: denial.reason,
480            }
481            .fail()
482        }
483    }
484}

486async fn write_frame<F: serde::Serialize + Frame>(
487    io: &mut impl BytesStreamSink,
488    frame: F,
489) -> Result<(), Error> {
490    let mut bytes = BytesMut::new();
491    trace!(frame_type = ?F::TAG, "Writing frame");
492    F::TAG.write_to(&mut bytes);
493    let bytes = postcard::to_io(&frame, bytes.writer())
494        .expect("serialization failed") // buffer can't become "full" without being a critical failure, datastructures shouldn't ever fail serialization
495        .into_inner()
496        .freeze();
497    io.send(bytes).await?;
498    io.flush().await?;
499    Ok(())
500}
501
502async fn read_frame(
503    io: &mut impl BytesStreamSink,
504    expected_types: &[FrameType],
505) -> Result<(FrameType, Bytes), Error> {
506    let mut payload = io
507        .try_next()
508        .await?
509        .ok_or_else(|| UnexpectedEndSnafu.build())?;
510
511    let frame_type = FrameType::from_bytes(&mut payload)?;
512    trace!(?frame_type, "Reading frame");
513    snafu::ensure!(
514        expected_types.contains(&frame_type),
515        UnexpectedFrameTypeSnafu {
516            frame_type,
517            expected_types: expected_types.to_vec()
518        }
519    );
520
521    Ok((frame_type, payload))
522}
523
524fn deserialize_frame<F: Frame + serde::de::DeserializeOwned>(frame: Bytes) -> Result<F, Error> {
525    postcard::from_bytes(&frame).context(DeserializationSnafu { frame_type: F::TAG })
526}

627
628        let mut client_io = Framed::new(client, LengthDelimitedCodec::new())
629            .map_ok(BytesMut::freeze)
630            .map_err(tokio_websockets::Error::Io)
631            .sink_map_err(tokio_websockets::Error::Io)
632            .with_shared_secret(client_shared_secret);
633        let mut server_io = Framed::new(server, LengthDelimitedCodec::new())
634            .map_ok(BytesMut::freeze)
635            .map_err(tokio_websockets::Error::Io)
636            .sink_map_err(tokio_websockets::Error::Io)
637            .with_shared_secret(server_shared_secret);
638
639        let client_auth_header = KeyMaterialClientAuth::new(secret_key, &client_io)
640            .map(KeyMaterialClientAuth::into_header_value);
641
642        n0_future::future::zip(
643            async {
644                super::clientside(&mut client_io, secret_key)
645                    .await
646                    .context("clientside")
647            }
648            .instrument(info_span!("clientside")),
649            async {
650                let auth_n =
651                    super::serverside(&mut server_io, client_auth_header, rand::rngs::OsRng)
652                        .await
653                        .context("serverside")?;
654                let mechanism = auth_n.mechanism;
655                let is_authorized = restricted_to.is_none_or(|key| key == auth_n.client_key);
656                let key = auth_n.authorize_if(is_authorized, &mut server_io).await?;
657                Ok((key, mechanism))
658            }
659            .instrument(info_span!("serverside")),
660        )
661        .await
662    }
